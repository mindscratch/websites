  1669		d := a - b
  1670		if d < 0 {
  1671			d = -d
  1672		}
  1673	
  1674		if a != 0 {
  1675			e = e * a
  1676			if e < 0 {
  1677				e = -e
  1678			}
  1679		}
  1680		return d < e
  1681	}
