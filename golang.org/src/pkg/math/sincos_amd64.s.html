     5	#include "../../cmd/ld/textflag.h"

     7	// The method is based on a paper by Naoki Shibata: "Efficient evaluation
     8	// methods of elementary functions suitable for SIMD computation", Proc.
     9	// of International Supercomputing Conference 2010 (ISC'10), pp. 25 -- 32
    10	// (May 2010). The paper is available at
    11	// http://www.springerlink.com/content/340228x165742104/
    12	//
    13	// The original code and the constants below are from the author's
    14	// implementation available at http://freshmeat.net/projects/sleef.
    15	// The README file says, "The software is in public domain.
    16	// You can use the software without any obligation."
    17	//
    18	// This code is a simplified version of the original.  The CMPSD
    19	// instruction, not generated by the compiler, eliminates jumps in the
    20	// body of the calculation.
    21	
    22	#define PosOne 0x3FF0000000000000
    23	#define PosInf 0x7FF0000000000000
    24	#define NaN    0x7FF8000000000001
    25	#define PI4A 0.7853981554508209228515625 // pi/4 split into three parts
    26	#define PI4B 0.794662735614792836713604629039764404296875e-8
    27	#define PI4C 0.306161699786838294306516483068750264552437361480769e-16
    28	#define M4PI 1.273239544735162542821171882678754627704620361328125 // 4/pi
    29	#define T0 1.0
    30	#define T1 -8.33333333333333333333333e-02 // (-1.0/12)
    31	#define T2 2.77777777777777777777778e-03 // (+1.0/360)
    32	#define T3 -4.96031746031746031746032e-05 // (-1.0/20160)
    33	#define T4 5.51146384479717813051146e-07 // (+1.0/1814400)
    34	
    35	// func Sincos(d float64) (sin, cos float64)
    36	TEXT ·Sincos(SB),NOSPLIT,$0
    37		// test for special cases
    38		MOVQ    $~(1<<63), DX // sign bit mask
    39		MOVQ    x+0(FP), BX
    40		ANDQ    BX, DX
    41		JEQ     isZero
    42		MOVQ    $PosInf, AX
    43		CMPQ    AX, DX
    44		JLE     isInfOrNaN
    45		// Reduce argument
    46		MOVQ    BX, X7 // x7= d
    47		MOVQ    DX, X0 // x0= |d|
    48		MOVSD   $M4PI, X2
    49		MULSD   X0, X2
    50		CVTTSD2SQ X2, BX // bx= q
    51		MOVQ    $1, AX
    52		ANDQ    BX, AX
    53		ADDQ    BX, AX
    54		CVTSQ2SD AX, X2
    55		MOVSD   $PI4A, X3
    56		MULSD   X2, X3
    57		SUBSD   X3, X0
    58		MOVSD   $PI4B, X3
    59		MULSD   X2, X3
    60		SUBSD   X3, X0
    61		MOVSD   $PI4C, X3
    62		MULSD   X2, X3
    63		SUBSD   X3, X0
    64		MULSD   $0.125, X0 // x0= x, x7= d, bx= q
    65		// Evaluate Taylor series
    66		MULSD   X0, X0
    67		MOVSD   $T4, X2
    68		MULSD   X0, X2
    69		ADDSD   $T3, X2
    70		MULSD   X0, X2
    71		ADDSD   $T2, X2
    72		MULSD   X0, X2
    73		ADDSD   $T1, X2
    74		MULSD   X0, X2
    75		ADDSD   $T0, X2
    76		MULSD   X2, X0 // x0= x, x7= d, bx= q
    77		// Apply double angle formula
    78		MOVSD   $4.0, X2
    79		SUBSD   X0, X2
    80		MULSD   X2, X0
    81		MOVSD   $4.0, X2
    82		SUBSD   X0, X2
    83		MULSD   X2, X0
    84		MOVSD   $4.0, X2
    85		SUBSD   X0, X2
    86		MULSD   X2, X0
    87		MULSD   $0.5, X0 // x0= x, x7= d, bx= q
    88		// sin = sqrt((2 - x) * x)
    89		MOVSD   $2.0, X2
    90		SUBSD   X0, X2
    91		MULSD   X0, X2
    92		SQRTSD  X2, X2 // x0= x, x2= z, x7= d, bx= q
    93		// cos = 1 - x
    94		MOVSD   $1.0, X1
    95		SUBSD   X0, X1 // x1= x, x2= z, x7= d, bx= q
    96		// if ((q + 1) & 2) != 0 { sin, cos = cos, sin }
    97		MOVQ    $1, DX
    98		ADDQ    BX, DX
    99		MOVQ    $2, AX
   100		ANDQ    AX, DX
   101		MOVQ    DX, X0
   102		MOVSD   $0.0, X3
   103		CMPSD   X0, X3, 0 // cmpeq; x1= x, x2= z, x3 = y, x7= d, bx= q
   104		// sin = (y & z) | (^y & x)
   105		MOVAPD  X2, X0
   106		ANDPD   X3, X0 // x0= sin
   107		MOVAPD  X3, X4
   108		ANDNPD  X1, X4
   109		ORPD    X4, X0 // x0= sin, x1= x, x2= z, x3= y, x7= d, bx= q
   110		// cos = (y & x) | (^y & z)
   111		ANDPD   X3, X1 // x1= cos
   112		ANDNPD  X2, X3
   113		ORPD    X3, X1 // x0= sin, x1= cos, x7= d, bx= q
   114		// if ((q & 4) != 0) != (d < 0) { sin = -sin }
   115		MOVQ    BX, AX
   116		MOVQ    $61, CX
   117		SHLQ    CX, AX
   118		MOVQ    AX, X3
   119		XORPD   X7, X3
   120		MOVQ    $(1<<63), AX
   121		MOVQ    AX, X2 // x2= -0.0
   122		ANDPD   X2, X3
   123		ORPD    X3, X0 // x0= sin, x1= cos, x2= -0.0, bx= q
   124		// if ((q + 2) & 4) != 0 { cos = -cos }
   125		MOVQ    $2, AX
   126		ADDQ    AX, BX
   127		MOVQ    $61, CX
   128		SHLQ    CX, BX
   129		MOVQ    BX, X3
   130		ANDPD   X2, X3
   131		ORPD    X3, X1 // x0= sin, x1= cos
   132		// return (sin, cos)
   133		MOVSD   X0, sin+8(FP)
   134		MOVSD   X1, cos+16(FP)
   135		RET
   136	isZero: // return (±0.0, 1.0)
   137		MOVQ    BX, sin+8(FP)
   138		MOVQ    $PosOne, AX
   139		MOVQ    AX, cos+16(FP)
   140		RET
   141	isInfOrNaN: // return (NaN, NaN)
   142		MOVQ    $NaN, AX
   143		MOVQ    AX, sin+8(FP)
   144		MOVQ    AX, cos+16(FP)
   145		RET
