    75	func size(name string, t *testing.T) int64 {
    76		file, err := Open(name)
    77		if err != nil {
    78			t.Fatal("open failed:", err)
    79		}
    80		defer file.Close()
    81		var buf [100]byte
    82		len := 0
    83		for {
    84			n, e := file.Read(buf[0:])
    85			len += n
    86			if e == io.EOF {
    87				break
    88			}
    89			if e != nil {
    90				t.Fatal("read failed:", err)
    91			}
    92		}
    93		return int64(len)
    94	}

   260	// Read the directory one entry at a time.
   261	func smallReaddirnames(file *File, length int, t *testing.T) []string {
   262		names := make([]string, length)
   263		count := 0
   264		for {
   265			d, err := file.Readdirnames(1)
   266			if err == io.EOF {
   267				break
   268			}
   269			if err != nil {
   270				t.Fatalf("readdirnames %q failed: %v", file.Name(), err)
   271			}
   272			if len(d) == 0 {
   273				t.Fatalf("readdirnames %q returned empty slice and no error", file.Name())
   274			}
   275			names[count] = d[0]
   276			count++
   277		}
   278		return names[0:count]
   279	}

   325		defer RemoveAll(dir)
   326		for i := 1; i <= 105; i++ {
   327			f, err := Create(filepath.Join(dir, fmt.Sprintf("%d", i)))
   328			if err != nil {

   332			f.Close()

   398	func TestReaddirStatFailures(t *testing.T) {
   399		switch runtime.GOOS {
   400		case "windows", "plan9":
   401			// Windows and Plan 9 already do this correctly,
   402			// but are structured with different syscalls such
   403			// that they don't use Lstat, so the hook below for
   404			// testing it wouldn't work.
   405			t.Skipf("skipping test on %v", runtime.GOOS)
   406		}
   407		dir, err := ioutil.TempDir("", "")
   408		if err != nil {
   409			t.Fatalf("TempDir: %v", err)
   410		}
   411		defer RemoveAll(dir)
   412		touch(t, filepath.Join(dir, "good1"))
   413		touch(t, filepath.Join(dir, "x")) // will disappear or have an error
   414		touch(t, filepath.Join(dir, "good2"))
   415		defer func() {
   416			*LstatP = Lstat
   417		}()
   418		var xerr error // error to return for x
   419		*LstatP = func(path string) (FileInfo, error) {
   420			if xerr != nil && strings.HasSuffix(path, "x") {
   421				return nil, xerr
   422			}
   423			return Lstat(path)
   424		}
   425		readDir := func() ([]FileInfo, error) {
   426			d, err := Open(dir)
   427			if err != nil {
   428				t.Fatal(err)
   429			}
   430			defer d.Close()
   431			return d.Readdir(-1)
   432		}
   433		mustReadDir := func(testName string) []FileInfo {
   434			fis, err := readDir()
   435			if err != nil {
   436				t.Fatalf("%s: Readdir: %v", testName, err)
   437			}
   438			return fis
   439		}
   440		names := func(fis []FileInfo) []string {
   441			s := make([]string, len(fis))
   442			for i, fi := range fis {
   443				s[i] = fi.Name()
   444			}
   445			sort.Strings(s)
   446			return s
   447		}
   448	
   449		if got, want := names(mustReadDir("inital readdir")),
   450			[]string{"good1", "good2", "x"}; !reflect.DeepEqual(got, want) {
   451			t.Errorf("initial readdir got %q; want %q", got, want)
   452		}
   453	
   454		xerr = ErrNotExist
   455		if got, want := names(mustReadDir("with x disappearing")),
   456			[]string{"good1", "good2"}; !reflect.DeepEqual(got, want) {
   457			t.Errorf("with x disappearing, got %q; want %q", got, want)
   458		}
   459	
   460		xerr = errors.New("some real error")
   461		if _, err := readDir(); err != xerr {
   462			t.Errorf("with a non-ErrNotExist error, got error %v; want %v", err, xerr)
   463		}
   464	}

   606	func exec(t *testing.T, dir, cmd string, args []string, expect string) {
   607		r, w, err := Pipe()
   608		if err != nil {
   609			t.Fatalf("Pipe: %v", err)
   610		}
   611		defer r.Close()
   612		attr := &ProcAttr{Dir: dir, Files: []*File{nil, w, Stderr}}
   613		p, err := StartProcess(cmd, args, attr)
   614		if err != nil {
   615			t.Fatalf("StartProcess: %v", err)
   616		}
   617		w.Close()
   618	
   619		var b bytes.Buffer
   620		io.Copy(&b, r)

   736	// Use TempDir() to make sure we're on a local file system,
   737	// so that timings are not distorted by latency and caching.
   738	// On NFS, timings can be off due to caching of meta-data on
   739	// NFS servers (Issue 848).
   740	func TestChtimes(t *testing.T) {
   741		f := newFile("TestChtimes", t)
   742		defer Remove(f.Name())
   743		defer f.Close()
   744	
   745		f.Write([]byte("hello, world\n"))
   746		f.Close()
   747	
   748		st, err := Stat(f.Name())
   749		if err != nil {
   750			t.Fatalf("Stat %s: %s", f.Name(), err)
   751		}
   752		preStat := st
   753	
   754		// Move access and modification time back a second
   755		at := Atime(preStat)
   756		mt := preStat.ModTime()
   757		err = Chtimes(f.Name(), at.Add(-time.Second), mt.Add(-time.Second))
   758		if err != nil {
   759			t.Fatalf("Chtimes %s: %s", f.Name(), err)
   760		}
   761	
   762		st, err = Stat(f.Name())
   763		if err != nil {
   764			t.Fatalf("second Stat %s: %s", f.Name(), err)
   765		}
   766		postStat := st
   767	
   768		/* Plan 9, NaCl:
   769			Mtime is the time of the last change of content.  Similarly, atime is set whenever the
   770		    contents are accessed; also, it is set whenever mtime is set.
   771		*/
   772		pat := Atime(postStat)
   773		pmt := postStat.ModTime()
   774		if !pat.Before(at) && runtime.GOOS != "plan9" && runtime.GOOS != "nacl" {
   775			t.Errorf("AccessTime didn't go backwards; was=%d, after=%d", at, pat)
   776		}
   777	
   778		if !pmt.Before(mt) {
   779			t.Errorf("ModTime didn't go backwards; was=%d, after=%d", mt, pmt)
   780		}
   781	}

   940		// Run /bin/hostname and collect output.
   941		r, w, err := Pipe()
   942		if err != nil {
   943			t.Fatal(err)
   944		}
   945		defer r.Close()
   946		p, err := StartProcess("/bin/hostname", []string{"hostname"}, &ProcAttr{Files: []*File{nil, w, Stderr}})
   947		if err != nil {

   955		if err != nil {

   958		err = p.Kill()
   959		if err == nil {
   960			t.Errorf("expected an error from Kill running 'hostname'")
   961		}
   962		output := b.String()
   963		if n := len(output); n > 0 && output[n-1] == '\n' {
   964			output = output[0 : n-1]
   965		}
   966		if output == "" {
   967			t.Fatalf("%v produced no output", cmd)
   968		}
   969	
   970		return output
   971	}
