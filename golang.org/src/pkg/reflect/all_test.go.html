   547		a := []int{1, 2, 3, 4, 10, 9, 8, 7}
   548		b := []int{11, 22, 33, 44, 1010, 99, 88, 77, 66, 55, 44}
   549		c := []int{11, 22, 33, 44, 1010, 99, 88, 77, 66, 55, 44}
   550		for i := 0; i < len(b); i++ {
   551			if b[i] != c[i] {
   552				t.Fatalf("b != c before test")
   553			}
   554		}
   555		a1 := a
   556		b1 := b
   557		aa := ValueOf(&a1).Elem()
   558		ab := ValueOf(&b1).Elem()
   559		for tocopy := 1; tocopy <= 7; tocopy++ {
   560			aa.SetLen(tocopy)
   561			Copy(ab, aa)
   562			aa.SetLen(8)
   563			for i := 0; i < tocopy; i++ {
   564				if a[i] != b[i] {
   565					t.Errorf("(i) tocopy=%d a[%d]=%d, b[%d]=%d",
   566						tocopy, i, a[i], i, b[i])
   567				}
   568			}
   569			for i := tocopy; i < len(b); i++ {
   570				if b[i] != c[i] {
   571					if i < len(a) {
   572						t.Errorf("(ii) tocopy=%d a[%d]=%d, b[%d]=%d, c[%d]=%d",
   573							tocopy, i, a[i], i, b[i], i, c[i])
   574					} else {
   575						t.Errorf("(iii) tocopy=%d b[%d]=%d, c[%d]=%d",
   576							tocopy, i, b[i], i, c[i])
   577					}
   578				} else {
   579					t.Logf("tocopy=%d elem %d is okay\n", tocopy, i)

  1262			// block forever, and maybe add it anyway.
  1263			numCanSelect := 0

  1290				helper = cas.helper
  1291				cas.canSelect = true
  1292				numCanSelect++
  1293			}

  1295			// Permute cases and case info.
  1296			// Doing too much here makes the exhaustive loop
  1297			// too exhausting, so just do two swaps.
  1298			for loop := 0; loop < 2; loop++ {
  1299				i := x.Choose(len(cases))
  1300				j := x.Choose(len(cases))
  1301				cases[i], cases[j] = cases[j], cases[i]
  1302				info[i], info[j] = info[j], info[i]

  1307				// The pause needs to be big enough to let the select block before
  1308				// we run the helper, but if we lose that race once in a while it's okay: the
  1309				// select will just proceed immediately. Not a big deal.
  1310				// For short tests we can grow [sic] the timeout a bit without fear of taking too long
  1311				pause := 10 * time.Microsecond
  1312				if testing.Short() {
  1313					pause = 100 * time.Microsecond
  1314				}

  1316			}

  1323			if panicErr == nil && canPanic && numCanSelect == 1 {

  1325			}

  1327				continue
  1328			}
  1329	
  1330			cas := info[i]
  1331			if !cas.canSelect {
  1332				recvStr := ""
  1333				if recv.IsValid() {
  1334					recvStr = fmt.Sprintf(", received %v, %v", recv.Interface(), recvOK)
  1335				}
  1336				t.Fatalf("%s\nselected #%d incorrectly%s", fmtSelect(info), i, recvStr)
  1337				continue
  1338			}
  1339			if cas.panic {
  1340				t.Fatalf("%s\nselected #%d incorrectly (case should panic)", fmtSelect(info), i)
  1341				continue
  1342			}
  1343	
  1344			if cases[i].Dir == SelectRecv {
  1345				if !recv.IsValid() {
  1346					t.Fatalf("%s\nselected #%d but got %v, %v, want %v, %v", fmtSelect(info), i, recv, recvOK, cas.recv.Interface(), !cas.closed)
  1347				}
  1348				if !cas.recv.IsValid() {
  1349					t.Fatalf("%s\nselected #%d but internal error: missing recv value", fmtSelect(info), i)
  1350				}
  1351				if recv.Interface() != cas.recv.Interface() || recvOK != !cas.closed {
  1352					if recv.Interface() == cas.recv.Interface() && recvOK == !cas.closed {
  1353						t.Fatalf("%s\nselected #%d, got %#v, %v, and DeepEqual is broken on %T", fmtSelect(info), i, recv.Interface(), recvOK, recv.Interface())
  1354					}
  1355					t.Fatalf("%s\nselected #%d but got %#v, %v, want %#v, %v", fmtSelect(info), i, recv.Interface(), recvOK, cas.recv.Interface(), !cas.closed)
  1356				}
  1357			} else {
  1358				if recv.IsValid() || recvOK {
  1359					t.Fatalf("%s\nselected #%d but got %v, %v, want %v, %v", fmtSelect(info), i, recv, recvOK, Value{}, false)
  1360				}
  1361			}
  1362		}
  1363	}

  1368	var selectWatch struct {

  1370		once sync.Once
  1371		now  time.Time

  1382			}

  1384		}

  3613	// An exhaustive is a mechanism for writing exhaustive or stochastic tests.
  3614	// The basic usage is:
  3615	//
  3616	//	for x.Next() {
  3617	//		... code using x.Maybe() or x.Choice(n) to create test cases ...
  3618	//	}
  3619	//
  3620	// Each iteration of the loop returns a different set of results, until all
  3621	// possible result sets have been explored. It is okay for different code paths
  3622	// to make different method call sequences on x, but there must be no
  3623	// other source of non-determinism in the call sequences.
  3624	//
  3625	// When faced with a new decision, x chooses randomly. Future explorations
  3626	// of that path will choose successive values for the result. Thus, stopping
  3627	// the loop after a fixed number of iterations gives somewhat stochastic
  3628	// testing.
  3629	//
  3630	// Example:
  3631	//
  3632	//	for x.Next() {
  3633	//		v := make([]bool, x.Choose(4))
  3634	//		for i := range v {
  3635	//			v[i] = x.Maybe()
  3636	//		}
  3637	//		fmt.Println(v)
  3638	//	}
  3639	//
  3640	// prints (in some order):
  3641	//
  3642	//	[]
  3643	//	[false]
  3644	//	[true]
  3645	//	[false false]
  3646	//	[false true]
  3647	//	...
  3648	//	[true true]
  3649	//	[false false false]
  3650	//	...
  3651	//	[true true true]
  3652	//	[false false false false]
  3653	//	...
  3654	//	[true true true true]
  3655	//
  3656	type exhaustive struct {
  3657		r    *rand.Rand
  3658		pos  int
  3659		last []choice
  3660	}
  3661	
  3662	type choice struct {
  3663		off int
  3664		n   int
  3665		max int
  3666	}
  3667	
  3668	func (x *exhaustive) Next() bool {
  3669		if x.r == nil {
  3670			x.r = rand.New(rand.NewSource(time.Now().UnixNano()))
  3671		}
  3672		x.pos = 0
  3673		if x.last == nil {
  3674			x.last = []choice{}
  3675			return true
  3676		}
  3677		for i := len(x.last) - 1; i >= 0; i-- {
  3678			c := &x.last[i]
  3679			if c.n+1 < c.max {
  3680				c.n++
  3681				x.last = x.last[:i+1]
  3682				return true
  3683			}
  3684		}
  3685		return false
  3686	}
  3687	
  3688	func (x *exhaustive) Choose(max int) int {
  3689		if x.pos >= len(x.last) {
  3690			x.last = append(x.last, choice{x.r.Intn(max), 0, max})
  3691		}
  3692		c := &x.last[x.pos]
  3693		x.pos++
  3694		if c.max != max {
  3695			panic("inconsistent use of exhaustive tester")
  3696		}
  3697		return (c.n + c.off) % max
  3698	}

  3762	// Given
  3763	//	type Outer struct {
  3764	//		*Inner
  3765	//		...
  3766	//	}
  3767	// the compiler generates the implementation of (*Outer).M dispatching to the embedded Inner.
  3768	// The implementation is logically:
  3769	//	func (p *Outer) M() {
  3770	//		(p.Inner).M()
  3771	//	}
  3772	// but since the only change here is the replacement of one pointer receiver with another,
  3773	// the actual generated code overwrites the original receiver with the p.Inner pointer and
  3774	// then jumps to the M method expecting the *Inner receiver.
  3775	//
  3776	// During reflect.Value.Call, we create an argument frame and the associated data structures
  3777	// to describe it to the garbage collector, populate the frame, call reflect.call to
  3778	// run a function call using that frame, and then copy the results back out of the frame.
  3779	// The reflect.call function does a memmove of the frame structure onto the
  3780	// stack (to set up the inputs), runs the call, and the memmoves the stack back to
  3781	// the frame structure (to preserve the outputs).
  3782	//
  3783	// Originally reflect.call did not distinguish inputs from outputs: both memmoves
  3784	// were for the full stack frame. However, in the case where the called function was
  3785	// one of these wrappers, the rewritten receiver is almost certainly a different type
  3786	// than the original receiver. This is not a problem on the stack, where we use the
  3787	// program counter to determine the type information and understand that
  3788	// during (*Outer).M the receiver is an *Outer while during (*Inner).M the receiver in the same
  3789	// memory word is now an *Inner. But in the statically typed argument frame created
  3790	// by reflect, the receiver is always an *Outer. Copying the modified receiver pointer
  3791	// off the stack into the frame will store an *Inner there, and then if a garbage collection
  3792	// happens to scan that argument frame before it is discarded, it will scan the *Inner
  3793	// memory as if it were an *Outer. If the two have different memory layouts, the
  3794	// collection will intepret the memory incorrectly.
  3795	//
  3796	// One such possible incorrect interpretation is to treat two arbitrary memory words
  3797	// (Inner.P1 and Inner.P2 below) as an interface (Outer.R below). Because interpreting
  3798	// an interface requires dereferencing the itab word, the misinterpretation will try to
  3799	// deference Inner.P1, causing a crash during garbage collection.
  3800	//
  3801	// This came up in a real program in issue 7725.
  3802	
  3803	type Outer struct {
  3804		*Inner
  3805		R io.Reader
  3806	}
  3807	
  3808	type Inner struct {
  3809		X  *Outer
  3810		P1 uintptr
  3811		P2 uintptr
  3812	}

  3814	func (pi *Inner) M() {
  3815		// Clear references to pi so that the only way the
  3816		// garbage collection will find the pointer is in the
  3817		// argument frame, typed as a *Outer.
  3818		pi.X.Inner = nil
  3819	
  3820		// Set up an interface value that will cause a crash.
  3821		// P1 = 1 is a non-zero, so the interface looks non-nil.
  3822		// P2 = pi ensures that the data word points into the
  3823		// allocated heap; if not the collection skips the interface
  3824		// value as irrelevant, without dereferencing P1.
  3825		pi.P1 = 1
  3826		pi.P2 = uintptr(unsafe.Pointer(pi))
  3827	}
