     5	// Package reflect implements run-time reflection, allowing a program to
     6	// manipulate objects with arbitrary types.  The typical use is to take a value
     7	// with static type interface{} and extract its dynamic type information by
     8	// calling TypeOf, which returns a Type.
     9	//
    10	// A call to ValueOf returns a Value representing the run-time data.
    11	// Zero takes a Type and returns a Value representing a zero value
    12	// for that type.
    13	//
    14	// See "The Laws of Reflection" for an introduction to reflection in Go:
    15	// http://golang.org/doc/articles/laws_of_reflection.html
    16	package reflect

    25	// Type is the representation of a Go type.
    26	//
    27	// Not all methods apply to all kinds of types.  Restrictions,
    28	// if any, are noted in the documentation for each method.
    29	// Use the Kind method to find out the kind of type before
    30	// calling kind-specific methods.  Calling a method
    31	// inappropriate to the kind of type causes a run-time panic.
    32	type Type interface {
    33		// Methods applicable to all types.
    34	
    35		// Align returns the alignment in bytes of a value of
    36		// this type when allocated in memory.
    37		Align() int
    38	
    39		// FieldAlign returns the alignment in bytes of a value of
    40		// this type when used as a field in a struct.
    41		FieldAlign() int
    42	
    43		// Method returns the i'th method in the type's method set.
    44		// It panics if i is not in the range [0, NumMethod()).
    45		//
    46		// For a non-interface type T or *T, the returned Method's Type and Func
    47		// fields describe a function whose first argument is the receiver.
    48		//
    49		// For an interface type, the returned Method's Type field gives the
    50		// method signature, without a receiver, and the Func field is nil.
    51		Method(int) Method
    52	
    53		// MethodByName returns the method with that name in the type's
    54		// method set and a boolean indicating if the method was found.
    55		//
    56		// For a non-interface type T or *T, the returned Method's Type and Func
    57		// fields describe a function whose first argument is the receiver.
    58		//
    59		// For an interface type, the returned Method's Type field gives the
    60		// method signature, without a receiver, and the Func field is nil.
    61		MethodByName(string) (Method, bool)
    62	
    63		// NumMethod returns the number of methods in the type's method set.
    64		NumMethod() int
    65	
    66		// Name returns the type's name within its package.
    67		// It returns an empty string for unnamed types.
    68		Name() string
    69	
    70		// PkgPath returns a named type's package path, that is, the import path
    71		// that uniquely identifies the package, such as "encoding/base64".
    72		// If the type was predeclared (string, error) or unnamed (*T, struct{}, []int),
    73		// the package path will be the empty string.
    74		PkgPath() string
    75	
    76		// Size returns the number of bytes needed to store
    77		// a value of the given type; it is analogous to unsafe.Sizeof.
    78		Size() uintptr
    79	
    80		// String returns a string representation of the type.
    81		// The string representation may use shortened package names
    82		// (e.g., base64 instead of "encoding/base64") and is not
    83		// guaranteed to be unique among types.  To test for equality,
    84		// compare the Types directly.
    85		String() string
    86	
    87		// Kind returns the specific kind of this type.
    88		Kind() Kind
    89	
    90		// Implements returns true if the type implements the interface type u.
    91		Implements(u Type) bool
    92	
    93		// AssignableTo returns true if a value of the type is assignable to type u.
    94		AssignableTo(u Type) bool
    95	
    96		// ConvertibleTo returns true if a value of the type is convertible to type u.
    97		ConvertibleTo(u Type) bool
    98	
    99		// Methods applicable only to some types, depending on Kind.
   100		// The methods allowed for each kind are:
   101		//
   102		//	Int*, Uint*, Float*, Complex*: Bits
   103		//	Array: Elem, Len
   104		//	Chan: ChanDir, Elem
   105		//	Func: In, NumIn, Out, NumOut, IsVariadic.
   106		//	Map: Key, Elem
   107		//	Ptr: Elem
   108		//	Slice: Elem
   109		//	Struct: Field, FieldByIndex, FieldByName, FieldByNameFunc, NumField
   110	
   111		// Bits returns the size of the type in bits.
   112		// It panics if the type's Kind is not one of the
   113		// sized or unsized Int, Uint, Float, or Complex kinds.
   114		Bits() int
   115	
   116		// ChanDir returns a channel type's direction.
   117		// It panics if the type's Kind is not Chan.
   118		ChanDir() ChanDir
   119	
   120		// IsVariadic returns true if a function type's final input parameter
   121		// is a "..." parameter.  If so, t.In(t.NumIn() - 1) returns the parameter's
   122		// implicit actual type []T.
   123		//
   124		// For concreteness, if t represents func(x int, y ... float64), then
   125		//
   126		//	t.NumIn() == 2
   127		//	t.In(0) is the reflect.Type for "int"
   128		//	t.In(1) is the reflect.Type for "[]float64"
   129		//	t.IsVariadic() == true
   130		//
   131		// IsVariadic panics if the type's Kind is not Func.
   132		IsVariadic() bool
   133	
   134		// Elem returns a type's element type.
   135		// It panics if the type's Kind is not Array, Chan, Map, Ptr, or Slice.
   136		Elem() Type
   137	
   138		// Field returns a struct type's i'th field.
   139		// It panics if the type's Kind is not Struct.
   140		// It panics if i is not in the range [0, NumField()).
   141		Field(i int) StructField
   142	
   143		// FieldByIndex returns the nested field corresponding
   144		// to the index sequence.  It is equivalent to calling Field
   145		// successively for each index i.
   146		// It panics if the type's Kind is not Struct.
   147		FieldByIndex(index []int) StructField
   148	
   149		// FieldByName returns the struct field with the given name
   150		// and a boolean indicating if the field was found.
   151		FieldByName(name string) (StructField, bool)
   152	
   153		// FieldByNameFunc returns the first struct field with a name
   154		// that satisfies the match function and a boolean indicating if
   155		// the field was found.
   156		FieldByNameFunc(match func(string) bool) (StructField, bool)
   157	
   158		// In returns the type of a function type's i'th input parameter.
   159		// It panics if the type's Kind is not Func.
   160		// It panics if i is not in the range [0, NumIn()).
   161		In(i int) Type
   162	
   163		// Key returns a map type's key type.
   164		// It panics if the type's Kind is not Map.
   165		Key() Type
   166	
   167		// Len returns an array type's length.
   168		// It panics if the type's Kind is not Array.
   169		Len() int
   170	
   171		// NumField returns a struct type's field count.
   172		// It panics if the type's Kind is not Struct.
   173		NumField() int
   174	
   175		// NumIn returns a function type's input parameter count.
   176		// It panics if the type's Kind is not Func.
   177		NumIn() int
   178	
   179		// NumOut returns a function type's output parameter count.
   180		// It panics if the type's Kind is not Func.
   181		NumOut() int
   182	
   183		// Out returns the type of a function type's i'th output parameter.
   184		// It panics if the type's Kind is not Func.
   185		// It panics if i is not in the range [0, NumOut()).
   186		Out(i int) Type
   187	
   188		common() *rtype
   189		uncommon() *uncommonType
   190	}

   240	// rtype is the common implementation of most values.
   241	// It is embedded in other, public struct types, but always
   242	// with a unique tag like `reflect:"array"` or `reflect:"ptr"`
   243	// so that code cannot convert from, say, *arrayType to *ptrType.
   244	type rtype struct {
   245		size          uintptr        // size in bytes
   246		hash          uint32         // hash of type; avoids computation in hash tables
   247		_             uint8          // unused/padding
   248		align         uint8          // alignment of variable with this type
   249		fieldAlign    uint8          // alignment of struct field with this type
   250		kind          uint8          // enumeration for C
   251		alg           *uintptr       // algorithm table (../runtime/runtime.h:/Alg)
   252		gc            unsafe.Pointer // garbage collection data
   253		string        *string        // string form; unnecessary but undeniably useful
   254		*uncommonType                // (relatively) uncommon fields
   255		ptrToThis     *rtype         // type for pointer to this type, if used in binary or has methods
   256		zero          unsafe.Pointer // pointer to zero value
   257	}

   465	func (t *rtype) Bits() int {
   466		if t == nil {
   467			panic("reflect: Bits of nil Type")

   469		k := t.Kind()

   471			panic("reflect: Bits of non-arithmetic Type " + t.String())

   486	func (t *uncommonType) Method(i int) (m Method) {
   487		if t == nil || i < 0 || i >= len(t.methods) {
   488			panic("reflect: Method index out of range")
   489		}
   490		p := &t.methods[i]
   491		if p.name != nil {
   492			m.Name = *p.name
   493		}
   494		fl := flag(Func) << flagKindShift
   495		if p.pkgPath != nil {
   496			m.PkgPath = *p.pkgPath
   497			fl |= flagRO
   498		}
   499		mt := p.typ
   500		m.Type = mt
   501		fn := unsafe.Pointer(&p.tfn)
   502		m.Func = Value{mt, fn, 0, fl}
   503		m.Index = i
   504		return
   505	}

   514	func (t *uncommonType) MethodByName(name string) (m Method, ok bool) {
   515		if t == nil {
   516			return
   517		}
   518		var p *method
   519		for i := range t.methods {
   520			p = &t.methods[i]
   521			if p.name != nil && *p.name == name {
   522				return t.Method(i), true
   523			}
   524		}
   525		return
   526	}

   749	// A StructTag is the tag string in a struct field.
   750	//
   751	// By convention, tag strings are a concatenation of
   752	// optionally space-separated key:"value" pairs.
   753	// Each key is a non-empty string consisting of non-control
   754	// characters other than space (U+0020 ' '), quote (U+0022 '"'),
   755	// and colon (U+003A ':').  Each value is quoted using U+0022 '"'
   756	// characters and Go string literal syntax.
   757	type StructTag string
   758	
   759	// Get returns the value associated with key in the tag string.
   760	// If there is no such key in the tag, Get returns the empty string.
   761	// If the tag does not have the conventional format, the value
   762	// returned by Get is unspecified.
   763	func (tag StructTag) Get(key string) string {
   764		for tag != "" {
   765			// skip leading space
   766			i := 0
   767			for i < len(tag) && tag[i] == ' ' {
   768				i++
   769			}
   770			tag = tag[i:]
   771			if tag == "" {
   772				break
   773			}
   774	
   775			// scan to colon.
   776			// a space or a quote is a syntax error
   777			i = 0
   778			for i < len(tag) && tag[i] != ' ' && tag[i] != ':' && tag[i] != '"' {
   779				i++
   780			}
   781			if i+1 >= len(tag) || tag[i] != ':' || tag[i+1] != '"' {
   782				break
   783			}
   784			name := string(tag[:i])
   785			tag = tag[i+1:]
   786	
   787			// scan quoted string to find value
   788			i = 1
   789			for i < len(tag) && tag[i] != '"' {
   790				if tag[i] == '\\' {
   791					i++
   792				}
   793				i++
   794			}
   795			if i >= len(tag) {
   796				break
   797			}
   798			qvalue := string(tag[:i+1])
   799			tag = tag[i+1:]
   800	
   801			if key == name {
   802				value, _ := strconv.Unquote(qvalue)
   803				return value
   804			}
   805		}
   806		return ""
   807	}

   870	// FieldByNameFunc returns the struct field with a name that satisfies the
   871	// match function and a boolean to indicate if the field was found.
   872	func (t *structType) FieldByNameFunc(match func(string) bool) (result StructField, ok bool) {
   873		// This uses the same condition that the Go language does: there must be a unique instance
   874		// of the match at a given depth level. If there are multiple instances of a match at the
   875		// same depth, they annihilate each other and inhibit any possible match at a lower level.
   876		// The algorithm is breadth first search, one depth level at a time.
   877	
   878		// The current and next slices are work queues:
   879		// current lists the fields to visit on this depth level,
   880		// and next lists the fields on the next lower level.
   881		current := []fieldScan{}
   882		next := []fieldScan{{typ: t}}
   883	
   884		// nextCount records the number of times an embedded type has been
   885		// encountered and considered for queueing in the 'next' slice.
   886		// We only queue the first one, but we increment the count on each.
   887		// If a struct type T can be reached more than once at a given depth level,
   888		// then it annihilates itself and need not be considered at all when we
   889		// process that next depth level.
   890		var nextCount map[*structType]int
   891	
   892		// visited records the structs that have been considered already.
   893		// Embedded pointer fields can create cycles in the graph of
   894		// reachable embedded types; visited avoids following those cycles.
   895		// It also avoids duplicated effort: if we didn't find the field in an
   896		// embedded type T at level 2, we won't find it in one at level 4 either.
   897		visited := map[*structType]bool{}
   898	
   899		for len(next) > 0 {
   900			current, next = next, current[:0]
   901			count := nextCount
   902			nextCount = nil
   903	
   904			// Process all the fields at this depth, now listed in 'current'.
   905			// The loop queues embedded fields found in 'next', for processing during the next
   906			// iteration. The multiplicity of the 'current' field counts is recorded
   907			// in 'count'; the multiplicity of the 'next' field counts is recorded in 'nextCount'.
   908			for _, scan := range current {
   909				t := scan.typ
   910				if visited[t] {
   911					// We've looked through this type before, at a higher level.
   912					// That higher level would shadow the lower level we're now at,
   913					// so this one can't be useful to us. Ignore it.
   914					continue
   915				}
   916				visited[t] = true
   917				for i := range t.fields {
   918					f := &t.fields[i]
   919					// Find name and type for field f.
   920					var fname string
   921					var ntyp *rtype
   922					if f.name != nil {
   923						fname = *f.name
   924					} else {
   925						// Anonymous field of type T or *T.
   926						// Name taken from type.
   927						ntyp = f.typ
   928						if ntyp.Kind() == Ptr {
   929							ntyp = ntyp.Elem().common()
   930						}
   931						fname = ntyp.Name()
   932					}
   933	
   934					// Does it match?
   935					if match(fname) {
   936						// Potential match
   937						if count[t] > 1 || ok {
   938							// Name appeared multiple times at this level: annihilate.
   939							return StructField{}, false
   940						}
   941						result = t.Field(i)
   942						result.Index = nil
   943						result.Index = append(result.Index, scan.index...)
   944						result.Index = append(result.Index, i)
   945						ok = true
   946						continue
   947					}
   948	
   949					// Queue embedded struct fields for processing with next level,
   950					// but only if we haven't seen a match yet at this level and only
   951					// if the embedded types haven't already been queued.
   952					if ok || ntyp == nil || ntyp.Kind() != Struct {
   953						continue
   954					}
   955					styp := (*structType)(unsafe.Pointer(ntyp))
   956					if nextCount[styp] > 0 {
   957						nextCount[styp] = 2 // exact multiple doesn't matter
   958						continue
   959					}
   960					if nextCount == nil {
   961						nextCount = map[*structType]int{}
   962					}
   963					nextCount[styp] = 1
   964					if count[t] > 1 {
   965						nextCount[styp] = 2 // exact multiple doesn't matter
   966					}
   967					var index []int
   968					index = append(index, scan.index...)
   969					index = append(index, i)
   970					next = append(next, fieldScan{styp, index})
   971				}
   972			}
   973			if ok {
   974				break
   975			}
   976		}
   977		return
   978	}

  1048	func (t *rtype) ptrTo() *rtype {
  1049		if p := t.ptrToThis; p != nil {
  1050			return p
  1051		}
  1052	
  1053		// Otherwise, synthesize one.
  1054		// This only happens for pointers with no methods.
  1055		// We keep the mapping in a map on the side, because
  1056		// this operation is rare and a separate map lets us keep
  1057		// the type structures in read-only memory.
  1058		ptrMap.RLock()
  1059		if m := ptrMap.m; m != nil {
  1060			if p := m[t]; p != nil {
  1061				ptrMap.RUnlock()
  1062				return &p.rtype
  1063			}
  1064		}
  1065		ptrMap.RUnlock()
  1066		ptrMap.Lock()
  1067		if ptrMap.m == nil {
  1068			ptrMap.m = make(map[*rtype]*ptrType)
  1069		}
  1070		p := ptrMap.m[t]
  1071		if p != nil {
  1072			// some other goroutine won the race and created it
  1073			ptrMap.Unlock()
  1074			return &p.rtype
  1075		}
  1076	
  1077		// Create a new ptrType starting with the description
  1078		// of an *unsafe.Pointer.
  1079		p = new(ptrType)
  1080		var iptr interface{} = (*unsafe.Pointer)(nil)
  1081		prototype := *(**ptrType)(unsafe.Pointer(&iptr))
  1082		*p = *prototype
  1083	
  1084		s := "*" + *t.string
  1085		p.string = &s
  1086	
  1087		// For the type structures linked into the binary, the
  1088		// compiler provides a good hash of the string.
  1089		// Create a good hash for the new string by using
  1090		// the FNV-1 hash's mixing function to combine the
  1091		// old hash and the new "*".
  1092		p.hash = fnv1(t.hash, '*')
  1093	
  1094		p.uncommonType = nil
  1095		p.ptrToThis = nil
  1096		p.zero = unsafe.Pointer(&make([]byte, p.size)[0])
  1097		p.elem = t
  1098	
  1099		if t.kind&kindNoPointers != 0 {
  1100			p.gc = unsafe.Pointer(&ptrDataGCProg)
  1101		} else {
  1102			p.gc = unsafe.Pointer(&ptrGC{
  1103				width:  p.size,
  1104				op:     _GC_PTR,
  1105				off:    0,
  1106				elemgc: t.gc,
  1107				end:    _GC_END,
  1108			})
  1109		}
  1110		// INCORRECT. Uncomment to check that TestPtrToGC fails when p.gc is wrong.
  1111		//p.gc = unsafe.Pointer(&badGC{width: p.size, end: _GC_END})
  1112	
  1113		ptrMap.m[t] = p
  1114		ptrMap.Unlock()
  1115		return &p.rtype
  1116	}

  1152	// implements returns true if the type V implements the interface type T.
  1153	func implements(T, V *rtype) bool {
  1154		if T.Kind() != Interface {
  1155			return false
  1156		}
  1157		t := (*interfaceType)(unsafe.Pointer(T))
  1158		if len(t.methods) == 0 {
  1159			return true
  1160		}
  1161	
  1162		// The same algorithm applies in both cases, but the
  1163		// method tables for an interface type and a concrete type
  1164		// are different, so the code is duplicated.
  1165		// In both cases the algorithm is a linear scan over the two
  1166		// lists - T's methods and V's methods - simultaneously.
  1167		// Since method tables are stored in a unique sorted order
  1168		// (alphabetical, with no duplicate method names), the scan
  1169		// through V's methods must hit a match for each of T's
  1170		// methods along the way, or else V does not implement T.
  1171		// This lets us run the scan in overall linear time instead of
  1172		// the quadratic time  a naive search would require.
  1173		// See also ../runtime/iface.c.
  1174		if V.Kind() == Interface {
  1175			v := (*interfaceType)(unsafe.Pointer(V))
  1176			i := 0
  1177			for j := 0; j < len(v.methods); j++ {
  1178				tm := &t.methods[i]
  1179				vm := &v.methods[j]
  1180				if vm.name == tm.name && vm.pkgPath == tm.pkgPath && vm.typ == tm.typ {
  1181					if i++; i >= len(t.methods) {
  1182						return true
  1183					}
  1184				}
  1185			}
  1186			return false
  1187		}
  1188	
  1189		v := V.uncommon()
  1190		if v == nil {
  1191			return false
  1192		}
  1193		i := 0
  1194		for j := 0; j < len(v.methods); j++ {
  1195			tm := &t.methods[i]
  1196			vm := &v.methods[j]
  1197			if vm.name == tm.name && vm.pkgPath == tm.pkgPath && vm.mtyp == tm.typ {
  1198				if i++; i >= len(t.methods) {
  1199					return true
  1200				}
  1201			}
  1202		}
  1203		return false
  1204	}

  1206	// directlyAssignable returns true if a value x of type V can be directly
  1207	// assigned (using memmove) to a value of type T.
  1208	// http://golang.org/doc/go_spec.html#Assignability
  1209	// Ignoring the interface rules (implemented elsewhere)
  1210	// and the ideal constant rules (no ideal constants at run time).
  1211	func directlyAssignable(T, V *rtype) bool {
  1212		// x's type V is identical to T?
  1213		if T == V {
  1214			return true
  1215		}
  1216	
  1217		// Otherwise at least one of T and V must be unnamed
  1218		// and they must have the same kind.
  1219		if T.Name() != "" && V.Name() != "" || T.Kind() != V.Kind() {
  1220			return false
  1221		}
  1222	
  1223		// x's type T and V must  have identical underlying types.
  1224		return haveIdenticalUnderlyingType(T, V)
  1225	}

  1227	func haveIdenticalUnderlyingType(T, V *rtype) bool {
  1228		if T == V {
  1229			return true
  1230		}
  1231	
  1232		kind := T.Kind()
  1233		if kind != V.Kind() {
  1234			return false
  1235		}
  1236	
  1237		// Non-composite types of equal kind have same underlying type
  1238		// (the predefined instance of the type).
  1239		if Bool <= kind && kind <= Complex128 || kind == String || kind == UnsafePointer {
  1240			return true
  1241		}
  1242	
  1243		// Composite types.
  1244		switch kind {
  1245		case Array:
  1246			return T.Elem() == V.Elem() && T.Len() == V.Len()
  1247	
  1248		case Chan:
  1249			// Special case:
  1250			// x is a bidirectional channel value, T is a channel type,
  1251			// and x's type V and T have identical element types.
  1252			if V.ChanDir() == BothDir && T.Elem() == V.Elem() {
  1253				return true
  1254			}
  1255	
  1256			// Otherwise continue test for identical underlying type.
  1257			return V.ChanDir() == T.ChanDir() && T.Elem() == V.Elem()
  1258	
  1259		case Func:
  1260			t := (*funcType)(unsafe.Pointer(T))
  1261			v := (*funcType)(unsafe.Pointer(V))
  1262			if t.dotdotdot != v.dotdotdot || len(t.in) != len(v.in) || len(t.out) != len(v.out) {
  1263				return false
  1264			}
  1265			for i, typ := range t.in {
  1266				if typ != v.in[i] {
  1267					return false
  1268				}
  1269			}
  1270			for i, typ := range t.out {
  1271				if typ != v.out[i] {
  1272					return false
  1273				}
  1274			}
  1275			return true
  1276	
  1277		case Interface:
  1278			t := (*interfaceType)(unsafe.Pointer(T))
  1279			v := (*interfaceType)(unsafe.Pointer(V))
  1280			if len(t.methods) == 0 && len(v.methods) == 0 {
  1281				return true
  1282			}
  1283			// Might have the same methods but still
  1284			// need a run time conversion.
  1285			return false
  1286	
  1287		case Map:
  1288			return T.Key() == V.Key() && T.Elem() == V.Elem()
  1289	
  1290		case Ptr, Slice:
  1291			return T.Elem() == V.Elem()
  1292	
  1293		case Struct:
  1294			t := (*structType)(unsafe.Pointer(T))
  1295			v := (*structType)(unsafe.Pointer(V))
  1296			if len(t.fields) != len(v.fields) {
  1297				return false
  1298			}
  1299			for i := range t.fields {
  1300				tf := &t.fields[i]
  1301				vf := &v.fields[i]
  1302				if tf.name != vf.name && (tf.name == nil || vf.name == nil || *tf.name != *vf.name) {
  1303					return false
  1304				}
  1305				if tf.pkgPath != vf.pkgPath && (tf.pkgPath == nil || vf.pkgPath == nil || *tf.pkgPath != *vf.pkgPath) {
  1306					return false
  1307				}
  1308				if tf.typ != vf.typ {
  1309					return false
  1310				}
  1311				if tf.tag != vf.tag && (tf.tag == nil || vf.tag == nil || *tf.tag != *vf.tag) {
  1312					return false
  1313				}
  1314				if tf.offset != vf.offset {
  1315					return false
  1316				}
  1317			}
  1318			return true
  1319		}
  1320	
  1321		return false
  1322	}

  1324	// typelinks is implemented in package runtime.
  1325	// It returns a slice of all the 'typelink' information in the binary,
  1326	// which is to say a slice of known types, sorted by string.
  1327	// Note that strings are not unique identifiers for types:
  1328	// there can be more than one with a given string.
  1329	// Only types we might want to look up are included:
  1330	// channels, maps, slices, and arrays.
  1331	func typelinks() []*rtype
  1332	
  1333	// typesByString returns the subslice of typelinks() whose elements have
  1334	// the given string representation.
  1335	// It may be empty (no known types with that string) or may have
  1336	// multiple elements (multiple types with that string).
  1337	func typesByString(s string) []*rtype {
  1338		typ := typelinks()
  1339	
  1340		// We are looking for the first index i where the string becomes >= s.
  1341		// This is a copy of sort.Search, with f(h) replaced by (*typ[h].string >= s).
  1342		i, j := 0, len(typ)
  1343		for i < j {
  1344			h := i + (j-i)/2 // avoid overflow when computing h
  1345			// i ≤ h < j
  1346			if !(*typ[h].string >= s) {
  1347				i = h + 1 // preserves f(i-1) == false
  1348			} else {
  1349				j = h // preserves f(j) == true
  1350			}
  1351		}
  1352		// i == j, f(i-1) == false, and f(j) (= f(i)) == true  =>  answer is i.
  1353	
  1354		// Having found the first, linear scan forward to find the last.
  1355		// We could do a second binary search, but the caller is going
  1356		// to do a linear scan anyway.
  1357		j = i
  1358		for j < len(typ) && *typ[j].string == s {
  1359			j++
  1360		}
  1361	
  1362		// This slice will be empty if the string is not found.
  1363		return typ[i:j]
  1364	}

  1366	// The lookupCache caches ChanOf, MapOf, and SliceOf lookups.
  1367	var lookupCache struct {
  1368		sync.RWMutex
  1369		m map[cacheKey]*rtype
  1370	}

  1372	// A cacheKey is the key for use in the lookupCache.
  1373	// Four values describe any of the types we are looking for:
  1374	// type kind, one or two subtypes, and an extra integer.

  1376		kind  Kind
  1377		t1    *rtype
  1378		t2    *rtype
  1379		extra uintptr

  1382	// cacheGet looks for a type under the key k in the lookupCache.
  1383	// If it finds one, it returns that type.
  1384	// If not, it returns nil with the cache locked.
  1385	// The caller is expected to use cachePut to unlock the cache.
  1386	func cacheGet(k cacheKey) Type {
  1387		lookupCache.RLock()
  1388		t := lookupCache.m[k]
  1389		lookupCache.RUnlock()
  1390		if t != nil {
  1391			return t
  1392		}
  1393	
  1394		lookupCache.Lock()
  1395		t = lookupCache.m[k]
  1396		if t != nil {
  1397			lookupCache.Unlock()
  1398			return t
  1399		}
  1400	
  1401		if lookupCache.m == nil {
  1402			lookupCache.m = make(map[cacheKey]*rtype)
  1403		}
  1404	
  1405		return nil
  1406	}

  1408	// cachePut stores the given type in the cache, unlocks the cache,
  1409	// and returns the type. It is expected that the cache is locked
  1410	// because cacheGet returned nil.
  1411	func cachePut(k cacheKey, t *rtype) Type {
  1412		lookupCache.m[k] = t
  1413		lookupCache.Unlock()
  1414		return t
  1415	}

  1436	// If t's size is equal to or exceeds this limit, ChanOf panics.
  1437	func ChanOf(dir ChanDir, t Type) Type {
  1438		typ := t.(*rtype)
  1439	
  1440		// Look in cache.
  1441		ckey := cacheKey{Chan, typ, nil, uintptr(dir)}
  1442		if ch := cacheGet(ckey); ch != nil {
  1443			return ch
  1444		}
  1445	
  1446		// This restriction is imposed by the gc compiler and the runtime.
  1447		if typ.size >= 1<<16 {
  1448			lookupCache.Unlock()

  1450		}
  1451	
  1452		// Look in known types.
  1453		// TODO: Precedence when constructing string.
  1454		var s string
  1455		switch dir {
  1456		default:
  1457			lookupCache.Unlock()

  1459		case SendDir:
  1460			s = "chan<- " + *typ.string
  1461		case RecvDir:

  1463		case BothDir:
  1464			s = "chan " + *typ.string
  1465		}

  1477		*ch = *prototype
  1478		ch.string = &s

  1481		ch.uncommonType = nil
  1482		ch.ptrToThis = nil

  1569	func bucketOf(ktyp, etyp *rtype) *rtype {
  1570		if ktyp.size > _MAXKEYSIZE {
  1571			ktyp = PtrTo(ktyp).(*rtype)
  1572		}
  1573		if etyp.size > _MAXVALSIZE {
  1574			etyp = PtrTo(etyp).(*rtype)
  1575		}
  1576		ptrsize := unsafe.Sizeof(uintptr(0))
  1577	
  1578		gc := make([]uintptr, 1)                                       // first entry is size, filled in at the end
  1579		offset := _BUCKETSIZE * unsafe.Sizeof(uint8(0))                // topbits
  1580		gc = append(gc, _GC_PTR, offset, 0 /*self pointer set below*/) // overflow
  1581		offset += ptrsize
  1582	
  1583		if runtime.GOARCH == "amd64p32" {
  1584			offset += 4
  1585		}
  1586

  1590			gc = appendGCProgram(gc, ktyp)
  1591			gc = append(gc, _GC_ARRAY_NEXT)
  1592		}
  1593		offset += _BUCKETSIZE * ktyp.size
  1594	
  1595		// values
  1596		if etyp.kind&kindNoPointers == 0 {
  1597			gc = append(gc, _GC_ARRAY_START, offset, _BUCKETSIZE, etyp.size)
  1598			gc = appendGCProgram(gc, etyp)
  1599			gc = append(gc, _GC_ARRAY_NEXT)
  1600		}
  1601		offset += _BUCKETSIZE * etyp.size
  1602

  1604		gc[0] = offset
  1605		gc[3] = uintptr(unsafe.Pointer(&gc[0])) // set self pointer
  1606	
  1607		b := new(rtype)
  1608		b.size = offset
  1609		b.gc = unsafe.Pointer(&gc[0])

  1612		return b
  1613	}

  1615	// Take the GC program for "t" and append it to the GC program "gc".
  1616	func appendGCProgram(gc []uintptr, t *rtype) []uintptr {
  1617		p := t.gc
  1618		p = unsafe.Pointer(uintptr(p) + unsafe.Sizeof(uintptr(0))) // skip size
  1619	loop:
  1620		for {
  1621			var argcnt int
  1622			switch *(*uintptr)(p) {
  1623			case _GC_END:
  1624				// Note: _GC_END not included in append
  1625				break loop
  1626			case _GC_ARRAY_NEXT:
  1627				argcnt = 0
  1628			case _GC_APTR, _GC_STRING, _GC_EFACE, _GC_IFACE:
  1629				argcnt = 1
  1630			case _GC_PTR, _GC_CALL, _GC_CHAN_PTR, _GC_SLICE:
  1631				argcnt = 2
  1632			case _GC_ARRAY_START, _GC_REGION:
  1633				argcnt = 3
  1634			default:
  1635				panic("unknown GC program op for " + *t.string + ": " + strconv.FormatUint(*(*uint64)(p), 10))
  1636			}
  1637			for i := 0; i < argcnt+1; i++ {
  1638				gc = append(gc, *(*uintptr)(p))
  1639				p = unsafe.Pointer(uintptr(p) + unsafe.Sizeof(uintptr(0)))
  1640			}
  1641		}
  1642		return gc
  1643	}

  1776	
  1777		// Make an array type.
  1778		var iarray interface{} = [1]unsafe.Pointer{}
  1779		prototype := *(**arrayType)(unsafe.Pointer(&iarray))
  1780		array := new(arrayType)
  1781		*array = *prototype
  1782		array.string = &s
  1783		array.hash = fnv1(typ.hash, '[')
  1784		for n := uint32(count); n > 0; n >>= 8 {
  1785			array.hash = fnv1(array.hash, byte(n))
  1786		}
  1787		array.hash = fnv1(array.hash, ']')
  1788		array.elem = typ
  1789		max := ^uintptr(0) / typ.size
  1790		if uintptr(count) > max {
  1791			panic("reflect.ArrayOf: array size would exceed virtual address space")
  1792		}
  1793		array.size = typ.size * uintptr(count)
  1794		array.align = typ.align
  1795		array.fieldAlign = typ.fieldAlign
  1796		// TODO: array.alg
  1797		// TODO: array.gc
  1798		array.uncommonType = nil
  1799		array.ptrToThis = nil
  1800		array.zero = unsafe.Pointer(&make([]byte, array.size)[0])
  1801		array.len = uintptr(count)
  1802		array.slice = slice.(*rtype)
  1803

  1805	}

  1807	// toType converts from a *rtype to a Type that can be returned
  1808	// to the client of package reflect. In gc, the only concern is that
  1809	// a nil *rtype must be replaced by a nil Type, but in gccgo this
  1810	// function takes care of ensuring that multiple *rtype for the same
  1811	// type are coalesced into a single Type.
  1812	func toType(t *rtype) Type {
  1813		if t == nil {
  1814			return nil
  1815		}
  1816		return t
  1817	}

  1819	type layoutKey struct {
  1820		t    *rtype // function signature
  1821		rcvr *rtype // receiver type, or nil if none
  1822	}
  1823	
  1824	type layoutType struct {
  1825		t         *rtype
  1826		argSize   uintptr // size of arguments
  1827		retOffset uintptr // offset of return values.
  1828	}
  1829	
  1830	var layoutCache struct {
  1831		sync.RWMutex
  1832		m map[layoutKey]layoutType
  1833	}
  1834	
  1835	// funcLayout computes a struct type representing the layout of the
  1836	// function arguments and return values for the function type t.
  1837	// If rcvr != nil, rcvr specifies the type of the receiver.
  1838	// The returned type exists only for GC, so we only fill out GC relevant info.
  1839	// Currently, that's just size and the GC program.  We also fill in
  1840	// the name for possible debugging use.
  1841	func funcLayout(t *rtype, rcvr *rtype) (frametype *rtype, argSize, retOffset uintptr) {
  1842		if t.Kind() != Func {
  1843			panic("reflect: funcLayout of non-func type")
  1844		}
  1845		if rcvr != nil && rcvr.Kind() == Interface {
  1846			panic("reflect: funcLayout with interface receiver " + rcvr.String())
  1847		}
  1848		k := layoutKey{t, rcvr}
  1849		layoutCache.RLock()
  1850		if x := layoutCache.m[k]; x.t != nil {
  1851			layoutCache.RUnlock()
  1852			return x.t, x.argSize, x.retOffset
  1853		}
  1854		layoutCache.RUnlock()
  1855		layoutCache.Lock()
  1856		if x := layoutCache.m[k]; x.t != nil {
  1857			layoutCache.Unlock()
  1858			return x.t, x.argSize, x.retOffset
  1859		}
  1860	
  1861		tt := (*funcType)(unsafe.Pointer(t))
  1862	
  1863		// compute gc program for arguments
  1864		gc := make([]uintptr, 1) // first entry is size, filled in at the end
  1865		offset := uintptr(0)
  1866		if rcvr != nil {
  1867			// Reflect uses the "interface" calling convention for
  1868			// methods, where receivers take one word of argument
  1869			// space no matter how big they actually are.
  1870			if rcvr.size > ptrSize {
  1871				// we pass a pointer to the receiver.
  1872				gc = append(gc, _GC_PTR, offset, uintptr(rcvr.gc))
  1873			} else if rcvr.pointers() {
  1874				// rcvr is a one-word pointer object.  Its gc program
  1875				// is just what we need here.
  1876				gc = appendGCProgram(gc, rcvr)
  1877			}
  1878			offset += ptrSize
  1879		}
  1880		for _, arg := range tt.in {
  1881			offset = align(offset, uintptr(arg.align))
  1882			if arg.pointers() {
  1883				gc = append(gc, _GC_REGION, offset, arg.size, uintptr(arg.gc))
  1884			}
  1885			offset += arg.size
  1886		}
  1887		argSize = offset
  1888		if runtime.GOARCH == "amd64p32" {
  1889			offset = align(offset, 8)
  1890		}
  1891		offset = align(offset, ptrSize)
  1892		retOffset = offset
  1893		for _, res := range tt.out {
  1894			offset = align(offset, uintptr(res.align))
  1895			if res.pointers() {
  1896				gc = append(gc, _GC_REGION, offset, res.size, uintptr(res.gc))
  1897			}
  1898			offset += res.size
  1899		}
  1900		gc = append(gc, _GC_END)
  1901		gc[0] = offset
  1902	
  1903		// build dummy rtype holding gc program
  1904		x := new(rtype)
  1905		x.size = offset
  1906		x.gc = unsafe.Pointer(&gc[0])
  1907		var s string
  1908		if rcvr != nil {
  1909			s = "methodargs(" + *rcvr.string + ")(" + *t.string + ")"
  1910		} else {
  1911			s = "funcargs(" + *t.string + ")"
  1912		}
  1913		x.string = &s
  1914	
  1915		// cache result for future callers
  1916		if layoutCache.m == nil {
  1917			layoutCache.m = make(map[layoutKey]layoutType)
  1918		}
  1919		layoutCache.m[k] = layoutType{
  1920			t:         x,
  1921			argSize:   argSize,
  1922			retOffset: retOffset,
  1923		}
  1924		layoutCache.Unlock()
  1925		return x, argSize, retOffset
  1926	}
