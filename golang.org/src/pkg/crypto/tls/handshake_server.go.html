    19	// serverHandshakeState contains details of a server handshake in progress.
    20	// It's discarded once the handshake has completed.
    21	type serverHandshakeState struct {
    22		c               *Conn
    23		clientHello     *clientHelloMsg
    24		hello           *serverHelloMsg
    25		suite           *cipherSuite
    26		ellipticOk      bool
    27		ecdsaOk         bool
    28		sessionState    *sessionState
    29		finishedHash    finishedHash
    30		masterSecret    []byte
    31		certsFromClient [][]byte
    32		cert            *Certificate
    33	}

    35	// serverHandshake performs a TLS handshake as a server.
    36	func (c *Conn) serverHandshake() error {
    37		config := c.config
    38	
    39		// If this is the first server handshake, we generate a random key to
    40		// encrypt the tickets with.
    41		config.serverInitOnce.Do(config.serverInit)
    42	
    43		hs := serverHandshakeState{
    44			c: c,
    45		}
    46		isResume, err := hs.readClientHello()
    47		if err != nil {
    48			return err
    49		}
    50	
    51		// For an overview of TLS handshaking, see https://tools.ietf.org/html/rfc5246#section-7.3
    52		if isResume {
    53			// The client has included a session ticket and so we do an abbreviated handshake.
    54			if err := hs.doResumeHandshake(); err != nil {
    55				return err
    56			}
    57			if err := hs.establishKeys(); err != nil {
    58				return err
    59			}
    60			if err := hs.sendFinished(); err != nil {
    61				return err
    62			}
    63			if err := hs.readFinished(); err != nil {
    64				return err
    65			}
    66			c.didResume = true
    67		} else {
    68			// The client didn't include a session ticket, or it wasn't
    69			// valid so we do a full handshake.
    70			if err := hs.doFullHandshake(); err != nil {
    71				return err
    72			}
    73			if err := hs.establishKeys(); err != nil {
    74				return err
    75			}
    76			if err := hs.readFinished(); err != nil {
    77				return err
    78			}
    79			if err := hs.sendSessionTicket(); err != nil {
    80				return err
    81			}
    82			if err := hs.sendFinished(); err != nil {
    83				return err
    84			}
    85		}
    86		c.handshakeComplete = true
    87	
    88		return nil
    89	}

    91	// readClientHello reads a ClientHello message from the client and decides
    92	// whether we will perform session resumption.
    93	func (hs *serverHandshakeState) readClientHello() (isResume bool, err error) {
    94		config := hs.c.config
    95		c := hs.c
    96	
    97		msg, err := c.readHandshake()
    98		if err != nil {
    99			return false, err
   100		}
   101		var ok bool
   102		hs.clientHello, ok = msg.(*clientHelloMsg)
   103		if !ok {
   104			c.sendAlert(alertUnexpectedMessage)
   105			return false, unexpectedMessageError(hs.clientHello, msg)
   106		}
   107		c.vers, ok = config.mutualVersion(hs.clientHello.vers)
   108		if !ok {
   109			c.sendAlert(alertProtocolVersion)
   110			return false, fmt.Errorf("tls: client offered an unsupported, maximum protocol version of %x", hs.clientHello.vers)
   111		}
   112		c.haveVers = true
   113	
   114		hs.finishedHash = newFinishedHash(c.vers)
   115		hs.finishedHash.Write(hs.clientHello.marshal())
   116	
   117		hs.hello = new(serverHelloMsg)
   118	
   119		supportedCurve := false
   120		preferredCurves := config.curvePreferences()
   121	Curves:
   122		for _, curve := range hs.clientHello.supportedCurves {
   123			for _, supported := range preferredCurves {
   124				if supported == curve {
   125					supportedCurve = true
   126					break Curves
   127				}
   128			}
   129		}
   130	
   131		supportedPointFormat := false
   132		for _, pointFormat := range hs.clientHello.supportedPoints {
   133			if pointFormat == pointFormatUncompressed {
   134				supportedPointFormat = true
   135				break
   136			}
   137		}
   138		hs.ellipticOk = supportedCurve && supportedPointFormat
   139	
   140		foundCompression := false
   141		// We only support null compression, so check that the client offered it.
   142		for _, compression := range hs.clientHello.compressionMethods {
   143			if compression == compressionNone {
   144				foundCompression = true
   145				break
   146			}
   147		}
   148	
   149		if !foundCompression {
   150			c.sendAlert(alertHandshakeFailure)
   151			return false, errors.New("tls: client does not support uncompressed connections")
   152		}
   153	
   154		hs.hello.vers = c.vers
   155		hs.hello.random = make([]byte, 32)
   156		_, err = io.ReadFull(config.rand(), hs.hello.random)
   157		if err != nil {
   158			c.sendAlert(alertInternalError)
   159			return false, err
   160		}
   161		hs.hello.secureRenegotiation = hs.clientHello.secureRenegotiation
   162		hs.hello.compressionMethod = compressionNone
   163		if len(hs.clientHello.serverName) > 0 {
   164			c.serverName = hs.clientHello.serverName
   165		}
   166		// Although sending an empty NPN extension is reasonable, Firefox has
   167		// had a bug around this. Best to send nothing at all if
   168		// config.NextProtos is empty. See
   169		// https://code.google.com/p/go/issues/detail?id=5445.
   170		if hs.clientHello.nextProtoNeg && len(config.NextProtos) > 0 {
   171			hs.hello.nextProtoNeg = true
   172			hs.hello.nextProtos = config.NextProtos
   173		}
   174	
   175		if len(config.Certificates) == 0 {
   176			c.sendAlert(alertInternalError)
   177			return false, errors.New("tls: no certificates configured")
   178		}
   179		hs.cert = &config.Certificates[0]
   180		if len(hs.clientHello.serverName) > 0 {
   181			hs.cert = config.getCertificateForName(hs.clientHello.serverName)
   182		}
   183	
   184		_, hs.ecdsaOk = hs.cert.PrivateKey.(*ecdsa.PrivateKey)
   185	
   186		if hs.checkForResumption() {
   187			return true, nil
   188		}
   189	
   190		var preferenceList, supportedList []uint16
   191		if c.config.PreferServerCipherSuites {
   192			preferenceList = c.config.cipherSuites()
   193			supportedList = hs.clientHello.cipherSuites
   194		} else {
   195			preferenceList = hs.clientHello.cipherSuites
   196			supportedList = c.config.cipherSuites()
   197		}
   198	
   199		for _, id := range preferenceList {
   200			if hs.suite = c.tryCipherSuite(id, supportedList, c.vers, hs.ellipticOk, hs.ecdsaOk); hs.suite != nil {
   201				break
   202			}
   203		}
   204	
   205		if hs.suite == nil {
   206			c.sendAlert(alertHandshakeFailure)
   207			return false, errors.New("tls: no cipher suite supported by both client and server")
   208		}
   209	
   210		return false, nil
   211	}

   213	// checkForResumption returns true if we should perform resumption on this connection.
   214	func (hs *serverHandshakeState) checkForResumption() bool {
   215		c := hs.c
   216	
   217		if c.config.SessionTicketsDisabled {
   218			return false
   219		}
   220	
   221		var ok bool
   222		if hs.sessionState, ok = c.decryptTicket(hs.clientHello.sessionTicket); !ok {
   223			return false
   224		}
   225	
   226		if hs.sessionState.vers > hs.clientHello.vers {
   227			return false
   228		}
   229		if vers, ok := c.config.mutualVersion(hs.sessionState.vers); !ok || vers != hs.sessionState.vers {
   230			return false
   231		}
   232	
   233		cipherSuiteOk := false
   234		// Check that the client is still offering the ciphersuite in the session.
   235		for _, id := range hs.clientHello.cipherSuites {
   236			if id == hs.sessionState.cipherSuite {
   237				cipherSuiteOk = true
   238				break
   239			}
   240		}
   241		if !cipherSuiteOk {
   242			return false
   243		}
   244	
   245		// Check that we also support the ciphersuite from the session.
   246		hs.suite = c.tryCipherSuite(hs.sessionState.cipherSuite, c.config.cipherSuites(), hs.sessionState.vers, hs.ellipticOk, hs.ecdsaOk)
   247		if hs.suite == nil {
   248			return false
   249		}
   250	
   251		sessionHasClientCerts := len(hs.sessionState.certificates) != 0
   252		needClientCerts := c.config.ClientAuth == RequireAnyClientCert || c.config.ClientAuth == RequireAndVerifyClientCert
   253		if needClientCerts && !sessionHasClientCerts {
   254			return false
   255		}
   256		if sessionHasClientCerts && c.config.ClientAuth == NoClientCert {
   257			return false
   258		}
   259	
   260		return true
   261	}

   263	func (hs *serverHandshakeState) doResumeHandshake() error {
   264		c := hs.c
   265	
   266		hs.hello.cipherSuite = hs.suite.id
   267		// We echo the client's session ID in the ServerHello to let it know
   268		// that we're doing a resumption.
   269		hs.hello.sessionId = hs.clientHello.sessionId
   270		hs.finishedHash.Write(hs.hello.marshal())
   271		c.writeRecord(recordTypeHandshake, hs.hello.marshal())
   272	
   273		if len(hs.sessionState.certificates) > 0 {
   274			if _, err := hs.processCertsFromClient(hs.sessionState.certificates); err != nil {
   275				return err
   276			}
   277		}
   278	
   279		hs.masterSecret = hs.sessionState.masterSecret
   280	
   281		return nil
   282	}

   284	func (hs *serverHandshakeState) doFullHandshake() error {
   285		config := hs.c.config
   286		c := hs.c
   287	
   288		if hs.clientHello.ocspStapling && len(hs.cert.OCSPStaple) > 0 {
   289			hs.hello.ocspStapling = true
   290		}
   291	
   292		hs.hello.ticketSupported = hs.clientHello.ticketSupported && !config.SessionTicketsDisabled
   293		hs.hello.cipherSuite = hs.suite.id
   294		hs.finishedHash.Write(hs.hello.marshal())
   295		c.writeRecord(recordTypeHandshake, hs.hello.marshal())
   296	
   297		certMsg := new(certificateMsg)
   298		certMsg.certificates = hs.cert.Certificate
   299		hs.finishedHash.Write(certMsg.marshal())
   300		c.writeRecord(recordTypeHandshake, certMsg.marshal())
   301	
   302		if hs.hello.ocspStapling {
   303			certStatus := new(certificateStatusMsg)
   304			certStatus.statusType = statusTypeOCSP
   305			certStatus.response = hs.cert.OCSPStaple
   306			hs.finishedHash.Write(certStatus.marshal())
   307			c.writeRecord(recordTypeHandshake, certStatus.marshal())
   308		}
   309	
   310		keyAgreement := hs.suite.ka(c.vers)
   311		skx, err := keyAgreement.generateServerKeyExchange(config, hs.cert, hs.clientHello, hs.hello)
   312		if err != nil {
   313			c.sendAlert(alertHandshakeFailure)
   314			return err
   315		}
   316		if skx != nil {
   317			hs.finishedHash.Write(skx.marshal())
   318			c.writeRecord(recordTypeHandshake, skx.marshal())
   319		}
   320	
   321		if config.ClientAuth >= RequestClientCert {
   322			// Request a client certificate
   323			certReq := new(certificateRequestMsg)
   324			certReq.certificateTypes = []byte{
   325				byte(certTypeRSASign),
   326				byte(certTypeECDSASign),
   327			}
   328			if c.vers >= VersionTLS12 {
   329				certReq.hasSignatureAndHash = true
   330				certReq.signatureAndHashes = supportedClientCertSignatureAlgorithms
   331			}
   332	
   333			// An empty list of certificateAuthorities signals to
   334			// the client that it may send any certificate in response
   335			// to our request. When we know the CAs we trust, then
   336			// we can send them down, so that the client can choose
   337			// an appropriate certificate to give to us.
   338			if config.ClientCAs != nil {
   339				certReq.certificateAuthorities = config.ClientCAs.Subjects()
   340			}
   341			hs.finishedHash.Write(certReq.marshal())
   342			c.writeRecord(recordTypeHandshake, certReq.marshal())
   343		}
   344	
   345		helloDone := new(serverHelloDoneMsg)
   346		hs.finishedHash.Write(helloDone.marshal())
   347		c.writeRecord(recordTypeHandshake, helloDone.marshal())
   348	
   349		var pub crypto.PublicKey // public key for client auth, if any
   350	
   351		msg, err := c.readHandshake()
   352		if err != nil {
   353			return err
   354		}
   355	
   356		var ok bool
   357		// If we requested a client certificate, then the client must send a
   358		// certificate message, even if it's empty.
   359		if config.ClientAuth >= RequestClientCert {
   360			if certMsg, ok = msg.(*certificateMsg); !ok {
   361				c.sendAlert(alertUnexpectedMessage)
   362				return unexpectedMessageError(certMsg, msg)
   363			}
   364			hs.finishedHash.Write(certMsg.marshal())
   365	
   366			if len(certMsg.certificates) == 0 {
   367				// The client didn't actually send a certificate
   368				switch config.ClientAuth {
   369				case RequireAnyClientCert, RequireAndVerifyClientCert:
   370					c.sendAlert(alertBadCertificate)
   371					return errors.New("tls: client didn't provide a certificate")
   372				}
   373			}
   374	
   375			pub, err = hs.processCertsFromClient(certMsg.certificates)
   376			if err != nil {
   377				return err
   378			}
   379	
   380			msg, err = c.readHandshake()
   381			if err != nil {
   382				return err
   383			}
   384		}
   385	
   386		// Get client key exchange
   387		ckx, ok := msg.(*clientKeyExchangeMsg)
   388		if !ok {
   389			c.sendAlert(alertUnexpectedMessage)
   390			return unexpectedMessageError(ckx, msg)
   391		}
   392		hs.finishedHash.Write(ckx.marshal())
   393	
   394		// If we received a client cert in response to our certificate request message,
   395		// the client will send us a certificateVerifyMsg immediately after the
   396		// clientKeyExchangeMsg.  This message is a digest of all preceding
   397		// handshake-layer messages that is signed using the private key corresponding
   398		// to the client's certificate. This allows us to verify that the client is in
   399		// possession of the private key of the certificate.
   400		if len(c.peerCertificates) > 0 {
   401			msg, err = c.readHandshake()
   402			if err != nil {
   403				return err
   404			}
   405			certVerify, ok := msg.(*certificateVerifyMsg)
   406			if !ok {
   407				c.sendAlert(alertUnexpectedMessage)
   408				return unexpectedMessageError(certVerify, msg)
   409			}
   410	
   411			switch key := pub.(type) {
   412			case *ecdsa.PublicKey:
   413				ecdsaSig := new(ecdsaSignature)
   414				if _, err = asn1.Unmarshal(certVerify.signature, ecdsaSig); err != nil {
   415					break
   416				}
   417				if ecdsaSig.R.Sign() <= 0 || ecdsaSig.S.Sign() <= 0 {
   418					err = errors.New("ECDSA signature contained zero or negative values")
   419					break
   420				}
   421				digest, _, _ := hs.finishedHash.hashForClientCertificate(signatureECDSA)
   422				if !ecdsa.Verify(key, digest, ecdsaSig.R, ecdsaSig.S) {
   423					err = errors.New("ECDSA verification failure")
   424					break
   425				}
   426			case *rsa.PublicKey:
   427				digest, hashFunc, _ := hs.finishedHash.hashForClientCertificate(signatureRSA)
   428				err = rsa.VerifyPKCS1v15(key, hashFunc, digest, certVerify.signature)
   429			}
   430			if err != nil {
   431				c.sendAlert(alertBadCertificate)
   432				return errors.New("could not validate signature of connection nonces: " + err.Error())
   433			}
   434	
   435			hs.finishedHash.Write(certVerify.marshal())
   436		}
   437	
   438		preMasterSecret, err := keyAgreement.processClientKeyExchange(config, hs.cert, ckx, c.vers)
   439		if err != nil {
   440			c.sendAlert(alertHandshakeFailure)
   441			return err
   442		}
   443		hs.masterSecret = masterFromPreMasterSecret(c.vers, preMasterSecret, hs.clientHello.random, hs.hello.random)
   444	
   445		return nil
   446	}

   448	func (hs *serverHandshakeState) establishKeys() error {
   449		c := hs.c
   450	
   451		clientMAC, serverMAC, clientKey, serverKey, clientIV, serverIV :=
   452			keysFromMasterSecret(c.vers, hs.masterSecret, hs.clientHello.random, hs.hello.random, hs.suite.macLen, hs.suite.keyLen, hs.suite.ivLen)
   453	
   454		var clientCipher, serverCipher interface{}
   455		var clientHash, serverHash macFunction
   456	
   457		if hs.suite.aead == nil {
   458			clientCipher = hs.suite.cipher(clientKey, clientIV, true /* for reading */)
   459			clientHash = hs.suite.mac(c.vers, clientMAC)
   460			serverCipher = hs.suite.cipher(serverKey, serverIV, false /* not for reading */)
   461			serverHash = hs.suite.mac(c.vers, serverMAC)
   462		} else {
   463			clientCipher = hs.suite.aead(clientKey, clientIV)
   464			serverCipher = hs.suite.aead(serverKey, serverIV)
   465		}
   466	
   467		c.in.prepareCipherSpec(c.vers, clientCipher, clientHash)
   468		c.out.prepareCipherSpec(c.vers, serverCipher, serverHash)
   469	
   470		return nil
   471	}

   473	func (hs *serverHandshakeState) readFinished() error {
   474		c := hs.c
   475	
   476		c.readRecord(recordTypeChangeCipherSpec)
   477		if err := c.in.error(); err != nil {
   478			return err
   479		}
   480	
   481		if hs.hello.nextProtoNeg {
   482			msg, err := c.readHandshake()
   483			if err != nil {
   484				return err
   485			}
   486			nextProto, ok := msg.(*nextProtoMsg)
   487			if !ok {
   488				c.sendAlert(alertUnexpectedMessage)
   489				return unexpectedMessageError(nextProto, msg)
   490			}
   491			hs.finishedHash.Write(nextProto.marshal())
   492			c.clientProtocol = nextProto.proto
   493		}
   494	
   495		msg, err := c.readHandshake()
   496		if err != nil {
   497			return err
   498		}
   499		clientFinished, ok := msg.(*finishedMsg)
   500		if !ok {
   501			c.sendAlert(alertUnexpectedMessage)
   502			return unexpectedMessageError(clientFinished, msg)
   503		}
   504	
   505		verify := hs.finishedHash.clientSum(hs.masterSecret)
   506		if len(verify) != len(clientFinished.verifyData) ||
   507			subtle.ConstantTimeCompare(verify, clientFinished.verifyData) != 1 {
   508			c.sendAlert(alertHandshakeFailure)
   509			return errors.New("tls: client's Finished message is incorrect")
   510		}
   511	
   512		hs.finishedHash.Write(clientFinished.marshal())
   513		return nil
   514	}

   516	func (hs *serverHandshakeState) sendSessionTicket() error {
   517		if !hs.hello.ticketSupported {
   518			return nil
   519		}
   520	
   521		c := hs.c
   522		m := new(newSessionTicketMsg)
   523	
   524		var err error
   525		state := sessionState{
   526			vers:         c.vers,
   527			cipherSuite:  hs.suite.id,
   528			masterSecret: hs.masterSecret,
   529			certificates: hs.certsFromClient,
   530		}
   531		m.ticket, err = c.encryptTicket(&state)
   532		if err != nil {
   533			return err
   534		}
   535	
   536		hs.finishedHash.Write(m.marshal())
   537		c.writeRecord(recordTypeHandshake, m.marshal())
   538	
   539		return nil
   540	}

   542	func (hs *serverHandshakeState) sendFinished() error {
   543		c := hs.c
   544	
   545		c.writeRecord(recordTypeChangeCipherSpec, []byte{1})
   546	
   547		finished := new(finishedMsg)
   548		finished.verifyData = hs.finishedHash.serverSum(hs.masterSecret)
   549		hs.finishedHash.Write(finished.marshal())
   550		c.writeRecord(recordTypeHandshake, finished.marshal())
   551	
   552		c.cipherSuite = hs.suite.id
   553	
   554		return nil
   555	}

   557	// processCertsFromClient takes a chain of client certificates either from a
   558	// Certificates message or from a sessionState and verifies them. It returns
   559	// the public key of the leaf certificate.
   560	func (hs *serverHandshakeState) processCertsFromClient(certificates [][]byte) (crypto.PublicKey, error) {
   561		c := hs.c
   562	
   563		hs.certsFromClient = certificates
   564		certs := make([]*x509.Certificate, len(certificates))
   565		var err error
   566		for i, asn1Data := range certificates {
   567			if certs[i], err = x509.ParseCertificate(asn1Data); err != nil {
   568				c.sendAlert(alertBadCertificate)
   569				return nil, errors.New("tls: failed to parse client certificate: " + err.Error())
   570			}
   571		}
   572	
   573		if c.config.ClientAuth >= VerifyClientCertIfGiven && len(certs) > 0 {
   574			opts := x509.VerifyOptions{
   575				Roots:         c.config.ClientCAs,
   576				CurrentTime:   c.config.time(),
   577				Intermediates: x509.NewCertPool(),
   578				KeyUsages:     []x509.ExtKeyUsage{x509.ExtKeyUsageClientAuth},
   579			}
   580	
   581			for _, cert := range certs[1:] {
   582				opts.Intermediates.AddCert(cert)
   583			}
   584	
   585			chains, err := certs[0].Verify(opts)
   586			if err != nil {
   587				c.sendAlert(alertBadCertificate)
   588				return nil, errors.New("tls: failed to verify client's certificate: " + err.Error())
   589			}
   590	
   591			ok := false
   592			for _, ku := range certs[0].ExtKeyUsage {
   593				if ku == x509.ExtKeyUsageClientAuth {
   594					ok = true
   595					break
   596				}
   597			}
   598			if !ok {
   599				c.sendAlert(alertHandshakeFailure)
   600				return nil, errors.New("tls: client's certificate's extended key usage doesn't permit it to be used for client authentication")
   601			}
   602	
   603			c.verifiedChains = chains
   604		}
   605	
   606		if len(certs) > 0 {
   607			var pub crypto.PublicKey
   608			switch key := certs[0].PublicKey.(type) {
   609			case *ecdsa.PublicKey, *rsa.PublicKey:
   610				pub = key
   611			default:
   612				c.sendAlert(alertUnsupportedCertificate)
   613				return nil, fmt.Errorf("tls: client's certificate contains an unsupported public key of type %T", certs[0].PublicKey)
   614			}
   615			c.peerCertificates = certs
   616			return pub, nil
   617		}
   618	
   619		return nil, nil
   620	}

   622	// tryCipherSuite returns a cipherSuite with the given id if that cipher suite
   623	// is acceptable to use.
   624	func (c *Conn) tryCipherSuite(id uint16, supportedCipherSuites []uint16, version uint16, ellipticOk, ecdsaOk bool) *cipherSuite {
   625		for _, supported := range supportedCipherSuites {
   626			if id == supported {
   627				var candidate *cipherSuite
   628	
   629				for _, s := range cipherSuites {
   630					if s.id == id {
   631						candidate = s
   632						break
   633					}
   634				}
   635				if candidate == nil {
   636					continue
   637				}
   638				// Don't select a ciphersuite which we can't
   639				// support for this client.
   640				if (candidate.flags&suiteECDHE != 0) && !ellipticOk {
   641					continue
   642				}
   643				if (candidate.flags&suiteECDSA != 0) != ecdsaOk {
   644					continue
   645				}
   646				if version < VersionTLS12 && candidate.flags&suiteTLS12 != 0 {
   647					continue
   648				}
   649				return candidate
   650			}
   651		}
   652	
   653		return nil
