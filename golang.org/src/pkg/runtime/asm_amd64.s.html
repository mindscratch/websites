   191	// Called during function prolog when more stack is needed.
   192	// Caller has already done get_tls(CX); MOVQ m(CX), BX.
   193	//
   194	// The traceback routines see morestack on a g0 as being
   195	// the top of a stack (for example, morestack calling newstack
   196	// calling the scheduler calling newm calling gc), so we must
   197	// record an argument size. For that purpose, it has no arguments.
   198	TEXT runtime·morestack(SB),NOSPLIT,$0-0
   199		// Cannot grow scheduler stack (m->g0).
   200		MOVQ	m_g0(BX), SI
   201		CMPQ	g(CX), SI
   202		JNE	2(PC)
   203		INT	$3
   204	
   205		// Called from f.
   206		// Set m->morebuf to f's caller.
   207		MOVQ	8(SP), AX	// f's caller's PC
   208		MOVQ	AX, (m_morebuf+gobuf_pc)(BX)
   209		LEAQ	16(SP), AX	// f's caller's SP
   210		MOVQ	AX, (m_morebuf+gobuf_sp)(BX)
   211		MOVQ	AX, m_moreargp(BX)
   212		get_tls(CX)
   213		MOVQ	g(CX), SI
   214		MOVQ	SI, (m_morebuf+gobuf_g)(BX)
   215	
   216		// Set g->sched to context in f.
   217		MOVQ	0(SP), AX // f's PC
   218		MOVQ	AX, (g_sched+gobuf_pc)(SI)
   219		MOVQ	SI, (g_sched+gobuf_g)(SI)
   220		LEAQ	8(SP), AX // f's SP
   221		MOVQ	AX, (g_sched+gobuf_sp)(SI)
   222		MOVQ	DX, (g_sched+gobuf_ctxt)(SI)
   223	
   224		// Call newstack on m->g0's stack.
   225		MOVQ	m_g0(BX), BP
   226		MOVQ	BP, g(CX)
   227		MOVQ	(g_sched+gobuf_sp)(BP), SP
   228		CALL	runtime·newstack(SB)
   229		MOVQ	$0, 0x1003	// crash if newstack returns
   230		RET

   232	// Called from panic.  Mimics morestack,
   233	// reuses stack growth code to create a frame
   234	// with the desired args running the desired function.
   235	//
   236	// func call(fn *byte, arg *byte, argsize uint32).
   237	TEXT runtime·newstackcall(SB), NOSPLIT, $0-20
   238		get_tls(CX)
   239		MOVQ	m(CX), BX
   240	
   241		// Save our caller's state as the PC and SP to
   242		// restore when returning from f.
   243		MOVQ	0(SP), AX	// our caller's PC
   244		MOVQ	AX, (m_morebuf+gobuf_pc)(BX)
   245		LEAQ	8(SP), AX	// our caller's SP
   246		MOVQ	AX, (m_morebuf+gobuf_sp)(BX)
   247		MOVQ	g(CX), AX
   248		MOVQ	AX, (m_morebuf+gobuf_g)(BX)
   249		
   250		// Save our own state as the PC and SP to restore
   251		// if this goroutine needs to be restarted.
   252		MOVQ	$runtime·newstackcall(SB), (g_sched+gobuf_pc)(AX)
   253		MOVQ	SP, (g_sched+gobuf_sp)(AX)
   254	
   255		// Set up morestack arguments to call f on a new stack.
   256		// We set f's frame size to 1, as a hint to newstack
   257		// that this is a call from runtime·newstackcall.
   258		// If it turns out that f needs a larger frame than
   259		// the default stack, f's usual stack growth prolog will
   260		// allocate a new segment (and recopy the arguments).
   261		MOVQ	8(SP), AX	// fn
   262		MOVQ	16(SP), DX	// arg frame
   263		MOVL	24(SP), CX	// arg size
   264	
   265		MOVQ	AX, m_cret(BX)	// f's PC
   266		MOVQ	DX, m_moreargp(BX)	// argument frame pointer
   267		MOVL	CX, m_moreargsize(BX)	// f's argument size
   268		MOVL	$1, m_moreframesize(BX)	// f's frame size
   269	
   270		// Call newstack on m->g0's stack.
   271		MOVQ	m_g0(BX), BP
   272		get_tls(CX)
   273		MOVQ	BP, g(CX)
   274		MOVQ	(g_sched+gobuf_sp)(BP), SP
   275		CALL	runtime·newstack(SB)
   276		MOVQ	$0, 0x1103	// crash if newstack returns
   277		RET
