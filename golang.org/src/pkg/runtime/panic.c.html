    16	// Each P holds pool for defers with arg sizes 8, 24, 40, 56 and 72 bytes.
    17	// Memory block is 40 (24 for 32 bits) bytes larger due to Defer header.
    18	// This maps exactly to malloc size classes.
    19	
    20	// defer size class for arg size sz
    21	#define DEFERCLASS(sz) (((sz)+7)>>4)
    22	// total size of memory block for defer with arg size sz
    23	#define TOTALSIZE(sz) (sizeof(Defer) - sizeof(((Defer*)nil)->args) + ROUND(sz, sizeof(uintptr)))
    24	
    25	// Allocate a Defer, usually using per-P pool.
    26	// Each defer must be released with freedefer.
    27	static Defer*
    28	newdefer(int32 siz)
    29	{
    30		int32 total, sc;
    31		Defer *d;
    32		P *p;
    33	
    34		d = nil;
    35		sc = DEFERCLASS(siz);
    36		if(sc < nelem(p->deferpool)) {
    37			p = m->p;
    38			d = p->deferpool[sc];
    39			if(d)
    40				p->deferpool[sc] = d->link;
    41		}
    42		if(d == nil) {
    43			// deferpool is empty or just a big defer
    44			total = TOTALSIZE(siz);
    45			d = runtime·malloc(total);
    46		}
    47		d->siz = siz;
    48		d->special = 0;
    49		d->link = g->defer;
    50		g->defer = d;
    51		return d;
    52	}

    54	// Free the given defer.
    55	// The defer cannot be used after this call.
    56	static void
    57	freedefer(Defer *d)
    58	{
    59		int32 sc;
    60		P *p;
    61	
    62		if(d->special)
    63			return;
    64		sc = DEFERCLASS(d->siz);
    65		if(sc < nelem(p->deferpool)) {
    66			p = m->p;
    67			d->link = p->deferpool[sc];
    68			p->deferpool[sc] = d;
    69			// No need to wipe out pointers in argp/pc/fn/args,
    70			// because we empty the pool before GC.
    71		} else
    72			runtime·free(d);
    73	}

    75	// Create a new deferred function fn with siz bytes of arguments.
    76	// The compiler turns a defer statement into a call to this.
    77	// Cannot split the stack because it assumes that the arguments
    78	// are available sequentially after &fn; they would not be
    79	// copied if a stack split occurred.  It's OK for this to call
    80	// functions that split the stack.
    81	#pragma textflag NOSPLIT
    82	uintptr
    83	runtime·deferproc(int32 siz, FuncVal *fn, ...)
    84	{
    85		Defer *d;
    86	
    87		d = newdefer(siz);
    88		d->fn = fn;
    89		d->pc = runtime·getcallerpc(&siz);
    90		if(thechar == '5')
    91			d->argp = (byte*)(&fn+2);  // skip caller's saved link register
    92		else
    93			d->argp = (byte*)(&fn+1);
    94		runtime·memmove(d->args, d->argp, d->siz);
    95	
    96		// deferproc returns 0 normally.
    97		// a deferred func that stops a panic
    98		// makes the deferproc return 1.
    99		// the code the compiler generates always
   100		// checks the return value and jumps to the
   101		// end of the function if deferproc returns != 0.
   102		return 0;
   103	}
   104	
   105	// Run a deferred function if there is one.
   106	// The compiler inserts a call to this at the end of any
   107	// function which calls defer.
   108	// If there is a deferred function, this will call runtime·jmpdefer,
   109	// which will jump to the deferred function such that it appears
   110	// to have been called by the caller of deferreturn at the point
   111	// just before deferreturn was called.  The effect is that deferreturn
   112	// is called again and again until there are no more deferred functions.
   113	// Cannot split the stack because we reuse the caller's frame to
   114	// call the deferred function.
   115	
   116	// The single argument isn't actually used - it just has its address
   117	// taken so it can be matched against pending defers.
   118	#pragma textflag NOSPLIT
   119	void
   120	runtime·deferreturn(uintptr arg0)
   121	{
   122		Defer *d;
   123		byte *argp;
   124		FuncVal *fn;
   125	
   126		d = g->defer;
   127		if(d == nil)
   128			return;
   129		argp = (byte*)&arg0;
   130		if(d->argp != argp)
   131			return;
   132	
   133		// Moving arguments around.
   134		// Do not allow preemption here, because the garbage collector
   135		// won't know the form of the arguments until the jmpdefer can
   136		// flip the PC over to fn.
   137		m->locks++;
   138		runtime·memmove(argp, d->args, d->siz);
   139		fn = d->fn;
   140		g->defer = d->link;
   141		freedefer(d);
   142		m->locks--;
   143		if(m->locks == 0 && g->preempt)
   144			g->stackguard0 = StackPreempt;
   145		runtime·jmpdefer(fn, argp);
   146	}

   148	// Ensure that defer arg sizes that map to the same defer size class
   149	// also map to the same malloc size class.
   150	void
   151	runtime·testdefersizes(void)
   152	{
   153		P *p;
   154		int32 i, siz, defersc, mallocsc;
   155		int32 map[nelem(p->deferpool)];
   156	
   157		for(i=0; i<nelem(p->deferpool); i++)
   158			map[i] = -1;
   159		for(i=0;; i++) {
   160			defersc = DEFERCLASS(i);
   161			if(defersc >= nelem(p->deferpool))
   162				break;
   163			siz = TOTALSIZE(i);
   164			mallocsc = runtime·SizeToClass(siz);
   165			siz = runtime·class_to_size[mallocsc];
   166			// runtime·printf("defer class %d: arg size %d, block size %d(%d)\n", defersc, i, siz, mallocsc);
   167			if(map[defersc] < 0) {
   168				map[defersc] = mallocsc;
   169				continue;
   170			}
   171			if(map[defersc] != mallocsc) {
   172				runtime·printf("bad defer size class: i=%d siz=%d mallocsc=%d/%d\n",
   173					i, siz, map[defersc], mallocsc);
   174				runtime·throw("bad defer size class");
   175			}
   176		}
   177	}

   179	// Run all deferred functions for the current goroutine.
   180	static void
   181	rundefer(void)
   182	{
   183		Defer *d;
   184	
   185		while((d = g->defer) != nil) {
   186			g->defer = d->link;
   187			reflect·call(d->fn, (byte*)d->args, d->siz, d->siz);
   188			freedefer(d);
   189		}
   190	}

   211	// The implementation of the predeclared function panic.
   212	void
   213	runtime·panic(Eface e)
   214	{
   215		Defer *d, dabort;
   216		Panic p;
   217		void *pc, *argp;
   218	
   219		runtime·memclr((byte*)&p, sizeof p);
   220		p.arg = e;
   221		p.link = g->panic;
   222		p.stackbase = g->stackbase;
   223		g->panic = &p;
   224	
   225		dabort.fn = &abortpanicV;
   226		dabort.siz = sizeof(&p);
   227		dabort.args[0] = &p;
   228		dabort.argp = NoArgs;
   229		dabort.special = true;
   230	
   231		for(;;) {
   232			d = g->defer;
   233			if(d == nil)
   234				break;
   235			// take defer off list in case of recursive panic
   236			g->defer = d->link;
   237			g->ispanic = true;	// rock for runtime·newstack, where runtime·newstackcall ends up
   238			argp = d->argp;
   239			pc = d->pc;
   240	
   241			// The deferred function may cause another panic,
   242			// so newstackcall may not return. Set up a defer
   243			// to mark this panic aborted if that happens.
   244			dabort.link = g->defer;
   245			g->defer = &dabort;
   246			p.defer = d;
   247	
   248			runtime·newstackcall(d->fn, (byte*)d->args, d->siz);
   249	
   250			// Newstackcall did not panic. Remove dabort.
   251			if(g->defer != &dabort)
   252				runtime·throw("bad defer entry in panic");
   253			g->defer = dabort.link;
   254	
   255			freedefer(d);
   256			if(p.recovered) {
   257				g->panic = p.link;
   258				// Aborted panics are marked but remain on the g->panic list.
   259				// Recovery will unwind the stack frames containing their Panic structs.
   260				// Remove them from the list and free the associated defers.
   261				while(g->panic && g->panic->aborted) {
   262					freedefer(g->panic->defer);
   263					g->panic = g->panic->link;
   264				}
   265				if(g->panic == nil)	// must be done with signal
   266					g->sig = 0;
   267				// Pass information about recovering frame to recovery.
   268				g->sigcode0 = (uintptr)argp;
   269				g->sigcode1 = (uintptr)pc;
   270				runtime·mcall(recovery);
   271				runtime·throw("recovery failed"); // mcall should not return
   272			}
   273		}
   274	
   275		// ran out of deferred calls - old-school panic now
   276		runtime·startpanic();
   277		printpanics(g->panic);
   278		runtime·dopanic(0);	// should not return
   279		runtime·exit(1);	// not reached
   280	}

   288	// Unwind the stack after a deferred function calls recover
   289	// after a panic.  Then arrange to continue running as though
   290	// the caller of the deferred function returned normally.
   291	static void
   292	recovery(G *gp)
   293	{
   294		void *argp;
   295		uintptr pc;
   296		
   297		// Info about defer passed in G struct.
   298		argp = (void*)gp->sigcode0;
   299		pc = (uintptr)gp->sigcode1;
   300	
   301		// Unwind to the stack frame with d's arguments in it.
   302		runtime·unwindstack(gp, argp);
   303	
   304		// Make the deferproc for this d return again,
   305		// this time returning 1.  The calling function will
   306		// jump to the standard return epilogue.
   307		// The -2*sizeof(uintptr) makes up for the
   308		// two extra words that are on the stack at
   309		// each call to deferproc.
   310		// (The pc we're returning to does pop pop
   311		// before it tests the return value.)
   312		// On the arm there are 2 saved LRs mixed in too.
   313		if(thechar == '5')
   314			gp->sched.sp = (uintptr)argp - 4*sizeof(uintptr);
   315		else
   316			gp->sched.sp = (uintptr)argp - 2*sizeof(uintptr);
   317		gp->sched.pc = pc;
   318		gp->sched.lr = 0;
   319		gp->sched.ret = 1;
   320		runtime·gogo(&gp->sched);
   321	}

   323	// Free stack frames until we hit the last one
   324	// or until we find the one that contains the sp.
   325	void
   326	runtime·unwindstack(G *gp, byte *sp)
   327	{
   328		Stktop *top;
   329		byte *stk;
   330	
   331		// Must be called from a different goroutine, usually m->g0.
   332		if(g == gp)
   333			runtime·throw("unwindstack on self");
   334	
   335		while((top = (Stktop*)gp->stackbase) != 0 && top->stackbase != 0) {
   336			stk = (byte*)gp->stackguard - StackGuard;
   337			if(stk <= sp && sp < (byte*)gp->stackbase)
   338				break;
   339			gp->stackbase = top->stackbase;
   340			gp->stackguard = top->stackguard;
   341			gp->stackguard0 = gp->stackguard;
   342			runtime·stackfree(gp, stk, top);
   343		}
   344	
   345		if(sp != nil && (sp < (byte*)gp->stackguard - StackGuard || (byte*)gp->stackbase < sp)) {
   346			runtime·printf("recover: %p not in [%p, %p]\n", sp, gp->stackguard - StackGuard, gp->stackbase);
   347			runtime·throw("bad unwindstack");
   348		}
   349	}

   351	// The implementation of the predeclared function recover.
   352	// Cannot split the stack because it needs to reliably
   353	// find the stack segment of its caller.
   354	#pragma textflag NOSPLIT
   355	void
   356	runtime·recover(byte *argp, GoOutput retbase, ...)
   357	{
   358		Panic *p;
   359		Stktop *top;
   360		Eface *ret;
   361	
   362		// Must be an unrecovered panic in progress.
   363		// Must be on a stack segment created for a deferred call during a panic.
   364		// Must be at the top of that segment, meaning the deferred call itself
   365		// and not something it called. The top frame in the segment will have
   366		// argument pointer argp == top - top->argsize.
   367		// The subtraction of g->panicwrap allows wrapper functions that
   368		// do not count as official calls to adjust what we consider the top frame
   369		// while they are active on the stack. The linker emits adjustments of
   370		// g->panicwrap in the prologue and epilogue of functions marked as wrappers.
   371		ret = (Eface*)&retbase;
   372		top = (Stktop*)g->stackbase;
   373		p = g->panic;
   374		if(p != nil && !p->recovered && top->panic && argp == (byte*)top - top->argsize - g->panicwrap) {
   375			p->recovered = 1;
   376			*ret = p->arg;
   377		} else {
   378			ret->type = nil;
   379			ret->data = nil;
   380		}
   381	}

   383	void
   384	runtime·startpanic(void)
   385	{
   386		if(runtime·mheap.cachealloc.size == 0) { // very early
   387			runtime·printf("runtime: panic before malloc heap initialized\n");
   388			m->mallocing = 1; // tell rest of panic not to try to malloc
   389		} else if(m->mcache == nil) // can happen if called from signal handler or throw
   390			m->mcache = runtime·allocmcache();
   391		switch(m->dying) {
   392		case 0:
   393			m->dying = 1;
   394			if(g != nil)
   395				g->writebuf = nil;
   396			runtime·xadd(&runtime·panicking, 1);
   397			runtime·lock(&paniclk);
   398			if(runtime·debug.schedtrace > 0 || runtime·debug.scheddetail > 0)
   399				runtime·schedtrace(true);
   400			runtime·freezetheworld();
   401			return;
   402		case 1:
   403			// Something failed while panicing, probably the print of the
   404			// argument to panic().  Just print a stack trace and exit.
   405			m->dying = 2;
   406			runtime·printf("panic during panic\n");
   407			runtime·dopanic(0);
   408			runtime·exit(3);
   409		case 2:
   410			// This is a genuine bug in the runtime, we couldn't even
   411			// print the stack trace successfully.
   412			m->dying = 3;
   413			runtime·printf("stack trace unavailable\n");
   414			runtime·exit(4);
   415		default:
   416			// Can't even print!  Just exit.
   417			runtime·exit(5);
   418		}
   419	}

   421	void
   422	runtime·dopanic(int32 unused)
   423	{
   424		static bool didothers;
   425		bool crash;
   426		int32 t;
   427	
   428		if(g->sig != 0)
   429			runtime·printf("[signal %x code=%p addr=%p pc=%p]\n",
   430				g->sig, g->sigcode0, g->sigcode1, g->sigpc);
   431	
   432		if((t = runtime·gotraceback(&crash)) > 0){
   433			if(g != m->g0) {
   434				runtime·printf("\n");
   435				runtime·goroutineheader(g);
   436				runtime·traceback((uintptr)runtime·getcallerpc(&unused), (uintptr)runtime·getcallersp(&unused), 0, g);
   437			} else if(t >= 2 || m->throwing > 0) {
   438				runtime·printf("\nruntime stack:\n");
   439				runtime·traceback((uintptr)runtime·getcallerpc(&unused), (uintptr)runtime·getcallersp(&unused), 0, g);
   440			}
   441			if(!didothers) {
   442				didothers = true;
   443				runtime·tracebackothers(g);
   444			}
   445		}
   446		runtime·unlock(&paniclk);
   447		if(runtime·xadd(&runtime·panicking, -1) != 0) {
   448			// Some other m is panicking too.
   449			// Let it print what it needs to print.
   450			// Wait forever without chewing up cpu.
   451			// It will exit when it's done.
   452			static Lock deadlock;
   453			runtime·lock(&deadlock);
   454			runtime·lock(&deadlock);
   455		}
   456		
   457		if(crash)
   458			runtime·crash();
   459	
   460		runtime·exit(2);
   461	}

   489	bool
   490	runtime·canpanic(G *gp)
   491	{
   492		byte g;
   493	
   494		USED(&g);  // don't use global g, it points to gsignal
   495	
   496		// Is it okay for gp to panic instead of crashing the program?
   497		// Yes, as long as it is running Go code, not runtime code,
   498		// and not stuck in a system call.
   499		if(gp == nil || gp != m->curg)
   500			return false;
   501		if(m->locks-m->softfloat != 0 || m->mallocing != 0 || m->throwing != 0 || m->gcing != 0 || m->dying != 0)
   502			return false;
   503		if(gp->status != Grunning || gp->syscallsp != 0)
   504			return false;
   505	#ifdef GOOS_windows
   506		if(m->libcallsp != 0)
   507			return false;
   508	#endif
   509		return true;
   510	}

   512	void
   513	runtime·throw(int8 *s)
   514	{
   515		if(m->throwing == 0)
   516			m->throwing = 1;
   517		runtime·startpanic();
   518		runtime·printf("fatal error: %s\n", s);
   519		runtime·dopanic(0);
   520		*(int32*)0 = 0;	// not reached
   521		runtime·exit(1);	// even more not reached
   522	}

   524	void
   525	runtime·panicstring(int8 *s)
   526	{
   527		Eface err;
   528	
   529		// m->softfloat is set during software floating point,
   530		// which might cause a fault during a memory load.
   531		// It increments m->locks to avoid preemption.
   532		// If we're panicking, the software floating point frames
   533		// will be unwound, so decrement m->locks as they would.
   534		if(m->softfloat) {
   535			m->locks--;
   536			m->softfloat = 0;
   537		}
   538	
   539		if(m->mallocing) {
   540			runtime·printf("panic: %s\n", s);
   541			runtime·throw("panic during malloc");
   542		}
   543		if(m->gcing) {
   544			runtime·printf("panic: %s\n", s);
   545			runtime·throw("panic during gc");
   546		}
   547		if(m->locks) {
   548			runtime·printf("panic: %s\n", s);
   549			runtime·throw("panic holding locks");
   550		}
   551		runtime·newErrorCString(s, &err);
   552		runtime·panic(err);
   553	}
