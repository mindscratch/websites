     5	// Page heap.
     6	//
     7	// See malloc.h for overview.
     8	//
     9	// When a MSpan is in the heap free list, state == MSpanFree
    10	// and heapmap(s->start) == span, heapmap(s->start+s->npages-1) == span.
    11	//
    12	// When a MSpan is allocated, state == MSpanInUse
    13	// and heapmap(i) == span for all s->start <= i < s->start+s->npages.

    25	static void
    26	RecordSpan(void *vh, byte *p)
    27	{
    28		MHeap *h;
    29		MSpan *s;
    30		MSpan **all;
    31		uint32 cap;
    32	
    33		h = vh;
    34		s = (MSpan*)p;
    35		if(h->nspan >= h->nspancap) {
    36			cap = 64*1024/sizeof(all[0]);
    37			if(cap < h->nspancap*3/2)
    38				cap = h->nspancap*3/2;
    39			all = (MSpan**)runtime·SysAlloc(cap*sizeof(all[0]), &mstats.other_sys);
    40			if(all == nil)
    41				runtime·throw("runtime: cannot allocate memory");
    42			if(h->allspans) {
    43				runtime·memmove(all, h->allspans, h->nspancap*sizeof(all[0]));
    44				// Don't free the old array if it's referenced by sweep.
    45				// See the comment in mgc0.c.
    46				if(h->allspans != runtime·mheap.sweepspans)
    47					runtime·SysFree(h->allspans, h->nspancap*sizeof(all[0]), &mstats.other_sys);
    48			}
    49			h->allspans = all;
    50			h->nspancap = cap;
    51		}
    52		h->allspans[h->nspan++] = s;
    53	}

    55	// Initialize the heap; fetch memory using alloc.
    56	void
    57	runtime·MHeap_Init(MHeap *h)
    58	{
    59		uint32 i;
    60	
    61		runtime·FixAlloc_Init(&h->spanalloc, sizeof(MSpan), RecordSpan, h, &mstats.mspan_sys);
    62		runtime·FixAlloc_Init(&h->cachealloc, sizeof(MCache), nil, nil, &mstats.mcache_sys);
    63		runtime·FixAlloc_Init(&h->specialfinalizeralloc, sizeof(SpecialFinalizer), nil, nil, &mstats.other_sys);
    64		runtime·FixAlloc_Init(&h->specialprofilealloc, sizeof(SpecialProfile), nil, nil, &mstats.other_sys);
    65		// h->mapcache needs no init
    66		for(i=0; i<nelem(h->free); i++) {
    67			runtime·MSpanList_Init(&h->free[i]);
    68			runtime·MSpanList_Init(&h->busy[i]);
    69		}
    70		runtime·MSpanList_Init(&h->freelarge);
    71		runtime·MSpanList_Init(&h->busylarge);
    72		for(i=0; i<nelem(h->central); i++)
    73			runtime·MCentral_Init(&h->central[i], i);
    74	}

    76	void
    77	runtime·MHeap_MapSpans(MHeap *h)
    78	{
    79		uintptr n;
    80	
    81		// Map spans array, PageSize at a time.
    82		n = (uintptr)h->arena_used;
    83		n -= (uintptr)h->arena_start;
    84		n = n / PageSize * sizeof(h->spans[0]);
    85		n = ROUND(n, PhysPageSize);
    86		if(h->spans_mapped >= n)
    87			return;
    88		runtime·SysMap((byte*)h->spans + h->spans_mapped, n - h->spans_mapped, h->arena_reserved, &mstats.other_sys);
    89		h->spans_mapped = n;
    90	}

    92	// Sweeps spans in list until reclaims at least npages into heap.
    93	// Returns the actual number of pages reclaimed.
    94	static uintptr
    95	MHeap_ReclaimList(MHeap *h, MSpan *list, uintptr npages)
    96	{
    97		MSpan *s;
    98		uintptr n;
    99		uint32 sg;
   100	
   101		n = 0;
   102		sg = runtime·mheap.sweepgen;
   103	retry:
   104		for(s = list->next; s != list; s = s->next) {
   105			if(s->sweepgen == sg-2 && runtime·cas(&s->sweepgen, sg-2, sg-1)) {
   106				runtime·MSpanList_Remove(s);
   107				// swept spans are at the end of the list
   108				runtime·MSpanList_InsertBack(list, s);
   109				runtime·unlock(h);
   110				n += runtime·MSpan_Sweep(s);
   111				runtime·lock(h);
   112				if(n >= npages)
   113					return n;
   114				// the span could have been moved elsewhere
   115				goto retry;
   116			}
   117			if(s->sweepgen == sg-1) {
   118				// the span is being sweept by background sweeper, skip
   119				continue;
   120			}
   121			// already swept empty span,
   122			// all subsequent ones must also be either swept or in process of sweeping
   123			break;
   124		}
   125		return n;
   126	}

   128	// Sweeps and reclaims at least npage pages into heap.
   129	// Called before allocating npage pages.
   130	static void
   131	MHeap_Reclaim(MHeap *h, uintptr npage)
   132	{
   133		uintptr reclaimed, n;
   134	
   135		// First try to sweep busy spans with large objects of size >= npage,
   136		// this has good chances of reclaiming the necessary space.
   137		for(n=npage; n < nelem(h->busy); n++) {
   138			if(MHeap_ReclaimList(h, &h->busy[n], npage))
   139				return;  // Bingo!
   140		}
   141	
   142		// Then -- even larger objects.
   143		if(MHeap_ReclaimList(h, &h->busylarge, npage))
   144			return;  // Bingo!
   145	
   146		// Now try smaller objects.
   147		// One such object is not enough, so we need to reclaim several of them.
   148		reclaimed = 0;
   149		for(n=0; n < npage && n < nelem(h->busy); n++) {
   150			reclaimed += MHeap_ReclaimList(h, &h->busy[n], npage-reclaimed);
   151			if(reclaimed >= npage)
   152				return;
   153		}
   154	
   155		// Now sweep everything that is not yet swept.
   156		runtime·unlock(h);
   157		for(;;) {
   158			n = runtime·sweepone();
   159			if(n == -1)  // all spans are swept
   160				break;
   161			reclaimed += n;
   162			if(reclaimed >= npage)
   163				break;
   164		}
   165		runtime·lock(h);
   166	}

   168	// Allocate a new span of npage pages from the heap
   169	// and record its size class in the HeapMap and HeapMapCache.
   170	MSpan*
   171	runtime·MHeap_Alloc(MHeap *h, uintptr npage, int32 sizeclass, bool large, bool needzero)
   172	{
   173		MSpan *s;
   174	
   175		runtime·lock(h);
   176		mstats.heap_alloc += m->mcache->local_cachealloc;
   177		m->mcache->local_cachealloc = 0;
   178		s = MHeap_AllocLocked(h, npage, sizeclass);
   179		if(s != nil) {
   180			mstats.heap_inuse += npage<<PageShift;
   181			if(large) {
   182				mstats.heap_objects++;
   183				mstats.heap_alloc += npage<<PageShift;
   184				// Swept spans are at the end of lists.
   185				if(s->npages < nelem(h->free))
   186					runtime·MSpanList_InsertBack(&h->busy[s->npages], s);
   187				else
   188					runtime·MSpanList_InsertBack(&h->busylarge, s);
   189			}
   190		}
   191		runtime·unlock(h);
   192		if(s != nil) {
   193			if(needzero && s->needzero)
   194				runtime·memclr((byte*)(s->start<<PageShift), s->npages<<PageShift);
   195			s->needzero = 0;
   196		}
   197		return s;
   198	}

   200	static MSpan*
   201	MHeap_AllocLocked(MHeap *h, uintptr npage, int32 sizeclass)
   202	{
   203		uintptr n;
   204		MSpan *s, *t;
   205		PageID p;
   206	
   207		// To prevent excessive heap growth, before allocating n pages
   208		// we need to sweep and reclaim at least n pages.
   209		if(!h->sweepdone)
   210			MHeap_Reclaim(h, npage);
   211	
   212		// Try in fixed-size lists up to max.
   213		for(n=npage; n < nelem(h->free); n++) {
   214			if(!runtime·MSpanList_IsEmpty(&h->free[n])) {
   215				s = h->free[n].next;
   216				goto HaveSpan;
   217			}
   218		}
   219	
   220		// Best fit in list of large spans.
   221		if((s = MHeap_AllocLarge(h, npage)) == nil) {
   222			if(!MHeap_Grow(h, npage))
   223				return nil;
   224			if((s = MHeap_AllocLarge(h, npage)) == nil)
   225				return nil;
   226		}
   227	
   228	HaveSpan:
   229		// Mark span in use.
   230		if(s->state != MSpanFree)
   231			runtime·throw("MHeap_AllocLocked - MSpan not free");
   232		if(s->npages < npage)
   233			runtime·throw("MHeap_AllocLocked - bad npages");
   234		runtime·MSpanList_Remove(s);
   235		runtime·atomicstore(&s->sweepgen, h->sweepgen);
   236		s->state = MSpanInUse;
   237		mstats.heap_idle -= s->npages<<PageShift;
   238		mstats.heap_released -= s->npreleased<<PageShift;
   239		if(s->npreleased > 0)
   240			runtime·SysUsed((void*)(s->start<<PageShift), s->npages<<PageShift);
   241		s->npreleased = 0;
   242	
   243		if(s->npages > npage) {
   244			// Trim extra and put it back in the heap.
   245			t = runtime·FixAlloc_Alloc(&h->spanalloc);
   246			runtime·MSpan_Init(t, s->start + npage, s->npages - npage);
   247			s->npages = npage;
   248			p = t->start;
   249			p -= ((uintptr)h->arena_start>>PageShift);
   250			if(p > 0)
   251				h->spans[p-1] = s;
   252			h->spans[p] = t;
   253			h->spans[p+t->npages-1] = t;
   254			t->needzero = s->needzero;
   255			runtime·atomicstore(&t->sweepgen, h->sweepgen);
   256			t->state = MSpanInUse;
   257			MHeap_FreeLocked(h, t);
   258			t->unusedsince = s->unusedsince; // preserve age
   259		}
   260		s->unusedsince = 0;
   261	
   262		// Record span info, because gc needs to be
   263		// able to map interior pointer to containing span.
   264		s->sizeclass = sizeclass;
   265		s->elemsize = (sizeclass==0 ? s->npages<<PageShift : runtime·class_to_size[sizeclass]);
   266		s->types.compression = MTypes_Empty;
   267		p = s->start;
   268		p -= ((uintptr)h->arena_start>>PageShift);
   269		for(n=0; n<npage; n++)
   270			h->spans[p+n] = s;
   271		return s;
   272	}

   281	// Search list for smallest span with >= npage pages.
   282	// If there are multiple smallest spans, take the one
   283	// with the earliest starting address.
   284	static MSpan*
   285	BestFit(MSpan *list, uintptr npage, MSpan *best)
   286	{
   287		MSpan *s;
   288	
   289		for(s=list->next; s != list; s=s->next) {
   290			if(s->npages < npage)
   291				continue;
   292			if(best == nil
   293			|| s->npages < best->npages
   294			|| (s->npages == best->npages && s->start < best->start))
   295				best = s;
   296		}
   297		return best;
   298	}

   300	// Try to add at least npage pages of memory to the heap,
   301	// returning whether it worked.
   302	static bool
   303	MHeap_Grow(MHeap *h, uintptr npage)
   304	{
   305		uintptr ask;
   306		void *v;
   307		MSpan *s;
   308		PageID p;
   309	
   310		// Ask for a big chunk, to reduce the number of mappings
   311		// the operating system needs to track; also amortizes
   312		// the overhead of an operating system mapping.
   313		// Allocate a multiple of 64kB (16 pages).
   314		npage = (npage+15)&~15;
   315		ask = npage<<PageShift;
   316		if(ask < HeapAllocChunk)
   317			ask = HeapAllocChunk;
   318	
   319		v = runtime·MHeap_SysAlloc(h, ask);
   320		if(v == nil) {
   321			if(ask > (npage<<PageShift)) {
   322				ask = npage<<PageShift;
   323				v = runtime·MHeap_SysAlloc(h, ask);
   324			}
   325			if(v == nil) {
   326				runtime·printf("runtime: out of memory: cannot allocate %D-byte block (%D in use)\n", (uint64)ask, mstats.heap_sys);
   327				return false;
   328			}
   329		}
   330	
   331		// Create a fake "in use" span and free it, so that the
   332		// right coalescing happens.
   333		s = runtime·FixAlloc_Alloc(&h->spanalloc);
   334		runtime·MSpan_Init(s, (uintptr)v>>PageShift, ask>>PageShift);
   335		p = s->start;
   336		p -= ((uintptr)h->arena_start>>PageShift);
   337		h->spans[p] = s;
   338		h->spans[p + s->npages - 1] = s;
   339		runtime·atomicstore(&s->sweepgen, h->sweepgen);
   340		s->state = MSpanInUse;
   341		MHeap_FreeLocked(h, s);
   342		return true;
   343	}

   345	// Look up the span at the given address.
   346	// Address is guaranteed to be in map
   347	// and is guaranteed to be start or end of span.
   348	MSpan*
   349	runtime·MHeap_Lookup(MHeap *h, void *v)
   350	{
   351		uintptr p;
   352		
   353		p = (uintptr)v;
   354		p -= (uintptr)h->arena_start;
   355		return h->spans[p >> PageShift];
   356	}
   357	
   358	// Look up the span at the given address.
   359	// Address is *not* guaranteed to be in map
   360	// and may be anywhere in the span.
   361	// Map entries for the middle of a span are only
   362	// valid for allocated spans.  Free spans may have
   363	// other garbage in their middles, so we have to
   364	// check for that.
   365	MSpan*
   366	runtime·MHeap_LookupMaybe(MHeap *h, void *v)
   367	{
   368		MSpan *s;
   369		PageID p, q;
   370	
   371		if((byte*)v < h->arena_start || (byte*)v >= h->arena_used)
   372			return nil;
   373		p = (uintptr)v>>PageShift;
   374		q = p;
   375		q -= (uintptr)h->arena_start >> PageShift;
   376		s = h->spans[q];
   377		if(s == nil || p < s->start || v >= s->limit || s->state != MSpanInUse)
   378			return nil;
   379		return s;
   380	}

   382	// Free the span back into the heap.
   383	void
   384	runtime·MHeap_Free(MHeap *h, MSpan *s, int32 acct)
   385	{
   386		runtime·lock(h);
   387		mstats.heap_alloc += m->mcache->local_cachealloc;
   388		m->mcache->local_cachealloc = 0;
   389		mstats.heap_inuse -= s->npages<<PageShift;
   390		if(acct) {
   391			mstats.heap_alloc -= s->npages<<PageShift;
   392			mstats.heap_objects--;
   393		}
   394		MHeap_FreeLocked(h, s);
   395		runtime·unlock(h);
   396	}

   398	static void
   399	MHeap_FreeLocked(MHeap *h, MSpan *s)
   400	{
   401		MSpan *t;
   402		PageID p;
   403	
   404		s->types.compression = MTypes_Empty;
   405	
   406		if(s->state != MSpanInUse || s->ref != 0 || s->sweepgen != h->sweepgen) {
   407			runtime·printf("MHeap_FreeLocked - span %p ptr %p state %d ref %d sweepgen %d/%d\n",
   408				s, s->start<<PageShift, s->state, s->ref, s->sweepgen, h->sweepgen);
   409			runtime·throw("MHeap_FreeLocked - invalid free");
   410		}
   411		mstats.heap_idle += s->npages<<PageShift;
   412		s->state = MSpanFree;
   413		runtime·MSpanList_Remove(s);
   414		// Stamp newly unused spans. The scavenger will use that
   415		// info to potentially give back some pages to the OS.
   416		s->unusedsince = runtime·nanotime();
   417		s->npreleased = 0;
   418	
   419		// Coalesce with earlier, later spans.
   420		p = s->start;
   421		p -= (uintptr)h->arena_start >> PageShift;
   422		if(p > 0 && (t = h->spans[p-1]) != nil && t->state != MSpanInUse) {
   423			s->start = t->start;
   424			s->npages += t->npages;
   425			s->npreleased = t->npreleased; // absorb released pages
   426			s->needzero |= t->needzero;
   427			p -= t->npages;
   428			h->spans[p] = s;
   429			runtime·MSpanList_Remove(t);
   430			t->state = MSpanDead;
   431			runtime·FixAlloc_Free(&h->spanalloc, t);
   432		}
   433		if((p+s->npages)*sizeof(h->spans[0]) < h->spans_mapped && (t = h->spans[p+s->npages]) != nil && t->state != MSpanInUse) {
   434			s->npages += t->npages;
   435			s->npreleased += t->npreleased;
   436			s->needzero |= t->needzero;
   437			h->spans[p + s->npages - 1] = s;
   438			runtime·MSpanList_Remove(t);
   439			t->state = MSpanDead;
   440			runtime·FixAlloc_Free(&h->spanalloc, t);
   441		}
   442	
   443		// Insert s into appropriate list.
   444		if(s->npages < nelem(h->free))
   445			runtime·MSpanList_Insert(&h->free[s->npages], s);
   446		else
   447			runtime·MSpanList_Insert(&h->freelarge, s);
   448	}

   457	static uintptr
   458	scavengelist(MSpan *list, uint64 now, uint64 limit)
   459	{
   460		uintptr released, sumreleased;
   461		MSpan *s;
   462	
   463		if(runtime·MSpanList_IsEmpty(list))
   464			return 0;
   465	
   466		sumreleased = 0;
   467		for(s=list->next; s != list; s=s->next) {
   468			if((now - s->unusedsince) > limit && s->npreleased != s->npages) {
   469				released = (s->npages - s->npreleased) << PageShift;
   470				mstats.heap_released += released;
   471				sumreleased += released;
   472				s->npreleased = s->npages;
   473				runtime·SysUnused((void*)(s->start << PageShift), s->npages << PageShift);
   474			}
   475		}
   476		return sumreleased;
   477	}

   479	static void
   480	scavenge(int32 k, uint64 now, uint64 limit)
   481	{
   482		uint32 i;
   483		uintptr sumreleased;
   484		MHeap *h;
   485		
   486		h = &runtime·mheap;
   487		sumreleased = 0;
   488		for(i=0; i < nelem(h->free); i++)
   489			sumreleased += scavengelist(&h->free[i], now, limit);
   490		sumreleased += scavengelist(&h->freelarge, now, limit);
   491	
   492		if(runtime·debug.gctrace > 0) {
   493			if(sumreleased > 0)
   494				runtime·printf("scvg%d: %D MB released\n", k, (uint64)sumreleased>>20);
   495			runtime·printf("scvg%d: inuse: %D, idle: %D, sys: %D, released: %D, consumed: %D (MB)\n",
   496				k, mstats.heap_inuse>>20, mstats.heap_idle>>20, mstats.heap_sys>>20,
   497				mstats.heap_released>>20, (mstats.heap_sys - mstats.heap_released)>>20);
   498		}
   499	}

   503	// Release (part of) unused memory to OS.
   504	// Goroutine created at startup.
   505	// Loop forever.
   506	void
   507	runtime·MHeap_Scavenger(void)
   508	{
   509		MHeap *h;
   510		uint64 tick, now, forcegc, limit;
   511		int64 unixnow;
   512		int32 k;
   513		Note note, *notep;
   514	
   515		g->issystem = true;
   516		g->isbackground = true;
   517	
   518		// If we go two minutes without a garbage collection, force one to run.
   519		forcegc = 2*60*1e9;
   520		// If a span goes unused for 5 minutes after a garbage collection,
   521		// we hand it back to the operating system.
   522		limit = 5*60*1e9;
   523		// Make wake-up period small enough for the sampling to be correct.
   524		if(forcegc < limit)
   525			tick = forcegc/2;
   526		else
   527			tick = limit/2;
   528	
   529		h = &runtime·mheap;
   530		for(k=0;; k++) {
   531			runtime·noteclear(&note);
   532			runtime·notetsleepg(&note, tick);
   533	
   534			runtime·lock(h);
   535			unixnow = runtime·unixnanotime();
   536			if(unixnow - mstats.last_gc > forcegc) {
   537				runtime·unlock(h);
   538				// The scavenger can not block other goroutines,
   539				// otherwise deadlock detector can fire spuriously.
   540				// GC blocks other goroutines via the runtime·worldsema.
   541				runtime·noteclear(&note);
   542				notep = &note;
   543				runtime·newproc1(&forcegchelperv, (byte*)&notep, sizeof(notep), 0, runtime·MHeap_Scavenger);
   544				runtime·notetsleepg(&note, -1);
   545				if(runtime·debug.gctrace > 0)
   546					runtime·printf("scvg%d: GC forced\n", k);

   639	// Adds the special record s to the list of special records for
   640	// the object p.  All fields of s should be filled in except for
   641	// offset & next, which this routine will fill in.
   642	// Returns true if the special was successfully added, false otherwise.
   643	// (The add will fail only if a record with the same p and s->kind
   644	//  already exists.)
   645	static bool
   646	addspecial(void *p, Special *s)
   647	{
   648		MSpan *span;
   649		Special **t, *x;
   650		uintptr offset;
   651		byte kind;
   652	
   653		span = runtime·MHeap_LookupMaybe(&runtime·mheap, p);
   654		if(span == nil)
   655			runtime·throw("addspecial on invalid pointer");
   656	
   657		// Ensure that the span is swept.
   658		// GC accesses specials list w/o locks. And it's just much safer.
   659		m->locks++;
   660		runtime·MSpan_EnsureSwept(span);
   661	
   662		offset = (uintptr)p - (span->start << PageShift);
   663		kind = s->kind;
   664	
   665		runtime·lock(&span->specialLock);
   666	
   667		// Find splice point, check for existing record.
   668		t = &span->specials;
   669		while((x = *t) != nil) {
   670			if(offset == x->offset && kind == x->kind) {
   671				runtime·unlock(&span->specialLock);
   672				m->locks--;
   673				return false; // already exists
   674			}
   675			if(offset < x->offset || (offset == x->offset && kind < x->kind))
   676				break;
   677			t = &x->next;
   678		}
   679		// Splice in record, fill in offset.
   680		s->offset = offset;
   681		s->next = x;
   682		*t = s;
   683		runtime·unlock(&span->specialLock);
   684		m->locks--;
   685		return true;
   686	}

   688	// Removes the Special record of the given kind for the object p.
   689	// Returns the record if the record existed, nil otherwise.
   690	// The caller must FixAlloc_Free the result.
   691	static Special*
   692	removespecial(void *p, byte kind)
   693	{
   694		MSpan *span;
   695		Special *s, **t;
   696		uintptr offset;
   697	
   698		span = runtime·MHeap_LookupMaybe(&runtime·mheap, p);
   699		if(span == nil)
   700			runtime·throw("removespecial on invalid pointer");
   701	
   702		// Ensure that the span is swept.
   703		// GC accesses specials list w/o locks. And it's just much safer.
   704		m->locks++;
   705		runtime·MSpan_EnsureSwept(span);
   706	
   707		offset = (uintptr)p - (span->start << PageShift);
   708	
   709		runtime·lock(&span->specialLock);
   710		t = &span->specials;
   711		while((s = *t) != nil) {
   712			// This function is used for finalizers only, so we don't check for
   713			// "interior" specials (p must be exactly equal to s->offset).
   714			if(offset == s->offset && kind == s->kind) {
   715				*t = s->next;
   716				runtime·unlock(&span->specialLock);
   717				m->locks--;
   718				return s;
   719			}
   720			t = &s->next;
   721		}
   722		runtime·unlock(&span->specialLock);
   723		m->locks--;
   724		return nil;
   725	}

   727	// Adds a finalizer to the object p.  Returns true if it succeeded.
   728	bool
   729	runtime·addfinalizer(void *p, FuncVal *f, uintptr nret, Type *fint, PtrType *ot)
   730	{
   731		SpecialFinalizer *s;
   732	
   733		runtime·lock(&runtime·mheap.speciallock);
   734		s = runtime·FixAlloc_Alloc(&runtime·mheap.specialfinalizeralloc);
   735		runtime·unlock(&runtime·mheap.speciallock);
   736		s->kind = KindSpecialFinalizer;
   737		s->fn = f;
   738		s->nret = nret;
   739		s->fint = fint;
   740		s->ot = ot;
   741		if(addspecial(p, s))
   742			return true;
   743	
   744		// There was an old finalizer
   745		runtime·lock(&runtime·mheap.speciallock);
   746		runtime·FixAlloc_Free(&runtime·mheap.specialfinalizeralloc, s);
   747		runtime·unlock(&runtime·mheap.speciallock);
   748		return false;
   749	}

   780	// Do whatever cleanup needs to be done to deallocate s.  It has
   781	// already been unlinked from the MSpan specials list.
   782	// Returns true if we should keep working on deallocating p.
   783	bool
   784	runtime·freespecial(Special *s, void *p, uintptr size, bool freed)
   785	{
   786		SpecialFinalizer *sf;
   787		SpecialProfile *sp;
   788	
   789		switch(s->kind) {
   790		case KindSpecialFinalizer:
   791			sf = (SpecialFinalizer*)s;
   792			runtime·queuefinalizer(p, sf->fn, sf->nret, sf->fint, sf->ot);
   793			runtime·lock(&runtime·mheap.speciallock);
   794			runtime·FixAlloc_Free(&runtime·mheap.specialfinalizeralloc, sf);
   795			runtime·unlock(&runtime·mheap.speciallock);
   796			return false; // don't free p until finalizer is done
   797		case KindSpecialProfile:
   798			sp = (SpecialProfile*)s;
   799			runtime·MProf_Free(sp->b, size, freed);
   800			runtime·lock(&runtime·mheap.speciallock);
   801			runtime·FixAlloc_Free(&runtime·mheap.specialprofilealloc, sp);
   802			runtime·unlock(&runtime·mheap.speciallock);
   803			return true;
   804		default:
   805			runtime·throw("bad special kind");
   806			return true;
   807		}
   808	}

   810	// Free all special records for p.
   811	void
   812	runtime·freeallspecials(MSpan *span, void *p, uintptr size)
   813	{
   814		Special *s, **t, *list;
   815		uintptr offset;
   816	
   817		if(span->sweepgen != runtime·mheap.sweepgen)
   818			runtime·throw("runtime: freeallspecials: unswept span");
   819		// first, collect all specials into the list; then, free them
   820		// this is required to not cause deadlock between span->specialLock and proflock
   821		list = nil;
   822		offset = (uintptr)p - (span->start << PageShift);
   823		runtime·lock(&span->specialLock);
   824		t = &span->specials;
   825		while((s = *t) != nil) {
   826			if(offset + size <= s->offset)
   827				break;
   828			if(offset <= s->offset) {
   829				*t = s->next;
   830				s->next = list;
   831				list = s;
   832			} else
   833				t = &s->next;
   834		}
   835		runtime·unlock(&span->specialLock);
   836	
   837		while(list != nil) {
   838			s = list;
   839			list = s->next;
   840			if(!runtime·freespecial(s, p, size, true))
   841				runtime·throw("can't explicitly free an object with a finalizer");
   842		}
   843	}

   845	// Split an allocated span into two equal parts.
   846	void
   847	runtime·MHeap_SplitSpan(MHeap *h, MSpan *s)
   848	{
   849		MSpan *t;
   850		MCentral *c;
   851		uintptr i;
   852		uintptr npages;
   853		PageID p;
   854	
   855		if(s->state != MSpanInUse)
   856			runtime·throw("MHeap_SplitSpan on a free span");
   857		if(s->sizeclass != 0 && s->ref != 1)
   858			runtime·throw("MHeap_SplitSpan doesn't have an allocated object");
   859		npages = s->npages;
   860	
   861		// remove the span from whatever list it is in now
   862		if(s->sizeclass > 0) {
   863			// must be in h->central[x].empty
   864			c = &h->central[s->sizeclass];
   865			runtime·lock(c);
   866			runtime·MSpanList_Remove(s);
   867			runtime·unlock(c);
   868			runtime·lock(h);
   869		} else {
   870			// must be in h->busy/busylarge
   871			runtime·lock(h);
   872			runtime·MSpanList_Remove(s);
   873		}
   874		// heap is locked now
   875	
   876		if(npages == 1) {
   877			// convert span of 1 PageSize object to a span of 2 PageSize/2 objects.
   878			s->ref = 2;
   879			s->sizeclass = runtime·SizeToClass(PageSize/2);
   880			s->elemsize = PageSize/2;
   881		} else {
   882			// convert span of n>1 pages into two spans of n/2 pages each.
   883			if((s->npages & 1) != 0)
   884				runtime·throw("MHeap_SplitSpan on an odd size span");
   885	
   886			// compute position in h->spans
   887			p = s->start;
   888			p -= (uintptr)h->arena_start >> PageShift;
   889	
   890			// Allocate a new span for the first half.
   891			t = runtime·FixAlloc_Alloc(&h->spanalloc);
   892			runtime·MSpan_Init(t, s->start, npages/2);
   893			t->limit = (byte*)((t->start + npages/2) << PageShift);
   894			t->state = MSpanInUse;
   895			t->elemsize = npages << (PageShift - 1);
   896			t->sweepgen = s->sweepgen;
   897			if(t->elemsize <= MaxSmallSize) {
   898				t->sizeclass = runtime·SizeToClass(t->elemsize);
   899				t->ref = 1;
   900			}
   901	
   902			// the old span holds the second half.
   903			s->start += npages/2;
   904			s->npages = npages/2;
   905			s->elemsize = npages << (PageShift - 1);
   906			if(s->elemsize <= MaxSmallSize) {
   907				s->sizeclass = runtime·SizeToClass(s->elemsize);
   908				s->ref = 1;
   909			}
   910	
   911			// update span lookup table
   912			for(i = p; i < p + npages/2; i++)
   913				h->spans[i] = t;
   914		}
   915	
   916		// place the span into a new list
   917		if(s->sizeclass > 0) {
   918			runtime·unlock(h);
   919			c = &h->central[s->sizeclass];
   920			runtime·lock(c);
   921			// swept spans are at the end of the list
   922			runtime·MSpanList_InsertBack(&c->empty, s);
   923			runtime·unlock(c);
   924		} else {
   925			// Swept spans are at the end of lists.
   926			if(s->npages < nelem(h->free))
   927				runtime·MSpanList_InsertBack(&h->busy[s->npages], s);
   928			else
   929				runtime·MSpanList_InsertBack(&h->busylarge, s);
   930			runtime·unlock(h);
   931		}
   932	}
