     9	#include "malloc.h"
    10	
    11	enum
    12	{
    13		ENOMEM = 12,
    14	};
    15	
    16	void*
    17	runtime·SysAlloc(uintptr n, uint64 *stat)
    18	{
    19		void *v;
    20	
    21		v = runtime·mmap(nil, n, PROT_READ|PROT_WRITE, MAP_ANON|MAP_PRIVATE, -1, 0);
    22		if(v < (void*)4096)
    23			return nil;
    24		runtime·xadd64(stat, n);
    25		return v;
    26	}
    27	
    28	void
    29	runtime·SysUnused(void *v, uintptr n)
    30	{
    31		USED(v);
    32		USED(n);
    33	}
    34	
    35	void
    36	runtime·SysUsed(void *v, uintptr n)
    37	{
    38		USED(v);
    39		USED(n);
    40	}
    41	
    42	void
    43	runtime·SysFree(void *v, uintptr n, uint64 *stat)
    44	{
    45		runtime·xadd64(stat, -(uint64)n);
    46		runtime·munmap(v, n);
    47	}
    48	
    49	void
    50	runtime·SysFault(void *v, uintptr n)
    51	{
    52		runtime·mmap(v, n, PROT_NONE, MAP_ANON|MAP_PRIVATE|MAP_FIXED, -1, 0);
    53	}
    54	
    55	void*
    56	runtime·SysReserve(void *v, uintptr n, bool *reserved)
    57	{
    58		void *p;
    59	
    60		// On 64-bit, people with ulimit -v set complain if we reserve too
    61		// much address space.  Instead, assume that the reservation is okay
    62		// and check the assumption in SysMap.
    63		if(sizeof(void*) == 8 && n > 1LL<<32) {
    64			*reserved = false;
    65			return v;
    66		}
    67		
    68		p = runtime·mmap(v, n, PROT_NONE, MAP_ANON|MAP_PRIVATE, -1, 0);
    69		if(p < (void*)4096)
    70			return nil;
    71		*reserved = true;
    72		return p;
    73	}
    74	
    75	void
    76	runtime·SysMap(void *v, uintptr n, bool reserved, uint64 *stat)
    77	{
    78		void *p;
    79		
    80		runtime·xadd64(stat, n);
    81	
    82		// On 64-bit, we don't actually have v reserved, so tread carefully.
    83		if(!reserved) {
    84			p = runtime·mmap(v, n, PROT_READ|PROT_WRITE, MAP_ANON|MAP_PRIVATE, -1, 0);
    85			if(p == (void*)ENOMEM)
    86				runtime·throw("runtime: out of memory");
    87			if(p != v) {
    88				runtime·printf("runtime: address space conflict: map(%p) = %p\n", v, p);
    89				runtime·throw("runtime: address space conflict");
    90			}
    91			return;
    92		}
    93	
    94		p = runtime·mmap(v, n, PROT_READ|PROT_WRITE, MAP_ANON|MAP_FIXED|MAP_PRIVATE, -1, 0);
    95		if(p == (void*)ENOMEM)
    96			runtime·throw("runtime: out of memory");
    97		if(p != v)
    98			runtime·throw("runtime: cannot map pages in arena address space");
    99	}
