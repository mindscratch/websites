    12	// Exit the entire program (like C exit)
    13	TEXT runtime·exit(SB),NOSPLIT,$-4
    14		MOVW 0(FP), R0	// arg 1 exit status
    15		SWI $0xa00001
    16		MOVW.CS $0, R8	// crash on syscall failure
    17		MOVW.CS R8, (R8)
    18		RET

   177	TEXT runtime·sigreturn_tramp(SB),NOSPLIT,$-4
   178		// on entry, SP points to siginfo, we add sizeof(ucontext)
   179		// to SP to get a pointer to ucontext.
   180		ADD $0x80, R13, R0 // 0x80 == sizeof(UcontextT)
   181		SWI $0xa00134	// sys_setcontext
   182		// something failed, we have to exit
   183		MOVW $0x4242, R0 // magic return number
   184		SWI $0xa00001	// sys_exit
   185		B -2(PC)	// continue exit

   187	TEXT runtime·sigaction(SB),NOSPLIT,$4
   188		MOVW 0(FP), R0	// arg 1 - signum
   189		MOVW 4(FP), R1	// arg 2 - nsa
   190		MOVW 8(FP), R2	// arg 3 - osa
   191		MOVW $runtime·sigreturn_tramp(SB), R3	// arg 4 - tramp
   192		MOVW $2, R4	// arg 5 - vers
   193		MOVW R4, 4(R13)
   194		ADD $4, R13	// pass arg 5 on stack
   195		SWI $0xa00154	// sys___sigaction_sigtramp
   196		SUB $4, R13
   197		MOVW.CS $3, R8	// crash on syscall failure
   198		MOVW.CS R8, (R8)
   199		RET

   201	TEXT runtime·sigtramp(SB),NOSPLIT,$24
   202		// this might be called in external code context,
   203		// where g and m are not set.
   204		// first save R0, because runtime·load_gm will clobber it
   205		MOVW	R0, 4(R13) // signum
   206		MOVB	runtime·iscgo(SB), R0
   207		CMP 	$0, R0
   208		BL.NE	runtime·load_gm(SB)
   209	
   210		CMP $0, m
   211		BNE 4(PC)
   212		// signal number is already prepared in 4(R13)
   213		MOVW $runtime·badsignal(SB), R11
   214		BL (R11)
   215		RET
   216	
   217		// save g
   218		MOVW g, R4
   219		MOVW g, 20(R13)
   220	
   221		// g = m->signal
   222		MOVW m_gsignal(m), g
   223	
   224		// R0 is already saved
   225		MOVW R1, 8(R13) // info
   226		MOVW R2, 12(R13) // context
   227		MOVW R4, 16(R13) // gp
   228	
   229		BL runtime·sighandler(SB)
   230	
   231		// restore g
   232		MOVW 20(R13), g
   233		RET

   235	TEXT runtime·mmap(SB),NOSPLIT,$12
   236		MOVW 0(FP), R0	// arg 1 - addr
   237		MOVW 4(FP), R1	// arg 2 - len
   238		MOVW 8(FP), R2	// arg 3 - prot
   239		MOVW 12(FP), R3	// arg 4 - flags
   240		// arg 5 (fid) and arg6 (offset_lo, offset_hi) are passed on stack
   241		// note the C runtime only passes the 32-bit offset_lo to us
   242		MOVW 16(FP), R4		// arg 5
   243		MOVW R4, 4(R13)
   244		MOVW 20(FP), R5		// arg 6 lower 32-bit
   245		MOVW R5, 8(R13)
   246		MOVW $0, R6 // higher 32-bit for arg 6
   247		MOVW R6, 12(R13)
   248		ADD $4, R13 // pass arg 5 and arg 6 on stack
   249		SWI $0xa000c5	// sys_mmap
   250		SUB $4, R13
   251		RET
