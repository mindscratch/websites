    67	// buffer of pending write data
    68	enum {
    69		BufSize = 4096,
    70	};
    71	#pragma dataflag NOPTR
    72	static byte buf[BufSize];
    73	static uintptr nbuf;
    74	
    75	static void
    76	write(byte *data, uintptr len)
    77	{
    78		if(len + nbuf <= BufSize) {
    79			runtime·memmove(buf + nbuf, data, len);
    80			nbuf += len;
    81			return;
    82		}
    83		runtime·write(dumpfd, buf, nbuf);
    84		if(len >= BufSize) {
    85			runtime·write(dumpfd, data, len);
    86			nbuf = 0;
    87		} else {
    88			runtime·memmove(buf, data, len);
    89			nbuf = len;
    90		}
    91	}
    92	
    93	static void
    94	flush(void)
    95	{
    96		runtime·write(dumpfd, buf, nbuf);
    97		nbuf = 0;
    98	}

   100	// Cache of types that have been serialized already.
   101	// We use a type's hash field to pick a bucket.
   102	// Inside a bucket, we keep a list of types that
   103	// have been serialized so far, most recently used first.
   104	// Note: when a bucket overflows we may end up
   105	// serializing a type more than once.  That's ok.
   106	enum {
   107		TypeCacheBuckets = 256, // must be a power of 2
   108		TypeCacheAssoc = 4,
   109	};
   110	typedef struct TypeCacheBucket TypeCacheBucket;
   111	struct TypeCacheBucket {
   112		Type *t[TypeCacheAssoc];
   113	};
   114	static TypeCacheBucket typecache[TypeCacheBuckets];

   116	// dump a uint64 in a varint format parseable by encoding/binary
   117	static void
   118	dumpint(uint64 v)
   119	{
   120		byte buf[10];
   121		int32 n;
   122		n = 0;
   123		while(v >= 0x80) {
   124			buf[n++] = v | 0x80;
   125			v >>= 7;
   126		}
   127		buf[n++] = v;
   128		write(buf, n);
   129	}

   157	// dump information for a type
   158	static void
   159	dumptype(Type *t)
   160	{
   161		TypeCacheBucket *b;
   162		int32 i, j;
   163	
   164		if(t == nil) {
   165			return;
   166		}
   167	
   168		// If we've definitely serialized the type before,
   169		// no need to do it again.
   170		b = &typecache[t->hash & (TypeCacheBuckets-1)];
   171		if(t == b->t[0]) return;
   172		for(i = 1; i < TypeCacheAssoc; i++) {
   173			if(t == b->t[i]) {
   174				// Move-to-front
   175				for(j = i; j > 0; j--) {
   176					b->t[j] = b->t[j-1];
   177				}
   178				b->t[0] = t;
   179				return;
   180			}
   181		}
   182		// Might not have been dumped yet.  Dump it and
   183		// remember we did so.
   184		for(j = TypeCacheAssoc-1; j > 0; j--) {
   185			b->t[j] = b->t[j-1];
   186		}
   187		b->t[0] = t;
   188		
   189		// dump the type
   190		dumpint(TagType);
   191		dumpint((uintptr)t);
   192		dumpint(t->size);
   193		if(t->x == nil || t->x->pkgPath == nil || t->x->name == nil) {
   194			dumpstr(*t->string);
   195		} else {
   196			dumpint(t->x->pkgPath->len + 1 + t->x->name->len);
   197			write(t->x->pkgPath->str, t->x->pkgPath->len);
   198			write((byte*)".", 1);
   199			write(t->x->name->str, t->x->name->len);
   200		}
   201		dumpbool(t->size > PtrSize || (t->kind & KindNoPointers) == 0);
   202		dumpfields((uintptr*)t->gc + 1);
   203	}

   205	// returns true if object is scannable
   206	static bool
   207	scannable(byte *obj)
   208	{
   209		uintptr *b, off, shift;
   210	
   211		off = (uintptr*)obj - (uintptr*)runtime·mheap.arena_start;  // word offset
   212		b = (uintptr*)runtime·mheap.arena_start - off/wordsPerBitmapWord - 1;
   213		shift = off % wordsPerBitmapWord;
   214		return ((*b >> shift) & bitScan) != 0;
   215	}

   252	typedef struct ChildInfo ChildInfo;
   253	struct ChildInfo {
   254		// Information passed up from the callee frame about
   255		// the layout of the outargs region.
   256		uintptr argoff;     // where the arguments start in the frame
   257		uintptr arglen;     // size of args region
   258		BitVector args;    // if args.n >= 0, pointer map of args region

   264	// dump kinds & offsets of interesting fields in bv
   265	static void
   266	dumpbv(BitVector *bv, uintptr offset)
   267	{
   268		uintptr i;
   269	
   270		for(i = 0; i < bv->n; i += BitsPerPointer) {
   271			switch(bv->data[i/32] >> i%32 & 3) {
   272			case BitsDead:
   273			case BitsScalar:
   274				break;
   275			case BitsPointer:
   276				dumpint(FieldKindPtr);
   277				dumpint(offset + i / BitsPerPointer * PtrSize);
   278				break;
   279			case BitsMultiWord:
   280				switch(bv->data[(i+BitsPerPointer)/32] >> (i+BitsPerPointer)%32 & 3) {
   281				case BitsString:
   282					dumpint(FieldKindString);
   283					dumpint(offset + i / BitsPerPointer * PtrSize);
   284					i += BitsPerPointer;
   285					break;
   286				case BitsSlice:
   287					dumpint(FieldKindSlice);
   288					dumpint(offset + i / BitsPerPointer * PtrSize);
   289					i += 2 * BitsPerPointer;
   290					break;
   291				case BitsIface:
   292					dumpint(FieldKindIface);
   293					dumpint(offset + i / BitsPerPointer * PtrSize);
   294					i += BitsPerPointer;
   295					break;
   296				case BitsEface:
   297					dumpint(FieldKindEface);
   298					dumpint(offset + i / BitsPerPointer * PtrSize);
   299					i += BitsPerPointer;
   300					break;
   301				}
   302			}
   303		}
   304	}

   306	static bool
   307	dumpframe(Stkframe *s, void *arg)
   308	{
   309		Func *f;
   310		ChildInfo *child;
   311		uintptr pc, off, size;
   312		int32 pcdata;
   313		StackMap *stackmap;
   314		int8 *name;
   315		BitVector bv;
   316	
   317		child = (ChildInfo*)arg;
   318		f = s->fn;
   319	
   320		// Figure out what we can about our stack map
   321		pc = s->pc;
   322		if(pc != f->entry)
   323			pc--;
   324		pcdata = runtime·pcdatavalue(f, PCDATA_StackMapIndex, pc);
   325		if(pcdata == -1) {
   326			// We do not have a valid pcdata value but there might be a
   327			// stackmap for this function.  It is likely that we are looking
   328			// at the function prologue, assume so and hope for the best.
   329			pcdata = 0;
   330		}
   331		stackmap = runtime·funcdata(f, FUNCDATA_LocalsPointerMaps);
   332	
   333		// Dump any types we will need to resolve Efaces.
   334		if(child->args.n >= 0)
   335			dumpbvtypes(&child->args, (byte*)s->sp + child->argoff);
   336		if(stackmap != nil && stackmap->n > 0) {
   337			bv = runtime·stackmapdata(stackmap, pcdata);
   338			dumpbvtypes(&bv, s->varp - bv.n / BitsPerPointer * PtrSize);
   339		} else {
   340			bv.n = -1;
   341		}
   342	
   343		// Dump main body of stack frame.
   344		dumpint(TagStackFrame);
   345		dumpint(s->sp); // lowest address in frame
   346		dumpint(child->depth); // # of frames deep on the stack
   347		dumpint((uintptr)child->sp); // sp of child, or 0 if bottom of stack
   348		dumpmemrange((byte*)s->sp, s->fp - s->sp);  // frame contents
   349		dumpint(f->entry);
   350		dumpint(s->pc);
   351		dumpint(s->continpc);
   352		name = runtime·funcname(f);
   353		if(name == nil)
   354			name = "unknown function";
   355		dumpcstr(name);
   356	
   357		// Dump fields in the outargs section
   358		if(child->args.n >= 0) {
   359			dumpbv(&child->args, child->argoff);
   360		} else {
   361			// conservative - everything might be a pointer
   362			for(off = child->argoff; off < child->argoff + child->arglen; off += PtrSize) {
   363				dumpint(FieldKindPtr);
   364				dumpint(off);
   365			}
   366		}
   367	
   368		// Dump fields in the local vars section
   369		if(stackmap == nil) {
   370			// No locals information, dump everything.
   371			for(off = child->arglen; off < s->varp - (byte*)s->sp; off += PtrSize) {
   372				dumpint(FieldKindPtr);
   373				dumpint(off);
   374			}
   375		} else if(stackmap->n < 0) {
   376			// Locals size information, dump just the locals.
   377			size = -stackmap->n;
   378			for(off = s->varp - size - (byte*)s->sp; off < s->varp - (byte*)s->sp; off += PtrSize) {
   379				dumpint(FieldKindPtr);
   380				dumpint(off);
   381			}
   382		} else if(stackmap->n > 0) {
   383			// Locals bitmap information, scan just the pointers in
   384			// locals.
   385			dumpbv(&bv, s->varp - bv.n / BitsPerPointer * PtrSize - (byte*)s->sp);
   386		}
   387		dumpint(FieldKindEol);
   388	
   389		// Record arg info for parent.
   390		child->argoff = s->argp - (byte*)s->fp;
   391		child->arglen = s->arglen;
   392		child->sp = (byte*)s->sp;
   393		child->depth++;
   394		stackmap = runtime·funcdata(f, FUNCDATA_ArgsPointerMaps);
   395		if(stackmap != nil)
   396			child->args = runtime·stackmapdata(stackmap, pcdata);
   397		else
   398			child->args.n = -1;
   399		return true;
   400	}

   402	static void
   403	dumpgoroutine(G *gp)
   404	{
   405		uintptr sp, pc, lr;
   406		ChildInfo child;
   407		Defer *d;
   408		Panic *p;
   409	
   410		if(gp->syscallstack != (uintptr)nil) {
   411			sp = gp->syscallsp;
   412			pc = gp->syscallpc;
   413			lr = 0;
   414		} else {
   415			sp = gp->sched.sp;
   416			pc = gp->sched.pc;
   417			lr = gp->sched.lr;
   418		}
   419	
   420		dumpint(TagGoRoutine);
   421		dumpint((uintptr)gp);
   422		dumpint((uintptr)sp);
   423		dumpint(gp->goid);
   424		dumpint(gp->gopc);
   425		dumpint(gp->status);
   426		dumpbool(gp->issystem);
   427		dumpbool(gp->isbackground);
   428		dumpint(gp->waitsince);
   429		dumpcstr(gp->waitreason);
   430		dumpint((uintptr)gp->sched.ctxt);
   431		dumpint((uintptr)gp->m);
   432		dumpint((uintptr)gp->defer);
   433		dumpint((uintptr)gp->panic);
   434	
   435		// dump stack
   436		child.args.n = -1;
   437		child.arglen = 0;
   438		child.sp = nil;
   439		child.depth = 0;
   440		if(!ScanStackByFrames)
   441			runtime·throw("need frame info to dump stacks");
   442		runtime·gentraceback(pc, sp, lr, gp, 0, nil, 0x7fffffff, dumpframe, &child, false);
   443	
   444		// dump defer & panic records
   445		for(d = gp->defer; d != nil; d = d->link) {
   446			dumpint(TagDefer);
   447			dumpint((uintptr)d);
   448			dumpint((uintptr)gp);
   449			dumpint((uintptr)d->argp);
   450			dumpint((uintptr)d->pc);
   451			dumpint((uintptr)d->fn);
   452			dumpint((uintptr)d->fn->fn);
   453			dumpint((uintptr)d->link);
   454		}
   455		for (p = gp->panic; p != nil; p = p->link) {
   456			dumpint(TagPanic);
   457			dumpint((uintptr)p);
   458			dumpint((uintptr)gp);
   459			dumpint((uintptr)p->arg.type);
   460			dumpint((uintptr)p->arg.data);
   461			dumpint((uintptr)p->defer);
   462			dumpint((uintptr)p->link);
   463		}
   464	}

   466	static void
   467	dumpgs(void)
   468	{
   469		G *gp;
   470		uint32 i;
   471	
   472		// goroutines & stacks
   473		for(i = 0; i < runtime·allglen; i++) {
   474			gp = runtime·allg[i];
   475			switch(gp->status){
   476			default:
   477				runtime·printf("unexpected G.status %d\n", gp->status);
   478				runtime·throw("mark - bad status");
   479			case Gdead:
   480				break;
   481			case Grunnable:
   482			case Gsyscall:
   483			case Gwaiting:
   484				dumpgoroutine(gp);
   485				break;
   486			}
   487		}
   488	}

   503	static void
   504	dumproots(void)
   505	{
   506		MSpan *s, **allspans;
   507		uint32 spanidx;
   508		Special *sp;
   509		SpecialFinalizer *spf;
   510		byte *p;
   511	
   512		// data segment
   513		dumpint(TagData);
   514		dumpint((uintptr)data);
   515		dumpmemrange(data, edata - data);
   516		dumpfields((uintptr*)gcdata + 1);
   517	
   518		// bss segment
   519		dumpint(TagBss);
   520		dumpint((uintptr)bss);
   521		dumpmemrange(bss, ebss - bss);
   522		dumpfields((uintptr*)gcbss + 1);
   523		
   524		// MSpan.types
   525		allspans = runtime·mheap.allspans;
   526		for(spanidx=0; spanidx<runtime·mheap.nspan; spanidx++) {
   527			s = allspans[spanidx];
   528			if(s->state == MSpanInUse) {
   529				// The garbage collector ignores type pointers stored in MSpan.types:
   530				//  - Compiler-generated types are stored outside of heap.
   531				//  - The reflect package has runtime-generated types cached in its data structures.
   532				//    The garbage collector relies on finding the references via that cache.
   533				switch(s->types.compression) {
   534				case MTypes_Empty:
   535				case MTypes_Single:
   536					break;
   537				case MTypes_Words:
   538				case MTypes_Bytes:
   539					dumpotherroot("runtime type info", (byte*)s->types.data);
   540					break;
   541				}
   542	
   543				// Finalizers
   544				for(sp = s->specials; sp != nil; sp = sp->next) {
   545					if(sp->kind != KindSpecialFinalizer)
   546						continue;
   547					spf = (SpecialFinalizer*)sp;
   548					p = (byte*)((s->start << PageShift) + spf->offset);
   549					dumpfinalizer(p, spf->fn, spf->fint, spf->ot);
   550				}
   551			}
   552		}
   553	
   554		// Finalizer queue
   555		runtime·iterate_finq(finq_callback);
   556	}

   558	// Bit vector of free marks.
   559	// Needs to be as big as the largest number of objects per span.
   560	static byte free[PageSize/8];
   561	
   562	static void
   563	dumpobjs(void)
   564	{
   565		uintptr i, j, size, n, off, shift, *bitp, bits, ti, kind;
   566		MSpan *s;
   567		MLink *l;
   568		byte *p;
   569		Type *t;
   570	
   571		for(i = 0; i < runtime·mheap.nspan; i++) {
   572			s = runtime·mheap.allspans[i];
   573			if(s->state != MSpanInUse)
   574				continue;
   575			p = (byte*)(s->start << PageShift);
   576			size = s->elemsize;
   577			n = (s->npages << PageShift) / size;
   578			if(n > PageSize/8)
   579				runtime·throw("free array doesn't have enough entries");
   580			for(l = s->freelist; l != nil; l = l->next) {
   581				free[((byte*)l - p) / size] = true;
   582			}
   583			for(j = 0; j < n; j++, p += size) {
   584				if(free[j]) {
   585					free[j] = false;
   586					continue;
   587				}
   588				off = (uintptr*)p - (uintptr*)runtime·mheap.arena_start;
   589				bitp = (uintptr*)runtime·mheap.arena_start - off/wordsPerBitmapWord - 1;
   590				shift = off % wordsPerBitmapWord;
   591				bits = *bitp >> shift;
   592	
   593				// Skip FlagNoGC allocations (stacks)
   594				if((bits & bitAllocated) == 0)
   595					continue;
   596	
   597				// extract type and kind
   598				ti = runtime·gettype(p);
   599				t = (Type*)(ti & ~(uintptr)(PtrSize-1));
   600				kind = ti & (PtrSize-1);
   601				
   602				// dump it
   603				if(kind == TypeInfo_Chan)
   604					t = ((ChanType*)t)->elem; // use element type for chan encoding
   605				if(t == nil && scannable(p))
   606					kind = TypeInfo_Conservative; // special kind for conservatively scanned objects
   607				dumpobj(p, size, t, kind);
   608			}
   609		}
   610	}

   612	static void
   613	dumpparams(void)
   614	{
   615		byte *x;
   616	
   617		dumpint(TagParams);
   618		x = (byte*)1;
   619		if(*(byte*)&x == 1)
   620			dumpbool(false); // little-endian ptrs
   621		else
   622			dumpbool(true); // big-endian ptrs
   623		dumpint(PtrSize);
   624		dumpint(runtime·Hchansize);
   625		dumpint((uintptr)runtime·mheap.arena_start);

   698	dumpmemprof_callback(Bucket *b, uintptr nstk, uintptr *stk, uintptr size, uintptr allocs, uintptr frees)
   699	{
   700		uintptr i, pc;
   701		Func *f;
   702		byte buf[20];

   704		int32 line;

   706		dumpint(TagMemProf);
   707		dumpint((uintptr)b);

   711			pc = stk[i];

   714				runtime·snprintf(buf, sizeof(buf), "%X", (uint64)pc);

   717				dumpint(0);
   718			} else {
   719				dumpcstr(runtime·funcname(f));
   720				// TODO: Why do we need to back up to a call instruction here?
   721				// Maybe profiler should do this.
   722				if(i > 0 && pc > f->entry) {
   723					if(thechar == '6' || thechar == '8')
   724						pc--;
   725					else
   726						pc -= 4; // arm, etc
   727				}
   728				line = runtime·funcline(f, pc, &file);
   729				dumpstr(file);
   730				dumpint(line);
   731			}
   732		}
   733		dumpint(allocs);
   734		dumpint(frees);

   737	static void
   738	dumpmemprof(void)
   739	{
   740		MSpan *s, **allspans;
   741		uint32 spanidx;
   742		Special *sp;
   743		SpecialProfile *spp;
   744		byte *p;
   745	
   746		runtime·iterate_memprof(dumpmemprof_callback);
   747	
   748		allspans = runtime·mheap.allspans;
   749		for(spanidx=0; spanidx<runtime·mheap.nspan; spanidx++) {
   750			s = allspans[spanidx];
   751			if(s->state != MSpanInUse)
   752				continue;
   753			for(sp = s->specials; sp != nil; sp = sp->next) {
   754				if(sp->kind != KindSpecialProfile)
   755					continue;
   756				spp = (SpecialProfile*)sp;
   757				p = (byte*)((s->start << PageShift) + spp->offset);
   758				dumpint(TagAllocSample);
   759				dumpint((uintptr)p);
   760				dumpint((uintptr)spp->b);
   761			}
   762		}

   766	mdump(G *gp)
   767	{
   768		byte *hdr;
   769		uintptr i;
   770		MSpan *s;
   771

   799	runtime∕debug·WriteHeapDump(uintptr fd)
   800	{
   801		// Stop the world.
   802		runtime·semacquire(&runtime·worldsema, false);
   803		m->gcing = 1;
   804		m->locks++;
   805		runtime·stoptheworld();
   806	
   807		// Update stats so we can dump them.
   808		// As a side effect, flushes all the MCaches so the MSpan.freelist
   809		// lists contain all the free objects.
   810		runtime·updatememstats(nil);
   811	
   812		// Set dump file.
   813		dumpfd = fd;
   814	
   815		// Call dump routine on M stack.
   816		g->status = Gwaiting;
   817		g->waitreason = "dumping heap";
   818		runtime·mcall(mdump);
   819	
   820		// Reset dump file.
   821		dumpfd = 0;
   822	
   823		// Start up the world again.
   824		m->gcing = 0;
   825		runtime·semrelease(&runtime·worldsema);
   826		runtime·starttheworld();
   827		m->locks--;

   830	// Runs the specified gc program.  Calls the callback for every
   831	// pointer-like field specified by the program and passes to the
   832	// callback the kind and offset of that field within the object.
   833	// offset is the offset in the object of the start of the program.
   834	// Returns a pointer to the opcode that ended the gc program (either
   835	// GC_END or GC_ARRAY_NEXT).
   836	static uintptr*
   837	playgcprog(uintptr offset, uintptr *prog, void (*callback)(void*,uintptr,uintptr), void *arg)
   838	{
   839		uintptr len, elemsize, i, *end;
   840	
   841		for(;;) {
   842			switch(prog[0]) {
   843			case GC_END:
   844				return prog;
   845			case GC_PTR:
   846				callback(arg, FieldKindPtr, offset + prog[1]);
   847				prog += 3;
   848				break;
   849			case GC_APTR:
   850				callback(arg, FieldKindPtr, offset + prog[1]);
   851				prog += 2;
   852				break;
   853			case GC_ARRAY_START:
   854				len = prog[2];
   855				elemsize = prog[3];
   856				end = nil;
   857				for(i = 0; i < len; i++) {
   858					end = playgcprog(offset + prog[1] + i * elemsize, prog + 4, callback, arg);
   859					if(end[0] != GC_ARRAY_NEXT)
   860						runtime·throw("GC_ARRAY_START did not have matching GC_ARRAY_NEXT");
   861				}
   862				prog = end + 1;
   863				break;
   864			case GC_ARRAY_NEXT:
   865				return prog;
   866			case GC_CALL:
   867				playgcprog(offset + prog[1], (uintptr*)((byte*)prog + *(int32*)&prog[2]), callback, arg);
   868				prog += 3;
   869				break;
   870			case GC_CHAN_PTR:
   871				callback(arg, FieldKindPtr, offset + prog[1]);
   872				prog += 3;
   873				break;
   874			case GC_STRING:
   875				callback(arg, FieldKindString, offset + prog[1]);
   876				prog += 2;
   877				break;
   878			case GC_EFACE:
   879				callback(arg, FieldKindEface, offset + prog[1]);
   880				prog += 2;
   881				break;
   882			case GC_IFACE:
   883				callback(arg, FieldKindIface, offset + prog[1]);
   884				prog += 2;
   885				break;
   886			case GC_SLICE:
   887				callback(arg, FieldKindSlice, offset + prog[1]);
   888				prog += 3;
   889				break;
   890			case GC_REGION:
   891				playgcprog(offset + prog[1], (uintptr*)prog[3] + 1, callback, arg);
   892				prog += 4;
   893				break;
   894			default:
   895				runtime·printf("%D\n", (uint64)prog[0]);
   896				runtime·throw("bad gc op");
   897			}
   898		}
   899	}

   928	// The heap dump reader needs to be able to disambiguate
   929	// Eface entries.  So it needs to know every type that might
   930	// appear in such an entry.  The following two routines accomplish
   931	// that.
   932	
   933	// Dump all the types that appear in the type field of
   934	// any Eface contained in obj.
   935	static void
   936	dumpefacetypes(void *obj, uintptr size, Type *type, uintptr kind)
   937	{
   938		uintptr i;
   939	
   940		switch(kind) {
   941		case TypeInfo_SingleObject:
   942			playgcprog(0, (uintptr*)type->gc + 1, dumpeface_callback, obj);
   943			break;
   944		case TypeInfo_Array:
   945			for(i = 0; i <= size - type->size; i += type->size)
   946				playgcprog(i, (uintptr*)type->gc + 1, dumpeface_callback, obj);
   947			break;
   948		case TypeInfo_Chan:
   949			if(type->size == 0) // channels may have zero-sized objects in them
   950				break;
   951			for(i = runtime·Hchansize; i <= size - type->size; i += type->size)
   952				playgcprog(i, (uintptr*)type->gc + 1, dumpeface_callback, obj);
   953			break;
   954		}
   955	}

   957	// Dump all the types that appear in the type field of
   958	// any Eface described by this bit vector.
   959	static void
   960	dumpbvtypes(BitVector *bv, byte *base)
   961	{
   962		uintptr i;
   963	
   964		for(i = 0; i < bv->n; i += BitsPerPointer) {
   965			if((bv->data[i/32] >> i%32 & 3) != BitsMultiWord)
   966				continue;
   967			switch(bv->data[(i+BitsPerPointer)/32] >> (i+BitsPerPointer)%32 & 3) {
   968			case BitsString:
   969			case BitsIface:
   970				i += BitsPerPointer;
   971				break;
   972			case BitsSlice:
   973				i += 2 * BitsPerPointer;
   974				break;
   975			case BitsEface:
   976				dumptype(*(Type**)(base + i / BitsPerPointer * PtrSize));
   977				i += BitsPerPointer;
   978				break;
   979			}
   980		}
