    36	// stackcacherefill/stackcacherelease implement a global cache of stack segments.
    37	// The cache is required to prevent unlimited growth of per-thread caches.
    38	static void
    39	stackcacherefill(void)
    40	{
    41		StackCacheNode *n;
    42		int32 i, pos;
    43	
    44		runtime·lock(&stackcachemu);
    45		n = stackcache;
    46		if(n)
    47			stackcache = n->next;
    48		runtime·unlock(&stackcachemu);
    49		if(n == nil) {
    50			n = (StackCacheNode*)runtime·SysAlloc(FixedStack*StackCacheBatch, &mstats.stacks_sys);
    51			if(n == nil)
    52				runtime·throw("out of memory (stackcacherefill)");
    53			for(i = 0; i < StackCacheBatch-1; i++)
    54				n->batch[i] = (byte*)n + (i+1)*FixedStack;
    55		}
    56		pos = m->stackcachepos;
    57		for(i = 0; i < StackCacheBatch-1; i++) {
    58			m->stackcache[pos] = n->batch[i];
    59			pos = (pos + 1) % StackCacheSize;
    60		}
    61		m->stackcache[pos] = n;
    62		pos = (pos + 1) % StackCacheSize;
    63		m->stackcachepos = pos;
    64		m->stackcachecnt += StackCacheBatch;
    65	}

    67	static void
    68	stackcacherelease(void)
    69	{
    70		StackCacheNode *n;
    71		uint32 i, pos;
    72	
    73		pos = (m->stackcachepos - m->stackcachecnt) % StackCacheSize;
    74		n = (StackCacheNode*)m->stackcache[pos];
    75		pos = (pos + 1) % StackCacheSize;
    76		for(i = 0; i < StackCacheBatch-1; i++) {
    77			n->batch[i] = m->stackcache[pos];
    78			pos = (pos + 1) % StackCacheSize;
    79		}
    80		m->stackcachecnt -= StackCacheBatch;
    81		runtime·lock(&stackcachemu);
    82		n->next = stackcache;
    83		stackcache = n;
    84		runtime·unlock(&stackcachemu);
    85	}

    87	void*
    88	runtime·stackalloc(G *gp, uint32 n)
    89	{
    90		uint32 pos;
    91		void *v;
    92		bool malloced;
    93		Stktop *top;
    94	
    95		// Stackalloc must be called on scheduler stack, so that we
    96		// never try to grow the stack during the code that stackalloc runs.
    97		// Doing so would cause a deadlock (issue 1547).
    98		if(g != m->g0)
    99			runtime·throw("stackalloc not on scheduler stack");
   100		if((n & (n-1)) != 0)
   101			runtime·throw("stack size not a power of 2");
   102		if(StackDebug >= 1)
   103			runtime·printf("stackalloc %d\n", n);
   104	
   105		gp->stacksize += n;
   106		if(runtime·debug.efence || StackFromSystem) {
   107			v = runtime·SysAlloc(ROUND(n, PageSize), &mstats.stacks_sys);
   108			if(v == nil)
   109				runtime·throw("out of memory (stackalloc)");
   110			return v;
   111		}
   112	
   113		// Minimum-sized stacks are allocated with a fixed-size free-list allocator,
   114		// but if we need a stack of a bigger size, we fall back on malloc
   115		// (assuming that inside malloc all the stack frames are small,
   116		// so that we do not deadlock).
   117		malloced = true;
   118		if(n == FixedStack || m->mallocing) {
   119			if(n != FixedStack) {
   120				runtime·printf("stackalloc: in malloc, size=%d want %d\n", FixedStack, n);
   121				runtime·throw("stackalloc");
   122			}
   123			if(m->stackcachecnt == 0)
   124				stackcacherefill();
   125			pos = m->stackcachepos;
   126			pos = (pos - 1) % StackCacheSize;
   127			v = m->stackcache[pos];
   128			m->stackcachepos = pos;
   129			m->stackcachecnt--;
   130			m->stackinuse++;
   131			malloced = false;
   132		} else
   133			v = runtime·mallocgc(n, 0, FlagNoProfiling|FlagNoGC|FlagNoZero|FlagNoInvokeGC);
   134	
   135		top = (Stktop*)((byte*)v+n-sizeof(Stktop));
   136		runtime·memclr((byte*)top, sizeof(*top));
   137		top->malloced = malloced;
   138		return v;
   139	}

   141	void
   142	runtime·stackfree(G *gp, void *v, Stktop *top)
   143	{
   144		uint32 pos;
   145		uintptr n;
   146	
   147		n = (uintptr)(top+1) - (uintptr)v;
   148		if(StackDebug >= 1)
   149			runtime·printf("stackfree %p %d\n", v, (int32)n);
   150		gp->stacksize -= n;
   151		if(runtime·debug.efence || StackFromSystem) {
   152			if(runtime·debug.efence || StackFaultOnFree)
   153				runtime·SysFault(v, n);
   154			else
   155				runtime·SysFree(v, n, &mstats.stacks_sys);
   156			return;
   157		}
   158		if(top->malloced) {
   159			runtime·free(v);
   160			return;
   161		}
   162		if(n != FixedStack)
   163			runtime·throw("stackfree: bad fixed size");
   164		if(m->stackcachecnt == StackCacheSize)
   165			stackcacherelease();
   166		pos = m->stackcachepos;
   167		m->stackcache[pos] = v;
   168		m->stackcachepos = (pos + 1) % StackCacheSize;
   169		m->stackcachecnt++;
   170		m->stackinuse--;
   171	}

   173	// Called from runtime·lessstack when returning from a function which
   174	// allocated a new stack segment.  The function's return value is in
   175	// m->cret.
   176	void
   177	runtime·oldstack(void)
   178	{
   179		Stktop *top;
   180		uint32 argsize;
   181		byte *sp, *old;
   182		uintptr *src, *dst, *dstend;
   183		G *gp;
   184		int64 goid;
   185		int32 oldstatus;
   186	
   187		gp = m->curg;
   188		top = (Stktop*)gp->stackbase;
   189		old = (byte*)gp->stackguard - StackGuard;
   190		sp = (byte*)top;
   191		argsize = top->argsize;
   192	
   193		if(StackDebug >= 1) {
   194			runtime·printf("runtime: oldstack gobuf={pc:%p sp:%p lr:%p} cret=%p argsize=%p\n",
   195				top->gobuf.pc, top->gobuf.sp, top->gobuf.lr, (uintptr)m->cret, (uintptr)argsize);
   196		}
   197	
   198		// gp->status is usually Grunning, but it could be Gsyscall if a stack overflow
   199		// happens during a function call inside entersyscall.
   200		oldstatus = gp->status;
   201		
   202		gp->sched = top->gobuf;
   203		gp->sched.ret = m->cret;
   204		m->cret = 0; // drop reference
   205		gp->status = Gwaiting;
   206		gp->waitreason = "stack unsplit";
   207	
   208		if(argsize > 0) {
   209			sp -= argsize;
   210			dst = (uintptr*)top->argp;
   211			dstend = dst + argsize/sizeof(*dst);
   212			src = (uintptr*)sp;
   213			while(dst < dstend)
   214				*dst++ = *src++;
   215		}
   216		goid = top->gobuf.g->goid;	// fault if g is bad, before gogo
   217		USED(goid);
   218	
   219		gp->stackbase = top->stackbase;
   220		gp->stackguard = top->stackguard;
   221		gp->stackguard0 = gp->stackguard;
   222		gp->panicwrap = top->panicwrap;
   223		runtime·stackfree(gp, old, top);
   224	
   225		gp->status = oldstatus;
   226		runtime·gogo(&gp->sched);
   227	}

   255	struct CopyableInfo {
   256		byte *stk;	// bottom address of segment

   263	static bool
   264	checkframecopy(Stkframe *frame, void *arg)
   265	{
   266		CopyableInfo *cinfo;
   267		Func *f;
   268		StackMap *stackmap;
   269	
   270		cinfo = arg;
   271		f = frame->fn;
   272		if(StackDebug >= 2)
   273			runtime·printf("    checking %s frame=[%p,%p] stk=[%p,%p]\n", runtime·funcname(f), frame->sp, frame->fp, cinfo->stk, cinfo->base);
   274		// if we're not in the segment any more, return immediately.
   275		if(frame->varp < cinfo->stk || frame->varp >= cinfo->base) {
   276			if(StackDebug >= 2)
   277				runtime·printf("    <next segment>\n");
   278			return false; // stop traceback
   279		}
   280		if(f->entry == (uintptr)runtime·main) {
   281			// A special routine at the TOS of the main routine.
   282			// We will allow it to be copied even though we don't
   283			// have full GC info for it (because it is written in C).
   284			cinfo->frames++;
   285			return false; // stop traceback
   286		}
   287		if(frame->varp != (byte*)frame->sp) { // not in prologue (and has at least one local or outarg)
   288			stackmap = runtime·funcdata(f, FUNCDATA_LocalsPointerMaps);
   289			if(stackmap == nil) {
   290				cinfo->frames = -1;
   291				if(StackDebug >= 1)
   292					runtime·printf("copystack: no locals info for %s\n", runtime·funcname(f));
   293				return false;
   294			}
   295			if(stackmap->n <= 0) {
   296				cinfo->frames = -1;
   297				if(StackDebug >= 1)
   298					runtime·printf("copystack: locals size info only for %s\n", runtime·funcname(f));
   299				return false;
   300			}
   301		}
   302		if(frame->arglen != 0) {
   303			stackmap = runtime·funcdata(f, FUNCDATA_ArgsPointerMaps);
   304			if(stackmap == nil) {
   305				cinfo->frames = -1;
   306				if(StackDebug >= 1)
   307					runtime·printf("copystack: no arg info for %s\n", runtime·funcname(f));
   308				return false;
   309			}
   310		}
   311		cinfo->frames++;
   312		return true; // this frame is ok; keep going
   313	}

   315	// If the top segment of the stack contains an uncopyable
   316	// frame, return -1.  Otherwise return the number of frames
   317	// in the top segment, all of which are copyable.
   318	static int32
   319	copyabletopsegment(G *gp)
   320	{
   321		CopyableInfo cinfo;
   322		Defer *d;
   323		Func *f;
   324		FuncVal *fn;
   325		StackMap *stackmap;
   326	
   327		cinfo.stk = (byte*)gp->stackguard - StackGuard;
   328		cinfo.base = (byte*)gp->stackbase + sizeof(Stktop);
   329		cinfo.frames = 0;
   330	
   331		// Check that each frame is copyable.  As a side effect,
   332		// count the frames.
   333		runtime·gentraceback(~(uintptr)0, ~(uintptr)0, 0, gp, 0, nil, 0x7fffffff, checkframecopy, &cinfo, false);
   334		if(StackDebug >= 1 && cinfo.frames != -1)
   335			runtime·printf("copystack: %d copyable frames\n", cinfo.frames);
   336	
   337		// Check to make sure all Defers are copyable
   338		for(d = gp->defer; d != nil; d = d->link) {
   339			if(cinfo.stk <= (byte*)d && (byte*)d < cinfo.base) {
   340				// Defer is on the stack.  Its copyableness has
   341				// been established during stack walking.
   342				// For now, this only happens with the Defer in runtime.main.
   343				continue;
   344			}
   345			if(d->argp < cinfo.stk || cinfo.base <= d->argp)
   346				break; // a defer for the next segment
   347			fn = d->fn;
   348			if(fn == nil) // See issue 8047
   349				continue;
   350			f = runtime·findfunc((uintptr)fn->fn);
   351			if(f == nil)
   352				return -1;
   353	
   354			// Check to make sure we have an args pointer map for the defer's args.
   355			// We only need the args map, but we check
   356			// for the locals map also, because when the locals map
   357			// isn't provided it means the ptr map came from C and
   358			// C (particularly, cgo) lies to us.  See issue 7695.
   359			stackmap = runtime·funcdata(f, FUNCDATA_ArgsPointerMaps);
   360			if(stackmap == nil || stackmap->n <= 0)
   361				return -1;
   362			stackmap = runtime·funcdata(f, FUNCDATA_LocalsPointerMaps);
   363			if(stackmap == nil || stackmap->n <= 0)
   364				return -1;
   365	
   366			if(cinfo.stk <= (byte*)fn && (byte*)fn < cinfo.base) {
   367				// FuncVal is on the stack.  Again, its copyableness
   368				// was established during stack walking.
   369				continue;
   370			}
   371			// The FuncVal may have pointers in it, but fortunately for us
   372			// the compiler won't put pointers into the stack in a
   373			// heap-allocated FuncVal.
   374			// One day if we do need to check this, we'll need maps of the
   375			// pointerness of the closure args.  The only place we have that map
   376			// right now is in the gc program for the FuncVal.  Ugh.
   377		}
   378	
   379		return cinfo.frames;
   380	}

   382	typedef struct AdjustInfo AdjustInfo;
   383	struct AdjustInfo {
   384		byte *oldstk;	// bottom address of segment
   385		byte *oldbase;	// top address of segment (after Stktop)
   386		uintptr delta;  // ptr distance from old to new stack (newbase - oldbase)
   387	};

   389	// bv describes the memory starting at address scanp.
   390	// Adjust any pointers contained therein.
   391	static void
   392	adjustpointers(byte **scanp, BitVector *bv, AdjustInfo *adjinfo, Func *f)
   393	{
   394		uintptr delta;
   395		int32 num, i;
   396		byte *p, *minp, *maxp;
   397		Type *t;
   398		Itab *tab;
   399		
   400		minp = adjinfo->oldstk;
   401		maxp = adjinfo->oldbase;
   402		delta = adjinfo->delta;
   403		num = bv->n / BitsPerPointer;
   404		for(i = 0; i < num; i++) {
   405			if(StackDebug >= 4)
   406				runtime·printf("        %p:%s:%p\n", &scanp[i], mapnames[bv->data[i / (32 / BitsPerPointer)] >> (i * BitsPerPointer & 31) & 3], scanp[i]);
   407			switch(bv->data[i / (32 / BitsPerPointer)] >> (i * BitsPerPointer & 31) & 3) {
   408			case BitsDead:
   409				if(runtime·debug.gcdead)
   410					scanp[i] = (byte*)PoisonStack;
   411				break;
   412			case BitsScalar:
   413				break;
   414			case BitsPointer:
   415				p = scanp[i];
   416				if(f != nil && (byte*)0 < p && (p < (byte*)PageSize || (uintptr)p == PoisonGC || (uintptr)p == PoisonStack)) {
   417					// Looks like a junk value in a pointer slot.
   418					// Live analysis wrong?
   419					m->traceback = 2;
   420					runtime·printf("runtime: bad pointer in frame %s at %p: %p\n", runtime·funcname(f), &scanp[i], p);
   421					runtime·throw("bad pointer!");
   422				}
   423				if(minp <= p && p < maxp) {
   424					if(StackDebug >= 3)
   425						runtime·printf("adjust ptr %p %s\n", p, runtime·funcname(f));
   426					scanp[i] = p + delta;
   427				}
   428				break;
   429			case BitsMultiWord:
   430				switch(bv->data[(i+1) / (32 / BitsPerPointer)] >> ((i+1) * BitsPerPointer & 31) & 3) {
   431				case BitsString:
   432					// string referents are never on the stack, never need to be adjusted
   433					i++; // skip len
   434					break;
   435				case BitsSlice:
   436					p = scanp[i];
   437					if(minp <= p && p < maxp) {
   438						if(StackDebug >= 3)
   439							runtime·printf("adjust slice %p\n", p);
   440						scanp[i] = p + delta;
   441					}
   442					i += 2; // skip len, cap
   443					break;
   444				case BitsEface:
   445					t = (Type*)scanp[i];
   446					if(t != nil && (t->size > PtrSize || (t->kind & KindNoPointers) == 0)) {
   447						p = scanp[i+1];
   448						if(minp <= p && p < maxp) {
   449							if(StackDebug >= 3)
   450								runtime·printf("adjust eface %p\n", p);
   451							if(t->size > PtrSize) // currently we always allocate such objects on the heap
   452								runtime·throw("large interface value found on stack");
   453							scanp[i+1] = p + delta;
   454						}
   455					}
   456					i++;
   457					break;
   458				case BitsIface:
   459					tab = (Itab*)scanp[i];
   460					if(tab != nil) {
   461						t = tab->type;
   462						//runtime·printf("          type=%p\n", t);
   463						if(t->size > PtrSize || (t->kind & KindNoPointers) == 0) {
   464							p = scanp[i+1];
   465							if(minp <= p && p < maxp) {
   466								if(StackDebug >= 3)
   467									runtime·printf("adjust iface %p\n", p);
   468								if(t->size > PtrSize) // currently we always allocate such objects on the heap
   469									runtime·throw("large interface value found on stack");
   470								scanp[i+1] = p + delta;
   471							}
   472						}
   473					}
   474					i++;
   475					break;
   476				}
   477				break;
   478			}
   479		}
   480	}

   482	// Note: the argument/return area is adjusted by the callee.
   483	static bool
   484	adjustframe(Stkframe *frame, void *arg)
   485	{
   486		AdjustInfo *adjinfo;
   487		Func *f;
   488		StackMap *stackmap;
   489		int32 pcdata;
   490		BitVector bv;
   491		uintptr targetpc;
   492	
   493		adjinfo = arg;
   494		f = frame->fn;
   495		if(StackDebug >= 2)
   496			runtime·printf("    adjusting %s frame=[%p,%p] pc=%p continpc=%p\n", runtime·funcname(f), frame->sp, frame->fp, frame->pc, frame->continpc);
   497		if(f->entry == (uintptr)runtime·main)
   498			return true;
   499		targetpc = frame->continpc;
   500		if(targetpc == 0) {
   501			// Frame is dead.
   502			return true;
   503		}
   504		if(targetpc != f->entry)
   505			targetpc--;
   506		pcdata = runtime·pcdatavalue(f, PCDATA_StackMapIndex, targetpc);
   507		if(pcdata == -1)
   508			pcdata = 0; // in prologue
   509	
   510		// adjust local pointers
   511		if(frame->varp != (byte*)frame->sp) {
   512			stackmap = runtime·funcdata(f, FUNCDATA_LocalsPointerMaps);
   513			if(stackmap == nil)
   514				runtime·throw("no locals info");
   515			if(stackmap->n <= 0)
   516				runtime·throw("locals size info only");
   517			bv = runtime·stackmapdata(stackmap, pcdata);
   518			if(StackDebug >= 3)
   519				runtime·printf("      locals\n");
   520			adjustpointers((byte**)frame->varp - bv.n / BitsPerPointer, &bv, adjinfo, f);
   521		}
   522		// adjust inargs and outargs
   523		if(frame->arglen != 0) {
   524			stackmap = runtime·funcdata(f, FUNCDATA_ArgsPointerMaps);
   525			if(stackmap == nil)
   526				runtime·throw("no arg info");
   527			bv = runtime·stackmapdata(stackmap, pcdata);
   528			if(StackDebug >= 3)
   529				runtime·printf("      args\n");
   530			adjustpointers((byte**)frame->argp, &bv, adjinfo, nil);
   531		}
   532		return true;
   533	}

   535	static void
   536	adjustctxt(G *gp, AdjustInfo *adjinfo)
   537	{
   538		if(adjinfo->oldstk <= (byte*)gp->sched.ctxt && (byte*)gp->sched.ctxt < adjinfo->oldbase)
   539			gp->sched.ctxt = (byte*)gp->sched.ctxt + adjinfo->delta;

   542	static void
   543	adjustdefers(G *gp, AdjustInfo *adjinfo)
   544	{
   545		Defer *d, **dp;
   546		Func *f;
   547		FuncVal *fn;
   548		StackMap *stackmap;
   549		BitVector bv;
   550	
   551		for(dp = &gp->defer, d = *dp; d != nil; dp = &d->link, d = *dp) {
   552			if(adjinfo->oldstk <= (byte*)d && (byte*)d < adjinfo->oldbase) {
   553				// The Defer record is on the stack.  Its fields will
   554				// get adjusted appropriately.
   555				// This only happens for runtime.main now, but a compiler
   556				// optimization could do more of this.
   557				*dp = (Defer*)((byte*)d + adjinfo->delta);
   558				continue;
   559			}
   560			if(d->argp < adjinfo->oldstk || adjinfo->oldbase <= d->argp)
   561				break; // a defer for the next segment
   562			fn = d->fn;
   563			if(fn == nil) {
   564				// Defer of nil function.  It will panic when run, and there
   565				// aren't any args to adjust.  See issue 8047.
   566				d->argp += adjinfo->delta;
   567				continue;
   568			}
   569			f = runtime·findfunc((uintptr)fn->fn);
   570			if(f == nil)
   571				runtime·throw("can't adjust unknown defer");
   572			if(StackDebug >= 4)
   573				runtime·printf("  checking defer %s\n", runtime·funcname(f));
   574			// Defer's FuncVal might be on the stack
   575			if(adjinfo->oldstk <= (byte*)fn && (byte*)fn < adjinfo->oldbase) {
   576				if(StackDebug >= 3)
   577					runtime·printf("    adjust defer fn %s\n", runtime·funcname(f));
   578				d->fn = (FuncVal*)((byte*)fn + adjinfo->delta);
   579			} else {
   580				// deferred function's args might point into the stack.
   581				if(StackDebug >= 3)
   582					runtime·printf("    adjust deferred args for %s\n", runtime·funcname(f));
   583				stackmap = runtime·funcdata(f, FUNCDATA_ArgsPointerMaps);
   584				if(stackmap == nil)
   585					runtime·throw("runtime: deferred function has no arg ptr map");
   586				bv = runtime·stackmapdata(stackmap, 0);
   587				adjustpointers(d->args, &bv, adjinfo, f);
   588			}
   589			d->argp += adjinfo->delta;
   590		}
   591	}

   593	// Copies the top stack segment of gp to a new stack segment of a
   594	// different size.  The top segment must contain nframes frames.
   595	static void
   596	copystack(G *gp, uintptr nframes, uintptr newsize)
   597	{
   598		byte *oldstk, *oldbase, *newstk, *newbase;
   599		uintptr oldsize, used;
   600		AdjustInfo adjinfo;
   601		Stktop *oldtop, *newtop;
   602		bool malloced;
   603	
   604		if(gp->syscallstack != 0)
   605			runtime·throw("can't handle stack copy in syscall yet");
   606		oldstk = (byte*)gp->stackguard - StackGuard;
   607		oldbase = (byte*)gp->stackbase + sizeof(Stktop);
   608		oldsize = oldbase - oldstk;
   609		used = oldbase - (byte*)gp->sched.sp;
   610		oldtop = (Stktop*)gp->stackbase;
   611	
   612		// allocate new stack
   613		newstk = runtime·stackalloc(gp, newsize);
   614		newbase = newstk + newsize;
   615		newtop = (Stktop*)(newbase - sizeof(Stktop));
   616		malloced = newtop->malloced;
   617	
   618		if(StackDebug >= 1)
   619			runtime·printf("copystack [%p %p]/%d -> [%p %p]/%d\n", oldstk, oldbase, (int32)oldsize, newstk, newbase, (int32)newsize);
   620		USED(oldsize);
   621		
   622		// adjust pointers in the to-be-copied frames
   623		adjinfo.oldstk = oldstk;
   624		adjinfo.oldbase = oldbase;
   625		adjinfo.delta = newbase - oldbase;
   626		runtime·gentraceback(~(uintptr)0, ~(uintptr)0, 0, gp, 0, nil, nframes, adjustframe, &adjinfo, false);
   627		
   628		// adjust other miscellaneous things that have pointers into stacks.
   629		adjustctxt(gp, &adjinfo);
   630		adjustdefers(gp, &adjinfo);
   631		
   632		// copy the stack (including Stktop) to the new location
   633		runtime·memmove(newbase - used, oldbase - used, used);
   634		newtop->malloced = malloced;
   635		
   636		// Swap out old stack for new one
   637		gp->stackbase = (uintptr)newtop;
   638		gp->stackguard = (uintptr)newstk + StackGuard;
   639		gp->stackguard0 = (uintptr)newstk + StackGuard; // NOTE: might clobber a preempt request
   640		if(gp->stack0 == (uintptr)oldstk)
   641			gp->stack0 = (uintptr)newstk;
   642		gp->sched.sp = (uintptr)(newbase - used);
   643	
   644		// free old stack
   645		runtime·stackfree(gp, oldstk, oldtop);
   646	}

   648	// round x up to a power of 2.
   649	int32
   650	runtime·round2(int32 x)
   651	{
   652		int32 s;
   653	
   654		s = 0;
   655		while((1 << s) < x)
   656			s++;
   657		return 1 << s;
   658	}

   660	// Called from runtime·newstackcall or from runtime·morestack when a new
   661	// stack segment is needed.  Allocate a new stack big enough for
   662	// m->moreframesize bytes, copy m->moreargsize bytes to the new frame,
   663	// and then act as though runtime·lessstack called the function at
   664	// m->morepc.
   665	void
   666	runtime·newstack(void)
   667	{
   668		int32 framesize, argsize, oldstatus, oldsize, newsize, nframes;
   669		Stktop *top, *oldtop;
   670		byte *stk, *oldstk, *oldbase;
   671		uintptr sp;
   672		uintptr *src, *dst, *dstend;
   673		G *gp;
   674		Gobuf label, morebuf;
   675		void *moreargp;
   676		bool newstackcall;
   677	
   678		if(m->forkstackguard)
   679			runtime·throw("split stack after fork");
   680		if(m->morebuf.g != m->curg) {
   681			runtime·printf("runtime: newstack called from g=%p\n"
   682				"\tm=%p m->curg=%p m->g0=%p m->gsignal=%p\n",
   683				m->morebuf.g, m, m->curg, m->g0, m->gsignal);
   684			runtime·throw("runtime: wrong goroutine in newstack");
   685		}
   686	
   687		// gp->status is usually Grunning, but it could be Gsyscall if a stack overflow
   688		// happens during a function call inside entersyscall.
   689		gp = m->curg;
   690		oldstatus = gp->status;
   691	
   692		framesize = m->moreframesize;
   693		argsize = m->moreargsize;
   694		moreargp = m->moreargp;
   695		m->moreargp = nil;
   696		morebuf = m->morebuf;
   697		m->morebuf.pc = (uintptr)nil;
   698		m->morebuf.lr = (uintptr)nil;
   699		m->morebuf.sp = (uintptr)nil;
   700		gp->status = Gwaiting;
   701		gp->waitreason = "stack growth";
   702		newstackcall = framesize==1;
   703		if(newstackcall)
   704			framesize = 0;
   705	
   706		// For newstackcall the context already points to beginning of runtime·newstackcall.
   707		if(!newstackcall)
   708			runtime·rewindmorestack(&gp->sched);
   709	
   710		sp = gp->sched.sp;
   711		if(thechar == '6' || thechar == '8') {
   712			// The call to morestack cost a word.
   713			sp -= sizeof(uintptr);
   714		}
   715		if(StackDebug >= 1 || sp < gp->stackguard - StackGuard) {
   716			runtime·printf("runtime: newstack framesize=%p argsize=%p sp=%p stack=[%p, %p]\n"
   717				"\tmorebuf={pc:%p sp:%p lr:%p}\n"
   718				"\tsched={pc:%p sp:%p lr:%p ctxt:%p}\n",
   719				(uintptr)framesize, (uintptr)argsize, sp, gp->stackguard - StackGuard, gp->stackbase,
   720				m->morebuf.pc, m->morebuf.sp, m->morebuf.lr,
   721				gp->sched.pc, gp->sched.sp, gp->sched.lr, gp->sched.ctxt);
   722		}
   723		if(sp < gp->stackguard - StackGuard) {
   724			runtime·printf("runtime: split stack overflow: %p < %p\n", sp, gp->stackguard - StackGuard);
   725			runtime·throw("runtime: split stack overflow");
   726		}
   727	
   728		if(argsize % sizeof(uintptr) != 0) {
   729			runtime·printf("runtime: stack growth with misaligned argsize %d\n", argsize);
   730			runtime·throw("runtime: stack growth argsize");
   731		}
   732	
   733		if(gp->stackguard0 == (uintptr)StackPreempt) {
   734			if(gp == m->g0)
   735				runtime·throw("runtime: preempt g0");
   736			if(oldstatus == Grunning && m->p == nil && m->locks == 0)
   737				runtime·throw("runtime: g is running but p is not");
   738			if(oldstatus == Gsyscall && m->locks == 0)
   739				runtime·throw("runtime: stack growth during syscall");
   740			// Be conservative about where we preempt.
   741			// We are interested in preempting user Go code, not runtime code.
   742			if(oldstatus != Grunning || m->locks || m->mallocing || m->gcing || m->p->status != Prunning) {
   743				// Let the goroutine keep running for now.
   744				// gp->preempt is set, so it will be preempted next time.
   745				gp->stackguard0 = gp->stackguard;
   746				gp->status = oldstatus;
   747				runtime·gogo(&gp->sched);	// never return
   748			}
   749			// Act like goroutine called runtime.Gosched.
   750			gp->status = oldstatus;
   751			runtime·gosched0(gp);	// never return
   752		}
   753	
   754		// If every frame on the top segment is copyable, allocate a bigger segment
   755		// and move the segment instead of allocating a new segment.
   756		if(runtime·copystack) {
   757			if(!runtime·precisestack)
   758				runtime·throw("can't copy stacks without precise stacks");
   759			nframes = copyabletopsegment(gp);
   760			if(nframes != -1) {
   761				oldstk = (byte*)gp->stackguard - StackGuard;
   762				oldbase = (byte*)gp->stackbase + sizeof(Stktop);
   763				oldsize = oldbase - oldstk;
   764				newsize = oldsize * 2;
   765				copystack(gp, nframes, newsize);
   766				if(StackDebug >= 1)
   767					runtime·printf("stack grow done\n");
   768				if(gp->stacksize > runtime·maxstacksize) {
   769					runtime·printf("runtime: goroutine stack exceeds %D-byte limit\n", (uint64)runtime·maxstacksize);
   770					runtime·throw("stack overflow");
   771				}
   772				gp->status = oldstatus;
   773				runtime·gogo(&gp->sched);
   774			}
   775			// TODO: if stack is uncopyable because we're in C code, patch return value at
   776			// end of C code to trigger a copy as soon as C code exits.  That way, we'll
   777			// have stack available if we get this deep again.
   778		}
   779	
   780		// allocate new segment.
   781		framesize += argsize;
   782		framesize += StackExtra;	// room for more functions, Stktop.
   783		if(framesize < StackMin)
   784			framesize = StackMin;
   785		framesize += StackSystem;
   786		framesize = runtime·round2(framesize);
   787		stk = runtime·stackalloc(gp, framesize);
   788		if(gp->stacksize > runtime·maxstacksize) {
   789			runtime·printf("runtime: goroutine stack exceeds %D-byte limit\n", (uint64)runtime·maxstacksize);
   790			runtime·throw("stack overflow");
   791		}
   792		top = (Stktop*)(stk+framesize-sizeof(*top));
   793	
   794		if(StackDebug >= 1) {
   795			runtime·printf("\t-> new stack [%p, %p]\n", stk, top);
   796		}
   797	
   798		top->stackbase = gp->stackbase;
   799		top->stackguard = gp->stackguard;
   800		top->gobuf = morebuf;
   801		top->argp = moreargp;
   802		top->argsize = argsize;
   803	
   804		// copy flag from panic
   805		top->panic = gp->ispanic;
   806		gp->ispanic = false;
   807		
   808		// if this isn't a panic, maybe we're splitting the stack for a panic.
   809		// if we're splitting in the top frame, propagate the panic flag
   810		// forward so that recover will know we're in a panic.
   811		oldtop = (Stktop*)top->stackbase;
   812		if(oldtop != nil && oldtop->panic && top->argp == (byte*)oldtop - oldtop->argsize - gp->panicwrap)
   813			top->panic = true;
   814	
   815		top->panicwrap = gp->panicwrap;
   816		gp->panicwrap = 0;
   817	
   818		gp->stackbase = (uintptr)top;
   819		gp->stackguard = (uintptr)stk + StackGuard;
   820		gp->stackguard0 = gp->stackguard;
   821	
   822		sp = (uintptr)top;
   823		if(argsize > 0) {
   824			sp -= argsize;
   825			dst = (uintptr*)sp;
   826			dstend = dst + argsize/sizeof(*dst);
   827			src = (uintptr*)top->argp;
   828			while(dst < dstend)
   829				*dst++ = *src++;
   830		}
   831		if(thechar == '5') {
   832			// caller would have saved its LR below args.
   833			sp -= sizeof(void*);
   834			*(void**)sp = nil;
   835		}
   836	
   837		// Continue as if lessstack had just called m->morepc
   838		// (the PC that decided to grow the stack).
   839		runtime·memclr((byte*)&label, sizeof label);
   840		label.sp = sp;
   841		label.pc = (uintptr)runtime·lessstack;
   842		label.g = m->curg;
   843		if(newstackcall)
   844			runtime·gostartcallfn(&label, (FuncVal*)m->cret);
   845		else {
   846			runtime·gostartcall(&label, (void(*)(void))gp->sched.pc, gp->sched.ctxt);
   847			gp->sched.ctxt = nil;
   848		}
   849		gp->status = oldstatus;
   850		runtime·gogo(&label);
   851	
   852		*(int32*)345 = 123;	// never return
   853	}

   876	// Maybe shrink the stack being used by gp.
   877	// Called at garbage collection time.
   878	void
   879	runtime·shrinkstack(G *gp)
   880	{
   881		int32 nframes;
   882		byte *oldstk, *oldbase;
   883		uintptr used, oldsize, newsize;
   884		MSpan *span;
   885	
   886		if(!runtime·copystack)
   887			return;
   888		oldstk = (byte*)gp->stackguard - StackGuard;
   889		oldbase = (byte*)gp->stackbase + sizeof(Stktop);
   890		oldsize = oldbase - oldstk;
   891		newsize = oldsize / 2;
   892		if(newsize < FixedStack)
   893			return; // don't shrink below the minimum-sized stack
   894		used = oldbase - (byte*)gp->sched.sp;
   895		if(used >= oldsize / 4)
   896			return; // still using at least 1/4 of the segment.
   897	
   898		// To shrink to less than 1/2 a page, we need to copy.
   899		if(newsize < PageSize/2) {
   900			if(gp->syscallstack != (uintptr)nil) // TODO: can we handle this case?
   901				return;
   902	#ifdef GOOS_windows
   903			if(gp->m != nil && gp->m->libcallsp != 0)
   904				return;
   905	#endif
   906			nframes = copyabletopsegment(gp);
   907			if(nframes == -1)
   908				return;
   909			copystack(gp, nframes, newsize);
   910			return;
   911		}
   912	
   913		// To shrink a stack of one page size or more, we can shrink it
   914		// without copying.  Just deallocate the lower half.
   915		span = runtime·MHeap_LookupMaybe(&runtime·mheap, oldstk);
   916		if(span == nil)
   917			return; // stack allocated outside heap.  Can't shrink it.  Can happen if stack is allocated while inside malloc.  TODO: shrink by copying?
   918		if(span->elemsize != oldsize)
   919			runtime·throw("span element size doesn't match stack size");
   920		if((uintptr)oldstk != span->start << PageShift)
   921			runtime·throw("stack not at start of span");
   922	
   923		if(StackDebug)
   924			runtime·printf("shrinking stack in place %p %X->%X\n", oldstk, oldsize, newsize);
   925	
   926		// new stack guard for smaller stack
   927		gp->stackguard = (uintptr)oldstk + newsize + StackGuard;
   928		gp->stackguard0 = (uintptr)oldstk + newsize + StackGuard;
   929		if(gp->stack0 == (uintptr)oldstk)
   930			gp->stack0 = (uintptr)oldstk + newsize;
   931		gp->stacksize -= oldsize - newsize;
   932	
   933		// Free bottom half of the stack.
   934		if(runtime·debug.efence || StackFromSystem) {
   935			if(runtime·debug.efence || StackFaultOnFree)
   936				runtime·SysFault(oldstk, newsize);
   937			else
   938				runtime·SysFree(oldstk, newsize, &mstats.stacks_sys);
   939			return;
   940		}
   941		// First, we trick malloc into thinking
   942		// we allocated the stack as two separate half-size allocs.  Then the
   943		// free() call does the rest of the work for us.
   944		runtime·MSpan_EnsureSwept(span);
   945		runtime·MHeap_SplitSpan(&runtime·mheap, span);
   946		runtime·free(oldstk);
   947	}
