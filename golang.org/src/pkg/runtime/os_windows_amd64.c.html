    35	#define DBG_PRINTEXCEPTION_C 0x40010006
    36	
    37	// Called by sigtramp from Windows VEH handler.
    38	// Return value signals whether the exception has been handled (-1)
    39	// or should be made available to other handlers in the chain (0).
    40	uint32
    41	runtime·sighandler(ExceptionRecord *info, Context *r, G *gp)
    42	{
    43		bool crash;
    44		uintptr *sp;
    45		extern byte text[], etext[];
    46	
    47		if(info->ExceptionCode == DBG_PRINTEXCEPTION_C) {
    48			// This exception is intended to be caught by debuggers.
    49			// There is a not-very-informational message like
    50			// "Invalid parameter passed to C runtime function"
    51			// sitting at info->ExceptionInformation[0] (a wchar_t*),
    52			// with length info->ExceptionInformation[1].
    53			// The default behavior is to ignore this exception,
    54			// but somehow returning 0 here (meaning keep going)
    55			// makes the program crash instead. Maybe Windows has no
    56			// other handler registered? In any event, ignore it.
    57			return -1;
    58		}
    59	
    60		// Only handle exception if executing instructions in Go binary
    61		// (not Windows library code). 
    62		if(r->Rip < (uint64)text || (uint64)etext < r->Rip)
    63			return 0;
    64	
    65		switch(info->ExceptionCode) {
    66		case EXCEPTION_BREAKPOINT:
    67			// It is unclear whether this is needed, unclear whether it
    68			// would work, and unclear how to test it. Leave out for now.
    69			// This only handles breakpoint instructions written in the
    70			// assembly sources, not breakpoints set by a debugger, and
    71			// there are very few of the former.
    72			break;
    73		}
    74	
    75		if(gp != nil && runtime·issigpanic(info->ExceptionCode)) {
    76			// Make it look like a call to the signal func.
    77			// Have to pass arguments out of band since
    78			// augmenting the stack frame would break
    79			// the unwinding code.
    80			gp->sig = info->ExceptionCode;
    81			gp->sigcode0 = info->ExceptionInformation[0];
    82			gp->sigcode1 = info->ExceptionInformation[1];
    83			gp->sigpc = r->Rip;
    84	
    85			// Only push runtime·sigpanic if r->rip != 0.
    86			// If r->rip == 0, probably panicked because of a
    87			// call to a nil func.  Not pushing that onto sp will
    88			// make the trace look like a call to runtime·sigpanic instead.
    89			// (Otherwise the trace will end at runtime·sigpanic and we
    90			// won't get to see who faulted.)
    91			if(r->Rip != 0) {
    92				sp = (uintptr*)r->Rsp;
    93				*--sp = r->Rip;
    94				r->Rsp = (uintptr)sp;
    95			}
    96			r->Rip = (uintptr)runtime·sigpanic;
    97			return -1;
    98		}
    99	
   100		if(runtime·panicking)	// traceback already printed
   101			runtime·exit(2);
   102		runtime·panicking = 1;
   103	
   104		runtime·printf("Exception %x %p %p %p\n", info->ExceptionCode,
   105			info->ExceptionInformation[0], info->ExceptionInformation[1], r->Rip);
   106	
   107	
   108		runtime·printf("PC=%X\n", r->Rip);
   109		if(m->lockedg != nil && m->ncgo > 0 && gp == m->g0) {
   110			runtime·printf("signal arrived during cgo execution\n");
   111			gp = m->lockedg;
   112		}
   113		runtime·printf("\n");
   114	
   115		if(runtime·gotraceback(&crash)){
   116			runtime·traceback(r->Rip, r->Rsp, 0, gp);
   117			runtime·tracebackothers(gp);
   118			runtime·dumpregs(r);
   119		}
   120		
   121		if(crash)
   122			runtime·crash();
   123	
   124		runtime·exit(2);
   125		return -1; // not reached
   126	}
