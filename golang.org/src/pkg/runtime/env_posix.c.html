    13	byte*
    14	runtime·getenv(int8 *s)
    15	{
    16		int32 i, j;
    17		intgo len;
    18		byte *v, *bs;
    19		String* envv;
    20		int32 envc;
    21	
    22		bs = (byte*)s;
    23		len = runtime·findnull(bs);
    24		envv = (String*)syscall·envs.array;
    25		envc = syscall·envs.len;
    26		for(i=0; i<envc; i++){
    27			if(envv[i].len <= len)
    28				continue;
    29			v = envv[i].str;
    30			for(j=0; j<len; j++)
    31				if(bs[j] != v[j])
    32					goto nomatch;
    33			if(v[len] != '=')
    34				goto nomatch;
    35			return v+len+1;
    36		nomatch:;
    37		}
    38		return nil;
    39	}

    41	void (*_cgo_setenv)(byte**);

    43	// Update the C environment if cgo is loaded.
    44	// Called from syscall.Setenv.
    45	void
    46	syscall·setenv_c(String k, String v)
    47	{
    48		byte *arg[2];
    49		uintptr len;
    50	
    51		if(_cgo_setenv == nil)
    52			return;
    53	
    54		// Objects that are explicitly freed must be at least 16 bytes in size,
    55		// so that they are not allocated using tiny alloc.
    56		len = k.len + 1;
    57		if(len < TinySize)
    58			len = TinySize;
    59		arg[0] = runtime·malloc(len);
    60		runtime·memmove(arg[0], k.str, k.len);
    61		arg[0][k.len] = 0;
    62	
    63		len = v.len + 1;
    64		if(len < TinySize)
    65			len = TinySize;
    66		arg[1] = runtime·malloc(len);
    67		runtime·memmove(arg[1], v.str, v.len);
    68		arg[1][v.len] = 0;
    69	
    70		runtime·asmcgocall((void*)_cgo_setenv, arg);
    71		runtime·free(arg[0]);
    72		runtime·free(arg[1]);
    73	}
