   111		MOVQ	$0, gobuf_ret(AX)

   119	TEXT runtime·gogo(SB), NOSPLIT, $0-4
   120		MOVL	b+0(FP), BX		// gobuf
   121		MOVL	gobuf_g(BX), DX

   123		get_tls(CX)

   134	// void mcall(void (*fn)(G*))
   135	// Switch to m->g0's stack, call fn(g).
   136	// Fn must never return.  It should gogo(&g->sched)
   137	// to keep running g.
   138	TEXT runtime·mcall(SB), NOSPLIT, $0-4
   139		MOVL	fn+0(FP), DI
   140		
   141		get_tls(CX)
   142		MOVL	g(CX), AX	// save state in g->sched
   143		MOVL	0(SP), BX	// caller's PC
   144		MOVL	BX, (g_sched+gobuf_pc)(AX)
   145		LEAL	fn+0(FP), BX	// caller's SP
   146		MOVL	BX, (g_sched+gobuf_sp)(AX)
   147		MOVL	AX, (g_sched+gobuf_g)(AX)
   148	
   149		// switch to m->g0 & its stack, call fn
   150		MOVL	m(CX), BX
   151		MOVL	m_g0(BX), SI
   152		CMPL	SI, AX	// if g == m->g0 call badmcall
   153		JNE	3(PC)
   154		MOVL	$runtime·badmcall(SB), AX
   155		JMP	AX
   156		MOVL	SI, g(CX)	// g = m->g0
   157		MOVL	(g_sched+gobuf_sp)(SI), SP	// sp = m->g0->sched.sp
   158		PUSHQ	AX
   159		ARGSIZE(8)
   160		CALL	DI
   161		POPQ	AX
   162		MOVL	$runtime·badmcall2(SB), AX
   163		JMP	AX
   164		RET

   170	// Called during function prolog when more stack is needed.
   171	// Caller has already done get_tls(CX); MOVQ m(CX), BX.
   172	//
   173	// The traceback routines see morestack on a g0 as being
   174	// the top of a stack (for example, morestack calling newstack
   175	// calling the scheduler calling newm calling gc), so we must
   176	// record an argument size. For that purpose, it has no arguments.
   177	TEXT runtime·morestack(SB),NOSPLIT,$0-0
   178		// Cannot grow scheduler stack (m->g0).
   179		MOVL	m_g0(BX), SI
   180		CMPL	g(CX), SI
   181		JNE	2(PC)
   182		MOVL	0, AX
   183	
   184		// Called from f.
   185		// Set m->morebuf to f's caller.
   186		MOVL	8(SP), AX	// f's caller's PC
   187		MOVL	AX, (m_morebuf+gobuf_pc)(BX)
   188		LEAL	16(SP), AX	// f's caller's SP
   189		MOVL	AX, (m_morebuf+gobuf_sp)(BX)
   190		MOVL	AX, m_moreargp(BX)
   191		get_tls(CX)
   192		MOVL	g(CX), SI
   193		MOVL	SI, (m_morebuf+gobuf_g)(BX)
   194	
   195		// Set g->sched to context in f.
   196		MOVL	0(SP), AX // f's PC
   197		MOVL	AX, (g_sched+gobuf_pc)(SI)
   198		MOVL	SI, (g_sched+gobuf_g)(SI)
   199		LEAL	8(SP), AX // f's SP
   200		MOVL	AX, (g_sched+gobuf_sp)(SI)
   201		MOVL	DX, (g_sched+gobuf_ctxt)(SI)
   202	
   203		// Call newstack on m->g0's stack.
   204		MOVL	m_g0(BX), BX
   205		MOVL	BX, g(CX)
   206		MOVL	(g_sched+gobuf_sp)(BX), SP
   207		CALL	runtime·newstack(SB)
   208		MOVL	$0, 0x1003	// crash if newstack returns
   209		RET

   211	// Called from panic.  Mimics morestack,
   212	// reuses stack growth code to create a frame
   213	// with the desired args running the desired function.
   214	//
   215	// func call(fn *byte, arg *byte, argsize uint32).
   216	TEXT runtime·newstackcall(SB), NOSPLIT, $0-20
   217		get_tls(CX)
   218		MOVL	m(CX), BX
   219	
   220		// Save our caller's state as the PC and SP to
   221		// restore when returning from f.
   222		MOVL	0(SP), AX	// our caller's PC
   223		MOVL	AX, (m_morebuf+gobuf_pc)(BX)
   224		LEAL	8(SP), AX	// our caller's SP
   225		MOVL	AX, (m_morebuf+gobuf_sp)(BX)
   226		MOVL	g(CX), AX
   227		MOVL	AX, (m_morebuf+gobuf_g)(BX)
   228		
   229		// Save our own state as the PC and SP to restore
   230		// if this goroutine needs to be restarted.
   231		MOVL	$runtime·newstackcall(SB), DI
   232		MOVL	DI, (g_sched+gobuf_pc)(AX)
   233		MOVL	SP, (g_sched+gobuf_sp)(AX)
   234	
   235		// Set up morestack arguments to call f on a new stack.
   236		// We set f's frame size to 1, as a hint to newstack
   237		// that this is a call from runtime·newstackcall.
   238		// If it turns out that f needs a larger frame than
   239		// the default stack, f's usual stack growth prolog will
   240		// allocate a new segment (and recopy the arguments).
   241		MOVL	8(SP), AX	// fn
   242		MOVL	12(SP), DX	// arg frame
   243		MOVL	16(SP), CX	// arg size
   244	
   245		MOVQ	AX, m_cret(BX)	// f's PC
   246		MOVL	DX, m_moreargp(BX)	// argument frame pointer
   247		MOVL	CX, m_moreargsize(BX)	// f's argument size
   248		MOVL	$1, m_moreframesize(BX)	// f's frame size
   249	
   250		// Call newstack on m->g0's stack.
   251		MOVL	m_g0(BX), BX
   252		get_tls(CX)
   253		MOVL	BX, g(CX)
   254		MOVL	(g_sched+gobuf_sp)(BX), SP
   255		CALL	runtime·newstack(SB)
   256		MOVL	$0, 0x1103	// crash if newstack returns
   257		RET

   305	TEXT runtime·NAME(SB), WRAPPER, $MAXSIZE-16;		\

   484	// bool cas(int32 *val, int32 old, int32 new)
   485	// Atomically:
   486	//	if(*val == old){
   487	//		*val = new;
   488	//		return 1;
   489	//	} else
   490	//		return 0;
   491	TEXT runtime·cas(SB), NOSPLIT, $0-12
   492		MOVL	val+0(FP), BX
   493		MOVL	old+4(FP), AX
   494		MOVL	new+8(FP), CX
   495		LOCK
   496		CMPXCHGL	CX, 0(BX)
   497		JZ 3(PC)
   498		MOVL	$0, AX
   499		RET
   500		MOVL	$1, AX
   501		RET

   503	// bool	runtime·cas64(uint64 *val, uint64 old, uint64 new)
   504	// Atomically:
   505	//	if(*val == *old){
   506	//		*val = new;
   507	//		return 1;
   508	//	} else {
   509	//		return 0;
   510	//	}
   511	TEXT runtime·cas64(SB), NOSPLIT, $0-24
   512		MOVL	val+0(FP), BX
   513		MOVQ	old+8(FP), AX
   514		MOVQ	new+16(FP), CX
   515		LOCK
   516		CMPXCHGQ	CX, 0(BX)
   517		JNZ	cas64_fail
   518		MOVL	$1, AX
   519		RET

   521		MOVL	$0, AX
   522		RET

   833	
   834	// input:
   835	//   SI = a
   836	//   DI = b
   837	//   BX = alen
   838	//   DX = blen
   839	// output:
   840	//   AX = 1/0/-1
   841	TEXT runtime·cmpbody(SB),NOSPLIT,$0-0
   842		CMPQ	SI, DI
   843		JEQ	cmp_allsame
   844		CMPQ	BX, DX
   845		MOVQ	DX, R8
   846		CMOVQLT	BX, R8 // R8 = min(alen, blen) = # of bytes to compare
   847		CMPQ	R8, $8
   848		JB	cmp_small
   849	
   850	cmp_loop:
   851		CMPQ	R8, $16
   852		JBE	cmp_0through16
   853		MOVOU	(SI), X0
   854		MOVOU	(DI), X1
   855		PCMPEQB X0, X1
   856		PMOVMSKB X1, AX
   857		XORQ	$0xffff, AX	// convert EQ to NE
   858		JNE	cmp_diff16	// branch if at least one byte is not equal
   859		ADDQ	$16, SI
   860		ADDQ	$16, DI
   861		SUBQ	$16, R8
   862		JMP	cmp_loop
   863		
   864		// AX = bit mask of differences
   865	cmp_diff16:
   866		BSFQ	AX, BX	// index of first byte that differs
   867		XORQ	AX, AX
   868		ADDQ	BX, SI
   869		MOVB	(SI), CX
   870		ADDQ	BX, DI
   871		CMPB	CX, (DI)
   872		SETHI	AX
   873		LEAQ	-1(AX*2), AX	// convert 1/0 to +1/-1
   874		RET
   875	
   876		// 0 through 16 bytes left, alen>=8, blen>=8
   877	cmp_0through16:
   878		CMPQ	R8, $8
   879		JBE	cmp_0through8
   880		MOVQ	(SI), AX
   881		MOVQ	(DI), CX
   882		CMPQ	AX, CX
   883		JNE	cmp_diff8
   884	cmp_0through8:
   885		ADDQ	R8, SI
   886		ADDQ	R8, DI
   887		MOVQ	-8(SI), AX
   888		MOVQ	-8(DI), CX
   889		CMPQ	AX, CX
   890		JEQ	cmp_allsame
   891	
   892		// AX and CX contain parts of a and b that differ.
   893	cmp_diff8:
   894		BSWAPQ	AX	// reverse order of bytes
   895		BSWAPQ	CX
   896		XORQ	AX, CX
   897		BSRQ	CX, CX	// index of highest bit difference
   898		SHRQ	CX, AX	// move a's bit to bottom
   899		ANDQ	$1, AX	// mask bit
   900		LEAQ	-1(AX*2), AX // 1/0 => +1/-1
   901		RET

   903		// 0-7 bytes in common

   905		LEAQ	(R8*8), CX	// bytes left -> bits left
   906		NEGQ	CX		//  - bits lift (== 64 - bits left mod 64)
   907		JEQ	cmp_allsame
   908	
   909		// load bytes of a into high bytes of AX
   910		CMPB	SI, $0xf8

   912		MOVQ	(SI), SI
   913		JMP	cmp_si_finish

   969	//   SI: data
   970	//   BX: data len
   971	//   AL: byte sought
   972	// output:
   973	//   AX
   974	TEXT runtime·indexbytebody(SB),NOSPLIT,$0
   975		MOVL SI, DI
   976	
   977		CMPL BX, $16
   978		JLT indexbyte_small
   979	
   980		// round up to first 16-byte boundary
   981		TESTL $15, SI
   982		JZ aligned
   983		MOVL SI, CX
   984		ANDL $~15, CX
   985		ADDL $16, CX
   986	
   987		// search the beginning
   988		SUBL SI, CX
   989		REPN; SCASB
   990		JZ success
   991	
   992	// DI is 16-byte aligned; get ready to search using SSE instructions
   993	aligned:
   994		// round down to last 16-byte boundary
   995		MOVL BX, R11
   996		ADDL SI, R11
   997		ANDL $~15, R11
   998	
   999		// shuffle X0 around so that each byte contains c
  1000		MOVD AX, X0
  1001		PUNPCKLBW X0, X0
  1002		PUNPCKLBW X0, X0
  1003		PSHUFL $0, X0, X0
  1004		JMP condition
  1005	
  1006	sse:
  1007		// move the next 16-byte chunk of the buffer into X1
  1008		MOVO (DI), X1
  1009		// compare bytes in X0 to X1
  1010		PCMPEQB X0, X1
  1011		// take the top bit of each byte in X1 and put the result in DX
  1012		PMOVMSKB X1, DX
  1013		TESTL DX, DX
  1014		JNZ ssesuccess
  1015		ADDL $16, DI
  1016	
  1017	condition:
  1018		CMPL DI, R11
  1019		JLT sse
  1020	
  1021		// search the end

  1023		ADDL BX, CX

  1026		// returning a false success

  1028		REPN; SCASB
  1029		JZ success
