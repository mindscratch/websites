     5	// Lock-free stack.
     6	
     7	package runtime
     8	#include "runtime.h"
     9	#include "arch_GOARCH.h"
    10	
    11	#ifdef _64BIT
    12	// Amd64 uses 48-bit virtual addresses, 47-th bit is used as kernel/user flag.
    13	// So we use 17msb of pointers as ABA counter.
    14	# define PTR_BITS 47
    15	#else
    16	# define PTR_BITS 32
    17	#endif
    18	#define PTR_MASK ((1ull<<PTR_BITS)-1)
    19	#define CNT_MASK (0ull-1)
    20	
    21	#ifdef _64BIT
    22	#ifdef GOOS_solaris
    23	// SPARC64 and Solaris on AMD64 uses all 64 bits of virtual addresses.
    24	// Use low-order three bits as ABA counter.
    25	// http://docs.oracle.com/cd/E19120-01/open.solaris/816-5138/6mba6ua5p/index.html
    26	#undef PTR_BITS
    27	#undef CNT_MASK
    28	#undef PTR_MASK
    29	#define PTR_BITS 0
    30	#define CNT_MASK 7
    31	#define PTR_MASK ((0ull-1)<<3)
    32	#endif
    33	#endif
    34	
    35	void
    36	runtime·lfstackpush(uint64 *head, LFNode *node)
    37	{
    38		uint64 old, new;
    39	
    40		if((uintptr)node != ((uintptr)node&PTR_MASK)) {
    41			runtime·printf("p=%p\n", node);
    42			runtime·throw("runtime·lfstackpush: invalid pointer");
    43		}
    44	
    45		node->pushcnt++;
    46		new = (uint64)(uintptr)node|(((uint64)node->pushcnt&CNT_MASK)<<PTR_BITS);
    47		for(;;) {
    48			old = runtime·atomicload64(head);
    49			node->next = (LFNode*)(uintptr)(old&PTR_MASK);
    50			if(runtime·cas64(head, old, new))
    51				break;
    52		}
    53	}
    54	
    55	LFNode*
    56	runtime·lfstackpop(uint64 *head)
    57	{
    58		LFNode *node, *node2;
    59		uint64 old, new;
    60	
    61		for(;;) {
    62			old = runtime·atomicload64(head);
    63			if(old == 0)
    64				return nil;
    65			node = (LFNode*)(uintptr)(old&PTR_MASK);
    66			node2 = runtime·atomicloadp(&node->next);
    67			new = 0;
    68			if(node2 != nil)
    69				new = (uint64)(uintptr)node2|(((uint64)node2->pushcnt&CNT_MASK)<<PTR_BITS);
    70			if(runtime·cas64(head, old, new))
    71				return node;
    72		}
    73	}
    74	
    75	func lfstackpush_go(head *uint64, node *LFNode) {
    76		runtime·lfstackpush(head, node);
    77	}
    78	
    79	func lfstackpop_go(head *uint64) (node *LFNode) {
    80		node = runtime·lfstackpop(head);
    81	}
