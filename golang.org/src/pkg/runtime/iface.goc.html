    24	static Itab*
    25	itab(InterfaceType *inter, Type *type, int32 canfail)
    26	{
    27		int32 locked;
    28		int32 ni;
    29		Method *t, *et;
    30		IMethod *i, *ei;
    31		uint32 h;
    32		String *iname, *ipkgPath;
    33		Itab *m;
    34		UncommonType *x;
    35		Type *itype;
    36		Eface err;
    37	
    38		if(inter->mhdr.len == 0)
    39			runtime·throw("internal error - misuse of itab");
    40	
    41		locked = 0;
    42	
    43		// easy case
    44		x = type->x;
    45		if(x == nil) {
    46			if(canfail)
    47				return nil;
    48			iname = inter->m[0].name;
    49			goto throw;
    50		}
    51	
    52		// compiler has provided some good hash codes for us.
    53		h = inter->hash;
    54		h += 17 * type->hash;
    55		// TODO(rsc): h += 23 * x->mhash ?
    56		h %= nelem(hash);
    57	
    58		// look twice - once without lock, once with.
    59		// common case will be no lock contention.
    60		for(locked=0; locked<2; locked++) {
    61			if(locked)
    62				runtime·lock(&ifacelock);
    63			for(m=runtime·atomicloadp(&hash[h]); m!=nil; m=m->link) {
    64				if(m->inter == inter && m->type == type) {
    65					if(m->bad) {
    66						m = nil;
    67						if(!canfail) {
    68							// this can only happen if the conversion
    69							// was already done once using the , ok form
    70							// and we have a cached negative result.
    71							// the cached result doesn't record which
    72							// interface function was missing, so jump
    73							// down to the interface check, which will
    74							// do more work but give a better error.
    75							goto search;
    76						}
    77					}
    78					if(locked)
    79						runtime·unlock(&ifacelock);
    80					return m;
    81				}
    82			}
    83		}
    84	
    85		ni = inter->mhdr.len;
    86		m = runtime·persistentalloc(sizeof(*m) + ni*sizeof m->fun[0], 0, &mstats.other_sys);
    87		m->inter = inter;
    88		m->type = type;
    89	
    90	search:
    91		// both inter and type have method sorted by name,
    92		// and interface names are unique,
    93		// so can iterate over both in lock step;
    94		// the loop is O(ni+nt) not O(ni*nt).
    95		i = inter->m;
    96		ei = i + inter->mhdr.len;
    97		t = x->m;
    98		et = t + x->mhdr.len;
    99		for(; i < ei; i++) {
   100			itype = i->type;
   101			iname = i->name;
   102			ipkgPath = i->pkgPath;
   103			for(;; t++) {
   104				if(t >= et) {
   105					if(!canfail) {
   106					throw:
   107						// didn't find method
   108						runtime·newTypeAssertionError(
   109							nil, type->string, inter->string,
   110							iname, &err);
   111						if(locked)
   112							runtime·unlock(&ifacelock);
   113						runtime·panic(err);
   114						return nil;	// not reached
   115					}
   116					m->bad = 1;
   117					goto out;
   118				}
   119				if(t->mtyp == itype && t->name == iname && t->pkgPath == ipkgPath)
   120					break;
   121			}
   122			if(m)
   123				m->fun[i - inter->m] = t->ifn;
   124		}
   125	
   126	out:
   127		if(!locked)
   128			runtime·panicstring("invalid itab locking");
   129		m->link = hash[h];
   130		runtime·atomicstorep(&hash[h], m);
   131		runtime·unlock(&ifacelock);
   132		if(m->bad)
   133			return nil;
   134		return m;
   135	}

   137	// call the callback for every itab that is currently allocated.
   138	void
   139	runtime·iterate_itabs(void (*callback)(Itab*))
   140	{
   141		int32 i;
   142		Itab *tab;
   143	
   144		for(i = 0; i < nelem(hash); i++) {
   145			for(tab = hash[i]; tab != nil; tab = tab->link) {
   146				callback(tab);
   147			}
   148		}
   149	}

   151	static void
   152	copyin(Type *t, void *src, void **dst)
   153	{
   154		uintptr size;
   155		void *p;
   156		Alg *alg;
   157	
   158		size = t->size;
   159		alg = t->alg;
   160	
   161		if(size <= sizeof(*dst))
   162			alg->copy(size, dst, src);
   163		else {
   164			p = runtime·cnew(t);
   165			alg->copy(size, p, src);
   166			*dst = p;
   167		}
   168	}

   170	static void
   171	copyout(Type *t, void **src, void *dst)
   172	{
   173		uintptr size;
   174		Alg *alg;
   175	
   176		size = t->size;
   177		alg = t->alg;
   178	
   179		if(size <= sizeof(*src))
   180			alg->copy(size, dst, src);
   181		else
   182			alg->copy(size, dst, *src);
   183	}

   212	/*
   213	 * NOTE: Cannot use 'func' here, because we have to declare
   214	 * a return value, the only types we have are at least 1 byte large,
   215	 * goc2c will zero the return value, and the actual return value
   216	 * might have size 0 bytes, in which case the zeroing of the
   217	 * 1 or more bytes would be wrong.

   227	static void
   228	assertI2Tret(Type *t, Iface i, byte *ret)
   229	{
   230		Itab *tab;
   231		Eface err;
   232	
   233		tab = i.tab;
   234		if(tab == nil) {
   235			runtime·newTypeAssertionError(
   236				nil, nil, t->string,
   237				nil, &err);
   238			runtime·panic(err);
   239		}
   240		if(tab->type != t) {
   241			runtime·newTypeAssertionError(
   242				tab->inter->string, tab->type->string, t->string,
   243				nil, &err);
   244			runtime·panic(err);
   245		}
   246		copyout(t, &i.data, ret);
   247	}

   250	func assertI2T2(t *Type, i Iface) (ret byte, ...) {
   251		bool *ok;
   252		int32 wid;
   253	
   254		wid = t->size;
   255		ok = (bool*)(&ret + wid);
   256	
   257		if(i.tab == nil || i.tab->type != t) {
   258			*ok = false;

   260			return;
   261		}
   262	
   263		*ok = true;
   264		copyout(t, &i.data, &ret);

   486	static uintptr
   487	ifacehash1(void *data, Type *t, uintptr h)
   488	{
   489		Alg *alg;
   490		uintptr size;
   491		Eface err;
   492	
   493		if(t == nil)
   494			return 0;
   495	
   496		alg = t->alg;
   497		size = t->size;
   498		if(alg->hash == runtime·nohash) {
   499			// calling nohash will panic too,
   500			// but we can print a better error.
   501			runtime·newErrorString(runtime·catstring(runtime·gostringnocopy((byte*)"hash of unhashable type "), *t->string), &err);
   502			runtime·panic(err);
   503		}
   504		if(size <= sizeof(data))
   505			alg->hash(&h, size, &data);
   506		else
   507			alg->hash(&h, size, data);
   508		return h;
   509	}

   525	static bool
   526	ifaceeq1(void *data1, void *data2, Type *t)
   527	{
   528		uintptr size;
   529		Alg *alg;
   530		Eface err;
   531		bool eq;
   532	
   533		alg = t->alg;
   534		size = t->size;
   535	
   536		if(alg->equal == runtime·noequal) {
   537			// calling noequal will panic too,
   538			// but we can print a better error.
   539			runtime·newErrorString(runtime·catstring(runtime·gostringnocopy((byte*)"comparing uncomparable type "), *t->string), &err);
   540			runtime·panic(err);
   541		}
   542	
   543		eq = 0;
   544		if(size <= sizeof(data1))
   545			alg->equal(&eq, size, &data1, &data2);
   546		else
   547			alg->equal(&eq, size, data1, data2);
   548		return eq;
   549	}
