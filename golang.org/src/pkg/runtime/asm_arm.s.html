   171	// Called during function prolog when more stack is needed.
   172	// R1 frame size
   173	// R2 arg size
   174	// R3 prolog's LR
   175	// NB. we do not save R0 because we've forced 5c to pass all arguments
   176	// on the stack.
   177	// using frame size $-4 means do not save LR on stack.
   178	//
   179	// The traceback routines see morestack on a g0 as being
   180	// the top of a stack (for example, morestack calling newstack
   181	// calling the scheduler calling newm calling gc), so we must
   182	// record an argument size. For that purpose, it has no arguments.
   183	TEXT runtime·morestack(SB),NOSPLIT,$-4-0

   185		MOVW	m_g0(m), R4

   220	// Called from panic.  Mimics morestack,
   221	// reuses stack growth code to create a frame
   222	// with the desired args running the desired function.
   223	//
   224	// func call(fn *byte, arg *byte, argsize uint32).
   225	TEXT runtime·newstackcall(SB), NOSPLIT, $-4-12
   226		// Save our caller's state as the PC and SP to
   227		// restore when returning from f.
   228		MOVW	LR, (m_morebuf+gobuf_pc)(m)	// our caller's PC
   229		MOVW	SP, (m_morebuf+gobuf_sp)(m)	// our caller's SP
   230		MOVW	g,  (m_morebuf+gobuf_g)(m)
   231	
   232		// Save our own state as the PC and SP to restore
   233		// if this goroutine needs to be restarted.
   234		MOVW	$runtime·newstackcall(SB), R11
   235		MOVW	R11, (g_sched+gobuf_pc)(g)
   236		MOVW	LR, (g_sched+gobuf_lr)(g)
   237		MOVW	SP, (g_sched+gobuf_sp)(g)

   239		// Set up morestack arguments to call f on a new stack.
   240		// We set f's frame size to 1, as a hint to newstack
   241		// that this is a call from runtime·newstackcall.
   242		// If it turns out that f needs a larger frame than
   243		// the default stack, f's usual stack growth prolog will
   244		// allocate a new segment (and recopy the arguments).
   245		MOVW	4(SP), R0			// fn
   246		MOVW	8(SP), R1			// arg frame
   247		MOVW	12(SP), R2			// arg size

   249		MOVW	R0, m_cret(m)			// f's PC
   250		MOVW	R1, m_moreargp(m)		// f's argument pointer
   251		MOVW	R2, m_moreargsize(m)		// f's argument size
   252		MOVW	$1, R3
   253		MOVW	R3, m_moreframesize(m)		// f's frame size

   257		MOVW	(g_sched+gobuf_sp)(g), SP

   393	// called from deferreturn.
   394	// 1. grab stored LR for caller

   420	// asmcgocall(void(*fn)(void*), void *arg)
   421	// Call fn(arg) on the scheduler stack,
   422	// aligned appropriately for the gcc ABI.
   423	// See cgocall.c for more details.
   424	TEXT	runtime·asmcgocall(SB),NOSPLIT,$0-8
   425		MOVW	fn+0(FP), R1
   426		MOVW	arg+4(FP), R0
   427		MOVW	R13, R2
   428		MOVW	g, R5
   429	
   430		// Figure out if we need to switch to m->g0 stack.
   431		// We get called to create new OS threads too, and those
   432		// come in on the m->g0 stack already.
   433		MOVW	m_g0(m), R3
   434		CMP	R3, g
   435		BEQ	4(PC)
   436		BL	gosave<>(SB)
   437		MOVW	R3, g
   438		MOVW	(g_sched+gobuf_sp)(g), R13
   439	
   440		// Now on a scheduling stack (a pthread-created stack).
   441		SUB	$24, R13
   442		BIC	$0x7, R13	// alignment for gcc ABI
   443		MOVW	R5, 20(R13) // save old g
   444		MOVW	R2, 16(R13)	// save old SP
   445		// R0 already contains the first argument
   446		BL	(R1)
   447	
   448		// Restore registers, g, stack pointer.
   449		MOVW	20(R13), g
   450		MOVW	16(R13), R13
   451		RET

   469	TEXT	runtime·cgocallback_gofunc(SB),NOSPLIT,$8-12
   470		// Load m and g from thread-local storage.
   471		MOVB	runtime·iscgo(SB), R0
   472		CMP	$0, R0
   473		BL.NE	runtime·load_gm(SB)

   475		// If m is nil, Go did not create the current thread.
   476		// Call needm to obtain one for temporary use.
   477		// In this case, we're running on the thread stack, so there's
   478		// lots of space, but the linker doesn't know. Hide the call from
   479		// the linker analysis by using an indirect call.
   480		MOVW	m, savedm-4(SP)
   481		CMP	$0, m
   482		B.NE	havem
   483		MOVW	$runtime·needm(SB), R0
   484		BL	(R0)

   486	havem:
   487		// Now there's a valid m, and we're running on its m->g0.
   488		// Save current m->g0->sched.sp on stack and then set it to SP.
   489		// Save current sp in m->g0->sched.sp in preparation for
   490		// switch back to m->curg stack.
   491		// NOTE: unwindm knows that the saved g->sched.sp is at 4(R13) aka savedsp-8(SP).
   492		MOVW	m_g0(m), R3
   493		MOVW	(g_sched+gobuf_sp)(R3), R4
   494		MOVW	R4, savedsp-8(SP)
   495		MOVW	R13, (g_sched+gobuf_sp)(R3)
   496	
   497		// Switch to m->curg stack and call runtime.cgocallbackg.
   498		// Because we are taking over the execution of m->curg
   499		// but *not* resuming what had been running, we need to
   500		// save that information (m->curg->sched) so we can restore it.
   501		// We can restore m->curg->sched.sp easily, because calling
   502		// runtime.cgocallbackg leaves SP unchanged upon return.
   503		// To save m->curg->sched.pc, we push it onto the stack.
   504		// This has the added benefit that it looks to the traceback
   505		// routine like cgocallbackg is going to return to that
   506		// PC (because the frame we allocate below has the same
   507		// size as cgocallback_gofunc's frame declared above)
   508		// so that the traceback will seamlessly trace back into
   509		// the earlier calls.
   510		//
   511		// In the new goroutine, -8(SP) and -4(SP) are unused.
   512		MOVW	fn+4(FP), R0
   513		MOVW	frame+8(FP), R1
   514		MOVW	framesize+12(FP), R2
   515		MOVW	m_curg(m), g
   516		MOVW	(g_sched+gobuf_sp)(g), R4 // prepare stack as R4
   517		MOVW	(g_sched+gobuf_pc)(g), R5
   518		MOVW	R5, -12(R4)
   519		MOVW	$-12(R4), R13
   520		BL	runtime·cgocallbackg(SB)
   521	
   522		// Restore g->sched (== m->curg->sched) from saved values.
   523		MOVW	0(R13), R5
   524		MOVW	R5, (g_sched+gobuf_pc)(g)
   525		MOVW	$12(R13), R4
   526		MOVW	R4, (g_sched+gobuf_sp)(g)
   527	
   528		// Switch back to m->g0's stack and restore m->g0->sched.sp.
   529		// (Unlike m->curg, the g0 goroutine never uses sched.pc,
   530		// so we do not have to restore it.)
   531		MOVW	m_g0(m), g
   532		MOVW	(g_sched+gobuf_sp)(g), R13
   533		MOVW	savedsp-8(SP), R4
   534		MOVW	R4, (g_sched+gobuf_sp)(g)
   535	
   536		// If the m on entry was nil, we called needm above to borrow an m
   537		// for the duration of the call. Since the call is over, return it with dropm.
   538		MOVW	savedm-4(SP), R6
   539		CMP	$0, R6
   540		B.NE	3(PC)
   541		MOVW	$runtime·dropm(SB), R0
   542		BL	(R0)
   543	
   544		// Done!
   545		RET
