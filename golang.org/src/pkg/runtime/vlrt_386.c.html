    68	void
    69	_d2v(Vlong *y, double d)
    70	{
    71		union { double d; struct Vlong; } x;
    72		ulong xhi, xlo, ylo, yhi;
    73		int sh;
    74	
    75		x.d = d;
    76	
    77		xhi = (x.hi & 0xfffff) | 0x100000;
    78		xlo = x.lo;
    79		sh = 1075 - ((x.hi >> 20) & 0x7ff);
    80	
    81		ylo = 0;
    82		yhi = 0;
    83		if(sh >= 0) {
    84			/* v = (hi||lo) >> sh */
    85			if(sh < 32) {
    86				if(sh == 0) {
    87					ylo = xlo;
    88					yhi = xhi;
    89				} else {
    90					ylo = (xlo >> sh) | (xhi << (32-sh));
    91					yhi = xhi >> sh;
    92				}
    93			} else {
    94				if(sh == 32) {
    95					ylo = xhi;
    96				} else
    97				if(sh < 64) {
    98					ylo = xhi >> (sh-32);
    99				}
   100			}
   101		} else {
   102			/* v = (hi||lo) << -sh */
   103			sh = -sh;
   104			if(sh <= 10) {
   105				ylo = xlo << sh;
   106				yhi = (xhi << sh) | (xlo >> (32-sh));
   107			} else {
   108				/* overflow */
   109				yhi = d;	/* causes something awful */
   110			}
   111		}
   112		if(x.hi & SIGN(32)) {
   113			if(ylo != 0) {
   114				ylo = -ylo;
   115				yhi = ~yhi;
   116			} else
   117				yhi = -yhi;
   118		}
   119	
   120		y->hi = yhi;
   121		y->lo = ylo;
   122	}

   131	double
   132	_v2d(Vlong x)
   133	{
   134		if(x.hi & SIGN(32)) {
   135			if(x.lo) {
   136				x.lo = -x.lo;
   137				x.hi = ~x.hi;
   138			} else
   139				x.hi = -x.hi;
   140			return -((long)x.hi*4294967296. + x.lo);
   141		}
   142		return (long)x.hi*4294967296. + x.lo;
   143	}

   154	static void
   155	slowdodiv(Vlong num, Vlong den, Vlong *q, Vlong *r)
   156	{
   157		ulong numlo, numhi, denhi, denlo, quohi, quolo, t;
   158		int i;
   159	
   160		numhi = num.hi;
   161		numlo = num.lo;
   162		denhi = den.hi;
   163		denlo = den.lo;
   164	
   165		/*
   166		 * get a divide by zero
   167		 */
   168		if(denlo==0 && denhi==0) {
   169			numlo = numlo / denlo;
   170		}
   171	
   172		/*
   173		 * set up the divisor and find the number of iterations needed
   174		 */
   175		if(numhi >= SIGN(32)) {
   176			quohi = SIGN(32);
   177			quolo = 0;
   178		} else {
   179			quohi = numhi;
   180			quolo = numlo;
   181		}
   182		i = 0;
   183		while(denhi < quohi || (denhi == quohi && denlo < quolo)) {
   184			denhi = (denhi<<1) | (denlo>>31);
   185			denlo <<= 1;
   186			i++;
   187		}
   188	
   189		quohi = 0;
   190		quolo = 0;
   191		for(; i >= 0; i--) {
   192			quohi = (quohi<<1) | (quolo>>31);
   193			quolo <<= 1;
   194			if(numhi > denhi || (numhi == denhi && numlo >= denlo)) {
   195				t = numlo;
   196				numlo -= denlo;
   197				if(numlo > t)
   198					numhi--;
   199				numhi -= denhi;
   200				quolo |= 1;
   201			}
   202			denlo = (denlo>>1) | (denhi<<31);
   203			denhi >>= 1;
   204		}
   205	
   206		if(q) {
   207			q->lo = quolo;
   208			q->hi = quohi;
   209		}
   210		if(r) {
   211			r->lo = numlo;
   212			r->hi = numhi;
   213		}
   214	}

   216	static void
   217	dodiv(Vlong num, Vlong den, Vlong *qp, Vlong *rp)
   218	{
   219		ulong n;
   220		Vlong x, q, r;
   221	
   222		if(den.hi > num.hi || (den.hi == num.hi && den.lo > num.lo)){
   223			if(qp) {
   224				qp->hi = 0;
   225				qp->lo = 0;
   226			}
   227			if(rp) {
   228				rp->hi = num.hi;
   229				rp->lo = num.lo;
   230			}
   231			return;
   232		}
   233	
   234		if(den.hi != 0){
   235			q.hi = 0;
   236			n = num.hi/den.hi;
   237			if(_mul64by32(&x, den, n) || x.hi > num.hi || (x.hi == num.hi && x.lo > num.lo))
   238				slowdodiv(num, den, &q, &r);
   239			else {
   240				q.lo = n;
   241				r.v = num.v - x.v;
   242			}
   243		} else {
   244			if(num.hi >= den.lo){
   245				if(den.lo == 0)
   246					runtime·panicdivide();
   247				q.hi = n = num.hi/den.lo;
   248				num.hi -= den.lo*n;
   249			} else {
   250				q.hi = 0;
   251			}
   252			q.lo = _div64by32(num, den.lo, &r.lo);
   253			r.hi = 0;
   254		}
   255		if(qp) {
   256			qp->lo = q.lo;
   257			qp->hi = q.hi;
   258		}
   259		if(rp) {
   260			rp->lo = r.lo;
   261			rp->hi = r.hi;
   262		}
   263	}

   265	void
   266	_divvu(Vlong *q, Vlong n, Vlong d)
   267	{
   268	
   269		if(n.hi == 0 && d.hi == 0) {
   270			if(d.lo == 0)
   271				runtime·panicdivide();
   272			q->hi = 0;
   273			q->lo = n.lo / d.lo;
   274			return;
   275		}
   276		dodiv(n, d, q, 0);
   277	}

   305	static void
   306	vneg(Vlong *v)
   307	{
   308	
   309		if(v->lo == 0) {
   310			v->hi = -v->hi;
   311			return;
   312		}
   313		v->lo = -v->lo;
   314		v->hi = ~v->hi;
   315	}

   317	void
   318	_divv(Vlong *q, Vlong n, Vlong d)
   319	{
   320		long nneg, dneg;
   321	
   322		if(n.hi == (((long)n.lo)>>31) && d.hi == (((long)d.lo)>>31)) {
   323			if((long)n.lo == -0x80000000 && (long)d.lo == -1) {
   324				// special case: 32-bit -0x80000000 / -1 causes divide error,
   325				// but it's okay in this 64-bit context.
   326				q->lo = 0x80000000;
   327				q->hi = 0;
   328				return;
   329			}
   330			if(d.lo == 0)
   331				runtime·panicdivide();
   332			q->lo = (long)n.lo / (long)d.lo;
   333			q->hi = ((long)q->lo) >> 31;
   334			return;
   335		}
   336		nneg = n.hi >> 31;
   337		if(nneg)
   338			vneg(&n);
   339		dneg = d.hi >> 31;
   340		if(dneg)
   341			vneg(&d);
   342		dodiv(n, d, q, 0);
   343		if(nneg != dneg)
   344			vneg(q);
   345	}

   353	void
   354	_modv(Vlong *r, Vlong n, Vlong d)
   355	{
   356		long nneg, dneg;
   357	
   358		if(n.hi == (((long)n.lo)>>31) && d.hi == (((long)d.lo)>>31)) {
   359			if((long)n.lo == -0x80000000 && (long)d.lo == -1) {
   360				// special case: 32-bit -0x80000000 % -1 causes divide error,
   361				// but it's okay in this 64-bit context.
   362				r->lo = 0;
   363				r->hi = 0;
   364				return;
   365			}
   366			if(d.lo == 0)
   367				runtime·panicdivide();
   368			r->lo = (long)n.lo % (long)d.lo;
   369			r->hi = ((long)r->lo) >> 31;
   370			return;
   371		}
   372		nneg = n.hi >> 31;
   373		if(nneg)
   374			vneg(&n);
   375		dneg = d.hi >> 31;
   376		if(dneg)
   377			vneg(&d);
   378		dodiv(n, d, 0, r);
   379		if(nneg)
   380			vneg(r);
   381	}

   389	void
   390	_rshav(Vlong *r, Vlong a, int b)
   391	{
   392		long t;
   393	
   394		t = a.hi;
   395		if(b >= 32) {
   396			r->hi = t>>31;
   397			if(b >= 64) {
   398				/* this is illegal re C standard */
   399				r->lo = t>>31;
   400				return;
   401			}
   402			r->lo = t >> (b-32);
   403			return;
   404		}
   405		if(b <= 0) {
   406			r->hi = t;
   407			r->lo = a.lo;
   408			return;
   409		}
   410		r->hi = t >> b;
   411		r->lo = (t << (32-b)) | (a.lo >> b);
   412	}

   530	void
   531	_vasop(Vlong *ret, void *lv, void fn(Vlong*, Vlong, Vlong), int type, Vlong rv)
   532	{
   533		Vlong t, u;
   534	
   535		u.lo = 0;
   536		u.hi = 0;
   537		switch(type) {
   538		default:
   539			runtime·abort();
   540			break;
   541	
   542		case 1:	/* schar */
   543			t.lo = *(schar*)lv;
   544			t.hi = t.lo >> 31;
   545			fn(&u, t, rv);
   546			*(schar*)lv = u.lo;
   547			break;
   548	
   549		case 2:	/* uchar */
   550			t.lo = *(uchar*)lv;
   551			t.hi = 0;
   552			fn(&u, t, rv);
   553			*(uchar*)lv = u.lo;
   554			break;
   555	
   556		case 3:	/* short */
   557			t.lo = *(short*)lv;
   558			t.hi = t.lo >> 31;
   559			fn(&u, t, rv);
   560			*(short*)lv = u.lo;
   561			break;
   562	
   563		case 4:	/* ushort */
   564			t.lo = *(ushort*)lv;
   565			t.hi = 0;
   566			fn(&u, t, rv);
   567			*(ushort*)lv = u.lo;
   568			break;
   569	
   570		case 9:	/* int */
   571			t.lo = *(int*)lv;
   572			t.hi = t.lo >> 31;
   573			fn(&u, t, rv);
   574			*(int*)lv = u.lo;
   575			break;
   576	
   577		case 10:	/* uint */
   578			t.lo = *(uint*)lv;
   579			t.hi = 0;
   580			fn(&u, t, rv);
   581			*(uint*)lv = u.lo;
   582			break;
   583	
   584		case 5:	/* long */
   585			t.lo = *(long*)lv;
   586			t.hi = t.lo >> 31;
   587			fn(&u, t, rv);
   588			*(long*)lv = u.lo;
   589			break;
   590	
   591		case 6:	/* ulong */
   592			t.lo = *(ulong*)lv;
   593			t.hi = 0;
   594			fn(&u, t, rv);
   595			*(ulong*)lv = u.lo;
   596			break;
   597	
   598		case 7:	/* vlong */
   599		case 8:	/* uvlong */
   600			fn(&u, *(Vlong*)lv, rv);
   601			*(Vlong*)lv = u;
   602			break;
   603		}
   604		*ret = u;
   605	}
