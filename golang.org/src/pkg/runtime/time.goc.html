    32	// runtimeNano returns the current value of the runtime clock in nanoseconds.
    33	func runtimeNano() (ns int64) {

    35	}

    39		runtime·tsleep(ns, "sleep");

    42	// startTimer adds t to the timer heap.
    43	func startTimer(t *Timer) {
    44		if(raceenabled)
    45			runtime·racerelease(t);
    46		runtime·addtimer(t);
    47	}
    48	
    49	// stopTimer removes t from the timer heap if it is there.
    50	// It returns true if t was removed, false if t wasn't even there.
    51	func stopTimer(t *Timer) (stopped bool) {
    52		stopped = runtime·deltimer(t);
    53	}

    80	static FuncVal readyv = {(void(*)(void))ready};

    82	// Put the current goroutine to sleep for ns nanoseconds.
    83	void
    84	runtime·tsleep(int64 ns, int8 *reason)
    85	{
    86		Timer t;
    87	
    88		if(ns <= 0)
    89			return;
    90	
    91		t.when = runtime·nanotime() + ns;
    92		t.period = 0;
    93		t.fv = &readyv;
    94		t.arg.data = g;
    95		runtime·lock(&timers);
    96		addtimer(&t);
    97		runtime·parkunlock(&timers, reason);
    98	}

   103	runtime·addtimer(Timer *t)
   104	{
   105		runtime·lock(&timers);
   106		addtimer(t);
   107		runtime·unlock(&timers);
   108	}

   110	// Add a timer to the heap and start or kick the timer proc
   111	// if the new timer is earlier than any of the others.
   112	static void
   113	addtimer(Timer *t)
   114	{
   115		int32 n;
   116		Timer **nt;
   117	
   118		// when must never be negative; otherwise timerproc will overflow
   119		// during its delta calculation and never expire other timers.
   120		if(t->when < 0)
   121			t->when = (1LL<<63)-1;
   122	
   123		if(timers.len >= timers.cap) {
   124			// Grow slice.
   125			n = 16;
   126			if(n <= timers.cap)
   127				n = timers.cap*3 / 2;
   128			nt = runtime·malloc(n*sizeof nt[0]);
   129			runtime·memmove(nt, timers.t, timers.len*sizeof nt[0]);
   130			runtime·free(timers.t);
   131			timers.t = nt;
   132			timers.cap = n;
   133		}
   134		t->i = timers.len++;
   135		timers.t[t->i] = t;
   136		siftup(t->i);
   137		if(t->i == 0) {
   138			// siftup moved to top: new earliest deadline.
   139			if(timers.sleeping) {
   140				timers.sleeping = false;
   141				runtime·notewakeup(&timers.waitnote);
   142			}
   143			if(timers.rescheduling) {
   144				timers.rescheduling = false;
   145				runtime·ready(timers.timerproc);
   146			}
   147		}
   148		if(timers.timerproc == nil) {
   149			timers.timerproc = runtime·newproc1(&timerprocv, nil, 0, 0, addtimer);
   150			timers.timerproc->issystem = true;
   151		}
   152		if(debug)
   153			dumptimers("addtimer");
   154	}

   156	// Delete timer t from the heap.
   157	// Do not need to update the timerproc:
   158	// if it wakes up early, no big deal.
   159	bool
   160	runtime·deltimer(Timer *t)
   161	{
   162		int32 i;
   163	
   164		// Dereference t so that any panic happens before the lock is held.
   165		// Discard result, because t might be moving in the heap.
   166		i = t->i;
   167		USED(i);
   168	
   169		runtime·lock(&timers);
   170	
   171		// t may not be registered anymore and may have
   172		// a bogus i (typically 0, if generated by Go).
   173		// Verify it before proceeding.
   174		i = t->i;
   175		if(i < 0 || i >= timers.len || timers.t[i] != t) {
   176			runtime·unlock(&timers);
   177			return false;
   178		}
   179	
   180		timers.len--;
   181		if(i == timers.len) {
   182			timers.t[i] = nil;
   183		} else {
   184			timers.t[i] = timers.t[timers.len];
   185			timers.t[timers.len] = nil;
   186			timers.t[i]->i = i;
   187			siftup(i);
   188			siftdown(i);
   189		}
   190		if(debug)
   191			dumptimers("deltimer");
   192		runtime·unlock(&timers);
   193		return true;
   194	}

   196	// Timerproc runs the time-driven events.
   197	// It sleeps until the next event in the timers heap.
   198	// If addtimer inserts a new earlier event, addtimer
   199	// wakes timerproc early.
   200	static void
   201	timerproc(void)
   202	{
   203		int64 delta, now;
   204		Timer *t;
   205		void (*f)(int64, Eface);
   206		Eface arg;
   207	
   208		for(;;) {
   209			runtime·lock(&timers);
   210			timers.sleeping = false;
   211			now = runtime·nanotime();
   212			for(;;) {
   213				if(timers.len == 0) {
   214					delta = -1;
   215					break;
   216				}
   217				t = timers.t[0];
   218				delta = t->when - now;
   219				if(delta > 0)
   220					break;
   221				if(t->period > 0) {
   222					// leave in heap but adjust next time to fire
   223					t->when += t->period * (1 + -delta/t->period);
   224					siftdown(0);
   225				} else {
   226					// remove from heap
   227					timers.t[0] = timers.t[--timers.len];
   228					timers.t[0]->i = 0;
   229					siftdown(0);
   230					t->i = -1;  // mark as removed
   231				}
   232				f = (void*)t->fv->fn;
   233				arg = t->arg;
   234				runtime·unlock(&timers);
   235				if(raceenabled)
   236					runtime·raceacquire(t);
   237				f(now, arg);
   238	
   239				// clear f and arg to avoid leak while sleeping for next timer
   240				f = nil;
   241				USED(f);
   242				arg.type = nil;
   243				arg.data = nil;
   244				USED(&arg);
   245	
   246				runtime·lock(&timers);
   247			}
   248			if(delta < 0) {
   249				// No timers left - put goroutine to sleep.
   250				timers.rescheduling = true;
   251				g->isbackground = true;
   252				runtime·parkunlock(&timers, "timer goroutine (idle)");
   253				g->isbackground = false;
   254				continue;
   255			}
   256			// At least one timer pending.  Sleep until then.
   257			timers.sleeping = true;
   258			runtime·noteclear(&timers.waitnote);
   259			runtime·unlock(&timers);
   260			runtime·notetsleepg(&timers.waitnote, delta);
   261		}
   262	}

   264	// heap maintenance algorithms.
   265	
   266	static void
   267	siftup(int32 i)
   268	{
   269		int32 p;
   270		int64 when;
   271		Timer **t, *tmp;
   272	
   273		t = timers.t;
   274		when = t[i]->when;
   275		tmp = t[i];
   276		while(i > 0) {
   277			p = (i-1)/4;  // parent
   278			if(when >= t[p]->when)
   279				break;
   280			t[i] = t[p];
   281			t[i]->i = i;
   282			t[p] = tmp;
   283			tmp->i = p;
   284			i = p;
   285		}
   286	}
   287	
   288	static void
   289	siftdown(int32 i)
   290	{
   291		int32 c, c3, len;
   292		int64 when, w, w3;
   293		Timer **t, *tmp;
   294	
   295		t = timers.t;
   296		len = timers.len;
   297		when = t[i]->when;
   298		tmp = t[i];
   299		for(;;) {
   300			c = i*4 + 1;  // left child
   301			c3 = c + 2;  // mid child
   302			if(c >= len) {
   303				break;
   304			}
   305			w = t[c]->when;
   306			if(c+1 < len && t[c+1]->when < w) {
   307				w = t[c+1]->when;
   308				c++;
   309			}
   310			if(c3 < len) {
   311				w3 = t[c3]->when;
   312				if(c3+1 < len && t[c3+1]->when < w3) {
   313					w3 = t[c3+1]->when;
   314					c3++;
   315				}
   316				if(w3 < w) {
   317					w = w3;
   318					c = c3;
   319				}
   320			}
   321			if(w >= when)
   322				break;
   323			t[i] = t[c];
   324			t[i]->i = i;
   325			t[c] = tmp;
   326			tmp->i = c;
   327			i = c;
   328		}
   329	}

   331	static void
   332	dumptimers(int8 *msg)
   333	{
   334		Timer *t;
   335		int32 i;
   336	
   337		runtime·printf("timers: %s\n", msg);
   338		for(i = 0; i < timers.len; i++) {
   339			t = timers.t[i];
   340			runtime·printf("\t%d\t%p:\ti %d when %D period %D fn %p\n",
   341					i, t, t->i, t->when, t->period, t->fv->fn);
   342		}
   343		runtime·printf("\n");
   344	}
