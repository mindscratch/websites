     5	// Garbage collector (GC).
     6	//
     7	// GC is:
     8	// - mark&sweep
     9	// - mostly precise (with the exception of some C-allocated objects, assembly frames/arguments, etc)
    10	// - parallel (up to MaxGcproc threads)
    11	// - partially concurrent (mark is stop-the-world, while sweep is concurrent)
    12	// - non-moving/non-compacting
    13	// - full (non-partial)
    14	//
    15	// GC rate.
    16	// Next GC is after we've allocated an extra amount of memory proportional to
    17	// the amount already in use. The proportion is controlled by GOGC environment variable
    18	// (100 by default). If GOGC=100 and we're using 4M, we'll GC again when we get to 8M
    19	// (this mark is tracked in next_gc variable). This keeps the GC cost in linear
    20	// proportion to the allocation cost. Adjusting GOGC just changes the linear constant
    21	// (and also the amount of extra memory used).
    22	//
    23	// Concurrent sweep.
    24	// The sweep phase proceeds concurrently with normal program execution.
    25	// The heap is swept span-by-span both lazily (when a goroutine needs another span)
    26	// and concurrently in a background goroutine (this helps programs that are not CPU bound).
    27	// However, at the end of the stop-the-world GC phase we don't know the size of the live heap,
    28	// and so next_gc calculation is tricky and happens as follows.
    29	// At the end of the stop-the-world phase next_gc is conservatively set based on total
    30	// heap size; all spans are marked as "needs sweeping".
    31	// Whenever a span is swept, next_gc is decremented by GOGC*newly_freed_memory.
    32	// The background sweeper goroutine simply sweeps spans one-by-one bringing next_gc
    33	// closer to the target value. However, this is not enough to avoid over-allocating memory.
    34	// Consider that a goroutine wants to allocate a new span for a large object and
    35	// there are no free swept spans, but there are small-object unswept spans.
    36	// If the goroutine naively allocates a new span, it can surpass the yet-unknown
    37	// target next_gc value. In order to prevent such cases (1) when a goroutine needs
    38	// to allocate a new small-object span, it sweeps small-object spans for the same
    39	// object size until it frees at least one object; (2) when a goroutine needs to
    40	// allocate large-object span from heap, it sweeps spans until it frees at least
    41	// that many pages into heap. Together these two measures ensure that we don't surpass
    42	// target next_gc value by a large margin. There is an exception: if a goroutine sweeps
    43	// and frees two nonadjacent one-page spans to the heap, it will allocate a new two-page span,
    44	// but there can still be other one-page unswept spans which could be combined into a two-page span.
    45	// It's critical to ensure that no operations proceed on unswept spans (that would corrupt
    46	// mark bits in GC bitmap). During GC all mcaches are flushed into the central cache,
    47	// so they are empty. When a goroutine grabs a new span into mcache, it sweeps it.
    48	// When a goroutine explicitly frees an object or sets a finalizer, it ensures that
    49	// the span is swept (either by sweeping it, or by waiting for the concurrent sweep to finish).
    50	// The finalizer goroutine is kicked off only when all spans are swept.
    51	// When the next GC starts, it sweeps all not-yet-swept spans (if any).
    52

   266	// markonly marks an object. It returns true if the object
   267	// has been marked by this function, false otherwise.
   268	// This function doesn't append the object to any buffer.
   269	static bool
   270	markonly(void *obj)
   271	{
   272		byte *p;
   273		uintptr *bitp, bits, shift, x, xbits, off, j;
   274		MSpan *s;
   275		PageID k;
   276	
   277		// Words outside the arena cannot be pointers.
   278		if(obj < runtime·mheap.arena_start || obj >= runtime·mheap.arena_used)
   279			return false;
   280	
   281		// obj may be a pointer to a live object.
   282		// Try to find the beginning of the object.
   283	
   284		// Round down to word boundary.
   285		obj = (void*)((uintptr)obj & ~((uintptr)PtrSize-1));
   286	
   287		// Find bits for this word.
   288		off = (uintptr*)obj - (uintptr*)runtime·mheap.arena_start;
   289		bitp = (uintptr*)runtime·mheap.arena_start - off/wordsPerBitmapWord - 1;
   290		shift = off % wordsPerBitmapWord;
   291		xbits = *bitp;
   292		bits = xbits >> shift;
   293	
   294		// Pointing at the beginning of a block?
   295		if((bits & (bitAllocated|bitBlockBoundary)) != 0) {
   296			if(CollectStats)
   297				runtime·xadd64(&gcstats.markonly.foundbit, 1);
   298			goto found;
   299		}
   300	
   301		// Pointing just past the beginning?
   302		// Scan backward a little to find a block boundary.
   303		for(j=shift; j-->0; ) {
   304			if(((xbits>>j) & (bitAllocated|bitBlockBoundary)) != 0) {
   305				shift = j;
   306				bits = xbits>>shift;
   307				if(CollectStats)
   308					runtime·xadd64(&gcstats.markonly.foundword, 1);
   309				goto found;
   310			}
   311		}
   312	
   313		// Otherwise consult span table to find beginning.
   314		// (Manually inlined copy of MHeap_LookupMaybe.)
   315		k = (uintptr)obj>>PageShift;
   316		x = k;
   317		x -= (uintptr)runtime·mheap.arena_start>>PageShift;
   318		s = runtime·mheap.spans[x];
   319		if(s == nil || k < s->start || obj >= s->limit || s->state != MSpanInUse)
   320			return false;
   321		p = (byte*)((uintptr)s->start<<PageShift);
   322		if(s->sizeclass == 0) {
   323			obj = p;
   324		} else {
   325			uintptr size = s->elemsize;
   326			int32 i = ((byte*)obj - p)/size;
   327			obj = p+i*size;
   328		}
   329	
   330		// Now that we know the object header, reload bits.
   331		off = (uintptr*)obj - (uintptr*)runtime·mheap.arena_start;
   332		bitp = (uintptr*)runtime·mheap.arena_start - off/wordsPerBitmapWord - 1;
   333		shift = off % wordsPerBitmapWord;
   334		xbits = *bitp;
   335		bits = xbits >> shift;
   336		if(CollectStats)
   337			runtime·xadd64(&gcstats.markonly.foundspan, 1);
   338	
   339	found:
   340		// Now we have bits, bitp, and shift correct for
   341		// obj pointing at the base of the object.
   342		// Only care about allocated and not marked.
   343		if((bits & (bitAllocated|bitMarked)) != bitAllocated)
   344			return false;
   345		if(work.nproc == 1)
   346			*bitp |= bitMarked<<shift;
   347		else {
   348			for(;;) {
   349				x = *bitp;
   350				if(x & (bitMarked<<shift))
   351					return false;
   352				if(runtime·casp((void**)bitp, (void*)x, (void*)(x|(bitMarked<<shift))))
   353					break;
   354			}
   355		}
   356	
   357		// The object is now marked
   358		return true;
   359	}

   406	// flushptrbuf moves data from the PtrTarget buffer to the work buffer.
   407	// The PtrTarget buffer contains blocks irrespective of whether the blocks have been marked or scanned,
   408	// while the work buffer contains blocks which have been marked
   409	// and are prepared to be scanned by the garbage collector.
   410	//
   411	// _wp, _wbuf, _nobj are input/output parameters and are specifying the work buffer.
   412	//
   413	// A simplified drawing explaining how the todo-list moves from a structure to another:
   414	//
   415	//     scanblock
   416	//  (find pointers)
   417	//    Obj ------> PtrTarget (pointer targets)
   418	//     ↑          |
   419	//     |          |
   420	//     `----------'
   421	//     flushptrbuf
   422	//  (find block start, mark and enqueue)
   423	static void
   424	flushptrbuf(Scanbuf *sbuf)
   425	{
   426		byte *p, *arena_start, *obj;
   427		uintptr size, *bitp, bits, shift, j, x, xbits, off, nobj, ti, n;
   428		MSpan *s;
   429		PageID k;
   430		Obj *wp;
   431		Workbuf *wbuf;
   432		PtrTarget *ptrbuf;
   433		PtrTarget *ptrbuf_end;
   434	
   435		arena_start = runtime·mheap.arena_start;
   436	
   437		wp = sbuf->wp;
   438		wbuf = sbuf->wbuf;
   439		nobj = sbuf->nobj;
   440	
   441		ptrbuf = sbuf->ptr.begin;
   442		ptrbuf_end = sbuf->ptr.pos;
   443		n = ptrbuf_end - sbuf->ptr.begin;
   444		sbuf->ptr.pos = sbuf->ptr.begin;
   445	
   446		if(CollectStats) {
   447			runtime·xadd64(&gcstats.ptr.sum, n);
   448			runtime·xadd64(&gcstats.ptr.cnt, 1);
   449		}
   450	
   451		// If buffer is nearly full, get a new one.
   452		if(wbuf == nil || nobj+n >= nelem(wbuf->obj)) {
   453			if(wbuf != nil)
   454				wbuf->nobj = nobj;
   455			wbuf = getempty(wbuf);
   456			wp = wbuf->obj;
   457			nobj = 0;
   458	
   459			if(n >= nelem(wbuf->obj))
   460				runtime·throw("ptrbuf has to be smaller than WorkBuf");
   461		}
   462	
   463		while(ptrbuf < ptrbuf_end) {
   464			obj = ptrbuf->p;
   465			ti = ptrbuf->ti;
   466			ptrbuf++;
   467	
   468			// obj belongs to interval [mheap.arena_start, mheap.arena_used).
   469			if(Debug > 1) {
   470				if(obj < runtime·mheap.arena_start || obj >= runtime·mheap.arena_used)
   471					runtime·throw("object is outside of mheap");
   472			}
   473	
   474			// obj may be a pointer to a live object.
   475			// Try to find the beginning of the object.
   476	
   477			// Round down to word boundary.
   478			if(((uintptr)obj & ((uintptr)PtrSize-1)) != 0) {
   479				obj = (void*)((uintptr)obj & ~((uintptr)PtrSize-1));
   480				ti = 0;
   481			}
   482	
   483			// Find bits for this word.
   484			off = (uintptr*)obj - (uintptr*)arena_start;
   485			bitp = (uintptr*)arena_start - off/wordsPerBitmapWord - 1;
   486			shift = off % wordsPerBitmapWord;
   487			xbits = *bitp;
   488			bits = xbits >> shift;
   489	
   490			// Pointing at the beginning of a block?
   491			if((bits & (bitAllocated|bitBlockBoundary)) != 0) {
   492				if(CollectStats)
   493					runtime·xadd64(&gcstats.flushptrbuf.foundbit, 1);
   494				goto found;
   495			}
   496	
   497			ti = 0;
   498	
   499			// Pointing just past the beginning?
   500			// Scan backward a little to find a block boundary.
   501			for(j=shift; j-->0; ) {
   502				if(((xbits>>j) & (bitAllocated|bitBlockBoundary)) != 0) {
   503					obj = (byte*)obj - (shift-j)*PtrSize;
   504					shift = j;
   505					bits = xbits>>shift;
   506					if(CollectStats)
   507						runtime·xadd64(&gcstats.flushptrbuf.foundword, 1);
   508					goto found;
   509				}
   510			}
   511	
   512			// Otherwise consult span table to find beginning.
   513			// (Manually inlined copy of MHeap_LookupMaybe.)
   514			k = (uintptr)obj>>PageShift;
   515			x = k;
   516			x -= (uintptr)arena_start>>PageShift;
   517			s = runtime·mheap.spans[x];
   518			if(s == nil || k < s->start || obj >= s->limit || s->state != MSpanInUse)
   519				continue;
   520			p = (byte*)((uintptr)s->start<<PageShift);
   521			if(s->sizeclass == 0) {
   522				obj = p;
   523			} else {
   524				size = s->elemsize;
   525				int32 i = ((byte*)obj - p)/size;
   526				obj = p+i*size;
   527			}
   528	
   529			// Now that we know the object header, reload bits.
   530			off = (uintptr*)obj - (uintptr*)arena_start;
   531			bitp = (uintptr*)arena_start - off/wordsPerBitmapWord - 1;
   532			shift = off % wordsPerBitmapWord;
   533			xbits = *bitp;
   534			bits = xbits >> shift;
   535			if(CollectStats)
   536				runtime·xadd64(&gcstats.flushptrbuf.foundspan, 1);
   537	
   538		found:
   539			// Now we have bits, bitp, and shift correct for
   540			// obj pointing at the base of the object.
   541			// Only care about allocated and not marked.
   542			if((bits & (bitAllocated|bitMarked)) != bitAllocated)
   543				continue;
   544			if(work.nproc == 1)
   545				*bitp |= bitMarked<<shift;
   546			else {
   547				for(;;) {
   548					x = *bitp;
   549					if(x & (bitMarked<<shift))
   550						goto continue_obj;
   551					if(runtime·casp((void**)bitp, (void*)x, (void*)(x|(bitMarked<<shift))))
   552						break;
   553				}
   554			}
   555	
   556			// If object has no pointers, don't need to scan further.
   557			if((bits & bitScan) == 0)
   558				continue;
   559	
   560			// Ask span about size class.
   561			// (Manually inlined copy of MHeap_Lookup.)
   562			x = (uintptr)obj >> PageShift;
   563			x -= (uintptr)arena_start>>PageShift;
   564			s = runtime·mheap.spans[x];
   565	
   566			PREFETCH(obj);
   567	
   568			*wp = (Obj){obj, s->elemsize, ti};
   569			wp++;
   570			nobj++;
   571		continue_obj:;
   572		}
   573	
   574		// If another proc wants a pointer, give it some.
   575		if(work.nwait > 0 && nobj > handoffThreshold && work.full == 0) {
   576			wbuf->nobj = nobj;
   577			wbuf = handoff(wbuf);
   578			nobj = wbuf->nobj;
   579			wp = wbuf->obj + nobj;
   580		}
   581	
   582		sbuf->wp = wp;
   583		sbuf->wbuf = wbuf;
   584		sbuf->nobj = nobj;
   585	}

   587	static void
   588	flushobjbuf(Scanbuf *sbuf)
   589	{
   590		uintptr nobj, off;
   591		Obj *wp, obj;
   592		Workbuf *wbuf;
   593		Obj *objbuf;
   594		Obj *objbuf_end;
   595	
   596		wp = sbuf->wp;
   597		wbuf = sbuf->wbuf;
   598		nobj = sbuf->nobj;
   599	
   600		objbuf = sbuf->obj.begin;
   601		objbuf_end = sbuf->obj.pos;
   602		sbuf->obj.pos = sbuf->obj.begin;
   603	
   604		while(objbuf < objbuf_end) {
   605			obj = *objbuf++;
   606	
   607			// Align obj.b to a word boundary.
   608			off = (uintptr)obj.p & (PtrSize-1);
   609			if(off != 0) {
   610				obj.p += PtrSize - off;
   611				obj.n -= PtrSize - off;
   612				obj.ti = 0;
   613			}
   614	
   615			if(obj.p == nil || obj.n == 0)
   616				continue;
   617	
   618			// If buffer is full, get a new one.
   619			if(wbuf == nil || nobj >= nelem(wbuf->obj)) {
   620				if(wbuf != nil)
   621					wbuf->nobj = nobj;
   622				wbuf = getempty(wbuf);
   623				wp = wbuf->obj;
   624				nobj = 0;
   625			}
   626	
   627			*wp = obj;
   628			wp++;
   629			nobj++;
   630		}
   631	
   632		// If another proc wants a pointer, give it some.
   633		if(work.nwait > 0 && nobj > handoffThreshold && work.full == 0) {
   634			wbuf->nobj = nobj;
   635			wbuf = handoff(wbuf);
   636			nobj = wbuf->nobj;
   637			wp = wbuf->obj + nobj;
   638		}
   639	
   640		sbuf->wp = wp;
   641		sbuf->wbuf = wbuf;
   642		sbuf->nobj = nobj;
   643	}

   645	// Program that scans the whole block and treats every block element as a potential pointer
   646	static uintptr defaultProg[2] = {PtrSize, GC_DEFAULT_PTR};
   647	
   648	// Hchan program
   649	static uintptr chanProg[2] = {0, GC_CHAN};
   650	
   651	// Local variables of a program fragment or loop
   652	typedef struct Frame Frame;
   653	struct Frame {
   654		uintptr count, elemsize, b;
   655		uintptr *loop_or_ret;
   656	};

   658	// Sanity check for the derived type info objti.
   659	static void
   660	checkptr(void *obj, uintptr objti)
   661	{
   662		uintptr *pc1, *pc2, type, tisize, i, j, x;
   663		byte *objstart;
   664		Type *t;
   665		MSpan *s;
   666	
   667		if(!Debug)
   668			runtime·throw("checkptr is debug only");
   669	
   670		if(obj < runtime·mheap.arena_start || obj >= runtime·mheap.arena_used)
   671			return;
   672		type = runtime·gettype(obj);
   673		t = (Type*)(type & ~(uintptr)(PtrSize-1));
   674		if(t == nil)
   675			return;
   676		x = (uintptr)obj >> PageShift;
   677		x -= (uintptr)(runtime·mheap.arena_start)>>PageShift;
   678		s = runtime·mheap.spans[x];
   679		objstart = (byte*)((uintptr)s->start<<PageShift);
   680		if(s->sizeclass != 0) {
   681			i = ((byte*)obj - objstart)/s->elemsize;
   682			objstart += i*s->elemsize;
   683		}
   684		tisize = *(uintptr*)objti;
   685		// Sanity check for object size: it should fit into the memory block.
   686		if((byte*)obj + tisize > objstart + s->elemsize) {
   687			runtime·printf("object of type '%S' at %p/%p does not fit in block %p/%p\n",
   688				       *t->string, obj, tisize, objstart, s->elemsize);
   689			runtime·throw("invalid gc type info");
   690		}
   691		if(obj != objstart)
   692			return;
   693		// If obj points to the beginning of the memory block,
   694		// check type info as well.
   695		if(t->string == nil ||
   696			// Gob allocates unsafe pointers for indirection.
   697			(runtime·strcmp(t->string->str, (byte*)"unsafe.Pointer") &&
   698			// Runtime and gc think differently about closures.
   699			runtime·strstr(t->string->str, (byte*)"struct { F uintptr") != t->string->str)) {
   700			pc1 = (uintptr*)objti;
   701			pc2 = (uintptr*)t->gc;
   702			// A simple best-effort check until first GC_END.
   703			for(j = 1; pc1[j] != GC_END && pc2[j] != GC_END; j++) {
   704				if(pc1[j] != pc2[j]) {
   705					runtime·printf("invalid gc type info for '%s', type info %p [%d]=%p, block info %p [%d]=%p\n",
   706						       t->string ? (int8*)t->string->str : (int8*)"?", pc1, (int32)j, pc1[j], pc2, (int32)j, pc2[j]);
   707					runtime·throw("invalid gc type info");
   708				}
   709			}
   710		}
   711	}

   713	// scanblock scans a block of n bytes starting at pointer b for references
   714	// to other objects, scanning any it finds recursively until there are no
   715	// unscanned objects left.  Instead of using an explicit recursion, it keeps
   716	// a work list in the Workbuf* structures and loops in the main function
   717	// body.  Keeping an explicit work list is easier on the stack allocator and
   718	// more efficient.
   719	static void
   720	scanblock(Workbuf *wbuf, bool keepworking)
   721	{
   722		byte *b, *arena_start, *arena_used;
   723		uintptr n, i, end_b, elemsize, size, ti, objti, count, type, nobj;
   724		uintptr *pc, precise_type, nominal_size;
   725		uintptr *chan_ret, chancap;
   726		void *obj;
   727		Type *t, *et;
   728		Slice *sliceptr;
   729		String *stringptr;
   730		Frame *stack_ptr, stack_top, stack[GC_STACK_CAPACITY+4];
   731		BufferList *scanbuffers;
   732		Scanbuf sbuf;
   733		Eface *eface;
   734		Iface *iface;
   735		Hchan *chan;
   736		ChanType *chantype;
   737		Obj *wp;
   738	
   739		if(sizeof(Workbuf) % WorkbufSize != 0)
   740			runtime·throw("scanblock: size of Workbuf is suboptimal");
   741	
   742		// Memory arena parameters.
   743		arena_start = runtime·mheap.arena_start;
   744		arena_used = runtime·mheap.arena_used;
   745	
   746		stack_ptr = stack+nelem(stack)-1;
   747	
   748		precise_type = false;
   749		nominal_size = 0;
   750	
   751		if(wbuf) {
   752			nobj = wbuf->nobj;
   753			wp = &wbuf->obj[nobj];
   754		} else {
   755			nobj = 0;
   756			wp = nil;
   757		}
   758	
   759		// Initialize sbuf
   760		scanbuffers = &bufferList[m->helpgc];
   761	
   762		sbuf.ptr.begin = sbuf.ptr.pos = &scanbuffers->ptrtarget[0];
   763		sbuf.ptr.end = sbuf.ptr.begin + nelem(scanbuffers->ptrtarget);
   764	
   765		sbuf.obj.begin = sbuf.obj.pos = &scanbuffers->obj[0];
   766		sbuf.obj.end = sbuf.obj.begin + nelem(scanbuffers->obj);
   767	
   768		sbuf.wbuf = wbuf;
   769		sbuf.wp = wp;
   770		sbuf.nobj = nobj;
   771	
   772		// (Silence the compiler)
   773		chan = nil;
   774		chantype = nil;
   775		chan_ret = nil;
   776	
   777		goto next_block;
   778	
   779		for(;;) {
   780			// Each iteration scans the block b of length n, queueing pointers in
   781			// the work buffer.
   782	
   783			if(CollectStats) {
   784				runtime·xadd64(&gcstats.nbytes, n);
   785				runtime·xadd64(&gcstats.obj.sum, sbuf.nobj);
   786				runtime·xadd64(&gcstats.obj.cnt, 1);
   787			}
   788	
   789			if(ti != 0) {
   790				if(Debug > 1) {
   791					runtime·printf("scanblock %p %D ti %p\n", b, (int64)n, ti);
   792				}
   793				pc = (uintptr*)(ti & ~(uintptr)PC_BITS);
   794				precise_type = (ti & PRECISE);
   795				stack_top.elemsize = pc[0];
   796				if(!precise_type)
   797					nominal_size = pc[0];
   798				if(ti & LOOP) {
   799					stack_top.count = 0;	// 0 means an infinite number of iterations
   800					stack_top.loop_or_ret = pc+1;
   801				} else {
   802					stack_top.count = 1;
   803				}
   804				if(Debug) {
   805					// Simple sanity check for provided type info ti:
   806					// The declared size of the object must be not larger than the actual size
   807					// (it can be smaller due to inferior pointers).
   808					// It's difficult to make a comprehensive check due to inferior pointers,
   809					// reflection, gob, etc.
   810					if(pc[0] > n) {
   811						runtime·printf("invalid gc type info: type info size %p, block size %p\n", pc[0], n);
   812						runtime·throw("invalid gc type info");
   813					}
   814				}
   815			} else if(UseSpanType) {
   816				if(CollectStats)
   817					runtime·xadd64(&gcstats.obj.notype, 1);
   818	
   819				type = runtime·gettype(b);
   820				if(type != 0) {
   821					if(CollectStats)
   822						runtime·xadd64(&gcstats.obj.typelookup, 1);
   823	
   824					t = (Type*)(type & ~(uintptr)(PtrSize-1));
   825					switch(type & (PtrSize-1)) {
   826					case TypeInfo_SingleObject:
   827						pc = (uintptr*)t->gc;
   828						precise_type = true;  // type information about 'b' is precise
   829						stack_top.count = 1;
   830						stack_top.elemsize = pc[0];
   831						break;
   832					case TypeInfo_Array:
   833						pc = (uintptr*)t->gc;
   834						if(pc[0] == 0)
   835							goto next_block;
   836						precise_type = true;  // type information about 'b' is precise
   837						stack_top.count = 0;  // 0 means an infinite number of iterations
   838						stack_top.elemsize = pc[0];
   839						stack_top.loop_or_ret = pc+1;
   840						break;
   841					case TypeInfo_Chan:
   842						chan = (Hchan*)b;
   843						chantype = (ChanType*)t;
   844						chan_ret = nil;
   845						pc = chanProg;
   846						break;
   847					default:
   848						if(Debug > 1)
   849							runtime·printf("scanblock %p %D type %p %S\n", b, (int64)n, type, *t->string);
   850						runtime·throw("scanblock: invalid type");
   851						return;
   852					}
   853					if(Debug > 1)
   854						runtime·printf("scanblock %p %D type %p %S pc=%p\n", b, (int64)n, type, *t->string, pc);
   855				} else {
   856					pc = defaultProg;
   857					if(Debug > 1)
   858						runtime·printf("scanblock %p %D unknown type\n", b, (int64)n);
   859				}
   860			} else {
   861				pc = defaultProg;
   862				if(Debug > 1)
   863					runtime·printf("scanblock %p %D no span types\n", b, (int64)n);
   864			}
   865	
   866			if(IgnorePreciseGC)
   867				pc = defaultProg;
   868	
   869			pc++;
   870			stack_top.b = (uintptr)b;
   871			end_b = (uintptr)b + n - PtrSize;
   872	
   873		for(;;) {
   874			if(CollectStats)
   875				runtime·xadd64(&gcstats.instr[pc[0]], 1);
   876	
   877			obj = nil;
   878			objti = 0;
   879			switch(pc[0]) {
   880			case GC_PTR:
   881				obj = *(void**)(stack_top.b + pc[1]);
   882				objti = pc[2];
   883				if(Debug > 2)
   884					runtime·printf("gc_ptr @%p: %p ti=%p\n", stack_top.b+pc[1], obj, objti);
   885				pc += 3;
   886				if(Debug)
   887					checkptr(obj, objti);
   888				break;
   889	
   890			case GC_SLICE:
   891				sliceptr = (Slice*)(stack_top.b + pc[1]);
   892				if(Debug > 2)
   893					runtime·printf("gc_slice @%p: %p/%D/%D\n", sliceptr, sliceptr->array, (int64)sliceptr->len, (int64)sliceptr->cap);
   894				if(sliceptr->cap != 0) {
   895					obj = sliceptr->array;
   896					// Can't use slice element type for scanning,
   897					// because if it points to an array embedded
   898					// in the beginning of a struct,
   899					// we will scan the whole struct as the slice.
   900					// So just obtain type info from heap.
   901				}
   902				pc += 3;
   903				break;
   904	
   905			case GC_APTR:
   906				obj = *(void**)(stack_top.b + pc[1]);
   907				if(Debug > 2)
   908					runtime·printf("gc_aptr @%p: %p\n", stack_top.b+pc[1], obj);
   909				pc += 2;
   910				break;
   911	
   912			case GC_STRING:
   913				stringptr = (String*)(stack_top.b + pc[1]);
   914				if(Debug > 2)
   915					runtime·printf("gc_string @%p: %p/%D\n", stack_top.b+pc[1], stringptr->str, (int64)stringptr->len);
   916				if(stringptr->len != 0)
   917					markonly(stringptr->str);
   918				pc += 2;
   919				continue;
   920	
   921			case GC_EFACE:
   922				eface = (Eface*)(stack_top.b + pc[1]);
   923				pc += 2;
   924				if(Debug > 2)
   925					runtime·printf("gc_eface @%p: %p %p\n", stack_top.b+pc[1], eface->type, eface->data);
   926				if(eface->type == nil)
   927					continue;
   928	
   929				// eface->type
   930				t = eface->type;
   931				if((void*)t >= arena_start && (void*)t < arena_used) {
   932					*sbuf.ptr.pos++ = (PtrTarget){t, 0};
   933					if(sbuf.ptr.pos == sbuf.ptr.end)
   934						flushptrbuf(&sbuf);
   935				}
   936	
   937				// eface->data
   938				if(eface->data >= arena_start && eface->data < arena_used) {
   939					if(t->size <= sizeof(void*)) {
   940						if((t->kind & KindNoPointers))
   941							continue;
   942	
   943						obj = eface->data;
   944						if((t->kind & ~KindNoPointers) == KindPtr) {
   945							// Only use type information if it is a pointer-containing type.
   946							// This matches the GC programs written by cmd/gc/reflect.c's
   947							// dgcsym1 in case TPTR32/case TPTR64. See rationale there.
   948							et = ((PtrType*)t)->elem;
   949							if(!(et->kind & KindNoPointers))
   950								objti = (uintptr)((PtrType*)t)->elem->gc;
   951						}
   952					} else {
   953						obj = eface->data;
   954						objti = (uintptr)t->gc;
   955					}
   956				}
   957				break;
   958	
   959			case GC_IFACE:
   960				iface = (Iface*)(stack_top.b + pc[1]);
   961				pc += 2;
   962				if(Debug > 2)
   963					runtime·printf("gc_iface @%p: %p/%p %p\n", stack_top.b+pc[1], iface->tab, nil, iface->data);
   964				if(iface->tab == nil)
   965					continue;
   966				
   967				// iface->tab
   968				if((void*)iface->tab >= arena_start && (void*)iface->tab < arena_used) {
   969					*sbuf.ptr.pos++ = (PtrTarget){iface->tab, (uintptr)itabtype->gc};
   970					if(sbuf.ptr.pos == sbuf.ptr.end)
   971						flushptrbuf(&sbuf);
   972				}
   973	
   974				// iface->data
   975				if(iface->data >= arena_start && iface->data < arena_used) {
   976					t = iface->tab->type;
   977					if(t->size <= sizeof(void*)) {
   978						if((t->kind & KindNoPointers))
   979							continue;
   980	
   981						obj = iface->data;
   982						if((t->kind & ~KindNoPointers) == KindPtr) {
   983							// Only use type information if it is a pointer-containing type.
   984							// This matches the GC programs written by cmd/gc/reflect.c's
   985							// dgcsym1 in case TPTR32/case TPTR64. See rationale there.
   986							et = ((PtrType*)t)->elem;
   987							if(!(et->kind & KindNoPointers))
   988								objti = (uintptr)((PtrType*)t)->elem->gc;
   989						}
   990					} else {
   991						obj = iface->data;
   992						objti = (uintptr)t->gc;
   993					}
   994				}
   995				break;
   996	
   997			case GC_DEFAULT_PTR:
   998				while(stack_top.b <= end_b) {
   999					obj = *(byte**)stack_top.b;
  1000					if(Debug > 2)
  1001						runtime·printf("gc_default_ptr @%p: %p\n", stack_top.b, obj);
  1002					stack_top.b += PtrSize;
  1003					if(obj >= arena_start && obj < arena_used) {
  1004						*sbuf.ptr.pos++ = (PtrTarget){obj, 0};
  1005						if(sbuf.ptr.pos == sbuf.ptr.end)
  1006							flushptrbuf(&sbuf);
  1007					}
  1008				}
  1009				goto next_block;
  1010	
  1011			case GC_END:
  1012				if(--stack_top.count != 0) {
  1013					// Next iteration of a loop if possible.
  1014					stack_top.b += stack_top.elemsize;
  1015					if(stack_top.b + stack_top.elemsize <= end_b+PtrSize) {
  1016						pc = stack_top.loop_or_ret;
  1017						continue;
  1018					}
  1019					i = stack_top.b;
  1020				} else {
  1021					// Stack pop if possible.
  1022					if(stack_ptr+1 < stack+nelem(stack)) {
  1023						pc = stack_top.loop_or_ret;
  1024						stack_top = *(++stack_ptr);
  1025						continue;
  1026					}
  1027					i = (uintptr)b + nominal_size;
  1028				}
  1029				if(!precise_type) {
  1030					// Quickly scan [b+i,b+n) for possible pointers.
  1031					for(; i<=end_b; i+=PtrSize) {
  1032						if(*(byte**)i != nil) {
  1033							// Found a value that may be a pointer.
  1034							// Do a rescan of the entire block.
  1035							enqueue((Obj){b, n, 0}, &sbuf.wbuf, &sbuf.wp, &sbuf.nobj);
  1036							if(CollectStats) {
  1037								runtime·xadd64(&gcstats.rescan, 1);
  1038								runtime·xadd64(&gcstats.rescanbytes, n);
  1039							}
  1040							break;
  1041						}
  1042					}
  1043				}
  1044				goto next_block;
  1045	
  1046			case GC_ARRAY_START:
  1047				i = stack_top.b + pc[1];
  1048				count = pc[2];
  1049				elemsize = pc[3];
  1050				pc += 4;
  1051	
  1052				// Stack push.
  1053				*stack_ptr-- = stack_top;
  1054				stack_top = (Frame){count, elemsize, i, pc};
  1055				continue;
  1056	
  1057			case GC_ARRAY_NEXT:
  1058				if(--stack_top.count != 0) {
  1059					stack_top.b += stack_top.elemsize;
  1060					pc = stack_top.loop_or_ret;
  1061				} else {
  1062					// Stack pop.
  1063					stack_top = *(++stack_ptr);
  1064					pc += 1;
  1065				}
  1066				continue;
  1067	
  1068			case GC_CALL:
  1069				// Stack push.
  1070				*stack_ptr-- = stack_top;
  1071				stack_top = (Frame){1, 0, stack_top.b + pc[1], pc+3 /*return address*/};
  1072				pc = (uintptr*)((byte*)pc + *(int32*)(pc+2));  // target of the CALL instruction
  1073				continue;
  1074	
  1075			case GC_REGION:
  1076				obj = (void*)(stack_top.b + pc[1]);
  1077				size = pc[2];
  1078				objti = pc[3];
  1079				pc += 4;
  1080	
  1081				if(Debug > 2)
  1082					runtime·printf("gc_region @%p: %D %p\n", stack_top.b+pc[1], (int64)size, objti);
  1083				*sbuf.obj.pos++ = (Obj){obj, size, objti};
  1084				if(sbuf.obj.pos == sbuf.obj.end)
  1085					flushobjbuf(&sbuf);
  1086				continue;
  1087	
  1088			case GC_CHAN_PTR:
  1089				chan = *(Hchan**)(stack_top.b + pc[1]);
  1090				if(Debug > 2 && chan != nil)
  1091					runtime·printf("gc_chan_ptr @%p: %p/%D/%D %p\n", stack_top.b+pc[1], chan, (int64)chan->qcount, (int64)chan->dataqsiz, pc[2]);
  1092				if(chan == nil) {
  1093					pc += 3;
  1094					continue;
  1095				}
  1096				if(markonly(chan)) {
  1097					chantype = (ChanType*)pc[2];
  1098					if(!(chantype->elem->kind & KindNoPointers)) {
  1099						// Start chanProg.
  1100						chan_ret = pc+3;
  1101						pc = chanProg+1;
  1102						continue;
  1103					}
  1104				}
  1105				pc += 3;
  1106				continue;
  1107	
  1108			case GC_CHAN:
  1109				// There are no heap pointers in struct Hchan,
  1110				// so we can ignore the leading sizeof(Hchan) bytes.
  1111				if(!(chantype->elem->kind & KindNoPointers)) {
  1112					// Channel's buffer follows Hchan immediately in memory.
  1113					// Size of buffer (cap(c)) is second int in the chan struct.
  1114					chancap = ((uintgo*)chan)[1];
  1115					if(chancap > 0) {
  1116						// TODO(atom): split into two chunks so that only the
  1117						// in-use part of the circular buffer is scanned.
  1118						// (Channel routines zero the unused part, so the current
  1119						// code does not lead to leaks, it's just a little inefficient.)
  1120						*sbuf.obj.pos++ = (Obj){(byte*)chan+runtime·Hchansize, chancap*chantype->elem->size,
  1121							(uintptr)chantype->elem->gc | PRECISE | LOOP};
  1122						if(sbuf.obj.pos == sbuf.obj.end)
  1123							flushobjbuf(&sbuf);
  1124					}
  1125				}
  1126				if(chan_ret == nil)
  1127					goto next_block;
  1128				pc = chan_ret;
  1129				continue;
  1130	
  1131			default:
  1132				runtime·printf("runtime: invalid GC instruction %p at %p\n", pc[0], pc);
  1133				runtime·throw("scanblock: invalid GC instruction");
  1134				return;
  1135			}
  1136	
  1137			if(obj >= arena_start && obj < arena_used) {
  1138				*sbuf.ptr.pos++ = (PtrTarget){obj, objti};
  1139				if(sbuf.ptr.pos == sbuf.ptr.end)
  1140					flushptrbuf(&sbuf);
  1141			}
  1142		}
  1143	
  1144		next_block:
  1145			// Done scanning [b, b+n).  Prepare for the next iteration of
  1146			// the loop by setting b, n, ti to the parameters for the next block.
  1147	
  1148			if(sbuf.nobj == 0) {
  1149				flushptrbuf(&sbuf);
  1150				flushobjbuf(&sbuf);
  1151	
  1152				if(sbuf.nobj == 0) {
  1153					if(!keepworking) {
  1154						if(sbuf.wbuf)
  1155							putempty(sbuf.wbuf);
  1156						return;
  1157					}
  1158					// Emptied our buffer: refill.
  1159					sbuf.wbuf = getfull(sbuf.wbuf);
  1160					if(sbuf.wbuf == nil)
  1161						return;
  1162					sbuf.nobj = sbuf.wbuf->nobj;
  1163					sbuf.wp = sbuf.wbuf->obj + sbuf.wbuf->nobj;
  1164				}
  1165			}
  1166	
  1167			// Fetch b from the work buffer.
  1168			--sbuf.wp;
  1169			b = sbuf.wp->p;
  1170			n = sbuf.wp->n;
  1171			ti = sbuf.wp->ti;
  1172			sbuf.nobj--;
  1173		}
  1174	}

  1176	// Append obj to the work buffer.
  1177	// _wbuf, _wp, _nobj are input/output parameters and are specifying the work buffer.
  1178	static void
  1179	enqueue(Obj obj, Workbuf **_wbuf, Obj **_wp, uintptr *_nobj)
  1180	{
  1181		uintptr nobj, off;
  1182		Obj *wp;
  1183		Workbuf *wbuf;
  1184	
  1185		if(Debug > 1)
  1186			runtime·printf("append obj(%p %D %p)\n", obj.p, (int64)obj.n, obj.ti);
  1187	
  1188		// Align obj.b to a word boundary.
  1189		off = (uintptr)obj.p & (PtrSize-1);
  1190		if(off != 0) {
  1191			obj.p += PtrSize - off;
  1192			obj.n -= PtrSize - off;
  1193			obj.ti = 0;
  1194		}
  1195	
  1196		if(obj.p == nil || obj.n == 0)
  1197			return;
  1198	
  1199		// Load work buffer state
  1200		wp = *_wp;
  1201		wbuf = *_wbuf;
  1202		nobj = *_nobj;
  1203	
  1204		// If another proc wants a pointer, give it some.
  1205		if(work.nwait > 0 && nobj > handoffThreshold && work.full == 0) {
  1206			wbuf->nobj = nobj;
  1207			wbuf = handoff(wbuf);
  1208			nobj = wbuf->nobj;
  1209			wp = wbuf->obj + nobj;
  1210		}
  1211	
  1212		// If buffer is full, get a new one.
  1213		if(wbuf == nil || nobj >= nelem(wbuf->obj)) {
  1214			if(wbuf != nil)
  1215				wbuf->nobj = nobj;
  1216			wbuf = getempty(wbuf);
  1217			wp = wbuf->obj;
  1218			nobj = 0;
  1219		}
  1220	
  1221		*wp = obj;
  1222		wp++;
  1223		nobj++;
  1224	
  1225		// Save work buffer state
  1226		*_wp = wp;
  1227		*_wbuf = wbuf;
  1228		*_nobj = nobj;
  1229	}

  1232	enqueue1(Workbuf **wbufp, Obj obj)
  1233	{
  1234		Workbuf *wbuf;
  1235	
  1236		wbuf = *wbufp;

  1238			*wbufp = wbuf = getempty(wbuf);
  1239		wbuf->obj[wbuf->nobj++] = obj;
  1240	}

  1242	static void
  1243	markroot(ParFor *desc, uint32 i)
  1244	{
  1245		Workbuf *wbuf;
  1246		FinBlock *fb;
  1247		MHeap *h;
  1248		MSpan **allspans, *s;
  1249		uint32 spanidx, sg;
  1250		G *gp;
  1251		void *p;
  1252	
  1253		USED(&desc);
  1254		wbuf = getempty(nil);
  1255		// Note: if you add a case here, please also update heapdump.c:dumproots.
  1256		switch(i) {
  1257		case RootData:
  1258			enqueue1(&wbuf, (Obj){data, edata - data, (uintptr)gcdata});
  1259			break;
  1260	
  1261		case RootBss:
  1262			enqueue1(&wbuf, (Obj){bss, ebss - bss, (uintptr)gcbss});
  1263			break;
  1264	
  1265		case RootFinalizers:
  1266			for(fb=allfin; fb; fb=fb->alllink)
  1267				enqueue1(&wbuf, (Obj){(byte*)fb->fin, fb->cnt*sizeof(fb->fin[0]), 0});
  1268			break;
  1269	
  1270		case RootSpanTypes:
  1271			// mark span types and MSpan.specials (to walk spans only once)
  1272			h = &runtime·mheap;
  1273			sg = h->sweepgen;
  1274			allspans = h->allspans;
  1275			for(spanidx=0; spanidx<runtime·mheap.nspan; spanidx++) {
  1276				Special *sp;
  1277				SpecialFinalizer *spf;
  1278	
  1279				s = allspans[spanidx];
  1280				if(s->sweepgen != sg) {
  1281					runtime·printf("sweep %d %d\n", s->sweepgen, sg);
  1282					runtime·throw("gc: unswept span");
  1283				}
  1284				if(s->state != MSpanInUse)
  1285					continue;
  1286				// The garbage collector ignores type pointers stored in MSpan.types:
  1287				//  - Compiler-generated types are stored outside of heap.
  1288				//  - The reflect package has runtime-generated types cached in its data structures.
  1289				//    The garbage collector relies on finding the references via that cache.
  1290				if(s->types.compression == MTypes_Words || s->types.compression == MTypes_Bytes)
  1291					markonly((byte*)s->types.data);
  1292				for(sp = s->specials; sp != nil; sp = sp->next) {
  1293					if(sp->kind != KindSpecialFinalizer)
  1294						continue;
  1295					// don't mark finalized object, but scan it so we
  1296					// retain everything it points to.
  1297					spf = (SpecialFinalizer*)sp;
  1298					// A finalizer can be set for an inner byte of an object, find object beginning.
  1299					p = (void*)((s->start << PageShift) + spf->offset/s->elemsize*s->elemsize);
  1300					enqueue1(&wbuf, (Obj){p, s->elemsize, 0});
  1301					enqueue1(&wbuf, (Obj){(void*)&spf->fn, PtrSize, 0});
  1302					enqueue1(&wbuf, (Obj){(void*)&spf->fint, PtrSize, 0});
  1303					enqueue1(&wbuf, (Obj){(void*)&spf->ot, PtrSize, 0});
  1304				}
  1305			}
  1306			break;
  1307	
  1308		case RootFlushCaches:
  1309			flushallmcaches();
  1310			break;
  1311	
  1312		default:
  1313			// the rest is scanning goroutine stacks
  1314			if(i - RootCount >= runtime·allglen)
  1315				runtime·throw("markroot: bad index");
  1316			gp = runtime·allg[i - RootCount];
  1317			// remember when we've first observed the G blocked
  1318			// needed only to output in traceback
  1319			if((gp->status == Gwaiting || gp->status == Gsyscall) && gp->waitsince == 0)
  1320				gp->waitsince = work.tstart;
  1321			addstackroots(gp, &wbuf);
  1322			break;
  1323			
  1324		}
  1325	
  1326		if(wbuf)
  1327			scanblock(wbuf, false);
  1328	}

  1330	// Get an empty work buffer off the work.empty list,
  1331	// allocating new buffers as needed.
  1332	static Workbuf*
  1333	getempty(Workbuf *b)
  1334	{
  1335		if(b != nil)
  1336			runtime·lfstackpush(&work.full, &b->node);
  1337		b = (Workbuf*)runtime·lfstackpop(&work.empty);
  1338		if(b == nil) {
  1339			// Need to allocate.
  1340			runtime·lock(&work);
  1341			if(work.nchunk < sizeof *b) {
  1342				work.nchunk = 1<<20;
  1343				work.chunk = runtime·SysAlloc(work.nchunk, &mstats.gc_sys);
  1344				if(work.chunk == nil)
  1345					runtime·throw("runtime: cannot allocate memory");
  1346			}
  1347			b = (Workbuf*)work.chunk;
  1348			work.chunk += sizeof *b;
  1349			work.nchunk -= sizeof *b;
  1350			runtime·unlock(&work);
  1351		}
  1352		b->nobj = 0;
  1353		return b;
  1354	}

  1365	// Get a full work buffer off the work.full list, or return nil.
  1366	static Workbuf*
  1367	getfull(Workbuf *b)
  1368	{
  1369		int32 i;
  1370	
  1371		if(CollectStats)
  1372			runtime·xadd64(&gcstats.getfull, 1);
  1373	
  1374		if(b != nil)
  1375			runtime·lfstackpush(&work.empty, &b->node);
  1376		b = (Workbuf*)runtime·lfstackpop(&work.full);
  1377		if(b != nil || work.nproc == 1)
  1378			return b;
  1379	
  1380		runtime·xadd(&work.nwait, +1);
  1381		for(i=0;; i++) {
  1382			if(work.full != 0) {
  1383				runtime·xadd(&work.nwait, -1);
  1384				b = (Workbuf*)runtime·lfstackpop(&work.full);
  1385				if(b != nil)
  1386					return b;
  1387				runtime·xadd(&work.nwait, +1);
  1388			}
  1389			if(work.nwait == work.nproc)
  1390				return nil;
  1391			if(i < 10) {
  1392				m->gcstats.nprocyield++;
  1393				runtime·procyield(20);
  1394			} else if(i < 20) {
  1395				m->gcstats.nosyield++;
  1396				runtime·osyield();
  1397			} else {
  1398				m->gcstats.nsleep++;
  1399				runtime·usleep(100);
  1400			}
  1401		}
  1402	}

  1404	static Workbuf*
  1405	handoff(Workbuf *b)
  1406	{
  1407		int32 n;
  1408		Workbuf *b1;
  1409	
  1410		// Make new buffer with half of b's pointers.
  1411		b1 = getempty(nil);
  1412		n = b->nobj/2;
  1413		b->nobj -= n;
  1414		b1->nobj = n;
  1415		runtime·memmove(b1->obj, b->obj+b->nobj, n*sizeof b1->obj[0]);
  1416		m->gcstats.nhandoff++;
  1417		m->gcstats.nhandoffcnt += n;
  1418	
  1419		// Put b on full list - let first half of b get stolen.
  1420		runtime·lfstackpush(&work.full, &b->node);
  1421		return b1;
  1422	}

  1456	// Starting from scanp, scans words corresponding to set bits.
  1457	static void
  1458	scanbitvector(Func *f, bool precise, byte *scanp, BitVector *bv, bool afterprologue, void *wbufp)
  1459	{
  1460		uintptr word, bits;
  1461		uint32 *wordp;
  1462		int32 i, remptrs;
  1463		byte *p;
  1464	
  1465		wordp = bv->data;
  1466		for(remptrs = bv->n; remptrs > 0; remptrs -= 32) {
  1467			word = *wordp++;
  1468			if(remptrs < 32)
  1469				i = remptrs;
  1470			else
  1471				i = 32;
  1472			i /= BitsPerPointer;
  1473			for(; i > 0; i--) {
  1474				bits = word & 3;
  1475				switch(bits) {
  1476				case BitsDead:
  1477					if(runtime·debug.gcdead)
  1478						*(uintptr*)scanp = PoisonGC;
  1479					break;
  1480				case BitsScalar:
  1481					break;
  1482				case BitsPointer:
  1483					p = *(byte**)scanp;
  1484					if(p != nil) {
  1485						if(Debug > 2)
  1486							runtime·printf("frame %s @%p: ptr %p\n", runtime·funcname(f), scanp, p);
  1487						if(precise && (p < (byte*)PageSize || (uintptr)p == PoisonGC || (uintptr)p == PoisonStack)) {
  1488							// Looks like a junk value in a pointer slot.
  1489							// Liveness analysis wrong?
  1490							m->traceback = 2;
  1491							runtime·printf("bad pointer in frame %s at %p: %p\n", runtime·funcname(f), scanp, p);
  1492							runtime·throw("bad pointer in scanbitvector");
  1493						}
  1494						enqueue1(wbufp, (Obj){scanp, PtrSize, 0});
  1495					}
  1496					break;
  1497				case BitsMultiWord:
  1498					p = scanp;
  1499					word >>= BitsPerPointer;
  1500					scanp += PtrSize;
  1501					i--;
  1502					if(i == 0) {
  1503						// Get next chunk of bits
  1504						remptrs -= 32;
  1505						word = *wordp++;
  1506						if(remptrs < 32)
  1507							i = remptrs;
  1508						else
  1509							i = 32;
  1510						i /= BitsPerPointer;
  1511					}
  1512					switch(word & 3) {
  1513					case BitsString:
  1514						if(Debug > 2)
  1515							runtime·printf("frame %s @%p: string %p/%D\n", runtime·funcname(f), p, ((String*)p)->str, (int64)((String*)p)->len);
  1516						if(((String*)p)->len != 0)
  1517							markonly(((String*)p)->str);
  1518						break;
  1519					case BitsSlice:
  1520						word >>= BitsPerPointer;
  1521						scanp += PtrSize;
  1522						i--;
  1523						if(i == 0) {
  1524							// Get next chunk of bits
  1525							remptrs -= 32;
  1526							word = *wordp++;
  1527							if(remptrs < 32)
  1528								i = remptrs;
  1529							else
  1530								i = 32;
  1531							i /= BitsPerPointer;
  1532						}
  1533						if(Debug > 2)
  1534							runtime·printf("frame %s @%p: slice %p/%D/%D\n", runtime·funcname(f), p, ((Slice*)p)->array, (int64)((Slice*)p)->len, (int64)((Slice*)p)->cap);
  1535						if(((Slice*)p)->cap < ((Slice*)p)->len) {
  1536							m->traceback = 2;
  1537							runtime·printf("bad slice in frame %s at %p: %p/%p/%p\n", runtime·funcname(f), p, ((byte**)p)[0], ((byte**)p)[1], ((byte**)p)[2]);
  1538							runtime·throw("slice capacity smaller than length");
  1539						}
  1540						if(((Slice*)p)->cap != 0)
  1541							enqueue1(wbufp, (Obj){p, PtrSize, 0});
  1542						break;
  1543					case BitsIface:
  1544					case BitsEface:
  1545						if(*(byte**)p != nil) {
  1546							if(Debug > 2) {
  1547								if((word&3) == BitsEface)
  1548									runtime·printf("frame %s @%p: eface %p %p\n", runtime·funcname(f), p, ((uintptr*)p)[0], ((uintptr*)p)[1]);
  1549								else
  1550									runtime·printf("frame %s @%p: iface %p %p\n", runtime·funcname(f), p, ((uintptr*)p)[0], ((uintptr*)p)[1]);
  1551							}
  1552							scaninterfacedata(word & 3, p, afterprologue, wbufp);
  1553						}
  1554						break;
  1555					}
  1556				}
  1557				word >>= BitsPerPointer;
  1558				scanp += PtrSize;
  1559			}
  1560		}
  1561	}

  1563	// Scan a stack frame: local variables and function arguments/results.
  1564	static bool
  1565	scanframe(Stkframe *frame, void *wbufp)
  1566	{
  1567		Func *f;
  1568		StackMap *stackmap;
  1569		BitVector bv;
  1570		uintptr size;
  1571		uintptr targetpc;
  1572		int32 pcdata;
  1573		bool afterprologue;
  1574		bool precise;
  1575	
  1576		f = frame->fn;
  1577		targetpc = frame->continpc;
  1578		if(targetpc == 0) {
  1579			// Frame is dead.
  1580			return true;
  1581		}
  1582		if(targetpc != f->entry)
  1583			targetpc--;
  1584		pcdata = runtime·pcdatavalue(f, PCDATA_StackMapIndex, targetpc);
  1585		if(pcdata == -1) {
  1586			// We do not have a valid pcdata value but there might be a
  1587			// stackmap for this function.  It is likely that we are looking
  1588			// at the function prologue, assume so and hope for the best.
  1589			pcdata = 0;
  1590		}
  1591	
  1592		// Scan local variables if stack frame has been allocated.
  1593		// Use pointer information if known.
  1594		afterprologue = (frame->varp > (byte*)frame->sp);
  1595		precise = false;
  1596		if(afterprologue) {
  1597			stackmap = runtime·funcdata(f, FUNCDATA_LocalsPointerMaps);
  1598			if(stackmap == nil) {
  1599				// No locals information, scan everything.
  1600				size = frame->varp - (byte*)frame->sp;
  1601				if(Debug > 2)
  1602					runtime·printf("frame %s unsized locals %p+%p\n", runtime·funcname(f), frame->varp-size, size);
  1603				enqueue1(wbufp, (Obj){frame->varp - size, size, 0});
  1604			} else if(stackmap->n < 0) {
  1605				// Locals size information, scan just the locals.
  1606				size = -stackmap->n;
  1607				if(Debug > 2)
  1608					runtime·printf("frame %s conservative locals %p+%p\n", runtime·funcname(f), frame->varp-size, size);
  1609				enqueue1(wbufp, (Obj){frame->varp - size, size, 0});
  1610			} else if(stackmap->n > 0) {
  1611				// Locals bitmap information, scan just the pointers in
  1612				// locals.
  1613				if(pcdata < 0 || pcdata >= stackmap->n) {
  1614					// don't know where we are
  1615					runtime·printf("pcdata is %d and %d stack map entries for %s (targetpc=%p)\n",
  1616						pcdata, stackmap->n, runtime·funcname(f), targetpc);
  1617					runtime·throw("scanframe: bad symbol table");
  1618				}
  1619				bv = runtime·stackmapdata(stackmap, pcdata);
  1620				size = (bv.n * PtrSize) / BitsPerPointer;
  1621				precise = true;
  1622				scanbitvector(f, true, frame->varp - size, &bv, afterprologue, wbufp);
  1623			}
  1624		}
  1625	
  1626		// Scan arguments.
  1627		// Use pointer information if known.
  1628		stackmap = runtime·funcdata(f, FUNCDATA_ArgsPointerMaps);
  1629		if(stackmap != nil) {
  1630			bv = runtime·stackmapdata(stackmap, pcdata);
  1631			scanbitvector(f, precise, frame->argp, &bv, true, wbufp);
  1632		} else {
  1633			if(Debug > 2)
  1634				runtime·printf("frame %s conservative args %p+%p\n", runtime·funcname(f), frame->argp, (uintptr)frame->arglen);
  1635			enqueue1(wbufp, (Obj){frame->argp, frame->arglen, 0});
  1636		}
  1637		return true;
  1638	}

  1640	static void
  1641	addstackroots(G *gp, Workbuf **wbufp)
  1642	{
  1643		M *mp;
  1644		int32 n;
  1645		Stktop *stk;
  1646		uintptr sp, guard;
  1647		void *base;
  1648		uintptr size;
  1649	
  1650		switch(gp->status){
  1651		default:
  1652			runtime·printf("unexpected G.status %d (goroutine %p %D)\n", gp->status, gp, gp->goid);
  1653			runtime·throw("mark - bad status");
  1654		case Gdead:
  1655			return;
  1656		case Grunning:
  1657			runtime·throw("mark - world not stopped");
  1658		case Grunnable:
  1659		case Gsyscall:
  1660		case Gwaiting:
  1661			break;
  1662		}
  1663	
  1664		if(gp == g)
  1665			runtime·throw("can't scan our own stack");
  1666		if((mp = gp->m) != nil && mp->helpgc)
  1667			runtime·throw("can't scan gchelper stack");
  1668	
  1669		if(gp->syscallstack != (uintptr)nil) {
  1670			// Scanning another goroutine that is about to enter or might
  1671			// have just exited a system call. It may be executing code such
  1672			// as schedlock and may have needed to start a new stack segment.
  1673			// Use the stack segment and stack pointer at the time of
  1674			// the system call instead, since that won't change underfoot.
  1675			sp = gp->syscallsp;
  1676			stk = (Stktop*)gp->syscallstack;
  1677			guard = gp->syscallguard;
  1678		} else {
  1679			// Scanning another goroutine's stack.
  1680			// The goroutine is usually asleep (the world is stopped).
  1681			sp = gp->sched.sp;
  1682			stk = (Stktop*)gp->stackbase;
  1683			guard = gp->stackguard;
  1684			// For function about to start, context argument is a root too.
  1685			if(gp->sched.ctxt != 0 && runtime·mlookup(gp->sched.ctxt, &base, &size, nil))
  1686				enqueue1(wbufp, (Obj){base, size, 0});
  1687		}
  1688		if(ScanStackByFrames) {
  1689			USED(sp);
  1690			USED(stk);
  1691			USED(guard);
  1692			runtime·gentraceback(~(uintptr)0, ~(uintptr)0, 0, gp, 0, nil, 0x7fffffff, scanframe, wbufp, false);
  1693		} else {
  1694			n = 0;
  1695			while(stk) {
  1696				if(sp < guard-StackGuard || (uintptr)stk < sp) {
  1697					runtime·printf("scanstack inconsistent: g%D#%d sp=%p not in [%p,%p]\n", gp->goid, n, sp, guard-StackGuard, stk);
  1698					runtime·throw("scanstack");
  1699				}
  1700				if(Debug > 2)
  1701					runtime·printf("conservative stack %p+%p\n", (byte*)sp, (uintptr)stk-sp);
  1702				enqueue1(wbufp, (Obj){(byte*)sp, (uintptr)stk - sp, (uintptr)defaultProg | PRECISE | LOOP});
  1703				sp = stk->gobuf.sp;
  1704				guard = stk->stackguard;
  1705				stk = (Stktop*)stk->stackbase;
  1706				n++;
  1707			}
  1708		}
  1709	}

  1756	void
  1757	runtime·MSpan_EnsureSwept(MSpan *s)
  1758	{
  1759		uint32 sg;
  1760	
  1761		// Caller must disable preemption.
  1762		// Otherwise when this function returns the span can become unswept again
  1763		// (if GC is triggered on another goroutine).
  1764		if(m->locks == 0 && m->mallocing == 0 && g != m->g0)
  1765			runtime·throw("MSpan_EnsureSwept: m is not locked");
  1766	
  1767		sg = runtime·mheap.sweepgen;
  1768		if(runtime·atomicload(&s->sweepgen) == sg)
  1769			return;
  1770		if(runtime·cas(&s->sweepgen, sg-2, sg-1)) {
  1771			runtime·MSpan_Sweep(s);
  1772			return;
  1773		}
  1774		// unfortunate condition, and we don't have efficient means to wait
  1775		while(runtime·atomicload(&s->sweepgen) != sg)
  1776			runtime·osyield();  
  1777	}

  1779	// Sweep frees or collects finalizers for blocks not marked in the mark phase.
  1780	// It clears the mark bits in preparation for the next GC round.
  1781	// Returns true if the span was returned to heap.
  1782	bool
  1783	runtime·MSpan_Sweep(MSpan *s)
  1784	{
  1785		int32 cl, n, npages, nfree;
  1786		uintptr size, off, *bitp, shift, bits;
  1787		uint32 sweepgen;
  1788		byte *p;
  1789		MCache *c;
  1790		byte *arena_start;
  1791		MLink head, *end;
  1792		byte *type_data;
  1793		byte compression;
  1794		uintptr type_data_inc;
  1795		MLink *x;
  1796		Special *special, **specialp, *y;
  1797		bool res, sweepgenset;
  1798	
  1799		// It's critical that we enter this function with preemption disabled,
  1800		// GC must not start while we are in the middle of this function.
  1801		if(m->locks == 0 && m->mallocing == 0 && g != m->g0)
  1802			runtime·throw("MSpan_Sweep: m is not locked");
  1803		sweepgen = runtime·mheap.sweepgen;
  1804		if(s->state != MSpanInUse || s->sweepgen != sweepgen-1) {
  1805			runtime·printf("MSpan_Sweep: state=%d sweepgen=%d mheap.sweepgen=%d\n",
  1806				s->state, s->sweepgen, sweepgen);
  1807			runtime·throw("MSpan_Sweep: bad span state");
  1808		}
  1809		arena_start = runtime·mheap.arena_start;
  1810		cl = s->sizeclass;
  1811		size = s->elemsize;
  1812		if(cl == 0) {
  1813			n = 1;
  1814		} else {
  1815			// Chunk full of small blocks.
  1816			npages = runtime·class_to_allocnpages[cl];
  1817			n = (npages << PageShift) / size;
  1818		}
  1819		res = false;
  1820		nfree = 0;
  1821		end = &head;
  1822		c = m->mcache;
  1823		sweepgenset = false;
  1824	
  1825		// mark any free objects in this span so we don't collect them
  1826		for(x = s->freelist; x != nil; x = x->next) {
  1827			// This is markonly(x) but faster because we don't need
  1828			// atomic access and we're guaranteed to be pointing at
  1829			// the head of a valid object.
  1830			off = (uintptr*)x - (uintptr*)runtime·mheap.arena_start;
  1831			bitp = (uintptr*)runtime·mheap.arena_start - off/wordsPerBitmapWord - 1;
  1832			shift = off % wordsPerBitmapWord;
  1833			*bitp |= bitMarked<<shift;
  1834		}
  1835	
  1836		// Unlink & free special records for any objects we're about to free.
  1837		specialp = &s->specials;
  1838		special = *specialp;
  1839		while(special != nil) {
  1840			// A finalizer can be set for an inner byte of an object, find object beginning.
  1841			p = (byte*)(s->start << PageShift) + special->offset/size*size;
  1842			off = (uintptr*)p - (uintptr*)arena_start;
  1843			bitp = (uintptr*)arena_start - off/wordsPerBitmapWord - 1;
  1844			shift = off % wordsPerBitmapWord;
  1845			bits = *bitp>>shift;
  1846			if((bits & (bitAllocated|bitMarked)) == bitAllocated) {
  1847				// Find the exact byte for which the special was setup
  1848				// (as opposed to object beginning).
  1849				p = (byte*)(s->start << PageShift) + special->offset;
  1850				// about to free object: splice out special record
  1851				y = special;
  1852				special = special->next;
  1853				*specialp = special;
  1854				if(!runtime·freespecial(y, p, size, false)) {
  1855					// stop freeing of object if it has a finalizer
  1856					*bitp |= bitMarked << shift;
  1857				}
  1858			} else {
  1859				// object is still live: keep special record
  1860				specialp = &special->next;
  1861				special = *specialp;
  1862			}
  1863		}
  1864	
  1865		type_data = (byte*)s->types.data;
  1866		type_data_inc = sizeof(uintptr);
  1867		compression = s->types.compression;
  1868		switch(compression) {
  1869		case MTypes_Bytes:
  1870			type_data += 8*sizeof(uintptr);
  1871			type_data_inc = 1;
  1872			break;
  1873		}
  1874	
  1875		// Sweep through n objects of given size starting at p.
  1876		// This thread owns the span now, so it can manipulate
  1877		// the block bitmap without atomic operations.
  1878		p = (byte*)(s->start << PageShift);
  1879		for(; n > 0; n--, p += size, type_data+=type_data_inc) {
  1880			off = (uintptr*)p - (uintptr*)arena_start;
  1881			bitp = (uintptr*)arena_start - off/wordsPerBitmapWord - 1;
  1882			shift = off % wordsPerBitmapWord;
  1883			bits = *bitp>>shift;
  1884	
  1885			if((bits & bitAllocated) == 0)
  1886				continue;
  1887	
  1888			if((bits & bitMarked) != 0) {
  1889				*bitp &= ~(bitMarked<<shift);
  1890				continue;
  1891			}
  1892	
  1893			if(runtime·debug.allocfreetrace)
  1894				runtime·tracefree(p, size);
  1895	
  1896			// Clear mark and scan bits.
  1897			*bitp &= ~((bitScan|bitMarked)<<shift);
  1898	
  1899			if(cl == 0) {
  1900				// Free large span.
  1901				runtime·unmarkspan(p, 1<<PageShift);
  1902				s->needzero = 1;
  1903				// important to set sweepgen before returning it to heap
  1904				runtime·atomicstore(&s->sweepgen, sweepgen);
  1905				sweepgenset = true;
  1906				// See note about SysFault vs SysFree in malloc.goc.
  1907				if(runtime·debug.efence)
  1908					runtime·SysFault(p, size);
  1909				else
  1910					runtime·MHeap_Free(&runtime·mheap, s, 1);
  1911				c->local_nlargefree++;
  1912				c->local_largefree += size;
  1913				runtime·xadd64(&mstats.next_gc, -(uint64)(size * (gcpercent + 100)/100));
  1914				res = true;
  1915			} else {
  1916				// Free small object.
  1917				switch(compression) {
  1918				case MTypes_Words:
  1919					*(uintptr*)type_data = 0;
  1920					break;
  1921				case MTypes_Bytes:
  1922					*(byte*)type_data = 0;
  1923					break;
  1924				}
  1925				if(size > 2*sizeof(uintptr))
  1926					((uintptr*)p)[1] = (uintptr)0xdeaddeaddeaddeadll;	// mark as "needs to be zeroed"
  1927				else if(size > sizeof(uintptr))
  1928					((uintptr*)p)[1] = 0;
  1929	
  1930				end->next = (MLink*)p;
  1931				end = (MLink*)p;
  1932				nfree++;
  1933			}
  1934		}
  1935	
  1936		// We need to set s->sweepgen = h->sweepgen only when all blocks are swept,
  1937		// because of the potential for a concurrent free/SetFinalizer.
  1938		// But we need to set it before we make the span available for allocation
  1939		// (return it to heap or mcentral), because allocation code assumes that a
  1940		// span is already swept if available for allocation.
  1941	
  1942		if(!sweepgenset && nfree == 0) {
  1943			// The span must be in our exclusive ownership until we update sweepgen,
  1944			// check for potential races.
  1945			if(s->state != MSpanInUse || s->sweepgen != sweepgen-1) {
  1946				runtime·printf("MSpan_Sweep: state=%d sweepgen=%d mheap.sweepgen=%d\n",
  1947					s->state, s->sweepgen, sweepgen);
  1948				runtime·throw("MSpan_Sweep: bad span state after sweep");
  1949			}
  1950			runtime·atomicstore(&s->sweepgen, sweepgen);
  1951		}
  1952		if(nfree > 0) {
  1953			c->local_nsmallfree[cl] += nfree;
  1954			c->local_cachealloc -= nfree * size;
  1955			runtime·xadd64(&mstats.next_gc, -(uint64)(nfree * size * (gcpercent + 100)/100));
  1956			res = runtime·MCentral_FreeSpan(&runtime·mheap.central[cl], s, nfree, head.next, end);
  1957			//MCentral_FreeSpan updates sweepgen
  1958		}
  1959		return res;
  1960	}

  1998	// sweeps one span
  1999	// returns number of pages returned to heap, or -1 if there is nothing to sweep
  2000	uintptr
  2001	runtime·sweepone(void)
  2002	{
  2003		MSpan *s;
  2004		uint32 idx, sg;
  2005		uintptr npages;
  2006	
  2007		// increment locks to ensure that the goroutine is not preempted
  2008		// in the middle of sweep thus leaving the span in an inconsistent state for next GC
  2009		m->locks++;
  2010		sg = runtime·mheap.sweepgen;
  2011		for(;;) {
  2012			idx = runtime·xadd(&sweep.spanidx, 1) - 1;
  2013			if(idx >= sweep.nspan) {
  2014				runtime·mheap.sweepdone = true;
  2015				m->locks--;
  2016				return -1;
  2017			}
  2018			s = sweep.spans[idx];
  2019			if(s->state != MSpanInUse) {
  2020				s->sweepgen = sg;
  2021				continue;
  2022			}
  2023			if(s->sweepgen != sg-2 || !runtime·cas(&s->sweepgen, sg-2, sg-1))
  2024				continue;
  2025			if(s->incache)
  2026				runtime·throw("sweep of incache span");
  2027			npages = s->npages;
  2028			if(!runtime·MSpan_Sweep(s))
  2029				npages = 0;
  2030			m->locks--;
  2031			return npages;
  2032		}
  2033	}

  2035	static void
  2036	dumpspan(uint32 idx)
  2037	{
  2038		int32 sizeclass, n, npages, i, column;
  2039		uintptr size;
  2040		byte *p;
  2041		byte *arena_start;
  2042		MSpan *s;
  2043		bool allocated;
  2044	
  2045		s = runtime·mheap.allspans[idx];
  2046		if(s->state != MSpanInUse)
  2047			return;
  2048		arena_start = runtime·mheap.arena_start;
  2049		p = (byte*)(s->start << PageShift);
  2050		sizeclass = s->sizeclass;
  2051		size = s->elemsize;
  2052		if(sizeclass == 0) {
  2053			n = 1;
  2054		} else {
  2055			npages = runtime·class_to_allocnpages[sizeclass];
  2056			n = (npages << PageShift) / size;
  2057		}
  2058		
  2059		runtime·printf("%p .. %p:\n", p, p+n*size);
  2060		column = 0;
  2061		for(; n>0; n--, p+=size) {
  2062			uintptr off, *bitp, shift, bits;
  2063	
  2064			off = (uintptr*)p - (uintptr*)arena_start;
  2065			bitp = (uintptr*)arena_start - off/wordsPerBitmapWord - 1;
  2066			shift = off % wordsPerBitmapWord;
  2067			bits = *bitp>>shift;
  2068	
  2069			allocated = ((bits & bitAllocated) != 0);
  2070	
  2071			for(i=0; i<size; i+=sizeof(void*)) {
  2072				if(column == 0) {
  2073					runtime·printf("\t");
  2074				}
  2075				if(i == 0) {
  2076					runtime·printf(allocated ? "(" : "[");
  2077					runtime·printf("%p: ", p+i);
  2078				} else {
  2079					runtime·printf(" ");
  2080				}
  2081	
  2082				runtime·printf("%p", *(void**)(p+i));
  2083	
  2084				if(i+sizeof(void*) >= size) {
  2085					runtime·printf(allocated ? ") " : "] ");
  2086				}
  2087	
  2088				column++;
  2089				if(column == 8) {
  2090					runtime·printf("\n");
  2091					column = 0;
  2092				}
  2093			}
  2094		}
  2095		runtime·printf("\n");
  2096	}

  2159	void
  2160	runtime·updatememstats(GCStats *stats)
  2161	{
  2162		M *mp;
  2163		MSpan *s;
  2164		int32 i;
  2165		uint64 stacks_inuse, smallfree;
  2166		uint64 *src, *dst;
  2167	
  2168		if(stats)
  2169			runtime·memclr((byte*)stats, sizeof(*stats));
  2170		stacks_inuse = 0;
  2171		for(mp=runtime·allm; mp; mp=mp->alllink) {
  2172			stacks_inuse += mp->stackinuse*FixedStack;
  2173			if(stats) {
  2174				src = (uint64*)&mp->gcstats;
  2175				dst = (uint64*)stats;
  2176				for(i=0; i<sizeof(*stats)/sizeof(uint64); i++)
  2177					dst[i] += src[i];
  2178				runtime·memclr((byte*)&mp->gcstats, sizeof(mp->gcstats));
  2179			}
  2180		}
  2181		mstats.stacks_inuse = stacks_inuse;
  2182		mstats.mcache_inuse = runtime·mheap.cachealloc.inuse;
  2183		mstats.mspan_inuse = runtime·mheap.spanalloc.inuse;
  2184		mstats.sys = mstats.heap_sys + mstats.stacks_sys + mstats.mspan_sys +
  2185			mstats.mcache_sys + mstats.buckhash_sys + mstats.gc_sys + mstats.other_sys;
  2186		
  2187		// Calculate memory allocator stats.
  2188		// During program execution we only count number of frees and amount of freed memory.
  2189		// Current number of alive object in the heap and amount of alive heap memory
  2190		// are calculated by scanning all spans.
  2191		// Total number of mallocs is calculated as number of frees plus number of alive objects.
  2192		// Similarly, total amount of allocated memory is calculated as amount of freed memory
  2193		// plus amount of alive heap memory.
  2194		mstats.alloc = 0;
  2195		mstats.total_alloc = 0;
  2196		mstats.nmalloc = 0;
  2197		mstats.nfree = 0;
  2198		for(i = 0; i < nelem(mstats.by_size); i++) {
  2199			mstats.by_size[i].nmalloc = 0;
  2200			mstats.by_size[i].nfree = 0;
  2201		}
  2202	
  2203		// Flush MCache's to MCentral.
  2204		flushallmcaches();
  2205	
  2206		// Aggregate local stats.
  2207		cachestats();
  2208	
  2209		// Scan all spans and count number of alive objects.
  2210		for(i = 0; i < runtime·mheap.nspan; i++) {
  2211			s = runtime·mheap.allspans[i];
  2212			if(s->state != MSpanInUse)
  2213				continue;
  2214			if(s->sizeclass == 0) {
  2215				mstats.nmalloc++;
  2216				mstats.alloc += s->elemsize;
  2217			} else {
  2218				mstats.nmalloc += s->ref;
  2219				mstats.by_size[s->sizeclass].nmalloc += s->ref;
  2220				mstats.alloc += s->ref*s->elemsize;
  2221			}
  2222		}
  2223	
  2224		// Aggregate by size class.
  2225		smallfree = 0;
  2226		mstats.nfree = runtime·mheap.nlargefree;

  2801	// mark the span of memory at v as having n blocks of the given size.
  2802	// if leftover is true, there is left over space at the end of the span.
  2803	void
  2804	runtime·markspan(void *v, uintptr size, uintptr n, bool leftover)
  2805	{
  2806		uintptr *b, *b0, off, shift, i, x;
  2807		byte *p;
  2808	
  2809		if((byte*)v+size*n > (byte*)runtime·mheap.arena_used || (byte*)v < runtime·mheap.arena_start)
  2810			runtime·throw("markspan: bad pointer");
  2811	
  2812		if(runtime·checking) {
  2813			// bits should be all zero at the start
  2814			off = (byte*)v + size - runtime·mheap.arena_start;
  2815			b = (uintptr*)(runtime·mheap.arena_start - off/wordsPerBitmapWord);
  2816			for(i = 0; i < size/PtrSize/wordsPerBitmapWord; i++) {
  2817				if(b[i] != 0)
  2818					runtime·throw("markspan: span bits not zero");
  2819			}
  2820		}
  2821	
  2822		p = v;
  2823		if(leftover)	// mark a boundary just past end of last block too
  2824			n++;
  2825	
  2826		b0 = nil;
  2827		x = 0;
  2828		for(; n-- > 0; p += size) {
  2829			// Okay to use non-atomic ops here, because we control
  2830			// the entire span, and each bitmap word has bits for only
  2831			// one span, so no other goroutines are changing these
  2832			// bitmap words.
  2833			off = (uintptr*)p - (uintptr*)runtime·mheap.arena_start;  // word offset
  2834			b = (uintptr*)runtime·mheap.arena_start - off/wordsPerBitmapWord - 1;
  2835			shift = off % wordsPerBitmapWord;
  2836			if(b0 != b) {
  2837				if(b0 != nil)
  2838					*b0 = x;
  2839				b0 = b;
  2840				x = 0;
  2841			}
  2842			x |= bitAllocated<<shift;
  2843		}
  2844		*b0 = x;
  2845	}

  2848	void
  2849	runtime·unmarkspan(void *v, uintptr n)
  2850	{
  2851		uintptr *p, *b, off;
  2852	
  2853		if((byte*)v+n > (byte*)runtime·mheap.arena_used || (byte*)v < runtime·mheap.arena_start)
  2854			runtime·throw("markspan: bad pointer");
  2855	
  2856		p = v;
  2857		off = p - (uintptr*)runtime·mheap.arena_start;  // word offset
  2858		if(off % wordsPerBitmapWord != 0)
  2859			runtime·throw("markspan: unaligned pointer");
  2860		b = (uintptr*)runtime·mheap.arena_start - off/wordsPerBitmapWord - 1;
  2861		n /= PtrSize;
  2862		if(n%wordsPerBitmapWord != 0)
  2863			runtime·throw("unmarkspan: unaligned length");
  2864		// Okay to use non-atomic ops here, because we control
  2865		// the entire span, and each bitmap word has bits for only
  2866		// one span, so no other goroutines are changing these
  2867		// bitmap words.
  2868		n /= wordsPerBitmapWord;
  2869		while(n-- > 0)
  2870			*b-- = 0;
  2871	}
