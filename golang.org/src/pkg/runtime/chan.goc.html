    22	static Hchan*
    23	makechan(ChanType *t, int64 hint)
    24	{
    25		Hchan *c;
    26		Type *elem;
    27	
    28		elem = t->elem;
    29	
    30		// compiler checks this but be safe.
    31		if(elem->size >= (1<<16))
    32			runtime·throw("makechan: invalid channel element type");
    33		if((sizeof(*c)%MAXALIGN) != 0 || elem->align > MAXALIGN)
    34			runtime·throw("makechan: bad alignment");
    35	
    36		if(hint < 0 || (intgo)hint != hint || (elem->size > 0 && hint > (MaxMem - sizeof(*c)) / elem->size))
    37			runtime·panicstring("makechan: size out of range");
    38	
    39		// allocate memory in one call
    40		c = (Hchan*)runtime·mallocgc(sizeof(*c) + hint*elem->size, (uintptr)t | TypeInfo_Chan, 0);
    41		c->elemsize = elem->size;
    42		c->elemtype = elem;
    43		c->dataqsiz = hint;
    44	
    45		if(debug)
    46			runtime·printf("makechan: chan=%p; elemsize=%D; elemalg=%p; dataqsiz=%D\n",
    47				c, (int64)elem->size, elem->alg, (int64)c->dataqsiz);
    48	
    49		return c;
    50	}

    60	/*
    61	 * generic single channel send/recv
    62	 * if the bool pointer is nil,
    63	 * then the full exchange will
    64	 * occur. if pres is not nil,
    65	 * then the protocol will not
    66	 * sleep but return if it could
    67	 * not complete.
    68	 *
    69	 * sleep can wake up with g->param == nil
    70	 * when a channel involved in the sleep has
    71	 * been closed.  it is easiest to loop and re-run
    72	 * the operation; we'll see that it's now closed.
    73	 */
    74	static bool
    75	chansend(ChanType *t, Hchan *c, byte *ep, bool block, void *pc)
    76	{
    77		SudoG *sg;
    78		SudoG mysg;
    79		G* gp;
    80		int64 t0;
    81	
    82		if(raceenabled)
    83			runtime·racereadobjectpc(ep, t->elem, runtime·getcallerpc(&t), chansend);
    84	
    85		if(c == nil) {
    86			USED(t);
    87			if(!block)
    88				return false;
    89			runtime·park(nil, nil, "chan send (nil chan)");
    90			return false;  // not reached
    91		}
    92	
    93		if(debug) {
    94			runtime·printf("chansend: chan=%p; elem=", c);
    95			c->elemtype->alg->print(c->elemsize, ep);
    96			runtime·prints("\n");
    97		}
    98	
    99		t0 = 0;
   100		mysg.releasetime = 0;
   101		if(runtime·blockprofilerate > 0) {
   102			t0 = runtime·cputicks();
   103			mysg.releasetime = -1;
   104		}
   105	
   106		runtime·lock(c);
   107		if(raceenabled)
   108			runtime·racereadpc(c, pc, chansend);
   109		if(c->closed)
   110			goto closed;
   111	
   112		if(c->dataqsiz > 0)
   113			goto asynch;
   114	
   115		sg = dequeue(&c->recvq);
   116		if(sg != nil) {
   117			if(raceenabled)
   118				racesync(c, sg);
   119			runtime·unlock(c);
   120	
   121			gp = sg->g;
   122			gp->param = sg;
   123			if(sg->elem != nil)
   124				c->elemtype->alg->copy(c->elemsize, sg->elem, ep);
   125			if(sg->releasetime)
   126				sg->releasetime = runtime·cputicks();
   127			runtime·ready(gp);
   128			return true;
   129		}
   130	
   131		if(!block) {
   132			runtime·unlock(c);
   133			return false;
   134		}
   135	
   136		mysg.elem = ep;
   137		mysg.g = g;
   138		mysg.selectdone = nil;
   139		g->param = nil;
   140		enqueue(&c->sendq, &mysg);
   141		runtime·parkunlock(c, "chan send");
   142	
   143		if(g->param == nil) {
   144			runtime·lock(c);
   145			if(!c->closed)
   146				runtime·throw("chansend: spurious wakeup");
   147			goto closed;
   148		}
   149	
   150		if(mysg.releasetime > 0)
   151			runtime·blockevent(mysg.releasetime - t0, 2);
   152	
   153		return true;
   154	
   155	asynch:
   156		if(c->closed)
   157			goto closed;
   158	
   159		if(c->qcount >= c->dataqsiz) {
   160			if(!block) {
   161				runtime·unlock(c);
   162				return false;
   163			}
   164			mysg.g = g;
   165			mysg.elem = nil;
   166			mysg.selectdone = nil;
   167			enqueue(&c->sendq, &mysg);
   168			runtime·parkunlock(c, "chan send");
   169	
   170			runtime·lock(c);
   171			goto asynch;
   172		}
   173	
   174		if(raceenabled) {
   175			runtime·raceacquire(chanbuf(c, c->sendx));
   176			runtime·racerelease(chanbuf(c, c->sendx));
   177		}
   178	
   179		c->elemtype->alg->copy(c->elemsize, chanbuf(c, c->sendx), ep);
   180		if(++c->sendx == c->dataqsiz)
   181			c->sendx = 0;
   182		c->qcount++;
   183	
   184		sg = dequeue(&c->recvq);
   185		if(sg != nil) {
   186			gp = sg->g;
   187			runtime·unlock(c);
   188			if(sg->releasetime)
   189				sg->releasetime = runtime·cputicks();
   190			runtime·ready(gp);
   191		} else
   192			runtime·unlock(c);
   193		if(mysg.releasetime > 0)
   194			runtime·blockevent(mysg.releasetime - t0, 2);
   195		return true;
   196	
   197	closed:
   198		runtime·unlock(c);
   199		runtime·panicstring("send on closed channel");
   200		return false;  // not reached
   201	}

   204	static bool
   205	chanrecv(ChanType *t, Hchan* c, byte *ep, bool block, bool *received)
   206	{
   207		SudoG *sg;
   208		SudoG mysg;
   209		G *gp;
   210		int64 t0;
   211	
   212		// raceenabled: don't need to check ep, as it is always on the stack.
   213	
   214		if(debug)
   215			runtime·printf("chanrecv: chan=%p\n", c);
   216	
   217		if(c == nil) {
   218			USED(t);
   219			if(!block)
   220				return false;
   221			runtime·park(nil, nil, "chan receive (nil chan)");
   222			return false;  // not reached
   223		}
   224	
   225		t0 = 0;
   226		mysg.releasetime = 0;
   227		if(runtime·blockprofilerate > 0) {
   228			t0 = runtime·cputicks();
   229			mysg.releasetime = -1;
   230		}
   231	
   232		runtime·lock(c);
   233		if(c->dataqsiz > 0)
   234			goto asynch;
   235	
   236		if(c->closed)
   237			goto closed;
   238	
   239		sg = dequeue(&c->sendq);
   240		if(sg != nil) {
   241			if(raceenabled)
   242				racesync(c, sg);
   243			runtime·unlock(c);
   244	
   245			if(ep != nil)
   246				c->elemtype->alg->copy(c->elemsize, ep, sg->elem);
   247			gp = sg->g;
   248			gp->param = sg;
   249			if(sg->releasetime)
   250				sg->releasetime = runtime·cputicks();
   251			runtime·ready(gp);
   252	
   253			if(received != nil)
   254				*received = true;
   255			return true;
   256		}
   257	
   258		if(!block) {
   259			runtime·unlock(c);
   260			return false;
   261		}
   262	
   263		mysg.elem = ep;
   264		mysg.g = g;
   265		mysg.selectdone = nil;
   266		g->param = nil;
   267		enqueue(&c->recvq, &mysg);
   268		runtime·parkunlock(c, "chan receive");
   269	
   270		if(g->param == nil) {
   271			runtime·lock(c);
   272			if(!c->closed)
   273				runtime·throw("chanrecv: spurious wakeup");
   274			goto closed;
   275		}
   276	
   277		if(received != nil)
   278			*received = true;
   279		if(mysg.releasetime > 0)
   280			runtime·blockevent(mysg.releasetime - t0, 2);
   281		return true;
   282	
   283	asynch:
   284		if(c->qcount <= 0) {
   285			if(c->closed)
   286				goto closed;
   287	
   288			if(!block) {
   289				runtime·unlock(c);
   290				if(received != nil)
   291					*received = false;
   292				return false;
   293			}
   294			mysg.g = g;
   295			mysg.elem = nil;
   296			mysg.selectdone = nil;
   297			enqueue(&c->recvq, &mysg);
   298			runtime·parkunlock(c, "chan receive");
   299	
   300			runtime·lock(c);
   301			goto asynch;
   302		}
   303	
   304		if(raceenabled) {
   305			runtime·raceacquire(chanbuf(c, c->recvx));
   306			runtime·racerelease(chanbuf(c, c->recvx));
   307		}
   308	
   309		if(ep != nil)
   310			c->elemtype->alg->copy(c->elemsize, ep, chanbuf(c, c->recvx));
   311		c->elemtype->alg->copy(c->elemsize, chanbuf(c, c->recvx), nil);
   312		if(++c->recvx == c->dataqsiz)
   313			c->recvx = 0;
   314		c->qcount--;
   315	
   316		sg = dequeue(&c->sendq);
   317		if(sg != nil) {
   318			gp = sg->g;
   319			runtime·unlock(c);
   320			if(sg->releasetime)
   321				sg->releasetime = runtime·cputicks();
   322			runtime·ready(gp);
   323		} else
   324			runtime·unlock(c);
   325	
   326		if(received != nil)
   327			*received = true;
   328		if(mysg.releasetime > 0)
   329			runtime·blockevent(mysg.releasetime - t0, 2);
   330		return true;
   331	
   332	closed:
   333		if(ep != nil)
   334			c->elemtype->alg->copy(c->elemsize, ep, nil);
   335		if(received != nil)
   336			*received = false;
   337		if(raceenabled)
   338			runtime·raceacquire(c);
   339		runtime·unlock(c);
   340		if(mysg.releasetime > 0)
   341			runtime·blockevent(mysg.releasetime - t0, 2);
   342		return true;
   343	}

   444	static Select*
   445	newselect(int32 size)
   446	{
   447		int32 n;
   448		Select *sel;
   449	
   450		n = 0;
   451		if(size > 1)
   452			n = size-1;
   453	
   454		// allocate all the memory we need in a single allocation
   455		// start with Select with size cases
   456		// then lockorder with size entries
   457		// then pollorder with size entries
   458		sel = runtime·mal(sizeof(*sel) +
   459			n*sizeof(sel->scase[0]) +
   460			size*sizeof(sel->lockorder[0]) +
   461			size*sizeof(sel->pollorder[0]));
   462	
   463		sel->tcase = size;
   464		sel->ncase = 0;
   465		sel->lockorder = (void*)(sel->scase + size);
   466		sel->pollorder = (void*)(sel->lockorder + size);
   467	
   468		if(debug)
   469			runtime·printf("newselect s=%p size=%d\n", sel, size);
   470		return sel;
   471	}

   486	selectsend(Select *sel, Hchan *c, void *pc, void *elem, int32 so)
   487	{
   488		int32 i;
   489		Scase *cas;
   490	
   491		i = sel->ncase;

   493			runtime·throw("selectsend: too many cases");
   494		sel->ncase = i+1;
   495		cas = &sel->scase[i];

   497		cas->pc = pc;
   498		cas->chan = c;
   499		cas->so = so;
   500		cas->kind = CaseSend;
   501		cas->sg.elem = elem;
   502	
   503		if(debug)
   504			runtime·printf("selectsend s=%p pc=%p chan=%p so=%d\n",
   505				sel, cas->pc, cas->chan, cas->so);

   584	static void
   585	sellock(Select *sel)
   586	{
   587		uint32 i;
   588		Hchan *c, *c0;
   589	
   590		c = nil;
   591		for(i=0; i<sel->ncase; i++) {
   592			c0 = sel->lockorder[i];
   593			if(c0 && c0 != c) {
   594				c = sel->lockorder[i];
   595				runtime·lock(c);
   596			}
   597		}
   598	}
   599	
   600	static void
   601	selunlock(Select *sel)
   602	{
   603		int32 i, n, r;
   604		Hchan *c;
   605	
   606		// We must be very careful here to not touch sel after we have unlocked
   607		// the last lock, because sel can be freed right after the last unlock.
   608		// Consider the following situation.
   609		// First M calls runtime·park() in runtime·selectgo() passing the sel.
   610		// Once runtime·park() has unlocked the last lock, another M makes
   611		// the G that calls select runnable again and schedules it for execution.
   612		// When the G runs on another M, it locks all the locks and frees sel.
   613		// Now if the first M touches sel, it will access freed memory.
   614		n = (int32)sel->ncase;
   615		r = 0;
   616		// skip the default case
   617		if(n>0 && sel->lockorder[0] == nil)
   618			r = 1;
   619		for(i = n-1; i >= r; i--) {
   620			c = sel->lockorder[i];
   621			if(i>0 && sel->lockorder[i-1] == c)
   622				continue;  // will unlock it on the next iteration
   623			runtime·unlock(c);
   624		}
   625	}

   631		selunlock(sel);

   650	static void*
   651	selectgo(Select **selp)
   652	{
   653		Select *sel;
   654		uint32 o, i, j, k, done;
   655		int64 t0;
   656		Scase *cas, *dfl;
   657		Hchan *c;
   658		SudoG *sg;
   659		G *gp;
   660		byte *as;
   661		void *pc;
   662	
   663		sel = *selp;
   664	
   665		if(debug)
   666			runtime·printf("select: sel=%p\n", sel);
   667	
   668		t0 = 0;
   669		if(runtime·blockprofilerate > 0) {
   670			t0 = runtime·cputicks();
   671			for(i=0; i<sel->ncase; i++)
   672				sel->scase[i].sg.releasetime = -1;
   673		}
   674	
   675		// The compiler rewrites selects that statically have
   676		// only 0 or 1 cases plus default into simpler constructs.
   677		// The only way we can end up with such small sel->ncase
   678		// values here is for a larger select in which most channels
   679		// have been nilled out.  The general code handles those
   680		// cases correctly, and they are rare enough not to bother
   681		// optimizing (and needing to test).
   682	
   683		// generate permuted order
   684		for(i=0; i<sel->ncase; i++)
   685			sel->pollorder[i] = i;
   686		for(i=1; i<sel->ncase; i++) {
   687			o = sel->pollorder[i];
   688			j = runtime·fastrand1()%(i+1);
   689			sel->pollorder[i] = sel->pollorder[j];
   690			sel->pollorder[j] = o;
   691		}
   692	
   693		// sort the cases by Hchan address to get the locking order.
   694		// simple heap sort, to guarantee n log n time and constant stack footprint.
   695		for(i=0; i<sel->ncase; i++) {
   696			j = i;
   697			c = sel->scase[j].chan;
   698			while(j > 0 && sel->lockorder[k=(j-1)/2] < c) {
   699				sel->lockorder[j] = sel->lockorder[k];
   700				j = k;
   701			}
   702			sel->lockorder[j] = c;
   703		}
   704		for(i=sel->ncase; i-->0; ) {
   705			c = sel->lockorder[i];
   706			sel->lockorder[i] = sel->lockorder[0];
   707			j = 0;
   708			for(;;) {
   709				k = j*2+1;
   710				if(k >= i)
   711					break;
   712				if(k+1 < i && sel->lockorder[k] < sel->lockorder[k+1])
   713					k++;
   714				if(c < sel->lockorder[k]) {
   715					sel->lockorder[j] = sel->lockorder[k];
   716					j = k;
   717					continue;
   718				}
   719				break;
   720			}
   721			sel->lockorder[j] = c;
   722		}
   723		/*
   724		for(i=0; i+1<sel->ncase; i++)
   725			if(sel->lockorder[i] > sel->lockorder[i+1]) {
   726				runtime·printf("i=%d %p %p\n", i, sel->lockorder[i], sel->lockorder[i+1]);
   727				runtime·throw("select: broken sort");
   728			}
   729		*/
   730		sellock(sel);
   731	
   732	loop:
   733		// pass 1 - look for something already waiting
   734		dfl = nil;
   735		for(i=0; i<sel->ncase; i++) {
   736			o = sel->pollorder[i];
   737			cas = &sel->scase[o];
   738			c = cas->chan;
   739	
   740			switch(cas->kind) {
   741			case CaseRecv:
   742				if(c->dataqsiz > 0) {
   743					if(c->qcount > 0)
   744						goto asyncrecv;
   745				} else {
   746					sg = dequeue(&c->sendq);
   747					if(sg != nil)
   748						goto syncrecv;
   749				}
   750				if(c->closed)
   751					goto rclose;
   752				break;
   753	
   754			case CaseSend:
   755				if(raceenabled)
   756					runtime·racereadpc(c, cas->pc, chansend);
   757				if(c->closed)
   758					goto sclose;
   759				if(c->dataqsiz > 0) {
   760					if(c->qcount < c->dataqsiz)
   761						goto asyncsend;
   762				} else {
   763					sg = dequeue(&c->recvq);
   764					if(sg != nil)
   765						goto syncsend;
   766				}
   767				break;
   768	
   769			case CaseDefault:
   770				dfl = cas;
   771				break;
   772			}
   773		}
   774	
   775		if(dfl != nil) {
   776			selunlock(sel);
   777			cas = dfl;
   778			goto retc;
   779		}
   780	
   781	
   782		// pass 2 - enqueue on all chans
   783		done = 0;
   784		for(i=0; i<sel->ncase; i++) {
   785			o = sel->pollorder[i];
   786			cas = &sel->scase[o];
   787			c = cas->chan;
   788			sg = &cas->sg;
   789			sg->g = g;
   790			sg->selectdone = &done;
   791	
   792			switch(cas->kind) {
   793			case CaseRecv:
   794				enqueue(&c->recvq, sg);
   795				break;
   796	
   797			case CaseSend:
   798				enqueue(&c->sendq, sg);
   799				break;
   800			}
   801		}
   802	
   803		g->param = nil;
   804		runtime·park(selparkcommit, sel, "select");
   805	
   806		sellock(sel);
   807		sg = g->param;
   808	
   809		// pass 3 - dequeue from unsuccessful chans
   810		// otherwise they stack up on quiet channels
   811		for(i=0; i<sel->ncase; i++) {
   812			cas = &sel->scase[i];
   813			if(cas != (Scase*)sg) {
   814				c = cas->chan;
   815				if(cas->kind == CaseSend)
   816					dequeueg(&c->sendq);
   817				else
   818					dequeueg(&c->recvq);
   819			}
   820		}
   821	
   822		if(sg == nil)
   823			goto loop;
   824	
   825		cas = (Scase*)sg;
   826		c = cas->chan;
   827	
   828		if(c->dataqsiz > 0)
   829			runtime·throw("selectgo: shouldn't happen");
   830	
   831		if(debug)
   832			runtime·printf("wait-return: sel=%p c=%p cas=%p kind=%d\n",
   833				sel, c, cas, cas->kind);
   834	
   835		if(cas->kind == CaseRecv) {
   836			if(cas->receivedp != nil)
   837				*cas->receivedp = true;
   838		}
   839	
   840		if(raceenabled) {
   841			if(cas->kind == CaseRecv && cas->sg.elem != nil)
   842				runtime·racewriteobjectpc(cas->sg.elem, c->elemtype, cas->pc, chanrecv);
   843			else if(cas->kind == CaseSend)
   844				runtime·racereadobjectpc(cas->sg.elem, c->elemtype, cas->pc, chansend);
   845		}
   846	
   847		selunlock(sel);
   848		goto retc;
   849	
   850	asyncrecv:
   851		// can receive from buffer
   852		if(raceenabled) {
   853			if(cas->sg.elem != nil)
   854				runtime·racewriteobjectpc(cas->sg.elem, c->elemtype, cas->pc, chanrecv);
   855			runtime·raceacquire(chanbuf(c, c->recvx));
   856			runtime·racerelease(chanbuf(c, c->recvx));
   857		}
   858		if(cas->receivedp != nil)
   859			*cas->receivedp = true;
   860		if(cas->sg.elem != nil)
   861			c->elemtype->alg->copy(c->elemsize, cas->sg.elem, chanbuf(c, c->recvx));
   862		c->elemtype->alg->copy(c->elemsize, chanbuf(c, c->recvx), nil);
   863		if(++c->recvx == c->dataqsiz)
   864			c->recvx = 0;
   865		c->qcount--;
   866		sg = dequeue(&c->sendq);
   867		if(sg != nil) {
   868			gp = sg->g;
   869			selunlock(sel);
   870			if(sg->releasetime)
   871				sg->releasetime = runtime·cputicks();
   872			runtime·ready(gp);
   873		} else {
   874			selunlock(sel);
   875		}
   876		goto retc;
   877	
   878	asyncsend:
   879		// can send to buffer
   880		if(raceenabled) {
   881			runtime·raceacquire(chanbuf(c, c->sendx));
   882			runtime·racerelease(chanbuf(c, c->sendx));
   883			runtime·racereadobjectpc(cas->sg.elem, c->elemtype, cas->pc, chansend);
   884		}
   885		c->elemtype->alg->copy(c->elemsize, chanbuf(c, c->sendx), cas->sg.elem);
   886		if(++c->sendx == c->dataqsiz)
   887			c->sendx = 0;
   888		c->qcount++;
   889		sg = dequeue(&c->recvq);
   890		if(sg != nil) {
   891			gp = sg->g;
   892			selunlock(sel);
   893			if(sg->releasetime)
   894				sg->releasetime = runtime·cputicks();
   895			runtime·ready(gp);
   896		} else {
   897			selunlock(sel);
   898		}
   899		goto retc;
   900	
   901	syncrecv:
   902		// can receive from sleeping sender (sg)
   903		if(raceenabled) {
   904			if(cas->sg.elem != nil)
   905				runtime·racewriteobjectpc(cas->sg.elem, c->elemtype, cas->pc, chanrecv);
   906			racesync(c, sg);
   907		}
   908		selunlock(sel);
   909		if(debug)
   910			runtime·printf("syncrecv: sel=%p c=%p o=%d\n", sel, c, o);
   911		if(cas->receivedp != nil)
   912			*cas->receivedp = true;
   913		if(cas->sg.elem != nil)
   914			c->elemtype->alg->copy(c->elemsize, cas->sg.elem, sg->elem);
   915		gp = sg->g;
   916		gp->param = sg;
   917		if(sg->releasetime)
   918			sg->releasetime = runtime·cputicks();
   919		runtime·ready(gp);
   920		goto retc;
   921	
   922	rclose:
   923		// read at end of closed channel
   924		selunlock(sel);
   925		if(cas->receivedp != nil)
   926			*cas->receivedp = false;
   927		if(cas->sg.elem != nil)
   928			c->elemtype->alg->copy(c->elemsize, cas->sg.elem, nil);
   929		if(raceenabled)
   930			runtime·raceacquire(c);
   931		goto retc;
   932	
   933	syncsend:
   934		// can send to sleeping receiver (sg)
   935		if(raceenabled) {
   936			runtime·racereadobjectpc(cas->sg.elem, c->elemtype, cas->pc, chansend);
   937			racesync(c, sg);
   938		}
   939		selunlock(sel);
   940		if(debug)
   941			runtime·printf("syncsend: sel=%p c=%p o=%d\n", sel, c, o);
   942		if(sg->elem != nil)
   943			c->elemtype->alg->copy(c->elemsize, sg->elem, cas->sg.elem);
   944		gp = sg->g;
   945		gp->param = sg;
   946		if(sg->releasetime)
   947			sg->releasetime = runtime·cputicks();
   948		runtime·ready(gp);
   949	
   950	retc:
   951		// return pc corresponding to chosen case.
   952		// Set boolean passed during select creation
   953		// (at offset selp + cas->so) to true.
   954		// If cas->so == 0, this is a reflect-driven select and we
   955		// don't need to update the boolean.
   956		pc = cas->pc;
   957		if(cas->so > 0) {
   958			as = (byte*)selp + cas->so;
   959			*as = true;
   960		}
   961		if(cas->sg.releasetime > 0)
   962			runtime·blockevent(cas->sg.releasetime - t0, 2);
   963		runtime·free(sel);
   964		return pc;
   965	
   966	sclose:
   967		// send on closed channel
   968		selunlock(sel);
   969		runtime·panicstring("send on closed channel");
   970		return nil;  // not reached
   971	}

   990	func reflect·rselect(cases Slice) (chosen int, recvOK bool) {
   991		int32 i;
   992		Select *sel;
   993		runtimeSelect* rcase, *rc;
   994	
   995		chosen = -1;
   996		recvOK = false;
   997	
   998		rcase = (runtimeSelect*)cases.array;
   999	
  1000		sel = newselect(cases.len);
  1001		for(i=0; i<cases.len; i++) {
  1002			rc = &rcase[i];
  1003			switch(rc->dir) {
  1004			case SelectDefault:
  1005				selectdefault(sel, (void*)i, 0);
  1006				break;
  1007			case SelectSend:
  1008				if(rc->ch == nil)
  1009					break;
  1010				selectsend(sel, rc->ch, (void*)i, rc->val, 0);
  1011				break;
  1012			case SelectRecv:
  1013				if(rc->ch == nil)
  1014					break;
  1015				selectrecv(sel, rc->ch, (void*)i, rc->val, &recvOK, 0);
  1016				break;
  1017			}
  1018		}
  1019	
  1020		chosen = (intgo)(uintptr)selectgo(&sel);
  1021	}

  1035	static void
  1036	closechan(Hchan *c, void *pc)
  1037	{
  1038		SudoG *sg;
  1039		G* gp;
  1040	
  1041		if(c == nil)
  1042			runtime·panicstring("close of nil channel");
  1043	
  1044		runtime·lock(c);
  1045		if(c->closed) {
  1046			runtime·unlock(c);
  1047			runtime·panicstring("close of closed channel");
  1048		}
  1049	
  1050		if(raceenabled) {
  1051			runtime·racewritepc(c, pc, runtime·closechan);
  1052			runtime·racerelease(c);
  1053		}
  1054	
  1055		c->closed = true;
  1056	
  1057		// release all readers
  1058		for(;;) {
  1059			sg = dequeue(&c->recvq);
  1060			if(sg == nil)
  1061				break;
  1062			gp = sg->g;
  1063			gp->param = nil;
  1064			if(sg->releasetime)
  1065				sg->releasetime = runtime·cputicks();
  1066			runtime·ready(gp);
  1067		}
  1068	
  1069		// release all writers
  1070		for(;;) {
  1071			sg = dequeue(&c->sendq);
  1072			if(sg == nil)
  1073				break;
  1074			gp = sg->g;
  1075			gp->param = nil;
  1076			if(sg->releasetime)
  1077				sg->releasetime = runtime·cputicks();
  1078			runtime·ready(gp);
  1079		}
  1080	
  1081		runtime·unlock(c);
  1082	}

  1098	static SudoG*
  1099	dequeue(WaitQ *q)
  1100	{
  1101		SudoG *sgp;
  1102	
  1103	loop:
  1104		sgp = q->first;
  1105		if(sgp == nil)
  1106			return nil;
  1107		q->first = sgp->link;
  1108	
  1109		// if sgp participates in a select and is already signaled, ignore it
  1110		if(sgp->selectdone != nil) {
  1111			// claim the right to signal
  1112			if(*sgp->selectdone != 0 || !runtime·cas(sgp->selectdone, 0, 1))
  1113				goto loop;
  1114		}
  1115	
  1116		return sgp;
  1117	}

  1119	static void
  1120	dequeueg(WaitQ *q)
  1121	{
  1122		SudoG **l, *sgp, *prevsgp;
  1123	
  1124		prevsgp = nil;
  1125		for(l=&q->first; (sgp=*l) != nil; l=&sgp->link, prevsgp=sgp) {
  1126			if(sgp->g == g) {
  1127				*l = sgp->link;
  1128				if(q->last == sgp)
  1129					q->last = prevsgp;
  1130				break;
  1131			}
  1132		}
  1133	}

  1135	static void
  1136	enqueue(WaitQ *q, SudoG *sgp)
  1137	{
  1138		sgp->link = nil;
  1139		if(q->first == nil) {
  1140			q->first = sgp;
  1141			q->last = sgp;
  1142			return;
  1143		}
  1144		q->last->link = sgp;
  1145		q->last = sgp;
  1146	}

  1149	racesync(Hchan *c, SudoG *sg)
  1150	{
  1151		runtime·racerelease(chanbuf(c, 0));

  1153		runtime·racereleaseg(sg->g, chanbuf(c, 0));
  1154		runtime·raceacquire(chanbuf(c, 0));
  1155	}
