    16	{
    17		docheck = 0,  // check invariants before and after every op.  Slow!!!

    21	static void
    22	check(MapType *t, Hmap *h)
    23	{
    24		uintptr bucket, oldbucket;
    25		Bucket *b;
    26		uintptr i;
    27		uintptr hash;
    28		uintgo cnt;
    29		uint8 top;
    30		bool eq;
    31		byte *k, *v;
    32	
    33		cnt = 0;
    34	
    35		// check buckets
    36		for(bucket = 0; bucket < (uintptr)1 << h->B; bucket++) {
    37			for(b = (Bucket*)(h->buckets + bucket * h->bucketsize); b != nil; b = b->overflow) {
    38				for(i = 0, k = (byte*)b->data, v = k + h->keysize * BUCKETSIZE; i < BUCKETSIZE; i++, k += h->keysize, v += h->valuesize) {
    39					if(b->tophash[i] == Empty)
    40						continue;
    41					if(b->tophash[i] > Empty && b->tophash[i] < MinTopHash)
    42						runtime·throw("evacuated cell in buckets");
    43					cnt++;
    44					t->key->alg->equal(&eq, t->key->size, IK(h, k), IK(h, k));
    45					if(!eq)
    46						continue; // NaN!
    47					hash = h->hash0;
    48					t->key->alg->hash(&hash, t->key->size, IK(h, k));
    49					top = hash >> (8*sizeof(uintptr) - 8);
    50					if(top < MinTopHash)
    51						top += MinTopHash;
    52					if(top != b->tophash[i])
    53						runtime·throw("bad hash");
    54				}
    55			}
    56		}
    57	
    58		// check oldbuckets
    59		if(h->oldbuckets != nil) {
    60			for(oldbucket = 0; oldbucket < (uintptr)1 << (h->B - 1); oldbucket++) {
    61				b = (Bucket*)(h->oldbuckets + oldbucket * h->bucketsize);
    62				for(; b != nil; b = b->overflow) {
    63					for(i = 0, k = (byte*)b->data, v = k + h->keysize * BUCKETSIZE; i < BUCKETSIZE; i++, k += h->keysize, v += h->valuesize) {
    64						if(b->tophash[i] < MinTopHash)
    65							continue;
    66						if(oldbucket < h->nevacuate)
    67							runtime·throw("unevacuated entry in an evacuated bucket");
    68						cnt++;
    69						t->key->alg->equal(&eq, t->key->size, IK(h, k), IK(h, k));
    70						if(!eq)
    71							continue; // NaN!
    72						hash = h->hash0;
    73						t->key->alg->hash(&hash, t->key->size, IK(h, k));
    74						top = hash >> (8*sizeof(uintptr) - 8);
    75						if(top < MinTopHash)
    76							top += MinTopHash;
    77						if(top != b->tophash[i])
    78							runtime·throw("bad hash (old)");
    79					}
    80				}
    81			}
    82		}
    83	
    84		if(cnt != h->count) {
    85			runtime·printf("%D %D\n", (uint64)cnt, (uint64)h->count);
    86			runtime·throw("entries missing");
    87		}
    88	}

    90	static void
    91	hash_init(MapType *t, Hmap *h, uint32 hint)
    92	{
    93		uint8 B;
    94		byte *buckets;
    95		uintptr keysize, valuesize, bucketsize;
    96		uint8 flags;
    97	
    98		flags = 0;
    99	
   100		// figure out how big we have to make everything
   101		keysize = t->key->size;
   102		if(keysize > MAXKEYSIZE) {
   103			flags |= IndirectKey;
   104			keysize = sizeof(byte*);
   105		}
   106		valuesize = t->elem->size;
   107		if(valuesize > MAXVALUESIZE) {
   108			flags |= IndirectValue;
   109			valuesize = sizeof(byte*);
   110		}
   111		bucketsize = offsetof(Bucket, data[0]) + (keysize + valuesize) * BUCKETSIZE;
   112		if(bucketsize != t->bucket->size) {
   113			runtime·printf("runtime: bucketsize=%p but t->bucket->size=%p; t=%S\n", bucketsize, t->bucket->size, *t->string);
   114			runtime·throw("bucketsize wrong");
   115		}
   116	
   117		// invariants we depend on.  We should probably check these at compile time
   118		// somewhere, but for now we'll do it here.
   119		if(t->key->align > BUCKETSIZE)
   120			runtime·throw("key align too big");
   121		if(t->elem->align > BUCKETSIZE)
   122			runtime·throw("value align too big");
   123		if(t->key->size % t->key->align != 0)
   124			runtime·throw("key size not a multiple of key align");
   125		if(t->elem->size % t->elem->align != 0)
   126			runtime·throw("value size not a multiple of value align");
   127		if(BUCKETSIZE < 8)
   128			runtime·throw("bucketsize too small for proper alignment");
   129		if((offsetof(Bucket, data[0]) & (t->key->align-1)) != 0)
   130			runtime·throw("need padding in bucket (key)");
   131		if((offsetof(Bucket, data[0]) & (t->elem->align-1)) != 0)
   132			runtime·throw("need padding in bucket (value)");
   133	
   134		// find size parameter which will hold the requested # of elements
   135		B = 0;
   136		while(hint > BUCKETSIZE && hint > LOAD * ((uintptr)1 << B))
   137			B++;
   138	
   139		// allocate initial hash table
   140		// If hint is large zeroing this memory could take a while.
   141		if(checkgc) mstats.next_gc = mstats.heap_alloc;
   142		if(B == 0) {
   143			// done lazily later.
   144			buckets = nil;
   145		} else {
   146			buckets = runtime·cnewarray(t->bucket, (uintptr)1 << B);
   147		}
   148	
   149		// initialize Hmap
   150		h->count = 0;
   151		h->B = B;
   152		h->flags = flags;
   153		h->keysize = keysize;
   154		h->valuesize = valuesize;
   155		h->bucketsize = bucketsize;
   156		h->hash0 = runtime·fastrand1();
   157		h->buckets = buckets;
   158		h->oldbuckets = nil;
   159		h->nevacuate = 0;
   160		if(docheck)
   161			check(t, h);
   162	}
   163	
   164	// Moves entries in oldbuckets[i] to buckets[i] and buckets[i+2^k].
   165	// We leave the original bucket intact, except for marking the topbits
   166	// entries as evacuated, so that iterators can still iterate through the old buckets.
   167	static void
   168	evacuate(MapType *t, Hmap *h, uintptr oldbucket)
   169	{
   170		Bucket *b;
   171		Bucket *x, *y;
   172		Bucket *newx, *newy;
   173		uintptr xi, yi;
   174		uintptr newbit;
   175		uintptr hash;
   176		uintptr i;
   177		byte *k, *v;
   178		byte *xk, *yk, *xv, *yv;
   179		uint8 top;
   180		bool eq;
   181	
   182		b = (Bucket*)(h->oldbuckets + oldbucket * h->bucketsize);
   183		newbit = (uintptr)1 << (h->B - 1);
   184	
   185		if(!evacuated(b)) {
   186			// TODO: reuse overflow buckets instead of using new ones, if there
   187			// is no iterator using the old buckets.  (If !OldIterator.)
   188	
   189			x = (Bucket*)(h->buckets + oldbucket * h->bucketsize);
   190			y = (Bucket*)(h->buckets + (oldbucket + newbit) * h->bucketsize);
   191			xi = 0;
   192			yi = 0;
   193			xk = (byte*)x->data;
   194			yk = (byte*)y->data;
   195			xv = xk + h->keysize * BUCKETSIZE;
   196			yv = yk + h->keysize * BUCKETSIZE;
   197			for(; b != nil; b = b->overflow) {
   198				for(i = 0, k = (byte*)b->data, v = k + h->keysize * BUCKETSIZE; i < BUCKETSIZE; i++, k += h->keysize, v += h->valuesize) {
   199					top = b->tophash[i];
   200					if(top == Empty) {
   201						b->tophash[i] = EvacuatedEmpty;
   202						continue;
   203					}
   204					if(top < MinTopHash)
   205						runtime·throw("bad state");
   206	
   207					// Compute hash to make our evacuation decision (whether we need
   208					// to send this key/value to bucket x or bucket y).
   209					hash = h->hash0;
   210					t->key->alg->hash(&hash, t->key->size, IK(h, k));
   211					if((h->flags & Iterator) != 0) {
   212						t->key->alg->equal(&eq, t->key->size, IK(h, k), IK(h, k));
   213						if(!eq) {
   214							// If key != key (NaNs), then the hash could be (and probably
   215							// will be) entirely different from the old hash.  Moreover,
   216							// it isn't reproducible.  Reproducibility is required in the
   217							// presence of iterators, as our evacuation decision must
   218							// match whatever decision the iterator made.
   219							// Fortunately, we have the freedom to send these keys either
   220							// way.  Also, tophash is meaningless for these kinds of keys.
   221							// We let the low bit of tophash drive the evacuation decision.
   222							// We recompute a new random tophash for the next level so
   223							// these keys will get evenly distributed across all buckets
   224							// after multiple grows.
   225							if((top & 1) != 0)
   226								hash |= newbit;
   227							else
   228								hash &= ~newbit;
   229							top = hash >> (8*sizeof(uintptr)-8);
   230							if(top < MinTopHash)
   231								top += MinTopHash;
   232						}
   233					}
   234	
   235					if((hash & newbit) == 0) {
   236						b->tophash[i] = EvacuatedX;
   237						if(xi == BUCKETSIZE) {
   238							if(checkgc) mstats.next_gc = mstats.heap_alloc;
   239							newx = runtime·cnew(t->bucket);
   240							x->overflow = newx;
   241							x = newx;
   242							xi = 0;
   243							xk = (byte*)x->data;
   244							xv = xk + h->keysize * BUCKETSIZE;
   245						}
   246						x->tophash[xi] = top;
   247						if((h->flags & IndirectKey) != 0) {
   248							*(byte**)xk = *(byte**)k;               // copy pointer
   249						} else {
   250							t->key->alg->copy(t->key->size, xk, k); // copy value
   251						}
   252						if((h->flags & IndirectValue) != 0) {
   253							*(byte**)xv = *(byte**)v;
   254						} else {
   255							t->elem->alg->copy(t->elem->size, xv, v);
   256						}
   257						xi++;
   258						xk += h->keysize;
   259						xv += h->valuesize;
   260					} else {
   261						b->tophash[i] = EvacuatedY;
   262						if(yi == BUCKETSIZE) {
   263							if(checkgc) mstats.next_gc = mstats.heap_alloc;
   264							newy = runtime·cnew(t->bucket);
   265							y->overflow = newy;
   266							y = newy;
   267							yi = 0;
   268							yk = (byte*)y->data;
   269							yv = yk + h->keysize * BUCKETSIZE;
   270						}
   271						y->tophash[yi] = top;
   272						if((h->flags & IndirectKey) != 0) {
   273							*(byte**)yk = *(byte**)k;
   274						} else {
   275							t->key->alg->copy(t->key->size, yk, k);
   276						}
   277						if((h->flags & IndirectValue) != 0) {
   278							*(byte**)yv = *(byte**)v;
   279						} else {
   280							t->elem->alg->copy(t->elem->size, yv, v);
   281						}
   282						yi++;
   283						yk += h->keysize;
   284						yv += h->valuesize;
   285					}
   286				}
   287			}
   288	
   289			// Unlink the overflow buckets & clear key/value to help GC.
   290			if((h->flags & OldIterator) == 0) {
   291				b = (Bucket*)(h->oldbuckets + oldbucket * h->bucketsize);
   292				b->overflow = nil;
   293				runtime·memclr((byte*)b->data, h->bucketsize - offsetof(Bucket, data[0]));
   294			}
   295		}
   296	
   297		// Advance evacuation mark
   298		if(oldbucket == h->nevacuate) {
   299			h->nevacuate = oldbucket + 1;
   300			if(oldbucket + 1 == newbit) // newbit == # of oldbuckets
   301				// Growing is all done.  Free old main bucket array.
   302				h->oldbuckets = nil;
   303		}
   304		if(docheck)
   305			check(t, h);
   306	}

   308	static void
   309	grow_work(MapType *t, Hmap *h, uintptr bucket)
   310	{
   311		uintptr noldbuckets;
   312	
   313		noldbuckets = (uintptr)1 << (h->B - 1);
   314	
   315		// make sure we evacuate the oldbucket corresponding
   316		// to the bucket we're about to use
   317		evacuate(t, h, bucket & (noldbuckets - 1));
   318	
   319		// evacuate one more oldbucket to make progress on growing
   320		if(h->oldbuckets != nil)
   321			evacuate(t, h, h->nevacuate);
   322	}

   324	static void
   325	hash_grow(MapType *t, Hmap *h)
   326	{
   327		byte *old_buckets;
   328		byte *new_buckets;
   329		uint8 flags;
   330	
   331		// allocate a bigger hash table
   332		if(h->oldbuckets != nil)
   333			runtime·throw("evacuation not done in time");
   334		old_buckets = h->buckets;
   335		if(checkgc) mstats.next_gc = mstats.heap_alloc;
   336		new_buckets = runtime·cnewarray(t->bucket, (uintptr)1 << (h->B + 1));
   337		flags = (h->flags & ~(Iterator | OldIterator));
   338		if((h->flags & Iterator) != 0)
   339			flags |= OldIterator;
   340	
   341		// commit the grow (atomic wrt gc)
   342		h->B++;
   343		h->flags = flags;
   344		h->oldbuckets = old_buckets;
   345		h->buckets = new_buckets;
   346		h->nevacuate = 0;
   347	
   348		// the actual copying of the hash table data is done incrementally
   349		// by grow_work() and evacuate().
   350		if(docheck)
   351			check(t, h);
   352	}

   354	// returns ptr to value associated with key *keyp, or nil if none.
   355	// if it returns non-nil, updates *keyp to point to the currently stored key.
   356	static byte*
   357	hash_lookup(MapType *t, Hmap *h, byte **keyp)
   358	{
   359		void *key;
   360		uintptr hash;
   361		uintptr bucket, oldbucket;
   362		Bucket *b;
   363		uint8 top;
   364		uintptr i;
   365		bool eq;
   366		byte *k, *k2, *v;
   367	
   368		key = *keyp;
   369		if(docheck)
   370			check(t, h);
   371		if(h->count == 0)
   372			return nil;
   373		hash = h->hash0;
   374		t->key->alg->hash(&hash, t->key->size, key);
   375		bucket = hash & (((uintptr)1 << h->B) - 1);
   376		if(h->oldbuckets != nil) {
   377			oldbucket = bucket & (((uintptr)1 << (h->B - 1)) - 1);
   378			b = (Bucket*)(h->oldbuckets + oldbucket * h->bucketsize);
   379			if(evacuated(b)) {
   380				b = (Bucket*)(h->buckets + bucket * h->bucketsize);
   381			}
   382		} else {
   383			b = (Bucket*)(h->buckets + bucket * h->bucketsize);
   384		}
   385		top = hash >> (sizeof(uintptr)*8 - 8);
   386		if(top < MinTopHash)
   387			top += MinTopHash;
   388		do {
   389			for(i = 0, k = (byte*)b->data, v = k + h->keysize * BUCKETSIZE; i < BUCKETSIZE; i++, k += h->keysize, v += h->valuesize) {
   390				if(b->tophash[i] == top) {
   391					k2 = IK(h, k);
   392					t->key->alg->equal(&eq, t->key->size, key, k2);
   393					if(eq) {
   394						*keyp = k2;
   395						return IV(h, v);
   396					}
   397				}
   398			}
   399			b = b->overflow;
   400		} while(b != nil);
   401		return nil;
   402	}

   474	static void
   475	hash_insert(MapType *t, Hmap *h, void *key, void *value)
   476	{
   477		uintptr hash;
   478		uintptr bucket;
   479		uintptr i;
   480		bool eq;
   481		Bucket *b;
   482		Bucket *newb;
   483		uint8 *inserti;
   484		byte *insertk, *insertv;
   485		uint8 top;
   486		byte *k, *v;
   487		byte *kmem, *vmem;
   488	
   489		if(docheck)
   490			check(t, h);
   491		hash = h->hash0;
   492		t->key->alg->hash(&hash, t->key->size, key);
   493		if(h->buckets == nil)
   494			h->buckets = runtime·cnewarray(t->bucket, 1);
   495	
   496	 again:
   497		bucket = hash & (((uintptr)1 << h->B) - 1);
   498		if(h->oldbuckets != nil)
   499			grow_work(t, h, bucket);
   500		b = (Bucket*)(h->buckets + bucket * h->bucketsize);
   501		top = hash >> (sizeof(uintptr)*8 - 8);
   502		if(top < MinTopHash)
   503			top += MinTopHash;
   504		inserti = nil;
   505		insertk = nil;
   506		insertv = nil;
   507		while(true) {
   508			for(i = 0, k = (byte*)b->data, v = k + h->keysize * BUCKETSIZE; i < BUCKETSIZE; i++, k += h->keysize, v += h->valuesize) {
   509				if(b->tophash[i] != top) {
   510					if(b->tophash[i] == Empty && inserti == nil) {
   511						inserti = &b->tophash[i];
   512						insertk = k;
   513						insertv = v;
   514					}
   515					continue;
   516				}
   517				t->key->alg->equal(&eq, t->key->size, key, IK(h, k));
   518				if(!eq)
   519					continue;
   520				// already have a mapping for key.  Update it.
   521				t->key->alg->copy(t->key->size, IK(h, k), key); // Need to update key for keys which are distinct but equal (e.g. +0.0 and -0.0)
   522				t->elem->alg->copy(t->elem->size, IV(h, v), value);
   523				if(docheck)
   524					check(t, h);
   525				return;
   526			}
   527			if(b->overflow == nil)
   528				break;
   529			b = b->overflow;
   530		}
   531	
   532		// did not find mapping for key.  Allocate new cell & add entry.
   533		if(h->count >= LOAD * ((uintptr)1 << h->B) && h->count >= BUCKETSIZE) {
   534			hash_grow(t, h);
   535			goto again; // Growing the table invalidates everything, so try again
   536		}
   537	
   538		if(inserti == nil) {
   539			// all current buckets are full, allocate a new one.
   540			if(checkgc) mstats.next_gc = mstats.heap_alloc;
   541			newb = runtime·cnew(t->bucket);
   542			b->overflow = newb;
   543			inserti = newb->tophash;
   544			insertk = (byte*)newb->data;
   545			insertv = insertk + h->keysize * BUCKETSIZE;
   546		}
   547	
   548		// store new key/value at insert position
   549		if((h->flags & IndirectKey) != 0) {
   550			if(checkgc) mstats.next_gc = mstats.heap_alloc;
   551			kmem = runtime·cnew(t->key);
   552			*(byte**)insertk = kmem;
   553			insertk = kmem;
   554		}
   555		if((h->flags & IndirectValue) != 0) {
   556			if(checkgc) mstats.next_gc = mstats.heap_alloc;
   557			vmem = runtime·cnew(t->elem);
   558			*(byte**)insertv = vmem;
   559			insertv = vmem;
   560		}
   561		t->key->alg->copy(t->key->size, insertk, key);
   562		t->elem->alg->copy(t->elem->size, insertv, value);
   563		*inserti = top;
   564		h->count++;
   565		if(docheck)
   566			check(t, h);
   567	}
   568	
   569	static void
   570	hash_remove(MapType *t, Hmap *h, void *key)
   571	{
   572		uintptr hash;
   573		uintptr bucket;
   574		Bucket *b;
   575		uint8 top;
   576		uintptr i;
   577		byte *k, *v;
   578		bool eq;
   579		
   580		if(docheck)
   581			check(t, h);
   582		if(h->count == 0)
   583			return;
   584		hash = h->hash0;
   585		t->key->alg->hash(&hash, t->key->size, key);
   586		bucket = hash & (((uintptr)1 << h->B) - 1);
   587		if(h->oldbuckets != nil)
   588			grow_work(t, h, bucket);
   589		b = (Bucket*)(h->buckets + bucket * h->bucketsize);
   590		top = hash >> (sizeof(uintptr)*8 - 8);
   591		if(top < MinTopHash)
   592			top += MinTopHash;
   593		do {
   594			for(i = 0, k = (byte*)b->data, v = k + h->keysize * BUCKETSIZE; i < BUCKETSIZE; i++, k += h->keysize, v += h->valuesize) {
   595				if(b->tophash[i] != top)
   596					continue;
   597				t->key->alg->equal(&eq, t->key->size, key, IK(h, k));
   598				if(!eq)
   599					continue;
   600	
   601				if((h->flags & IndirectKey) != 0) {
   602					*(byte**)k = nil;
   603				} else {
   604					t->key->alg->copy(t->key->size, k, nil);
   605				}
   606				if((h->flags & IndirectValue) != 0) {
   607					*(byte**)v = nil;
   608				} else {
   609					t->elem->alg->copy(t->elem->size, v, nil);
   610				}
   611	
   612				b->tophash[i] = Empty;
   613				h->count--;
   614				
   615				// TODO: consolidate buckets if they are mostly empty
   616				// can only consolidate if there are no live iterators at this size.
   617				if(docheck)
   618					check(t, h);
   619				return;
   620			}
   621			b = b->overflow;
   622		} while(b != nil);
   623	}

   627	// iterator state:
   628	// bucket: the current bucket ID
   629	// b: the current Bucket in the chain
   630	// i: the next offset to check in the current bucket
   631	static void
   632	hash_iter_init(MapType *t, Hmap *h, Hiter *it)
   633	{
   634		uint32 old;
   635	
   636		if(sizeof(Hiter) / sizeof(uintptr) != 10) {
   637			runtime·throw("hash_iter size incorrect"); // see ../../cmd/gc/reflect.c
   638		}
   639		it->t = t;
   640		it->h = h;
   641	
   642		// grab snapshot of bucket state
   643		it->B = h->B;
   644		it->buckets = h->buckets;
   645	
   646		// iterator state
   647		it->bucket = 0;
   648		it->offset = runtime·fastrand1() & (BUCKETSIZE - 1);
   649		it->done = false;
   650		it->bptr = nil;
   651	
   652		// Remember we have an iterator.
   653		// Can run concurrently with another hash_iter_init().
   654		for(;;) {
   655			old = h->flags;
   656			if((old&(Iterator|OldIterator)) == (Iterator|OldIterator))
   657				break;
   658			if(runtime·cas(&h->flags, old, old|Iterator|OldIterator))
   659				break;
   660		}
   661	
   662		if(h->buckets == nil) {
   663			// Empty map. Force next hash_next to exit without
   664			// evaluating h->bucket.
   665			it->done = true;
   666		}
   667	}
   668	
   669	// initializes it->key and it->value to the next key/value pair
   670	// in the iteration, or nil if we've reached the end.
   671	static void
   672	hash_next(Hiter *it)
   673	{
   674		Hmap *h;
   675		MapType *t;
   676		uintptr bucket, oldbucket;
   677		uintptr hash;
   678		Bucket *b;
   679		uintptr i, offi;
   680		intptr check_bucket;
   681		bool eq;
   682		byte *k, *v;
   683		byte *rk, *rv;
   684	
   685		h = it->h;
   686		t = it->t;
   687		bucket = it->bucket;
   688		b = it->bptr;
   689		i = it->i;
   690		check_bucket = it->check_bucket;
   691	
   692	next:
   693		if(b == nil) {
   694			if(it->done) {
   695				// end of iteration
   696				it->key = nil;
   697				it->value = nil;
   698				return;
   699			}
   700			if(h->oldbuckets != nil && it->B == h->B) {
   701				// Iterator was started in the middle of a grow, and the grow isn't done yet.
   702				// If the bucket we're looking at hasn't been filled in yet (i.e. the old
   703				// bucket hasn't been evacuated) then we need to iterate through the old
   704				// bucket and only return the ones that will be migrated to this bucket.
   705				oldbucket = bucket & (((uintptr)1 << (it->B - 1)) - 1);
   706				b = (Bucket*)(h->oldbuckets + oldbucket * h->bucketsize);
   707				if(!evacuated(b)) {
   708					check_bucket = bucket;
   709				} else {
   710					b = (Bucket*)(it->buckets + bucket * h->bucketsize);
   711					check_bucket = -1;
   712				}
   713			} else {
   714				b = (Bucket*)(it->buckets + bucket * h->bucketsize);
   715				check_bucket = -1;
   716			}
   717			bucket++;
   718			if(bucket == ((uintptr)1 << it->B)) {
   719				bucket = 0;
   720				it->done = true;
   721			}
   722			i = 0;
   723		}
   724		for(; i < BUCKETSIZE; i++) {
   725			offi = (i + it->offset) & (BUCKETSIZE - 1);
   726			k = (byte*)b->data + h->keysize * offi;
   727			v = (byte*)b->data + h->keysize * BUCKETSIZE + h->valuesize * offi;
   728			if(b->tophash[offi] != Empty && b->tophash[offi] != EvacuatedEmpty) {
   729				if(check_bucket >= 0) {
   730					// Special case: iterator was started during a grow and the
   731					// grow is not done yet.  We're working on a bucket whose
   732					// oldbucket has not been evacuated yet.  Or at least, it wasn't
   733					// evacuated when we started the bucket.  So we're iterating
   734					// through the oldbucket, skipping any keys that will go
   735					// to the other new bucket (each oldbucket expands to two
   736					// buckets during a grow).
   737					t->key->alg->equal(&eq, t->key->size, IK(h, k), IK(h, k));
   738					if(eq) {
   739						// If the item in the oldbucket is not destined for
   740						// the current new bucket in the iteration, skip it.
   741						hash = h->hash0;
   742						t->key->alg->hash(&hash, t->key->size, IK(h, k));
   743						if((hash & (((uintptr)1 << it->B) - 1)) != check_bucket) {
   744							continue;
   745						}
   746					} else {
   747						// Hash isn't repeatable if k != k (NaNs).  We need a
   748						// repeatable and randomish choice of which direction
   749						// to send NaNs during evacuation.  We'll use the low
   750						// bit of tophash to decide which way NaNs go.
   751						// NOTE: this case is why we need two evacuate tophash
   752						// values, evacuatedX and evacuatedY, that differ in
   753						// their low bit.
   754						if(check_bucket >> (it->B - 1) != (b->tophash[offi] & 1)) {
   755							continue;
   756						}
   757					}
   758				}
   759				if(b->tophash[offi] != EvacuatedX && b->tophash[offi] != EvacuatedY) {
   760					// this is the golden data, we can return it.
   761					it->key = IK(h, k);
   762					it->value = IV(h, v);
   763				} else {
   764					// The hash table has grown since the iterator was started.
   765					// The golden data for this key is now somewhere else.
   766					t->key->alg->equal(&eq, t->key->size, IK(h, k), IK(h, k));
   767					if(eq) {
   768						// Check the current hash table for the data.
   769						// This code handles the case where the key
   770						// has been deleted, updated, or deleted and reinserted.
   771						// NOTE: we need to regrab the key as it has potentially been
   772						// updated to an equal() but not identical key (e.g. +0.0 vs -0.0).
   773						rk = IK(h, k);
   774						rv = hash_lookup(t, it->h, &rk);
   775						if(rv == nil)
   776							continue; // key has been deleted
   777						it->key = rk;
   778						it->value = rv;
   779					} else {
   780						// if key!=key then the entry can't be deleted or
   781						// updated, so we can just return it.  That's lucky for
   782						// us because when key!=key we can't look it up
   783						// successfully in the current table.
   784						it->key = IK(h, k);
   785						it->value = IV(h, v);
   786					}
   787				}
   788				it->bucket = bucket;
   789				it->bptr = b;
   790				it->i = i + 1;
   791				it->check_bucket = check_bucket;
   792				return;
   793			}
   794		}
   795		b = b->overflow;
   796		i = 0;
   797		goto next;
   798	}
