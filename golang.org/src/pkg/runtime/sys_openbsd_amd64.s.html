    14	// int64 tfork(void *param, uintptr psize, M *mp, G *gp, void (*fn)(void));
    15	TEXT runtime·tfork(SB),NOSPLIT,$32
    16	
    17		// Copy mp, gp and fn off parent stack for use by child.
    18		MOVQ	mm+16(FP), R8
    19		MOVQ	gg+24(FP), R9
    20		MOVQ	fn+32(FP), R12
    21	
    22		MOVQ	param+0(FP), DI
    23		MOVQ	psize+8(FP), SI
    24		MOVL	$8, AX			// sys___tfork
    25		SYSCALL
    26	
    27		// Return if tfork syscall failed.
    28		JCC	3(PC)
    29		NEGQ	AX
    30		RET
    31	
    32		// In parent, return.
    33		CMPL	AX, $0
    34		JEQ	2(PC)
    35		RET
    36	
    37		// Set FS to point at m->tls.
    38		LEAQ	m_tls(R8), DI
    39		CALL	runtime·settls(SB)
    40	
    41		// In child, set up new stack.
    42		get_tls(CX)
    43		MOVQ	R8, m(CX)
    44		MOVQ	R9, g(CX)
    45		CALL	runtime·stackcheck(SB)
    46	
    47		// Call fn
    48		CALL	R12
    49	
    50		// It shouldn't return.  If it does, exit
    51		MOVQ	$0, DI			// arg 1 - notdead
    52		MOVL	$302, AX		// sys___threxit
    53		SYSCALL
    54		JMP	-3(PC)			// keep exiting
    55

    58		SYSCALL
    59		RET

   156	// func now() (sec int64, nsec int32)
   157	TEXT time·now(SB), NOSPLIT, $32
   158		MOVQ	$0, DI			// arg 1 - clock_id
   159		LEAQ	8(SP), SI		// arg 2 - tp
   160		MOVL	$87, AX			// sys_clock_gettime
   161		SYSCALL
   162		MOVQ	8(SP), AX		// sec
   163		MOVQ	16(SP), DX		// nsec
   164	
   165		// sec is in AX, nsec in DX
   166		MOVQ	AX, sec+0(FP)
   167		MOVL	DX, nsec+8(FP)
   168		RET

   170	TEXT runtime·nanotime(SB),NOSPLIT,$24
   171		MOVQ	CLOCK_MONOTONIC, DI	// arg 1 - clock_id
   172		LEAQ	8(SP), SI		// arg 2 - tp
   173		MOVL	$87, AX			// sys_clock_gettime
   174		SYSCALL
   175		MOVQ	8(SP), AX		// sec
   176		MOVQ	16(SP), DX		// nsec
   177	
   178		// sec is in AX, nsec in DX

   204	TEXT runtime·sigtramp(SB),NOSPLIT,$64
   205		get_tls(BX)
   206		
   207		// check that m exists
   208		MOVQ	m(BX), BP
   209		CMPQ	BP, $0
   210		JNE	5(PC)
   211		MOVQ	DI, 0(SP)
   212		MOVQ	$runtime·badsignal(SB), AX
   213		CALL	AX
   214		RET
   215	
   216		// save g
   217		MOVQ	g(BX), R10
   218		MOVQ	R10, 40(SP)
   219		
   220		// g = m->signal
   221		MOVQ	m_gsignal(BP), BP
   222		MOVQ	BP, g(BX)
   223		
   224		MOVQ	DI, 0(SP)
   225		MOVQ	SI, 8(SP)
   226		MOVQ	DX, 16(SP)
   227		MOVQ	R10, 24(SP)
   228		
   229		CALL	runtime·sighandler(SB)
   230	
   231		// restore g
   232		get_tls(BX)
   233		MOVQ	40(SP), R10
   234		MOVQ	R10, g(BX)
   235		RET
