    97	/*
    98	 * Per-CPU declaration.
    99	 *
   100	 * "extern register" is a special storage class implemented by 6c, 8c, etc.
   101	 * On the ARM, it is an actual register; elsewhere it is a slot in thread-
   102	 * local storage indexed by a pseudo-register TLS. See zasmhdr in
   103	 * src/cmd/dist/buildruntime.c for details, and be aware that the linker may
   104	 * make further OS-specific changes to the compiler's output. For example,
   105	 * 6l/linux rewrites 0(TLS) as -16(FS).
   106	 *
   107	 * Every C file linked into a Go program must include runtime.h so that the
   108	 * C compiler (6c, 8c, etc.) knows to avoid other uses of these dedicated
   109	 * registers. The Go compiler (6g, 8g, etc.) knows to avoid them.
   110	 */
   111	extern	register	G*	g;
   112	extern	register	M*	m;

   408	// The m->locked word holds two pieces of state counting active calls to LockOSThread/lockOSThread.
   409	// The low bit (LockExternal) is a boolean reporting whether any LockOSThread call is active.
   410	// External locks are not recursive; a second lock is silently ignored.
   411	// The upper bits of m->lockedcount record the nesting depth of calls to lockOSThread
   412	// (counting up by LockInternal), popped by unlockOSThread (counting down by LockInternal).
   413	// Internal locks can be recursive. For instance, a lock for cgo can occur while the main
   414	// goroutine is holding the lock during the initialization phase.
   415	enum
   416	{
   417		LockExternal = 1,
   418		LockInternal = 2,
   419	};

   995	/*
   996	 * sleep and wakeup on one-time events.
   997	 * before any calls to notesleep or notewakeup,
   998	 * must call noteclear to initialize the Note.
   999	 * then, exactly one thread can call notesleep
  1000	 * and exactly one thread can call notewakeup (once).
  1001	 * once notewakeup has been called, the notesleep
  1002	 * will return.  future notesleep will return immediately.
  1003	 * subsequent noteclear must be called only after
  1004	 * previous notesleep has returned, e.g. it's disallowed
  1005	 * to call noteclear straight after notewakeup.
  1006	 *
  1007	 * notetsleep is like notesleep but wakes up after
  1008	 * a given number of nanoseconds even if the event
  1009	 * has not yet happened.  if a goroutine uses notetsleep to
  1010	 * wake up early, it must wait to call noteclear until it
  1011	 * can be sure that no other goroutine is calling
  1012	 * notewakeup.
  1013	 *
  1014	 * notesleep/notetsleep are generally called on g0,
  1015	 * notetsleepg is similar to notetsleep but is called on user g.
