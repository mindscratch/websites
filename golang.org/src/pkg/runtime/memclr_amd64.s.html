     9	// void runtime·memclr(void*, uintptr)
    10	TEXT runtime·memclr(SB), NOSPLIT, $0-16
    11		MOVQ	ptr+0(FP), DI
    12		MOVQ	n+8(FP), BX
    13		XORQ	AX, AX
    14	
    15		// MOVOU seems always faster than REP STOSQ.
    16	clr_tail:
    17		TESTQ	BX, BX
    18		JEQ	clr_0
    19		CMPQ	BX, $2
    20		JBE	clr_1or2
    21		CMPQ	BX, $4
    22		JBE	clr_3or4
    23		CMPQ	BX, $8
    24		JBE	clr_5through8
    25		CMPQ	BX, $16
    26		JBE	clr_9through16
    27		PXOR	X0, X0
    28		CMPQ	BX, $32
    29		JBE	clr_17through32
    30		CMPQ	BX, $64
    31		JBE	clr_33through64
    32		CMPQ	BX, $128
    33		JBE	clr_65through128
    34		CMPQ	BX, $256
    35		JBE	clr_129through256
    36		// TODO: use branch table and BSR to make this just a single dispatch
    37		// TODO: for really big clears, use MOVNTDQ.
    38	
    39	clr_loop:
    40		MOVOU	X0, 0(DI)
    41		MOVOU	X0, 16(DI)
    42		MOVOU	X0, 32(DI)
    43		MOVOU	X0, 48(DI)
    44		MOVOU	X0, 64(DI)
    45		MOVOU	X0, 80(DI)
    46		MOVOU	X0, 96(DI)
    47		MOVOU	X0, 112(DI)
    48		MOVOU	X0, 128(DI)
    49		MOVOU	X0, 144(DI)
    50		MOVOU	X0, 160(DI)
    51		MOVOU	X0, 176(DI)
    52		MOVOU	X0, 192(DI)
    53		MOVOU	X0, 208(DI)
    54		MOVOU	X0, 224(DI)
    55		MOVOU	X0, 240(DI)
    56		SUBQ	$256, BX
    57		ADDQ	$256, DI
    58		CMPQ	BX, $256
    59		JAE	clr_loop
    60		JMP	clr_tail
    61	
    62	clr_1or2:
    63		MOVB	AX, (DI)
    64		MOVB	AX, -1(DI)(BX*1)
    65	clr_0:
    66		RET
    67	clr_3or4:
    68		MOVW	AX, (DI)
    69		MOVW	AX, -2(DI)(BX*1)
    70		RET
    71	clr_5through8:
    72		MOVL	AX, (DI)
    73		MOVL	AX, -4(DI)(BX*1)
    74		RET
    75	clr_9through16:
    76		MOVQ	AX, (DI)
    77		MOVQ	AX, -8(DI)(BX*1)
    78		RET
    79	clr_17through32:
    80		MOVOU	X0, (DI)
    81		MOVOU	X0, -16(DI)(BX*1)
    82		RET
    83	clr_33through64:
    84		MOVOU	X0, (DI)
    85		MOVOU	X0, 16(DI)
    86		MOVOU	X0, -32(DI)(BX*1)
    87		MOVOU	X0, -16(DI)(BX*1)
    88		RET
    89	clr_65through128:
    90		MOVOU	X0, (DI)
    91		MOVOU	X0, 16(DI)
    92		MOVOU	X0, 32(DI)
    93		MOVOU	X0, 48(DI)
    94		MOVOU	X0, -64(DI)(BX*1)
    95		MOVOU	X0, -48(DI)(BX*1)
    96		MOVOU	X0, -32(DI)(BX*1)
    97		MOVOU	X0, -16(DI)(BX*1)
    98		RET
    99	clr_129through256:
   100		MOVOU	X0, (DI)
   101		MOVOU	X0, 16(DI)
   102		MOVOU	X0, 32(DI)
   103		MOVOU	X0, 48(DI)
   104		MOVOU	X0, 64(DI)
   105		MOVOU	X0, 80(DI)
   106		MOVOU	X0, 96(DI)
   107		MOVOU	X0, 112(DI)
   108		MOVOU	X0, -128(DI)(BX*1)
   109		MOVOU	X0, -112(DI)(BX*1)
   110		MOVOU	X0, -96(DI)(BX*1)
   111		MOVOU	X0, -80(DI)(BX*1)
   112		MOVOU	X0, -64(DI)(BX*1)
   113		MOVOU	X0, -48(DI)(BX*1)
