    12	// A Dialer contains options for connecting to an address.
    13	//
    14	// The zero value for each field is equivalent to dialing
    15	// without that option. Dialing with the zero value of Dialer
    16	// is therefore equivalent to just calling the Dial function.
    17	type Dialer struct {
    18		// Timeout is the maximum amount of time a dial will wait for
    19		// a connect to complete. If Deadline is also set, it may fail
    20		// earlier.
    21		//
    22		// The default is no timeout.
    23		//
    24		// With or without a timeout, the operating system may impose
    25		// its own earlier timeout. For instance, TCP timeouts are
    26		// often around 3 minutes.
    27		Timeout time.Duration
    28	
    29		// Deadline is the absolute point in time after which dials
    30		// will fail. If Timeout is set, it may fail earlier.
    31		// Zero means no deadline, or dependent on the operating system
    32		// as with the Timeout option.
    33		Deadline time.Time
    34	
    35		// LocalAddr is the local address to use when dialing an
    36		// address. The address must be of a compatible type for the
    37		// network being dialed.
    38		// If nil, a local address is automatically chosen.
    39		LocalAddr Addr
    40	
    41		// DualStack allows a single dial to attempt to establish
    42		// multiple IPv4 and IPv6 connections and to return the first
    43		// established connection when the network is "tcp" and the
    44		// destination is a host name that has multiple address family
    45		// DNS records.
    46		DualStack bool
    47	
    48		// KeepAlive specifies the keep-alive period for an active
    49		// network connection.
    50		// If zero, keep-alives are not enabled. Network protocols
    51		// that do not support keep-alives ignore this field.
    52		KeepAlive time.Duration
    53	}
    54	
    55	// Return either now+Timeout or Deadline, whichever comes first.
    56	// Or zero, if neither is set.
    57	func (d *Dialer) deadline() time.Time {
    58		if d.Timeout == 0 {
    59			return d.Deadline
    60		}
    61		timeoutDeadline := time.Now().Add(d.Timeout)
    62		if d.Deadline.IsZero() || timeoutDeadline.Before(d.Deadline) {
    63			return timeoutDeadline
    64		} else {
    65			return d.Deadline
    66		}
    67	}

    69	func parseNetwork(net string) (afnet string, proto int, err error) {
    70		i := last(net, ':')
    71		if i < 0 { // no colon
    72			switch net {
    73			case "tcp", "tcp4", "tcp6":
    74			case "udp", "udp4", "udp6":
    75			case "ip", "ip4", "ip6":
    76			case "unix", "unixgram", "unixpacket":
    77			default:
    78				return "", 0, UnknownNetworkError(net)
    79			}
    80			return net, 0, nil
    81		}
    82		afnet = net[:i]
    83		switch afnet {
    84		case "ip", "ip4", "ip6":
    85			protostr := net[i+1:]
    86			proto, i, ok := dtoi(protostr, 0)
    87			if !ok || i != len(protostr) {
    88				proto, err = lookupProtocol(protostr)
    89				if err != nil {
    90					return "", 0, err
    91				}
    92			}
    93			return afnet, proto, nil
    94		}
    95		return "", 0, UnknownNetworkError(net)
    96	}

    98	func resolveAddr(op, net, addr string, deadline time.Time) (netaddr, error) {
    99		afnet, _, err := parseNetwork(net)
   100		if err != nil {
   101			return nil, err
   102		}
   103		if op == "dial" && addr == "" {
   104			return nil, errMissingAddress
   105		}
   106		switch afnet {
   107		case "unix", "unixgram", "unixpacket":
   108			return ResolveUnixAddr(afnet, addr)
   109		}
   110		return resolveInternetAddr(afnet, addr, deadline)
   111	}

   113	// Dial connects to the address on the named network.
   114	//
   115	// Known networks are "tcp", "tcp4" (IPv4-only), "tcp6" (IPv6-only),
   116	// "udp", "udp4" (IPv4-only), "udp6" (IPv6-only), "ip", "ip4"
   117	// (IPv4-only), "ip6" (IPv6-only), "unix", "unixgram" and
   118	// "unixpacket".
   119	//
   120	// For TCP and UDP networks, addresses have the form host:port.
   121	// If host is a literal IPv6 address or host name, it must be enclosed
   122	// in square brackets as in "[::1]:80", "[ipv6-host]:http" or
   123	// "[ipv6-host%zone]:80".
   124	// The functions JoinHostPort and SplitHostPort manipulate addresses
   125	// in this form.
   126	//
   127	// Examples:
   128	//	Dial("tcp", "12.34.56.78:80")
   129	//	Dial("tcp", "google.com:http")
   130	//	Dial("tcp", "[2001:db8::1]:http")
   131	//	Dial("tcp", "[fe80::1%lo0]:80")
   132	//
   133	// For IP networks, the network must be "ip", "ip4" or "ip6" followed
   134	// by a colon and a protocol number or name and the addr must be a
   135	// literal IP address.
   136	//
   137	// Examples:
   138	//	Dial("ip4:1", "127.0.0.1")
   139	//	Dial("ip6:ospf", "::1")
   140	//
   141	// For Unix networks, the address must be a file system path.
   142	func Dial(network, address string) (Conn, error) {
   143		var d Dialer
   144		return d.Dial(network, address)
   145	}

   147	// DialTimeout acts like Dial but takes a timeout.
   148	// The timeout includes name resolution, if required.
   149	func DialTimeout(network, address string, timeout time.Duration) (Conn, error) {
   150		d := Dialer{Timeout: timeout}
   151		return d.Dial(network, address)

   154	// Dial connects to the address on the named network.
   155	//
   156	// See func Dial for a description of the network and address
   157	// parameters.
   158	func (d *Dialer) Dial(network, address string) (Conn, error) {
   159		ra, err := resolveAddr("dial", network, address, d.deadline())
   160		if err != nil {
   161			return nil, &OpError{Op: "dial", Net: network, Addr: nil, Err: err}
   162		}
   163		dialer := func(deadline time.Time) (Conn, error) {
   164			return dialSingle(network, address, d.LocalAddr, ra.toAddr(), deadline)
   165		}
   166		if ras, ok := ra.(addrList); ok && d.DualStack && network == "tcp" {
   167			dialer = func(deadline time.Time) (Conn, error) {
   168				return dialMulti(network, address, d.LocalAddr, ras, deadline)
   169			}
   170		}
   171		c, err := dial(network, ra.toAddr(), dialer, d.deadline())
   172		if d.KeepAlive > 0 && err == nil {
   173			if tc, ok := c.(*TCPConn); ok {
   174				tc.SetKeepAlive(true)
   175				tc.SetKeepAlivePeriod(d.KeepAlive)
   176				testHookSetKeepAlive()
   177			}
   178		}
   179		return c, err
   180	}

   184	// dialMulti attempts to establish connections to each destination of
   185	// the list of addresses. It will return the first established
   186	// connection and close the other connections. Otherwise it returns
   187	// error on the last attempt.
   188	func dialMulti(net, addr string, la Addr, ras addrList, deadline time.Time) (Conn, error) {
   189		type racer struct {
   190			Conn
   191			error
   192		}
   193		// Sig controls the flow of dial results on lane. It passes a
   194		// token to the next racer and also indicates the end of flow
   195		// by using closed channel.
   196		sig := make(chan bool, 1)
   197		lane := make(chan racer, 1)
   198		for _, ra := range ras {
   199			go func(ra Addr) {
   200				c, err := dialSingle(net, addr, la, ra, deadline)
   201				if _, ok := <-sig; ok {
   202					lane <- racer{c, err}
   203				} else if err == nil {
   204					// We have to return the resources
   205					// that belong to the other
   206					// connections here for avoiding
   207					// unnecessary resource starvation.
   208					c.Close()
   209				}
   210			}(ra.toAddr())
   211		}
   212		defer close(sig)
   213		lastErr := errTimeout
   214		nracers := len(ras)
   215		for nracers > 0 {
   216			sig <- true
   217			select {
   218			case racer := <-lane:
   219				if racer.error == nil {
   220					return racer.Conn, nil
   221				}
   222				lastErr = racer.error
   223				nracers--
   224			}
   225		}
   226		return nil, lastErr
   227	}

   229	// dialSingle attempts to establish and returns a single connection to
   230	// the destination address.
   231	func dialSingle(net, addr string, la, ra Addr, deadline time.Time) (c Conn, err error) {
   232		if la != nil && la.Network() != ra.Network() {
   233			return nil, &OpError{Op: "dial", Net: net, Addr: ra, Err: errors.New("mismatched local address type " + la.Network())}
   234		}
   235		switch ra := ra.(type) {
   236		case *TCPAddr:
   237			la, _ := la.(*TCPAddr)
   238			c, err = dialTCP(net, la, ra, deadline)
   239		case *UDPAddr:
   240			la, _ := la.(*UDPAddr)
   241			c, err = dialUDP(net, la, ra, deadline)
   242		case *IPAddr:
   243			la, _ := la.(*IPAddr)
   244			c, err = dialIP(net, la, ra, deadline)
   245		case *UnixAddr:
   246			la, _ := la.(*UnixAddr)
   247			c, err = dialUnix(net, la, ra, deadline)
   248		default:
   249			return nil, &OpError{Op: "dial", Net: net, Addr: ra, Err: &AddrError{Err: "unexpected address type", Addr: addr}}
   250		}
   251		if err != nil {
   252			return nil, err // c is non-nil interface containing nil pointer
   253		}
   254		return c, nil
   255	}

   257	// Listen announces on the local network address laddr.
   258	// The network net must be a stream-oriented network: "tcp", "tcp4",
   259	// "tcp6", "unix" or "unixpacket".
   260	// See Dial for the syntax of laddr.
   261	func Listen(net, laddr string) (Listener, error) {
   262		la, err := resolveAddr("listen", net, laddr, noDeadline)
   263		if err != nil {
   264			return nil, &OpError{Op: "listen", Net: net, Addr: nil, Err: err}
   265		}
   266		var l Listener
   267		switch la := la.toAddr().(type) {
   268		case *TCPAddr:
   269			l, err = ListenTCP(net, la)
   270		case *UnixAddr:
   271			l, err = ListenUnix(net, la)
   272		default:
   273			return nil, &OpError{Op: "listen", Net: net, Addr: la, Err: &AddrError{Err: "unexpected address type", Addr: laddr}}
   274		}
   275		if err != nil {
   276			return nil, err // l is non-nil interface containing nil pointer
   277		}
   278		return l, nil
   279	}
