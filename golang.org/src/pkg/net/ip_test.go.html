    31	func TestParseIP(t *testing.T) {
    32		for _, tt := range parseIPTests {
    33			if out := ParseIP(tt.in); !reflect.DeepEqual(out, tt.out) {
    34				t.Errorf("ParseIP(%q) = %v, want %v", tt.in, out, tt.out)
    35			}
    36			if tt.in == "" {
    37				// Tested in TestMarshalEmptyIP below.
    38				continue
    39			}
    40			var out IP
    41			if err := out.UnmarshalText([]byte(tt.in)); !reflect.DeepEqual(out, tt.out) || (tt.out == nil) != (err != nil) {
    42				t.Errorf("IP.UnmarshalText(%q) = %v, %v, want %v", tt.in, out, err, tt.out)
    43			}
    44		}

    48	func TestMarshalEmptyIP(t *testing.T) {
    49		for _, in := range [][]byte{nil, []byte("")} {
    50			var out = IP{1, 2, 3, 4}
    51			if err := out.UnmarshalText(in); err != nil || out != nil {
    52				t.Errorf("UnmarshalText(%v) = %v, %v; want nil, nil", in, out, err)
    53			}
    54		}
    55		var ip IP
    56		got, err := ip.MarshalText()
    57		if err != nil {
    58			t.Fatal(err)
    59		}
    60		if !reflect.DeepEqual(got, []byte("")) {
    61			t.Errorf(`got %#v, want []byte("")`, got)
    62		}
    63	}
