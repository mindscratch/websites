   169	// to a test listener with various address families, different
   170	// listening address and same port.
   171	func TestDualStackTCPListener(t *testing.T) {
   172		if testing.Short() {
   173			t.Skip("skipping in -short mode, see issue 5001")
   174		}
   175		switch runtime.GOOS {
   176		case "plan9":
   177			t.Skipf("skipping test on %q", runtime.GOOS)
   178		}
   179		if !supportsIPv6 {
   180			t.Skip("ipv6 is not supported")
   181		}
   182	
   183		for _, tt := range dualStackListenerTests {
   184			if tt.wildcard && !*testExternal {
   185				continue
   186			}
   187			switch runtime.GOOS {
   188			case "openbsd":
   189				if tt.wildcard && differentWildcardAddr(tt.laddr1, tt.laddr2) {
   190					tt.xerr = nil
   191				}
   192			}
   193			l1, port := usableListenPort(t, tt.net1, tt.laddr1)
   194			laddr := tt.laddr1 + ":" + port
   195			checkFirstListener(t, tt.net1, laddr, l1)
   196			laddr = tt.laddr2 + ":" + port
   197			l2, err := Listen(tt.net2, laddr)
   198			checkDualStackSecondListener(t, tt.net2, laddr, tt.xerr, err, l2)
   199			l1.Close()
   200		}
   201	}

   249	func usableListenPort(t *testing.T, net, laddr string) (l Listener, port string) {
   250		var nladdr string
   251		var err error
   252		switch net {
   253		default:
   254			panic("usableListenPort net=" + net)
   255		case "tcp", "tcp4", "tcp6":
   256			l, err = Listen(net, laddr+":0")
   257			if err != nil {
   258				t.Fatalf("Probe Listen(%q, %q) failed: %v", net, laddr, err)
   259			}
   260			nladdr = l.(*TCPListener).Addr().String()
   261		}
   262		_, port, err = SplitHostPort(nladdr)
   263		if err != nil {
   264			t.Fatalf("SplitHostPort failed: %v", err)
   265		}
   266		return l, port
   267	}

   289	func differentWildcardAddr(i, j string) bool {
   290		if (i == "" || i == "0.0.0.0" || i == "::ffff:0.0.0.0") && (j == "" || j == "0.0.0.0" || j == "::ffff:0.0.0.0") {
   291			return false
   292		}
   293		if i == "[::]" && j == "[::]" {
   294			return false
   295		}
   296		return true
   297	}

   299	func checkFirstListener(t *testing.T, net, laddr string, l interface{}) {
   300		switch net {
   301		case "tcp":
   302			fd := l.(*TCPListener).fd
   303			checkDualStackAddrFamily(t, net, laddr, fd)
   304		case "tcp4":
   305			fd := l.(*TCPListener).fd
   306			if fd.family != syscall.AF_INET {
   307				t.Fatalf("First Listen(%q, %q) returns address family %v, expected %v", net, laddr, fd.family, syscall.AF_INET)
   308			}
   309		case "tcp6":
   310			fd := l.(*TCPListener).fd
   311			if fd.family != syscall.AF_INET6 {
   312				t.Fatalf("First Listen(%q, %q) returns address family %v, expected %v", net, laddr, fd.family, syscall.AF_INET6)
   313			}
   314		case "udp":
   315			fd := l.(*UDPConn).fd
   316			checkDualStackAddrFamily(t, net, laddr, fd)
   317		case "udp4":
   318			fd := l.(*UDPConn).fd
   319			if fd.family != syscall.AF_INET {
   320				t.Fatalf("First ListenPacket(%q, %q) returns address family %v, expected %v", net, laddr, fd.family, syscall.AF_INET)
   321			}
   322		case "udp6":
   323			fd := l.(*UDPConn).fd
   324			if fd.family != syscall.AF_INET6 {
   325				t.Fatalf("First ListenPacket(%q, %q) returns address family %v, expected %v", net, laddr, fd.family, syscall.AF_INET6)
   326			}
   327		default:
   328			t.Fatalf("Unexpected network: %q", net)
   329		}
   330	}

   370	func checkDualStackAddrFamily(t *testing.T, net, laddr string, fd *netFD) {
   371		switch a := fd.laddr.(type) {
   372		case *TCPAddr:
   373			// If a node under test supports both IPv6 capability
   374			// and IPv6 IPv4-mapping capability, we can assume
   375			// that the node listens on a wildcard address with an
   376			// AF_INET6 socket.
   377			if supportsIPv4map && fd.laddr.(*TCPAddr).isWildcard() {
   378				if fd.family != syscall.AF_INET6 {
   379					t.Fatalf("Listen(%q, %q) returns address family %v, expected %v", net, laddr, fd.family, syscall.AF_INET6)
   380				}
   381			} else {
   382				if fd.family != a.family() {
   383					t.Fatalf("Listen(%q, %q) returns address family %v, expected %v", net, laddr, fd.family, a.family())
   384				}
   385			}
   386		case *UDPAddr:
   387			// If a node under test supports both IPv6 capability
   388			// and IPv6 IPv4-mapping capability, we can assume
   389			// that the node listens on a wildcard address with an
   390			// AF_INET6 socket.
   391			if supportsIPv4map && fd.laddr.(*UDPAddr).isWildcard() {
   392				if fd.family != syscall.AF_INET6 {
   393					t.Fatalf("ListenPacket(%q, %q) returns address family %v, expected %v", net, laddr, fd.family, syscall.AF_INET6)
   394				}
   395			} else {
   396				if fd.family != a.family() {
   397					t.Fatalf("ListenPacket(%q, %q) returns address family %v, expected %v", net, laddr, fd.family, a.family())
   398				}
   399			}
   400		default:
   401			t.Fatalf("Unexpected protocol address type: %T", a)
   402		}
   403	}
