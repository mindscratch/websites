     7	import (
     8		"fmt"
     9		"path"

    11	
    12	func ExampleBase() {
    13		fmt.Println(path.Base("/a/b"))
    14		// Output: b
    15	}
    16	
    17	func ExampleClean() {
    18		paths := []string{
    19			"a/c",
    20			"a//c",
    21			"a/c/.",
    22			"a/c/b/..",
    23			"/../a/c",
    24			"/../a/b/../././/c",
    25		}
    26	
    27		for _, p := range paths {
    28			fmt.Printf("Clean(%q) = %q\n", p, path.Clean(p))
    29		}
    30	
    31		// Output:
    32		// Clean("a/c") = "a/c"
    33		// Clean("a//c") = "a/c"
    34		// Clean("a/c/.") = "a/c"
    35		// Clean("a/c/b/..") = "a/c"
    36		// Clean("/../a/c") = "/a/c"
    37		// Clean("/../a/b/../././/c") = "/a/c"
    38	}
    39	
    40	func ExampleDir() {
