    19	func eq(a, b []string) bool {
    20		if len(a) != len(b) {
    21			return false
    22		}
    23		for i := 0; i < len(a); i++ {
    24			if a[i] != b[i] {
    25				return false
    26			}
    27		}
    28		return true
    29	}

   228	func TestSplit(t *testing.T) {
   229		for _, tt := range splittests {
   230			a := SplitN(tt.s, tt.sep, tt.n)
   231			if !eq(a, tt.a) {
   232				t.Errorf("Split(%q, %q, %d) = %v; want %v", tt.s, tt.sep, tt.n, a, tt.a)
   233				continue
   234			}
   235			if tt.n == 0 {
   236				continue
   237			}
   238			s := Join(a, tt.sep)
   239			if s != tt.s {
   240				t.Errorf("Join(Split(%q, %q, %d), %q) = %q", tt.s, tt.sep, tt.n, tt.sep, s)
   241			}
   242			if tt.n < 0 {
   243				b := Split(tt.s, tt.sep)
   244				if !reflect.DeepEqual(a, b) {
   245					t.Errorf("Split disagrees with SplitN(%q, %q, %d) = %v; want %v", tt.s, tt.sep, tt.n, b, a)
   246				}
   247			}
   248		}

   393	func tenRunes(ch rune) string {
   394		r := make([]rune, 10)
   395		for i := range r {
   396			r[i] = ch
   397		}
   398		return string(r)
   399	}

   401	// User-defined self-inverse mapping function
   402	func rot13(r rune) rune {
   403		step := rune(13)
   404		if r >= 'a' && r <= 'z' {
   405			return ((r - 'a' + step) % 26) + 'a'
   406		}
   407		if r >= 'A' && r <= 'Z' {
   408			return ((r - 'A' + step) % 26) + 'A'
   409		}
   410		return r
   411	}

   413	func TestMap(t *testing.T) {
   414		// Run a couple of awful growth/shrinkage tests
   415		a := tenRunes('a')
   416		// 1.  Grow.  This triggers two reallocations in Map.
   417		maxRune := func(rune) rune { return unicode.MaxRune }
   418		m := Map(maxRune, a)
   419		expect := tenRunes(unicode.MaxRune)
   420		if m != expect {
   421			t.Errorf("growing: expected %q got %q", expect, m)
   422		}
   423	
   424		// 2. Shrink
   425		minRune := func(rune) rune { return 'a' }
   426		m = Map(minRune, tenRunes(unicode.MaxRune))
   427		expect = a
   428		if m != expect {
   429			t.Errorf("shrinking: expected %q got %q", expect, m)
   430		}
   431	
   432		// 3. Rot13
   433		m = Map(rot13, "a to zed")
   434		expect = "n gb mrq"
   435		if m != expect {
   436			t.Errorf("rot13: expected %q got %q", expect, m)
   437		}
   438	
   439		// 4. Rot13^2
   440		m = Map(rot13, Map(rot13, "a to zed"))
   441		expect = "a to zed"
   442		if m != expect {
   443			t.Errorf("rot13: expected %q got %q", expect, m)
   444		}
   445	
   446		// 5. Drop
   447		dropNotLatin := func(r rune) rune {
   448			if unicode.Is(unicode.Latin, r) {
   449				return r
   450			}
   451			return -1
   452		}
   453		m = Map(dropNotLatin, "Hello, 세계")
   454		expect = "Hello"
   455		if m != expect {
   456			t.Errorf("drop: expected %q got %q", expect, m)
   457		}
   458	
   459		// 6. Identity
   460		identity := func(r rune) rune {
   461			return r
   462		}
   463		orig := "Input string that we expect not to be copied."
   464		m = Map(identity, orig)
   465		if (*reflect.StringHeader)(unsafe.Pointer(&orig)).Data !=
   466			(*reflect.StringHeader)(unsafe.Pointer(&m)).Data {
   467			t.Error("unexpected copy during identity map")
   468		}
   469	}

   648	func equal(m string, s1, s2 string, t *testing.T) bool {
   649		if s1 == s2 {
   650			return true
   651		}
   652		e1 := Split(s1, "")
   653		e2 := Split(s2, "")
   654		for i, c1 := range e1 {
   655			if i >= len(e2) {
   656				break
   657			}
   658			r1, _ := utf8.DecodeRuneInString(c1)
   659			r2, _ := utf8.DecodeRuneInString(e2[i])
   660			if r1 != r2 {
   661				t.Errorf("%s diff at %d: U+%04X U+%04X", m, i, r1, r2)
   662			}
   663		}
   664		return false
   665	}

   667	func TestCaseConsistency(t *testing.T) {
   668		// Make a string of all the runes.
   669		numRunes := int(unicode.MaxRune + 1)
   670		if testing.Short() {
   671			numRunes = 1000
   672		}
   673		a := make([]rune, numRunes)
   674		for i := range a {
   675			a[i] = rune(i)
   676		}
   677		s := string(a)
   678		// convert the cases.
   679		upper := ToUpper(s)
   680		lower := ToLower(s)
   681	
   682		// Consistency checks
   683		if n := utf8.RuneCountInString(upper); n != numRunes {
   684			t.Error("rune count wrong in upper:", n)
   685		}
   686		if n := utf8.RuneCountInString(lower); n != numRunes {
   687			t.Error("rune count wrong in lower:", n)
   688		}
   689		if !equal("ToUpper(upper)", ToUpper(upper), upper, t) {
   690			t.Error("ToUpper(upper) consistency fail")
   691		}
   692		if !equal("ToLower(lower)", ToLower(lower), lower, t) {
   693			t.Error("ToLower(lower) consistency fail")
   694		}
   695		/*
   696			  These fail because of non-one-to-oneness of the data, such as multiple
   697			  upper case 'I' mapping to 'i'.  We comment them out but keep them for
   698			  interest.
   699			  For instance: CAPITAL LETTER I WITH DOT ABOVE:
   700				unicode.ToUpper(unicode.ToLower('\u0130')) != '\u0130'
   701	
   702			if !equal("ToUpper(lower)", ToUpper(lower), upper, t) {
   703				t.Error("ToUpper(lower) consistency fail");
   704			}
   705			if !equal("ToLower(upper)", ToLower(upper), lower, t) {
   706				t.Error("ToLower(upper) consistency fail");
   707			}
   708		*/
   709	}

   783			}
   784			var res bytes.Buffer
   785			for {
   786				b, e := reader.ReadByte()

   805				}
   806				if b1 != b {
   807					t.Errorf("Reading %q after unreading: want byte %q, got %q", s, b, b1)
   808					break
   809				}
   810			}
   811			if res.String() != s {
   812				t.Errorf("Reader(%q).ReadByte() produced %q", s, res.String())
