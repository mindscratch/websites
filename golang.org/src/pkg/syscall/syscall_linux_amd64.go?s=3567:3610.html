    64	func Gettimeofday(tv *Timeval) (err error) {
    65		errno := gettimeofday(tv)
    66		if errno != 0 {
    67			return errno
    68		}
    69		return nil
    70	}
    71	
    72	func Time(t *Time_t) (tt Time_t, err error) {
    73		var tv Timeval
    74		errno := gettimeofday(&tv)
    75		if errno != 0 {
    76			return 0, errno
    77		}
    78		if t != nil {
    79			*t = Time_t(tv.Sec)
    80		}
    81		return Time_t(tv.Sec), nil
    82	}

    95		nsec += 999 // round up to microsecond
